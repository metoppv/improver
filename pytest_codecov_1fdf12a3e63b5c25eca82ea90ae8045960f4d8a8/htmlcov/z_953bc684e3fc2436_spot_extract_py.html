11@cli.clizefy 

12@cli.with_output 

13def process( 

14 *cubes: cli.inputcube, 

15 apply_lapse_rate_correction: bool = False, 

16 fixed_lapse_rate: float = None, 

17 land_constraint: bool = False, 

18 similar_altitude: bool = False, 

19 extract_percentiles: cli.comma_separated_list = None, 

20 ignore_ecc_bounds_exceedance: bool = False, 

21 skip_ecc_bounds: bool = False, 

22 new_title: str = None, 

23 suppress_warnings: bool = False, 

24 realization_collapse: bool = False, 

25 subset_coord: str = None, 

26): 

27 """Module to run spot data extraction. 

28 

29 Extract diagnostic data from gridded fields for spot data sites. It is 

30 possible to apply a temperature lapse rate adjustment to temperature data 

31 that helps to account for differences between the spot site's real altitude 

32 and that of the grid point from which the temperature data is extracted. 

33 

34 Args: 

35 cubes (iris.cube.Cube): 

36 A list of cubes containing the diagnostic data to be extracted, 

37 the lapse rate (optional) and the neighbour cube. Where the lapse 

38 rate cube contains temperature lapse rates. If this cube is 

39 provided and a screen temperature cube is being processed, the 

40 lapse rates will be used to adjust the temperature to better 

41 represent each spot's site-altitude. 

42 And the neighbour cube is a cube of spot-data neighbours and 

43 the spot site information. 

44 apply_lapse_rate_correction (bool): 

45 Use to apply a lapse-rate correction to screen temperature 

46 forecasts so that they better represent the altitude of the 

47 spot site for which they have been extracted. This lapse rate 

48 will be applied for a fixed orographic difference between the 

49 site and grid point altitude. Differences in orography in 

50 excess of this fixed limit will use the Environmental Lapse 

51 Rate (also known as the Standard Atmosphere Lapse Rate). 

52 Lapse rate adjustment cannot be applied to existing spot 

53 forecasts that are passed in for subsetting. 

54 fixed_lapse_rate (float): 

55 If provided, use this fixed value as a lapse-rate for adjusting 

56 the forecast values if apply_lapse_rate_correction is True. This 

57 can be used instead of providing a lapse rate cube. Value is 

58 given in Kelvin / metre of temperature change with ascent. For 

59 example a dry adiabatic lapse rate would be given as -0.0098. 

60 This lapse rate will be applied for a fixed orographic difference 

61 between the site and gridpoint altitude. Differences in orography 

62 in excess of this fixed limit will use the Environmental Lapse 

63 Rate (also known as the Standard Atmosphere Lapse Rate). 

64 land_constraint (bool): 

65 Use to select the nearest-with-land-constraint neighbour-selection 

66 method from the neighbour_cube. This means that the grid points 

67 should be land points except for sites where none were found within 

68 the search radius when the neighbour cube was created. May be used 

69 with similar_altitude. 

70 similar_altitude (bool): 

71 Use to select the nearest-with-height-constraint 

72 neighbour-selection method from the neighbour_cube. These are grid 

73 points that were found to be the closest in altitude to the spot 

74 site within the search radius defined when the neighbour cube was 

75 created. May be used with land_constraint. 

76 extract_percentiles (list or int): 

77 If set to a percentile value or a list of percentile values, 

78 data corresponding to those percentiles will be returned. For 

79 example "25, 50, 75" will result in the 25th, 50th and 75th 

80 percentiles being returned from a cube of probabilities, 

81 percentiles or realizations. Deterministic input data will raise 

82 a warning message. 

83 Note that for percentile inputs, if the desired percentile(s) do 

84 not exist in the input cube the available percentiles will be 

85 resampled to produce those requested. 

86 ignore_ecc_bounds_exceedance (bool): 

87 Demotes exceptions where calculated percentiles are outside the ECC 

88 bounds range to warnings. 

89 skip_ecc_bounds (bool): 

90 If True, ECC bounds are not included when converting probabilities to 

91 percentiles or from one set of percentiles to another. This has the 

92 effect that percentiles outside of the range given by the input 

93 percentiles will be computed by nearest neighbour interpolation from 

94 the nearest available percentile, rather than using linear 

95 interpolation between the nearest available percentile and the ECC 

96 bound. 

97 new_title (str): 

98 New title for the spot-extracted data. If None, this attribute is 

99 removed from the output cube since it has no prescribed standard 

100 and may therefore contain grid information that is no longer 

101 correct after spot-extraction. 

102 suppress_warnings (bool): 

103 Suppress warning output. This option should only be used if it 

104 is known that warnings will be generated but they are not required. 

105 realization_collapse (bool): 

106 Triggers equal-weighting blending of the realization coord if required. 

107 Use this if a threshold coord is also present on the input cube. 

108 subset_coord (str): 

109 If a spot cube is provided as input this plugin can return a subset of 

110 the sites based on the sites specified in the neighbour cube. To 

111 achieve this the plugin needs the name of the site ID coordinate to be 

112 used for matching, e.g. wmo_id. If subset_coord is not provided, and a 

113 spot forecast is passed in, the entire spot cube will be processed and 

114 returned. The neighbour selection method options have no impact if a 

115 spot cube is passed in. 

116 

117 Returns: 

118 iris.cube.Cube: 

119 Cube of spot data. 

120 """ 

121 from improver.spotdata.spot_manipulation import SpotManipulation 

122 

123 return SpotManipulation( 

124 apply_lapse_rate_correction=apply_lapse_rate_correction, 

125 fixed_lapse_rate=fixed_lapse_rate, 

126 land_constraint=land_constraint, 

127 similar_altitude=similar_altitude, 

128 extract_percentiles=extract_percentiles, 

129 ignore_ecc_bounds_exceedance=ignore_ecc_bounds_exceedance, 

130 skip_ecc_bounds=skip_ecc_bounds, 

131 new_title=new_title, 

132 suppress_warnings=suppress_warnings, 

133 realization_collapse=realization_collapse, 

134 subset_coord=subset_coord, 

135 )(cubes)
