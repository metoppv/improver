6"""CLI to estimate the Generalized Additive Model (GAM) for Standardized Anomaly Model 

7Output Statistics (SAMOS).""" 

8 

9from improver import cli 

10 

11 

12@cli.clizefy 

13@cli.with_output 

14def process( 

15 *cubes: cli.inputcube, 

16 truth_attribute: str, 

17 gam_features: cli.comma_separated_list, 

18 model_specification: cli.inputjson, 

19 max_iterations: int = 100, 

20 tolerance: float = 0.0001, 

21 distribution: str = "normal", 

22 link: str = "identity", 

23 fit_intercept: bool = True, 

24 window_length: int = 11, 

25 unique_site_id_key: str = "wmo_id", 

26): 

27 """Estimate Generalized Additive Model (GAM) for SAMOS. 

28 

29 Args: 

30 cubes (list of iris.cube.Cube): 

31 A list of cubes containing the historical forecasts and 

32 corresponding truth used for calibration. They must have the same 

33 cube name and will be separated based on the truth attribute. The 

34 list may also contain additional features (static predictors) that 

35 will be provided when estimating the GAM. 

36 truth_attribute (str): 

37 An attribute and its value in the format of "attribute=value", 

38 which must be present on historical truth cubes. 

39 gam_features (list of str): 

40 A list of the names of the cubes that will be used as additional 

41 features in the GAM. 

42 model_specification (dict): 

43 A list containing three items (in order): 

44 1. a string containing a single pyGAM term; one of 'l' (linear), 

45 's' (spline), 'te' (tensor), or 'f' (factor) 

46 2. a list of integers which correspond to the features to be 

47 included in that term 

48 3. a dictionary of kwargs to be included when defining the term 

49 max_iterations (int): 

50 The maximum number of iterations to use when estimating the GAM 

51 coefficients. 

52 tolerance (float): 

53 The tolerance for the stopping criteria. 

54 distribution (str): 

55 The distribution to be used in the model. Valid options are normal, binomial, 

56 poisson, gamma, inv-gauss. 

57 link (str): 

58 The link function to be used in the model. Valid options are identity, logit, inverse, log 

59 or inverse-squared. 

60 fit_intercept (bool): 

61 Whether to include an intercept term in the model. Default is True. 

62 window_length (int): 

63 This must be an odd integer greater than 1. The length of the rolling 

64 window used to calculate the mean and standard deviation of the input cube 

65 when the input cube does not have a realization dimension coordinate. If a 

66 given window has fewer than half valid data points (not NaN) then the value 

67 returned for that window will be NaN and will be excluded from training. 

68 unique_site_id_key (str): 

69 If working with spot data and available, the name of the coordinate 

70 in the input cubes that contains unique site IDs, e.g. "wmo_id" if 

71 all sites have a valid wmo_id. For GAM estimation the default is 

72 "wmo_id" as we expect to have a training data set comprising matched 

73 obs and forecast sites. 

74 

75 Returns: 

76 List: 

77 A list containing the fitted GAMs for the forecast and truth cubes in that order. 

78 """ 

79 

80 from improver.calibration import split_cubes_for_samos 

81 from improver.calibration.samos_calibration import TrainGAMsForSAMOS 

82 

83 # Split the cubes into forecast and truth cubes, along with any additional fields 

84 # provided for the GAMs. 

85 ( 

86 forecast, 

87 truth, 

88 gam_additional_fields, 

89 _, 

90 _, 

91 _, 

92 ) = split_cubes_for_samos( 

93 cubes=cubes, 

94 gam_features=gam_features, 

95 truth_attribute=truth_attribute, 

96 expect_emos_coeffs=False, 

97 expect_emos_fields=False, 

98 ) 

99 

100 if forecast is None or truth is None: 

101 return 

102 

103 plugin = TrainGAMsForSAMOS( 

104 model_specification=model_specification, 

105 max_iter=max_iterations, 

106 tol=tolerance, 

107 distribution=distribution, 

108 link=link, 

109 fit_intercept=fit_intercept, 

110 window_length=window_length, 

111 unique_site_id_key=unique_site_id_key, 

112 ) 

113 

114 truth_gams = plugin.process( 

115 input_cube=truth, 

116 features=gam_features, 

117 additional_fields=gam_additional_fields, 

118 ) 

119 

120 forecast_gams = plugin.process( 

121 input_cube=forecast, 

122 features=gam_features, 

123 additional_fields=gam_additional_fields, 

124 ) 

125 

126 if forecast_gams is None or truth_gams is None: 

127 return 

128 

129 return [forecast_gams, truth_gams]
