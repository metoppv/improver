6"""CLI to enforce the sum total of a set of forecasts to be equal to a reference.""" 

7 

8from improver import cli 

9 

10 

11@cli.clizefy 

12@cli.with_output 

13def process( 

14 *cubes: cli.inputcubelist, 

15 reference_name: str, 

16 return_name: str, 

17 ignore_zero_total: bool = False, 

18): 

19 """Module to enforce that the sum of data in a list of cubes is equal to the 

20 corresponding data in a reference cube. Only one of the updated cubes is returned. 

21 

22 The data are updated as follows, if there are 2 cubes to be normalised containing 

23 data points a and b respectively with corresponding reference r, then: 

24 

25 a_new = r * (a / (a + b)) 

26 b_new = r * (b / (a + b)) 

27 

28 which ensures that r = a_new + b_new. 

29 

30 Args: 

31 cubes (List of iris.cube.Cube): A list of cubes containing both the cubes to be 

32 updated and the reference cube. The reference cube will be identified by 

33 matching the cube name to reference_name. 

34 reference_name (str): The name of the reference cube, this must match exactly 

35 one cube in cubes. 

36 return_name (str): The name of the cube to be returned. this must match exactly 

37 one cube in cubes. 

38 ignore_zero_total (bool): Determines whether an error will be raised in the 

39 instance where the total of non-reference cubes is zero but the 

40 corresponding value in reference is non-zero. If True this error will not be 

41 raised - instead the data in the updated cubes will remain as zero values, 

42 if False an error will be raised. 

43 

44 Returns: 

45 iris.cube.Cube: 

46 The cube in cubes with name matching return_name. The data in the cube will 

47 have been updated, but the metadata will be identical. 

48 

49 """ 

50 from improver.utilities.flatten import flatten 

51 from improver.utilities.forecast_reference_enforcement import ( 

52 normalise_to_reference, 

53 split_cubes_by_name, 

54 ) 

55 

56 cubes = flatten(cubes) 

57 

58 reference_cube, input_cubes = split_cubes_by_name(cubes, reference_name) 

59 

60 if len(reference_cube) == 1: 

61 reference_cube = reference_cube[0] 

62 else: 

63 msg = ( 

64 f"Exactly one cube with a name matching reference_name is required, but " 

65 f"{len(reference_cube)} were found." 

66 ) 

67 raise ValueError(msg) 

68 

69 output = normalise_to_reference(input_cubes, reference_cube, ignore_zero_total) 

70 

71 output = output.extract_cube(return_name) 

72 

73 return output
