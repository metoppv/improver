19class WindDirection(PostProcessingPlugin): 

20 """Plugin to calculate average wind direction from ensemble realizations. 

21 

22 Science background: 

23 Taking an average wind direction is tricky since an average of two wind 

24 directions at 10 and 350 degrees is 180 when it should be 0 degrees. 

25 Converting the wind direction angles to complex numbers allows us to 

26 find a useful numerical average. :: 

27 

28 z = a + bi 

29 a = r * Cos(theta) 

30 b = r * Sin(theta) 

31 r = radius 

32 

33 The average of two complex numbers is NOT the ANGLE between two points 

34 it is the MIDPOINT in cartesian space. 

35 Therefore if there are two data points with radius=1 at 90 and 270 degrees 

36 then the midpoint is at (0,0) with radius=0 and therefore its average angle 

37 is meaningless. :: 

38 

39 N 

40 | 

41 W---x------o------x---E 

42 | 

43 S 

44 

45 In the rare case that a meaningless complex average is calculated, the 

46 code rejects the calculated complex average and simply uses the wind 

47 direction taken from the first ensemble realization. 

48 

49 The steps are: 

50 

51 1) Take data from all ensemble realizations. 

52 2) Convert the wind direction angles to complex numbers. 

53 3) Find complex average and their radius values. 

54 4) Convert the complex average back into degrees. 

55 5) If any point has an radius of nearly zero - replace the 

56 calculated average with the wind direction from the first ensemble. 

57 

58 Args: 

59 backup_method: 

60 Backup method to use if the complex numbers approach has low 

61 confidence. 

62 "first_realization" uses the value of realization zero. 

63 "neighbourhood" (default) recalculates using the complex numbers 

64 approach with additional realizations extracted from neighbouring 

65 grid points from all available realizations. 

66 """ 

67 

68 def __init__(self, backup_method: str = "neighbourhood") -> None: 

69 """Initialise class.""" 

70 self.backup_methods = ["first_realization", "neighbourhood"] 

71 self.backup_method = backup_method 

72 if self.backup_method not in self.backup_methods: 

73 msg = "Invalid option for keyword backup_method ({})".format( 

74 self.backup_method 

75 ) 

76 raise ValueError(msg) 

77 

78 # Any points where the r-values are below the threshold is regarded as 

79 # containing ambigous data. 

80 self.r_thresh = 0.01 

81 

82 # Creates cubelists to hold data. 

83 self.wdir_cube_list = iris.cube.CubeList() 

84 self.r_vals_cube_list = iris.cube.CubeList() 

85 # Radius used in neighbourhood plugin as determined in IMPRO-491 

86 self.nb_radius = 6000.0 # metres 

87 # Initialise neighbourhood plugin ready for use 

88 self.nbhood = NeighbourhoodProcessing( 

89 "square", self.nb_radius, weighted_mode=False 

90 ) 

91 

92 def __repr__(self) -> str: 

93 """Represent the configured plugin instance as a string.""" 

94 return ( 

95 '<WindDirection: backup_method "{}"; neighbourhood radius "{}"m>' 

96 ).format(self.backup_method, self.nb_radius) 

97 

98 def _reset(self) -> None: 

99 """Empties working data objects""" 

100 self.realization_axis = None 

101 self.wdir_complex = None 

102 self.wdir_slice_mean = None 

103 self.wdir_mean_complex = None 

104 self.r_vals_slice = None 

105 

106 def calc_wind_dir_mean(self) -> None: 

107 """Find the mean wind direction using complex average which actually 

108 signifies a point between all of the data points in POLAR 

109 coordinates - NOT the average DEGREE ANGLE. 

110 

111 Uses: 

112 self.wdir_complex: 

113 3D array or float - wind direction angles in degrees. 

114 self.realization_axis: 

115 Axis to collapse over. 

116 

117 Defines: 

118 self.wdir_mean_complex: 

119 3D array or float - wind direction angles as complex numbers 

120 collapsed along an axis using np.mean(). 

121 self.wdir_slice_mean: 

122 3D array or float - wind direction angles in degrees collapsed 

123 along an axis using np.mean(). 

124 """ 

125 self.wdir_mean_complex = np.mean(self.wdir_complex, axis=self.realization_axis) 

126 self.wdir_slice_mean.data = complex_to_deg(self.wdir_mean_complex) 

127 

128 def find_r_values(self) -> None: 

129 """Find radius values from complex numbers. 

130 

131 Takes input wind direction in complex values and returns array 

132 containing r values using Pythagoras theorem. 

133 

134 Uses: 

135 self.wdir_mean_complex: 

136 3D array or float - wind direction angles in complex numbers. 

137 self.wdir_slice_mean: 

138 3D array or float - mean wind direction angles in complex 

139 numbers. 

140 

141 Defines: 

142 self.r_vals_slice: 

143 Contains r values and inherits meta-data from 

144 self.wdir_slice_mean. 

145 """ 

146 

147 r_vals = np.sqrt( 

148 np.square(self.wdir_mean_complex.real) 

149 + np.square(self.wdir_mean_complex.imag) 

150 ) 

151 self.r_vals_slice = self.wdir_slice_mean.copy(data=r_vals) 

152 

153 def wind_dir_decider(self, where_low_r: ndarray, wdir_cube: Cube) -> None: 

154 """If the wind direction is so widely scattered that the r value 

155 is nearly zero then this indicates that the average wind direction 

156 is essentially meaningless. 

157 We therefore substitute this meaningless average wind 

158 direction value for the wind direction calculated from a larger 

159 sample by smoothing across a neighbourhood of points before 

160 rerunning the main technique. 

161 This is invoked rarely (1 : 100 000) 

162 

163 Args: 

164 where_low_r: 

165 Array of boolean values. True where original wind direction 

166 estimate has low confidence. These points are replaced 

167 according to self.backup_method 

168 wdir_cube: 

169 Contains array of wind direction data (realization, y, x) 

170 

171 Uses: 

172 self.wdir_slice_mean: 

173 Containing average wind direction angle (in degrees). 

174 self.wdir_complex: 

175 3D array - wind direction angles from ensembles (in complex). 

176 self.r_vals_slice.data: 

177 2D array - Radius taken from average complex wind direction 

178 angle. 

179 self.r_thresh: 

180 Any r value below threshold is regarded as meaningless. 

181 self.realization_axis: 

182 Axis to collapse over. 

183 self.n_realizations: 

184 Number of realizations available in the plugin. Used to set the 

185 neighbourhood radius as this is used to adjust the radius again 

186 in the neighbourhooding plugin. 

187 

188 Defines: 

189 self.wdir_slice_mean.data: 

190 2D array - Wind direction degrees where ambigious values have 

191 been replaced with data from first ensemble realization. 

192 """ 

193 if self.backup_method == "neighbourhood": 

194 # Performs smoothing over a 6km square neighbourhood. 

195 # Then calculates the mean wind direction. 

196 child_class = WindDirection(backup_method="first_realization") 

197 child_class.wdir_complex = self.nbhood( 

198 wdir_cube.copy(data=self.wdir_complex) 

199 ).data 

200 child_class.realization_axis = self.realization_axis 

201 child_class.wdir_slice_mean = self.wdir_slice_mean.copy() 

202 child_class.calc_wind_dir_mean() 

203 improved_values = child_class.wdir_slice_mean.data 

204 else: 

205 # Takes realization zero (control member). 

206 improved_values = wdir_cube.extract(iris.Constraint(realization=0)).data 

207 

208 # If the r-value is low - substitute average wind direction value for 

209 # the wind direction taken from the first ensemble realization. 

210 self.wdir_slice_mean.data = np.where( 

211 where_low_r, improved_values, self.wdir_slice_mean.data 

212 ) 

213 

214 def process(self, cube_ens_wdir: Cube) -> Cube: 

215 """Create a cube containing the wind direction averaged over the 

216 ensemble realizations. 

217 

218 Args: 

219 cube_ens_wdir: 

220 Cube containing wind direction from multiple ensemble 

221 realizations. 

222 

223 Returns: 

224 - Cube containing the wind direction averaged from the 

225 ensemble realizations. 

226 

227 Raises: 

228 TypeError: If cube_wdir is not a cube. 

229 """ 

230 

231 if not isinstance(cube_ens_wdir, iris.cube.Cube): 

232 msg = "Wind direction input is not a cube, but {}" 

233 raise TypeError(msg.format(type(cube_ens_wdir))) 

234 

235 try: 

236 cube_ens_wdir.convert_units("degrees") 

237 except ValueError as err: 

238 msg = "Input cube cannot be converted to degrees: {}".format(err) 

239 raise ValueError(msg) 

240 

241 self.n_realizations = len(cube_ens_wdir.coord("realization").points) 

242 y_coord_name = cube_ens_wdir.coord(axis="y").name() 

243 x_coord_name = cube_ens_wdir.coord(axis="x").name() 

244 for wdir_slice in cube_ens_wdir.slices( 

245 ["realization", y_coord_name, x_coord_name] 

246 ): 

247 self._reset() 

248 # Extract wind direction data. 

249 self.wdir_complex = deg_to_complex(wdir_slice.data) 

250 (self.realization_axis,) = wdir_slice.coord_dims("realization") 

251 # Copies input cube and remove realization dimension to create 

252 # cubes for storing results. 

253 self.wdir_slice_mean = next(wdir_slice.slices_over("realization")) 

254 self.wdir_slice_mean.remove_coord("realization") 

255 

256 # Derive average wind direction. 

257 self.calc_wind_dir_mean() 

258 

259 # Find radius values for wind direction average. 

260 self.find_r_values() 

261 

262 # Finds any meaningless averages and substitute with 

263 # the wind direction taken from the first ensemble realization. 

264 # Mask True if r values below threshold. 

265 where_low_r = np.where(self.r_vals_slice.data < self.r_thresh, True, False) 

266 # If the any point in the array contains poor r-values, 

267 # trigger decider function. 

268 if where_low_r.any(): 

269 self.wind_dir_decider(where_low_r, wdir_slice) 

270 

271 # Append to cubelists. 

272 self.wdir_cube_list.append(self.wdir_slice_mean) 

273 

274 # Combine cubelists into cube. 

275 cube_mean_wdir = self.wdir_cube_list.merge_cube() 

276 

277 # Check that the dimensionality of coordinates of the output cube 

278 # matches the input cube. 

279 first_slice = next(cube_ens_wdir.slices_over(["realization"])) 

280 cube_mean_wdir = check_cube_coordinates(first_slice, cube_mean_wdir) 

281 

282 # Change cube identifiers. 

283 cube_mean_wdir.add_cell_method(CellMethod("mean", coords="realization")) 

284 return cube_mean_wdir
