18def nearest_with_mask_regrid( 

19 distances: ndarray, 

20 indexes: ndarray, 

21 surface_type_mask: ndarray, 

22 in_latlons: ndarray, 

23 out_latlons: ndarray, 

24 in_classified: ndarray, 

25 out_classified: ndarray, 

26 vicinity: float, 

27) -> Tuple[ndarray, ndarray]: 

28 """ 

29 Main regridding function for the nearest distance option. 

30 some input just for handling island-like points. 

31 

32 Args: 

33 distances: 

34 Distnace array from each target grid point to its source grid points. 

35 indexes: 

36 Source grid point indexes for each target grid point. 

37 surface_type_mask: 

38 Boolean true if source point type matches target point type. 

39 in_latlons: 

40 Source points's latitude-longitudes. 

41 out_latlons: 

42 Target points's latitude-longitudes. 

43 in_classified: 

44 Land/sea type for source grid points (land -> True). 

45 out_classified: 

46 Land/sea type for target grid points (land -> True). 

47 vicinity: 

48 Radius of specified searching domain, in meter. 

49 

50 Returns: 

51 - Updated distances - array from each target grid point to its source grid points. 

52 - Updated indexes - source grid point number for all target grid points. 

53 

54 """ 

55 # Check if there are output points with mismatched surface types 

56 matched_nearby_points_count = np.count_nonzero(surface_type_mask, axis=1) 

57 points_with_mismatches = (np.where(matched_nearby_points_count < 4))[0] 

58 # Look for nearest input points for the output points with mismatched surface 

59 indexes, distances, surface_type_mask = update_nearest_points( 

60 points_with_mismatches, 

61 in_latlons, 

62 out_latlons, 

63 indexes, 

64 distances, 

65 surface_type_mask, 

66 in_classified, 

67 out_classified, 

68 ) 

69 

70 # Handle island and lake like output points - find more distant same surface type input points 

71 # Note: surface_type_mask has been updated above 

72 matched_nearby_points_count = np.count_nonzero(surface_type_mask, axis=1) 

73 fully_mismatched_points = (np.where(matched_nearby_points_count == 0))[0] 

74 

75 if fully_mismatched_points.shape[0] > 0: 

76 indexes, surface_type_mask = lakes_islands( 

77 fully_mismatched_points, 

78 indexes, 

79 surface_type_mask, 

80 in_latlons, 

81 out_latlons, 

82 in_classified, 

83 out_classified, 

84 vicinity, 

85 ) 

86 

87 # Convert mask to be true where input points should not be considered 

88 inverse_surface_mask = np.logical_not(surface_type_mask) 

89 

90 # Replace distances with infinity where they should not be used 

91 masked_distances = np.where(inverse_surface_mask, np.float64(np.inf), distances) 

92 

93 # Distances and indexes have been prepared to handle the mask, so can now 

94 # call the non-masked regrid function in process 

95 return masked_distances, indexes

98def nearest_regrid(distances: ndarray, indexes: ndarray, in_values: ndarray) -> ndarray: 

99 """ 

100 Main regridding function for the nearest neighbour option. 

101 

102 Args: 

103 distances: 

104 Distance from each target grid point to its source grid points. 

105 indexes: 

106 Source grid point indexes for each target grid point. 

107 in_values: 

108 Input values with spatial dimensions flattened. 

109 

110 Returns: 

111 Regridded output values with spatial dimensions flattened. 

112 """ 

113 min_index = np.argmin(distances, axis=1) 

114 index0 = np.arange(min_index.shape[0]) 

115 index_in = indexes[index0, min_index] 

116 output = in_values[index_in] 

117 return output

120def update_nearest_points( 

121 points_with_mismatches: ndarray, 

122 in_latlons: ndarray, 

123 out_latlons: ndarray, 

124 indexes: ndarray, 

125 distances: ndarray, 

126 surface_type_mask: ndarray, 

127 in_classified: ndarray, 

128 out_classified: ndarray, 

129) -> Tuple[ndarray, ndarray, ndarray]: 

130 """ 

131 Update nearest source points and distances/surface_type to take into account 

132 surface type of nearby points. 

133 

134 Args: 

135 points_with_mismatches: 

136 Selected target points which will use Inverse Distance Weighting 

137 (idw) approach. These points will be processed by this function. 

138 in_latlons: 

139 Source points's latitude-longitudes. 

140 out_latlons: 

141 Target points's latitude-longitudes. 

142 indexes: 

143 Source grid point indexes for each target grid point. 

144 distances: 

145 Distance from each target grid point to its source grid points. 

146 surface_type_mask: 

147 Boolean true if source point type matches target point type. 

148 in_classified: 

149 Land/sea type for source grid points (land -> True). 

150 out_classified: 

151 Land/sea type for target grid points (land -> True). 

152 

153 Returns: 

154 - Updated indexes - source grid point number for all target grid points. 

155 - Updated distances - array from each target grid point to its source grid points. 

156 - Updated surface_type_mask - matching info between source/target point types. 

157 """ 

158 # Gather output points with mismatched surface type and find four nearest input 

159 # points via KDtree 

160 out_latlons_with_mismatches = out_latlons[points_with_mismatches] 

161 k_nearest = 4 

162 distances_updates, indexes_updates = nearest_input_pts( 

163 in_latlons, out_latlons_with_mismatches, k_nearest 

164 ) 

165 # Calculate update to surface classification at mismatched points 

166 out_classified_with_mismatches = out_classified[points_with_mismatches] 

167 surface_type_mask_updates = similar_surface_classify( 

168 in_classified, out_classified_with_mismatches, indexes_updates 

169 ) 

170 # Apply updates to indexes, distances and surface type mask 

171 indexes[points_with_mismatches] = indexes_updates 

172 distances[points_with_mismatches] = distances_updates 

173 surface_type_mask[points_with_mismatches] = surface_type_mask_updates 

174 return indexes, distances, surface_type_mask

177def lakes_islands( 

178 lake_island_indexes: ndarray, 

179 indexes: ndarray, 

180 surface_type_mask: ndarray, 

181 in_latlons: ndarray, 

182 out_latlons: ndarray, 

183 in_classified: ndarray, 

184 out_classified: ndarray, 

185 vicinity: float, 

186) -> Tuple[ndarray, ndarray]: 

187 """ 

188 Updating source points and weighting for 4-unmatching-source-point 

189 cases - water surrounded by land or land surrounded by water. 

190 This function searches nearest 8 points to check if any matching point exists. 

191 Note that a similar function can be found in bilinear.py for bilinear 

192 regridding rather than nearest neighbour regridding. 

193 

194 Args: 

195 lake_island_indexes: 

196 Indexes of points which are lakes/islands surrounded by mismatched surface type. 

197 These points will be processed by this function. 

198 in_latlons: 

199 Source points's latitude-longitudes. 

200 out_latlons: 

201 Target points's latitude-longitudes. 

202 surface_type_mask: 

203 Boolean true if source point type matches target point type. 

204 indexes: 

205 Source grid point indexes for each target grid point. 

206 in_classified: 

207 Land/sea type for source grid points (land -> True). 

208 out_classified: 

209 Land/sea type for target grid points (land -> True). 

210 vicinity: 

211 Radius of vicinity to search for a matching surface type, in metres. 

212 

213 Returns: 

214 - Updated indexes - source grid point number for all target grid points. 

215 - Updated surface_type_mask - matching info between source/target point types. 

216 """ 

217 

218 out_latlons_updates = out_latlons[lake_island_indexes] 

219 

220 # Consider a larger area of 8 nearest points to look for more distant same 

221 # surface type input points. 

222 # more than 8 points are within searching limits not considered here 

223 k_nearest = 8 

224 distances_updates, indexes_updates = nearest_input_pts( 

225 in_latlons, out_latlons_updates, k_nearest 

226 ) 

227 # Update output surface classification and surface type mask 

228 out_classified_updates = out_classified[lake_island_indexes] 

229 surface_type_mask_updates = similar_surface_classify( 

230 in_classified, out_classified_updates, indexes_updates 

231 ) 

232 

233 # Where distance is outside specified vicinity, set surface type to be mismatched 

234 # so that it will not be used, update surface type mask again 

235 distance_not_in_vicinity = distances_updates > vicinity 

236 surface_type_mask_updates = np.where( 

237 distance_not_in_vicinity, False, surface_type_mask_updates 

238 ) 

239 

240 count_matching_surface = np.count_nonzero(surface_type_mask_updates, axis=1) 

241 points_with_no_match = (np.where(count_matching_surface == 0))[0] 

242 if points_with_no_match.shape[0] > 0: 

243 # No improved input point has been found with the increase to 8 nearest points 

244 # Take the original nearest point, disregard the surface type 

245 no_match_indexes = lake_island_indexes[points_with_no_match] 

246 surface_type_mask[no_match_indexes, :] = True 

247 

248 # From the expansion to 8 nearby input points, a same surface type input has been found 

249 # Update the index and surface type mask to use the newly found same surface type input point 

250 points_with_match = (np.where(count_matching_surface > 0))[0] 

251 count_of_points_with_match = points_with_match.shape[0] 

252 

253 for point_idx in range(count_of_points_with_match): 

254 # Reset all input surface types to mismatched 

255 match_indexes = lake_island_indexes[points_with_match[point_idx]] 

256 surface_type_mask[match_indexes, :] = False 

257 # Loop through 8 nearest points found 

258 for i in range(k_nearest): 

259 # Look for an input point with same surface type as output 

260 if surface_type_mask_updates[points_with_match[point_idx], i]: 

261 # When found, update the indexes and surface mask to use that improved input point 

262 indexes[match_indexes, 0] = indexes_updates[ 

263 points_with_match[point_idx], i 

264 ] 

265 surface_type_mask[match_indexes, 0] = True 

266 break 

267 

268 return indexes, surface_type_mask
