19class SnowSplitter(BasePlugin): 

20 """A class to separate the contribution of rain or snow from the precipitation 

21 rate/accumulation. This is calculated using the probability of rain and snow 

22 at the surface to determine what fraction of the precipitation rate/accumulation 

23 is rain or snow. 

24 """ 

25 

26 def __init__(self, output_is_rain: bool): 

27 """ 

28 Sets up Class 

29 

30 Args: 

31 output_is_rain: 

32 A boolean where True means the plugin will output rain and False means the 

33 output is snow. 

34 """ 

35 self.output_is_rain = output_is_rain 

36 

37 @staticmethod 

38 def separate_input_cubes(cubes: CubeList) -> Tuple[Cube, Cube, Cube]: 

39 """Separate the input cubelist into cubes of rain and snow. 

40 

41 Args: 

42 cubes: 

43 containing: 

44 rain_cube: 

45 Cube of the probability of rain at the surface. 

46 snow_cube: 

47 Cube of the probability of snow at the surface. 

48 precip_cube: 

49 Cube of either precipitation rate or precipitation accumulation. 

50 

51 Returns: 

52 A tuple of rain_cube,snow_cube and precip_cube in that order. 

53 """ 

54 

55 rain_cube = cubes.extract_cube( 

56 Constraint(cube_func=lambda cube: "rain" in cube.name()) 

57 ) 

58 snow_cube = cubes.extract_cube( 

59 Constraint(cube_func=lambda cube: "snow" in cube.name()) 

60 ) 

61 precip_cube = cubes.extract_cube( 

62 Constraint(cube_func=lambda cube: "precipitation" in cube.name()) 

63 ) 

64 

65 return (rain_cube, snow_cube, precip_cube) 

66 

67 def process(self, *cubes: Union[Cube, CubeList]) -> Cube: 

68 """ 

69 Splits the precipitation cube data into a snow or rain contribution. 

70 

71 Whether the output is a rate or accumulation will depend on the precipitation_cube. 

72 output_is_rain will determine whether the outputted cube is a cube of snow or rain. 

73 

74 The probability of rain and snow at the surfaces should only contain 1's where the 

75 phase is present at the surface and 0's where the phase is not present 

76 at the surface. These cubes need to be consistent with each other such that both 

77 rain and snow can't be present at the surface (e.g. at no grid square can both 

78 diagnostics have a probability of 1). 

79 

80 A grid of coefficients is calculated by an arbitrary function that maps 

81 (0,0) -> 0.5, (1,0) -> 1, (0,1) -> 0 where the first coordinate is the probability 

82 for the variable that will be outputted. This grid of coefficients is then multiplied 

83 by the precipitation rate/accumulation to split out the contribution of the desired 

84 variable. 

85 

86 Args: 

87 cubes: 

88 containing: 

89 rain_cube: 

90 Cube of the probability of rain at the surface. 

91 snow_cube: 

92 Cube of the probability of snow at the surface. 

93 precip_cube: 

94 Cube of either precipitation rate or precipitation accumulation. 

95 

96 Returns: 

97 Cube of rain/snow (depending on self.output_is_rain) rate/accumulation (depending on 

98 precipitation cube). The name will be an updated version of precip_cube.name(). 

99 "precipitation" will be replaced with "rainfall" or "snowfall" and "lwe\_" will be 

100 removed for rain output. 

101 

102 Raises: 

103 ValueError: If, at some grid square, both snow_cube and rain_cube have a probability of 

104 0 

105 """ # noqa: W605 (flake8 objects to \_ in "lwe\_" that is required for Sphinx) 

106 cubes = as_cubelist(*cubes) 

107 rain_cube, snow_cube, precip_cube = self.separate_input_cubes(cubes) 

108 

109 assert_spatial_coords_match([rain_cube, snow_cube, precip_cube]) 

110 if np.any((rain_cube.data + snow_cube.data) > 1): 

111 raise ValueError( 

112 """There is at least 1 grid square where the probability of snow 

113 at the surface plus the probability of rain at the surface is greater 

114 1.""" 

115 ) 

116 

117 if self.output_is_rain: 

118 required_cube = rain_cube 

119 other_cube = snow_cube 

120 name = "rainfall" 

121 else: 

122 required_cube = snow_cube 

123 other_cube = rain_cube 

124 name = "snowfall" 

125 

126 # arbitrary function that maps combinations of rain and snow probabilities 

127 # to an appropriate coefficient 

128 coefficient_cube = (required_cube - other_cube + 1) / 2 

129 coefficient_cube.data = coefficient_cube.data.astype(np.float32) 

130 

131 new_name = precip_cube.name().replace("precipitation", name) 

132 if self.output_is_rain: 

133 new_name = new_name.replace("lwe_", "") 

134 output_cube = Combine(operation="*", new_name=new_name)( 

135 [precip_cube, coefficient_cube] 

136 ) 

137 

138 return output_cube
