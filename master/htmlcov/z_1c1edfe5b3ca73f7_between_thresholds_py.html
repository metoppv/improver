22class OccurrenceBetweenThresholds(PostProcessingPlugin): 

23 """Calculate the probability of occurrence between thresholds""" 

24 

25 def __init__( 

26 self, threshold_ranges: List[List[float]], threshold_units: str 

27 ) -> None: 

28 """ 

29 Initialise the class. Threshold ranges must be specified in a unit 

30 that is NOT sensitive to differences at the 1e-5 (float32) precision 

31 level. 

32 

33 Args: 

34 threshold_ranges: 

35 List of 2-item iterables specifying thresholds between which 

36 probabilities should be calculated 

37 threshold_units: 

38 Units in which the thresholds are specified 

39 

40 Raises: 

41 ValueError: 

42 If any of the specified thresholds are indistinguishable at the 

43 1e-5 (float32) precision level 

44 """ 

45 threshold_diffs = np.diff(threshold_ranges) 

46 if any(diff < 1e-5 for diff in threshold_diffs): 

47 raise ValueError( 

48 "Plugin cannot distinguish between thresholds at {} {}".format( 

49 threshold_ranges, threshold_units 

50 ) 

51 ) 

52 self.threshold_ranges = threshold_ranges 

53 self.threshold_units = threshold_units 

54 

55 def _slice_cube(self) -> List[List[Cube]]: 

56 """ 

57 Extract required slices from input cube 

58 

59 Returns: 

60 List of 2-item lists containing lower and upper 

61 threshold cubes 

62 

63 Raises: 

64 ValueError: 

65 If any of the required constraints returns None 

66 """ 

67 thresh_coord = self.cube.coord(self.thresh_coord.name()) 

68 error_string = ( 

69 f"{thresh_coord.name()} threshold {{}} {self.threshold_units} " 

70 "is not available\n" 

71 ) 

72 error_msg = "" 

73 

74 cubes = [] 

75 for t_range in self.threshold_ranges: 

76 t_range.sort() 

77 lower_constraint = iris.Constraint( 

78 coord_values={ 

79 thresh_coord: lambda t: np.isclose(t.point, t_range[0], atol=1e-5) 

80 } 

81 ) 

82 lower_cube = self.cube.extract(lower_constraint) 

83 if lower_cube is None: 

84 error_msg += error_string.format(t_range[0]) 

85 upper_constraint = iris.Constraint( 

86 coord_values={ 

87 thresh_coord: lambda t: np.isclose(t.point, t_range[1], atol=1e-5) 

88 } 

89 ) 

90 upper_cube = self.cube.extract(upper_constraint) 

91 if upper_cube is None: 

92 error_msg += error_string.format(t_range[1]) 

93 cubes.append([lower_cube, upper_cube]) 

94 

95 if error_msg: 

96 # if any thresholds were unavailable, raise errors together here 

97 raise ValueError(error_msg) 

98 

99 return cubes 

100 

101 def _get_multiplier(self) -> np.float32: 

102 """ 

103 Check whether the cube contains "above" or "below" threshold 

104 probabilities. For "above", the probability of occurrence between 

105 thresholds is the difference between probabilities at the lower 

106 and higher thresholds: P(lower) - P(higher). For "below" it is the 

107 inverse of this: P(higher) - P(lower), which is implemented by 

108 multiplying the difference by -1. 

109 

110 Returns: 

111 1. or -1. 

112 

113 Raises: 

114 ValueError: If the spp__relative_to_threshold attribute is 

115 not recognised 

116 """ 

117 relative_to_threshold = probability_is_above_or_below(self.cube) 

118 if relative_to_threshold == "above": 

119 multiplier = 1.0 

120 elif relative_to_threshold == "below": 

121 multiplier = -1.0 

122 else: 

123 raise ValueError( 

124 "Input cube must contain probabilities of " 

125 "occurrence above or below threshold" 

126 ) 

127 return np.float32(multiplier) 

128 

129 def _calculate_probabilities(self) -> Cube: 

130 """ 

131 Calculate between_threshold probabilities cube 

132 

133 Returns: 

134 Merged cube containing recalculated probabilities 

135 """ 

136 multiplier = self._get_multiplier() 

137 thresh_name = self.thresh_coord.name() 

138 

139 cubelist = iris.cube.CubeList([]) 

140 for lower_cube, upper_cube in self.cube_slices: 

141 # construct difference cube 

142 between_thresholds_data = (lower_cube.data - upper_cube.data) * multiplier 

143 between_thresholds_cube = upper_cube.copy(between_thresholds_data) 

144 

145 # add threshold coordinate bounds 

146 lower_threshold = lower_cube.coord(thresh_name).points[0] 

147 upper_threshold = upper_cube.coord(thresh_name).points[0] 

148 between_thresholds_cube.coord(thresh_name).bounds = [ 

149 lower_threshold, 

150 upper_threshold, 

151 ] 

152 

153 cubelist.append(between_thresholds_cube) 

154 

155 return cubelist.merge_cube() 

156 

157 def _update_metadata(self, output_cube: Cube, original_units: Unit) -> None: 

158 """ 

159 Update output cube name and threshold coordinate 

160 

161 Args: 

162 output_cube: 

163 Cube containing new "between_thresholds" probabilities 

164 original_units: 

165 Required threshold-type coordinate units 

166 """ 

167 new_name = self.cube.name().replace( 

168 "{}_threshold".format(probability_is_above_or_below(self.cube)), 

169 "between_thresholds", 

170 ) 

171 output_cube.rename(new_name) 

172 

173 new_thresh_coord = output_cube.coord(self.thresh_coord.name()) 

174 new_thresh_coord.convert_units(original_units) 

175 new_thresh_coord.attributes["spp__relative_to_threshold"] = "between_thresholds" 

176 

177 def process(self, cube: Cube) -> Cube: 

178 """ 

179 Calculate probabilities between thresholds for the input cube 

180 

181 Args: 

182 cube: 

183 Probability cube containing thresholded data (above or below) 

184 

185 Returns: 

186 Cube containing probability of occurrence between thresholds 

187 """ 

188 # if cube has no threshold-type coordinate, raise an error 

189 try: 

190 self.thresh_coord = find_threshold_coordinate(cube) 

191 except CoordinateNotFoundError: 

192 raise ValueError( 

193 "Input is not a probability cube (has no threshold-type coordinate)" 

194 ) 

195 self.cube = cube.copy() 

196 

197 # check input cube units and convert if needed 

198 original_units = self.thresh_coord.units 

199 if original_units != self.threshold_units: 

200 self.cube.coord(self.thresh_coord).convert_units(self.threshold_units) 

201 

202 # extract suitable cube slices 

203 self.cube_slices = self._slice_cube() 

204 

205 # generate "between thresholds" probabilities 

206 output_cube = self._calculate_probabilities() 

207 self._update_metadata(output_cube, original_units) 

208 return output_cube
