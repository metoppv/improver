33class TemperatureSaturatedAirParcel(BasePlugin): 

34 """ 

35 Plugin to calculate the temperature of a saturated air parcel as it rises 

36 adiabatically from the cloud condensation level (CCL) to a pressure level. 

37 The default is set at 500 hPa for the Lifted Index (LI) calculations. 

38 """ 

39 

40 def __init__(self, pressure_level: float = 50000.0): 

41 """ 

42 Set up class 

43 

44 Args: 

45 pressure_level: 

46 The pressure level that the air parcel is lifted to adiabatically 

47 from the cloud condensation level. (Pa) 

48 """ 

49 self.pressure_level = pressure_level 

50 self.temperature: Cube = None 

51 self.pressure: Cube = None 

52 

53 def parcel_temp_after_ascent(self) -> np.array: 

54 """Calculates the temperature of a saturated air parcel when it has been lifted 

55 from the CCL to a pressure level. This has been set at 500 hPa for the easy 

56 calculation of Lifted Index (LI). 

57 

58 Returns: 

59 Array of temperature of an air parcel at a pressure level (K) 

60 """ 

61 relative_humidity_cube = self.make_saturated_relative_humidity_cube() 

62 humidity_cube = HumidityMixingRatio()( 

63 [self.temperature, self.pressure, relative_humidity_cube] 

64 ) 

65 ccl_temperature_cube, ccl_pressure_cube = CloudCondensationLevel()( 

66 [self.temperature, self.pressure, humidity_cube] 

67 ) 

68 humidity_mixing_ratio_at_ccl = saturated_humidity( 

69 ccl_temperature_cube.data, ccl_pressure_cube.data 

70 ) 

71 dry_parcel_temperature_after_ascent = dry_adiabatic_temperature( 

72 ccl_temperature_cube.data, ccl_pressure_cube.data, self.pressure_level 

73 ) 

74 parcel_temperature_after_ascent, _ = adjust_for_latent_heat( 

75 dry_parcel_temperature_after_ascent, 

76 humidity_mixing_ratio_at_ccl, 

77 self.pressure_level, 

78 ) 

79 parcel_temperature_after_ascent = np.ma.masked_invalid( 

80 parcel_temperature_after_ascent 

81 ).astype(np.float32) 

82 return parcel_temperature_after_ascent 

83 

84 def make_saturated_relative_humidity_cube(self) -> Cube: 

85 """Creates a cube of relative humidity at the cloud condensation level (CCL) 

86 with a value of 1.0, as by definition the relative humidity is 100 

87 percent at the CCL. 

88 

89 The temperature cube is used as a template for the metadata of the relative humidity cube. 

90 Only the name and units will be replaced. 

91 

92 Returns: 

93 A cube of relative humidity at the CCL with a value of 1.0. 

94 """ 

95 relative_humidity_cube = self.temperature.copy( 

96 np.ones_like(self.temperature.data) 

97 ) 

98 relative_humidity_cube.rename("relative_humidity") 

99 relative_humidity_cube.units = "1" 

100 return relative_humidity_cube 

101 

102 def make_temperature_cube(self, temp_after_saturated_ascent: np.ndarray) -> Cube: 

103 """Puts the temperature information into a cube with appropriate metadata. 

104 Uses self.temperature as a template for the metadata of the temperature cube. 

105 Only the name and units will be replaced. 

106 

107 Args: 

108 temp_after_saturated_ascent: 

109 An n dimensional array of the temperature after a saturated air parcel has 

110 been lifted adiabatically from the cloud condensation level (CCL) to a 

111 pressure level (K). 

112 

113 Returns: 

114 A cube of the temperature of a saturated air parcel when it has been lifted 

115 adiabatically from the CCL to another pressure level (K) 

116 """ 

117 temp_cube = create_new_diagnostic_cube( 

118 name="parcel_temperature_after_saturated_ascent_from_ccl_to_pressure_level", 

119 units="K", 

120 template_cube=self.temperature, 

121 mandatory_attributes=generate_mandatory_attributes([self.temperature]), 

122 data=temp_after_saturated_ascent, 

123 ) 

124 temp_cube.add_aux_coord( 

125 DimCoord( 

126 self.pressure_level, 

127 long_name="pressure", 

128 units="Pa", 

129 attributes={"positive": "down"}, 

130 ) 

131 ) 

132 return temp_cube 

133 

134 def process(self, *cubes: Union[Cube, CubeList]) -> Cube: 

135 """ 

136 Calculates the temperature of a saturated air parcel that has risen adiabatically 

137 from the cloud condensation level to a pressure level. 

138 

139 Args: 

140 cubes: 

141 Cubes of temperature (K), pressure (Pa) at the cloud condensation level. 

142 

143 

144 Returns: 

145 Cube of parcel_temperature_after_saturated_ascent_from_ccl_to_pressure_level 

146 

147 """ 

148 cubes = as_cubelist(cubes) 

149 (self.temperature, self.pressure) = CubeList(cubes).extract( 

150 [ 

151 "air_temperature_at_condensation_level", 

152 "air_pressure_at_condensation_level", 

153 ] 

154 ) 

155 self.temperature.convert_units("K") 

156 self.pressure.convert_units("Pa") 

157 parcel_temp_at_pressure_level = self.parcel_temp_after_ascent() 

158 temp_cube = self.make_temperature_cube(parcel_temp_at_pressure_level) 

159 return temp_cube
