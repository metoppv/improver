19class EnforceConsistentForecasts(PostProcessingPlugin): 

20 """Enforce that the forecasts provided are no less than, no greater than, or between 

21 some linear function(s) of a reference forecast. For example, wind speed forecasts 

22 may be provided as the reference forecast for wind gust forecasts with a requirement 

23 that wind gusts are not less than 110% of the corresponding wind speed forecast. 

24 Wind speed forecasts of 10 m/s will mean that the resulting wind gust forecast will 

25 be at least 11 m/s. 

26 """ 

27 

28 def __init__( 

29 self, 

30 additive_amount: Union[float, List[float]] = 0.0, 

31 multiplicative_amount: Union[float, List[float]] = 1.0, 

32 comparison_operator: Union[str, List[str]] = ">=", 

33 diff_for_warning: Optional[float] = None, 

34 use_latest_update_time: Optional[bool] = False, 

35 ) -> None: 

36 """ 

37 Initialise class for enforcing a forecast to be either greater than or equal to, 

38 or less than or equal to a linear function of a reference forecast. Can also 

39 enforce that a forecast is between two bounds created from the reference 

40 forecast, if this is the case then a list of two elements must be provided for 

41 additive_amount, multiplicative_amount, and comparison_operator. 

42 

43 Args: 

44 additive_amount: The amount to be added to the reference forecast prior to 

45 enforcing consistency between the forecast and reference forecast. If 

46 both an additive_amount and multiplicative_amount are specified then 

47 addition occurs after multiplication. This option cannot be used for 

48 probability forecasts, if it is then an error will be raised. 

49 multiplicative_amount: The amount to multiply the reference forecast by 

50 prior to enforcing consistency between the forecast and reference 

51 forecast. If both an additive_amount and multiplicative_amount are 

52 specified then addition occurs after multiplication. This option cannot 

53 be used for probability forecasts, if it is then an error will be raised. 

54 comparison_operator: Determines whether the forecast is enforced to be not 

55 less than or not greater than the reference forecast. Valid choices are 

56 ">=", for not less than, and "<=" for not greater than. If provided as 

57 a list then each of ">=" and "<=" must be in the list exactly once. 

58 diff_for_warning: If assigned, the plugin will raise a warning if any 

59 absolute change in forecast value is greater than this value. 

60 use_latest_update_time: 

61 If True the returned cube will have a forecast_reference_time 

62 and/or blend_time that is the latest of these coordinate values 

63 on the input forecast and reference_forecast cubes. 

64 """ 

65 

66 self.additive_amount = additive_amount 

67 self.multiplicative_amount = multiplicative_amount 

68 self.comparison_operator = comparison_operator 

69 self.diff_for_warning = diff_for_warning 

70 self.use_latest_update_time = use_latest_update_time

72 @staticmethod 

73 def calculate_bound( 

74 cube: Cube, additive_amount: float, multiplicative_amount: float 

75 ) -> Cube: 

76 """ 

77 Function to calculate a linear transformation of the reference forecast. 

78 

79 Args: 

80 cube: An iris cube. 

81 additive_amount: The amount to be added to the cube. If both an 

82 additive_amount and multiplicative_amount are specified then addition 

83 occurs after multiplication. 

84 multiplicative_amount: The amount to multiply the cube by. If both an 

85 additive_amount and multiplicative_amount are specified then addition 

86 occurs after multiplication. 

87 

88 Returns: 

89 A cube with identical metadata to input cube but with transformed data. 

90 """ 

91 

92 output = cube.copy() 

93 output.data = multiplicative_amount * output.data 

94 output.data = additive_amount + output.data 

95 

96 return output

98 def process(self, forecast: Cube, reference_forecast: Cube) -> Cube: 

99 """ 

100 Function to enforce that the values in the forecast cube are not less than or 

101 not greater than a linear function of the corresponding values in 

102 reference_forecast, or between two bounds generated from two different linear 

103 functions of the reference_forecast. 

104 

105 Args: 

106 forecast: A forecast cube 

107 reference_forecast: A reference forecast cube used to determine the bound/s 

108 of the forecast cube. 

109 

110 Returns: 

111 A forecast cube with identical metadata to forecast but the forecasts are 

112 enforced to be within the calculated bounds. 

113 

114 Raises: 

115 ValueError: If units of forecast and reference cubes are different and 

116 cannot be converted to match. 

117 ValueError: If additive_amount and multiplicative_amount are not 0.0 and 1.0, 

118 respectively, when a probability forecast is input. 

119 ValueError: If incorrect comparison_operator is input. 

120 ValueError: If contradictory bounds are generated. 

121 ValueError: If any of additive_amount, multiplicative_amount, or 

122 comparison_operator are lists when they are not all lists. 

123 

124 Warns: 

125 Warning: If difference between generated bounds and forecast is greater than 

126 diff_for_warning. 

127 """ 

128 

129 # check forecast and reference units match 

130 try: 

131 reference_forecast.convert_units(forecast.units) 

132 except ValueError: 

133 if forecast.units != reference_forecast.units: 

134 msg = ( 

135 "The units in the forecast and reference cubes do not match and " 

136 "cannot be converted to match. The units of forecast were " 

137 f"{forecast.units}, the units of reference_forecast were " 

138 f"{reference_forecast.units}." 

139 ) 

140 raise ValueError(msg) 

141 

142 # linear transformation cannot be applied to probability forecasts 

143 if self.additive_amount != 0.0 or self.multiplicative_amount != 1.0: 

144 if is_probability(forecast): 

145 msg = ( 

146 "For probability data, additive_amount must be 0.0 and " 

147 "multiplicative_amount must be 1.0. The input additive_amount was " 

148 f"{self.additive_amount}, the input multiplicative_amount was " 

149 f"{self.multiplicative_amount}." 

150 ) 

151 raise ValueError(msg) 

152 

153 # calculate forecast_bound by applying specified linear transformation to 

154 # reference_forecast 

155 check_if_list = [ 

156 isinstance(item, list) 

157 for item in [ 

158 self.additive_amount, 

159 self.multiplicative_amount, 

160 self.comparison_operator, 

161 ] 

162 ] 

163 if all(check_if_list): 

164 lower_bound = self.calculate_bound( 

165 reference_forecast, 

166 self.additive_amount[0], 

167 self.multiplicative_amount[0], 

168 ).data 

169 upper_bound = self.calculate_bound( 

170 reference_forecast, 

171 self.additive_amount[1], 

172 self.multiplicative_amount[1], 

173 ).data 

174 if self.comparison_operator == ["<=", ">="]: 

175 upper_bound, lower_bound = lower_bound, upper_bound 

176 elif self.comparison_operator == [">=", "<="]: 

177 pass 

178 else: 

179 msg = ( 

180 "When comparison operators are provided as a list, the list must be " 

181 f"either ['>=', '<='] or ['<=', '>='], not {self.comparison_operator}." 

182 ) 

183 raise ValueError(msg) 

184 if np.any(lower_bound > upper_bound): 

185 msg = ( 

186 "The provided reference_cube, additive_amount and " 

187 "multiplicative_amount have created contradictory bounds. Some of" 

188 "the values in the lower bound are greater than the upper bound." 

189 ) 

190 raise ValueError(msg) 

191 elif any(check_if_list): 

192 msg = ( 

193 "If any of additive_amount, multiplicative_amount, or comparison_operator " 

194 "are input as a list, then they must all be input as a list of 2 elements. " 

195 ) 

196 raise ValueError(msg) 

197 else: 

198 bound = self.calculate_bound( 

199 reference_forecast, self.additive_amount, self.multiplicative_amount 

200 ) 

201 lower_bound = None 

202 upper_bound = None 

203 if self.comparison_operator == ">=": 

204 lower_bound = bound.data 

205 elif self.comparison_operator == "<=": 

206 upper_bound = bound.data 

207 else: 

208 msg = ( 

209 "When enforcing consistency with one bound, comparison_operator " 

210 f"must be either '>=' or '<=', not {self.comparison_operator}." 

211 ) 

212 raise ValueError(msg) 

213 

214 new_forecast = forecast.copy() 

215 required_dtype = new_forecast.data.dtype 

216 new_forecast.data = np.clip(new_forecast.data, lower_bound, upper_bound) 

217 new_forecast.data = new_forecast.data.astype(required_dtype) 

218 if self.use_latest_update_time: 

219 forecast_cycle_coords = [ 

220 crd 

221 for crd in ["forecast_reference_time", "blend_time"] 

222 if new_forecast.coords(crd) 

223 ] 

224 ref_cycle_coords = [ 

225 crd 

226 for crd in ["forecast_reference_time", "blend_time"] 

227 if reference_forecast.coords(crd) 

228 ] 

229 # If one cube has a blend_time and one does not the subsequent 

230 # tooling will not succeed, so raise an exception here. 

231 if set(forecast_cycle_coords) != set(ref_cycle_coords): 

232 raise ValueError( 

233 "Cubes do not include the same set of cycle time coordinates " 

234 "and cannot be updated to match as part of cube enforcement." 

235 ) 

236 latest_times = [] 

237 latest_times.extend( 

238 [new_forecast.coord(crd).cell(0).point for crd in forecast_cycle_coords] 

239 ) 

240 latest_times.extend( 

241 [ 

242 reference_forecast.coord(crd).cell(0).point 

243 for crd in ref_cycle_coords 

244 ] 

245 ) 

246 latest_time = max(latest_times) 

247 

248 new_forecast, _ = unify_cycletime( 

249 [new_forecast, reference_forecast], 

250 latest_time, 

251 target_coords=forecast_cycle_coords, 

252 ) 

253 

254 diff = new_forecast.data - forecast.data 

255 max_abs_diff = np.max(np.abs(diff)) 

256 if self.diff_for_warning is not None and max_abs_diff > self.diff_for_warning: 

257 warnings.warn( 

258 f"Inconsistency between forecast {forecast.name} and " 

259 f"{reference_forecast.name} is greater than {self.diff_for_warning}. " 

260 f"Maximum absolute difference reported was {max_abs_diff}" 

261 ) 

262 

263 return new_forecast

266def normalise_to_reference( 

267 cubes: CubeList, reference: Cube, ignore_zero_total: bool = False 

268) -> CubeList: 

269 """Update the data in cubes so that the sum of this data is equal to the reference 

270 cube. This is done by replacing the data in cubes with a fraction of the data in 

271 reference based upon the fraction that each cube contributes to the sum total of 

272 data in cubes. 

273 

274 Args: 

275 cubes: Cubelist containing the cubes to be updated. Must contain at least 2 

276 cubes. 

277 reference: Cube with data which the sum of cubes will be forced to be equal to. 

278 ignore_zero_total: If False, an error will be raised if the sum total of data 

279 in input_cubes is zero where the reference cube contains a non-zero value. 

280 If True, this case will be ignored, leaving the values in the cubelist as 

281 zero rather than ensuring their total equals the corresponding value in 

282 reference cube. 

283 

284 Raises: 

285 ValueError: If length of cubes is less than 2. 

286 ValueError: If any input cubes have a different number of dimensions to 

287 reference, or if the dimension coordinates in any of the input cubes do not 

288 match the dimension coordinates in reference. 

289 ValueError: If there are instances where the total of the input cubes is zero 

290 but the corresponding value in reference is non-zero. This error can be 

291 ignored if ignore_zero_total is true. 

292 

293 Returns: 

294 Cubelist with length equal to the length of cubes. Each cube in the returned 

295 cubelist will have metadata matching the cube in the same position in input 

296 cubes, but containing different data. 

297 """ 

298 if len(cubes) < 2: 

299 msg = ( 

300 f"The input cubes must be of at least length 2. The length of the input " 

301 f"cubes was {len(cubes)}" 

302 ) 

303 raise ValueError(msg) 

304 

305 # check cube compatibility 

306 reference_dim_coords = reference.coords(dim_coords=True) 

307 n_dims_mismatch = False 

308 coord_mismatch = False 

309 n_dims = [] 

310 mismatching_coords = {} 

311 for cube in cubes: 

312 cube_dim_coords = cube.coords(dim_coords=True) 

313 n_dims.append(len(cube_dim_coords)) 

314 if len(cube_dim_coords) != len(reference_dim_coords): 

315 n_dims_mismatch = True 

316 if not n_dims_mismatch: 

317 mismatching_coords[cube.name()] = [] 

318 for dim_coord in cube_dim_coords: 

319 try: 

320 reference_coord = reference.coord(dim_coord.name(), dim_coords=True) 

321 except iris.exceptions.CoordinateNotFoundError: 

322 coord_mismatch = True 

323 mismatching_coords[cube.name()].append(dim_coord.name()) 

324 continue 

325 if not dim_coord == reference_coord: 

326 coord_mismatch = True 

327 mismatching_coords[cube.name()].append(dim_coord.name()) 

328 

329 if n_dims_mismatch: 

330 msg = ( 

331 f"The number of dimensions in input cubes are not all the same as the " 

332 f"number of dimensions on the reference cube. The number of dimensions in " 

333 f"the input cubes were {n_dims}. The number of dimensions in the " 

334 f"reference cube was {len(reference_dim_coords)}." 

335 ) 

336 raise ValueError(msg) 

337 

338 if coord_mismatch and not is_probability(reference): 

339 msg = ( 

340 f"The dimension coordinates on the input cubes and the reference did not " 

341 f"all match. The following coordinates were found to differ: " 

342 f"{mismatching_coords}." 

343 ) 

344 raise ValueError(msg) 

345 

346 total = cubes[0].data.copy() 

347 if len(cubes) > 1: 

348 for cube in cubes[1:]: 

349 total += cube.data 

350 

351 # check for zeroes in total when reference is non-zero 

352 total_zeroes = total == 0.0 

353 reference_non_zero = reference.data != 0.0 

354 both_true = np.logical_and(total_zeroes, reference_non_zero) 

355 if np.any(both_true): 

356 if not ignore_zero_total: 

357 msg = ( 

358 "There are instances where the total of input cubes is zero but the " 

359 "corresponding value in reference is non-zero. The input cubes cannot " 

360 "be updated so that the total equals the value in the reference in " 

361 "these instances." 

362 ) 

363 raise ValueError(msg) 

364 

365 # update total where zero to avoid dividing by zero later. 

366 total[total_zeroes] = 1.0 

367 

368 output = iris.cube.CubeList() 

369 for index, cube in enumerate(cubes): 

370 output_cube = cube.copy(data=reference.data * cube.data / total) 

371 output.append(output_cube) 

372 

373 return output

376def split_cubes_by_name( 

377 cubes: Union[Iterable[Cube], CubeList], cube_names: Union[str, List[str]] = None 

378) -> tuple: 

379 """Split a list of cubes into two lists; one containing all cubes with names which 

380 match cube_names, and the other containing all the other cubes. 

381 

382 Args: 

383 cubes: List of cubes 

384 cube_names: the name of the cube/s to be used as reference. This can be either 

385 a single name or a list of names. If None, the first cubelist returned will 

386 contain all the input cubes and the second cubelist will be empty. 

387 

388 Returns: 

389 - A cubelist containing all cubes with names which match cube_names 

390 - A cubelist containing all cubes with names which do not match cube_names 

391 """ 

392 

393 desired_cubes = iris.cube.CubeList() 

394 other_cubes = iris.cube.CubeList() 

395 

396 if cube_names is None: 

397 desired_cubes = cubes 

398 else: 

399 if isinstance(cube_names, str): 

400 cube_names = [cube_names] 

401 for cube in cubes: 

402 if cube.name() in cube_names: 

403 desired_cubes.append(cube) 

404 else: 

405 other_cubes.append(cube) 

406 

407 return desired_cubes, other_cubes
