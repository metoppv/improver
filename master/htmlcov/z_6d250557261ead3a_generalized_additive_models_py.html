17class GAMFit(BasePlugin): 

18 """ 

19 Class for fitting Generalized Additive Models (GAMs) which predict the mean or 

20 standard deviation of input forecasts or observations. 

21 

22 This class uses functionality from pyGAM 

23 (https://pygam.readthedocs.io/en/latest/index.html) to fit the model. 

24 """ 

25 

26 def __init__( 

27 self, 

28 model_specification: List, 

29 max_iter: int = 100, 

30 tol: float = 0.0001, 

31 distribution: str = "normal", 

32 link: str = "identity", 

33 fit_intercept: bool = True, 

34 ): 

35 """ 

36 Initialize class for fitting GAMs using pyGAM. 

37 

38 Args: 

39 model_specification: 

40 A list containing lists of three items (in order): 

41 1. a string containing a single pyGAM term; one of 'linear', 

42 'spline', 'tensor', or 'factor' 

43 2. a list of indices of the features to be included in that term, 

44 corresponding to the index of those features in the predictor array 

45 3. a dictionary of kwargs to be included when defining the term 

46 max_iter: 

47 A pyGAM argument which determines the maximum iterations allowed when 

48 fitting the GAM. Defaults to 100. 

49 tol: 

50 A pyGAM argument determining the tolerance used to define the stopping 

51 criteria. Defaults to 0.0001. 

52 distribution: 

53 A pyGAM argument determining the distribution to be used in the model. 

54 The default is a normal distribution. 

55 link: 

56 A pyGAM argument determining the link function to be used in the model. 

57 Defaults to the identity link function, which implies a direct 

58 relationship between predictors and target. 

59 fit_intercept: 

60 A pyGAM argument determining whether to include an intercept term in 

61 the model. Default is True. 

62 """ 

63 self.model_specification = model_specification 

64 self.max_iter = max_iter 

65 self.tol = tol 

66 self.distribution = distribution 

67 self.link = link 

68 self.fit_intercept = fit_intercept 

69 

70 def create_pygam_model(self): 

71 """ 

72 Create a GAM model using pyGAM from the model_specification dictionary. 

73 

74 Returns: 

75 GAM model equation constructed using pyGAM model terms. 

76 """ 

77 # Import from pygam here to minimize dependencies 

78 from pygam import f, l, s, te 

79 

80 term = { 

81 "factor": f, 

82 "linear": l, 

83 "spline": s, 

84 "tensor": te, 

85 } # create dictionary of permissible pyGAM model terms 

86 

87 for index, config in enumerate(self.model_specification): 

88 # For each config in the list, parse the config to create a pyGAM term 

89 # from that config. The first term in the config defines the type of term, 

90 # the second defines which variables are included in that term, and the 

91 # third contains a dictionary of kwargs. 

92 if config[0] in term.keys(): 

93 new_term = term[config[0]](*config[1], **config[2]) 

94 else: 

95 msg = ( 

96 f"An unrecognised term has been included in the GAM model " 

97 f"specification. The term was {config[0]}, the accepted terms are " 

98 f"linear, spline, tensor, factor." 

99 ) 

100 raise ValueError(msg) 

101 

102 if index == 0: 

103 # Initialize the equation variable 

104 eqn = deepcopy(new_term) 

105 else: 

106 # Add new term to the existing equation 

107 eqn += new_term 

108 

109 return eqn

111 def process(self, predictors: np.ndarray, targets: np.ndarray): 

112 """ 

113 Fit a GAM model using pyGAM. 

114 

115 Args: 

116 predictors: A 2-D array of predictors. The index of each column (feature) is 

117 used in model_specification to determine which feature is included in each 

118 model term. 

119 targets: A 1-D array of target values associated with the predictors. 

120 

121 Returns: 

122 A fitted pyGAM GAM model. 

123 """ 

124 # Monkey patch for pyGAM due to handling of sparse arrays in some versions of 

125 # scipy. 

126 import scipy.sparse 

127 

128 def to_array(self): 

129 return self.toarray() 

130 

131 scipy.sparse.spmatrix.A = property(to_array) 

132 # Import from pygam here to minimize dependencies 

133 from pygam import GAM 

134 

135 # Remove nans from arrays. 

136 predictors = predictors[~np.isnan(targets)] 

137 targets = targets[~np.isnan(targets)] 

138 

139 # Check that there is data to fit after removing nans. 

140 if (len(predictors) == 0) or (len(targets) == 0): 

141 msg = ( 

142 "After removing NaN values from the input data, there are no " 

143 "remaining data points to fit the GAM model. No model has been fitted." 

144 ) 

145 warnings.warn(msg) 

146 return None 

147 

148 eqn = self.create_pygam_model() 

149 gam = GAM( 

150 eqn, 

151 max_iter=self.max_iter, 

152 tol=self.tol, 

153 distribution=self.distribution, 

154 link=self.link, 

155 fit_intercept=self.fit_intercept, 

156 ).fit(predictors, targets) 

157 

158 return gam

161class GAMPredict(BasePlugin): 

162 """Class for predicting new outputs from a fitted GAM given new input predictors.""" 

163 

164 def __init__(self, constant_extrapolation: bool = False): 

165 """Initialize class 

166 

167 Args: 

168 constant_extrapolation: 

169 If True, predictor values outside the range of those used to fit the GAM 

170 will be predicted using constant extrapolation (i.e. the nearest 

171 boundary value). If False, extrapolation extends the trend of each 

172 GAM term beyond the range of the training data. Default is False. 

173 """ 

174 self.constant_extrapolation = constant_extrapolation 

175 

176 @staticmethod 

177 def clip_predictors(gam, predictors: np.ndarray): 

178 """ 

179 Clip predictor values to be within the minimum and maximum values in the 

180 training data used to fit the GAM. This enables constant extrapolation when 

181 predicting new values. 

182 

183 Args: 

184 gam: A fitted pyGAM GAM model. 

185 predictors: A 2-D array of inputs to use to predict new values. Each 

186 feature (column) should have the same index as in the training dataset. 

187 The array is modified in place. 

188 """ 

189 # Create dictionary with keys corresponding to each column index and values 

190 # corresponding to the min and max values of that feature column in the training 

191 # data. 

192 bounds_dict = {} 

193 for term in gam.terms: 

194 if term.isintercept: 

195 continue 

196 if term.istensor: 

197 for tensor_term in term: 

198 bounds_dict[tensor_term.info["feature"]] = tensor_term.edge_knots_ 

199 else: 

200 bounds_dict[term.info["feature"]] = term.edge_knots_ 

201 

202 for i in bounds_dict.keys(): 

203 col_min, col_max = bounds_dict[i] 

204 predictors[:, i] = np.clip(predictors[:, i], col_min, col_max) 

205 

206 def process(self, gam, predictors: np.ndarray) -> np.ndarray: 

207 """ 

208 Use pyGAM functionality to predict values from a fitted GAM. 

209 

210 Args: 

211 gam: A fitted pyGAM GAM model. 

212 predictors: A 2-D array of inputs to use to predict new values. Each 

213 feature (column) should have the same index as in the training dataset. 

214 

215 Returns: 

216 A 1-D array of values predicted by the GAM with each value in the array 

217 corresponding to one row in the input predictors. 

218 """ 

219 if self.constant_extrapolation: 

220 self.clip_predictors(gam, predictors) 

221 

222 return gam.predict(predictors)
