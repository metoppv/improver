6"""Script to calculate temperature lapse rates for given temperature and 

7orogrophy datasets.""" 

8 

9from improver import cli 

10from improver.constants import DALR 

11 

12 

13@cli.clizefy 

14@cli.with_output 

15def process( 

16 temperature: cli.inputcube, 

17 orography: cli.inputcube = None, 

18 land_sea_mask: cli.inputcube = None, 

19 *, 

20 max_height_diff: float = 35, 

21 nbhood_radius: int = 7, 

22 max_lapse_rate: float = -3 * DALR, 

23 min_lapse_rate: float = DALR, 

24 dry_adiabatic=False, 

25 model_id_attr: str = None, 

26): 

27 """Calculate temperature lapse rates in units of K m-1 over orography grid. 

28 

29 Args: 

30 temperature (iris.cube.Cube): 

31 Air temperature data. This is required even when returning DALR, 

32 as this defines the grid on which lapse rates are required. 

33 orography (iris.cube.Cube): 

34 Orography data. 

35 land_sea_mask (iris.cube.Cube): 

36 Binary land-sea mask data. True for land-points, False for sea. 

37 max_height_diff (float): 

38 Maximum allowable height difference between the central point and 

39 points in the neighbourhood over which the lapse rate will be 

40 calculated. 

41 nbhood_radius (int): 

42 Radius of neighbourhood in grid points around each point. The 

43 neighbourhood is a square array with side length 

44 2*nbhood_radius + 1. The default value of 7 is from the reference 

45 paper (see plugin documentation). 

46 max_lapse_rate (float): 

47 Maximum lapse rate allowed, in K m-1. 

48 min_lapse_rate (float): 

49 Minimum lapse rate allowed, in K m-1. 

50 dry_adiabatic (bool): 

51 If True, returns a cube containing the dry adiabatic lapse rate 

52 rather than calculating the true lapse rate. 

53 model_id_attr (str): 

54 Name of the attribute used to identify the source model for 

55 blending. This is inherited from the input temperature cube. 

56 

57 Returns: 

58 iris.cube.Cube: 

59 Lapse rate (K m-1) 

60 

61 Raises: 

62 ValueError: If minimum lapse rate is greater than maximum. 

63 ValueError: If Maximum height difference is less than zero. 

64 ValueError: If neighbourhood radius is less than zero. 

65 RuntimeError: If calculating the true lapse rate and orography or 

66 land mask arguments are not given. 

67 """ 

68 import numpy as np 

69 

70 from improver.metadata.utilities import ( 

71 create_new_diagnostic_cube, 

72 generate_mandatory_attributes, 

73 ) 

74 from improver.temperature.lapse_rate import LapseRate 

75 

76 if dry_adiabatic: 

77 attributes = generate_mandatory_attributes( 

78 [temperature], model_id_attr=model_id_attr 

79 ) 

80 result = create_new_diagnostic_cube( 

81 "air_temperature_lapse_rate", 

82 "K m-1", 

83 temperature, 

84 attributes, 

85 data=np.full_like(temperature.data, DALR).astype(np.float32), 

86 ) 

87 return result 

88 

89 if min_lapse_rate > max_lapse_rate: 

90 msg = "Minimum lapse rate specified is greater than the maximum." 

91 raise ValueError(msg) 

92 

93 if max_height_diff < 0: 

94 msg = "Maximum height difference specified is less than zero." 

95 raise ValueError(msg) 

96 

97 if nbhood_radius < 0: 

98 msg = "Neighbourhood radius specified is less than zero." 

99 raise ValueError(msg) 

100 

101 if orography is None or land_sea_mask is None: 

102 msg = "Missing orography and/or land mask arguments." 

103 raise RuntimeError(msg) 

104 

105 result = LapseRate( 

106 max_height_diff=max_height_diff, 

107 nbhood_radius=nbhood_radius, 

108 max_lapse_rate=max_lapse_rate, 

109 min_lapse_rate=min_lapse_rate, 

110 )(temperature, orography, land_sea_mask, model_id_attr=model_id_attr) 

111 return result
