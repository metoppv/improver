6"""Script to extract a subset of input file data, given constraints."""

8from improver import cli

11 

12@cli.clizefy 

13@cli.with_output 

14def process( 

15 cube: cli.inputcube, 

16 *, 

17 constraints: parameters.multi(min=1), 

18 units: cli.comma_separated_list = None, 

19 ignore_failure=False, 

20): 

21 """Extract a subset of a single cube. 

22 

23 Extracts subset of data from a single cube, subject to equality-based 

24 constraints. 

25 Using a set of constraints, extract a sub-cube from the provided cube if it 

26 is available. 

27 

28 Args: 

29 cube (iris.cube.Cube): 

30 The Cube from which a sub-cube is extracted 

31 constraints (list): 

32 The constraint(s) to be applied. These must be of the form 

33 "key=value", eg "threshold=1". Multiple constraints can be provided 

34 by repeating this keyword before each. Scalars, boolean and string 

35 values are supported. Lists of values can be provided 

36 e.g. key=[value1, value2, value3]. Alternatively, ranges can also 

37 be specified e.g. key=[value1:value3]. 

38 When a range is specified, this is inclusive of the endpoints of 

39 the range. A range can also be specified with a step value, 

40 e.g. [value1:value2:step]. 

41 units (list): 

42 List of units as strings corresponding to each coordinate in the 

43 list of constraints. One or more "units" may be None and units may 

44 only be associated with coordinate constraints. The list should be 

45 entered as a comma separated list without spaces, e.g. mm/hr,K. 

46 ignore_failure (bool): 

47 Option to ignore constraint match failure and return the input 

48 cube. 

49 

50 Returns: 

51 iris.cube.Cube: 

52 A single cube matching the input constraints or None. If no 

53 sub-cube is found within the cube that matches the constraints. 

54 """ 

55 from improver.utilities.cube_extraction import ExtractSubCube 

56 

57 plugin = ExtractSubCube(constraints, units=units, ignore_failure=ignore_failure) 

58 result = plugin.process(cube) 

59 return result
