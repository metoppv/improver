5"""Module containing class for recalibrating blended probabilities. 

6 

7.. See the documentation for a more detailed discussion of this plugin. 

8.. include:: extended_documentation/calibration/beta_recalibration/ 

9 beta_recalibration.rst 

10""" 

11 

12from typing import Any, Dict 

13 

14import cf_units 

15import iris 

16import numpy as np 

17from iris.exceptions import CoordinateNotFoundError 

18from scipy.stats import beta 

19 

20from improver import PostProcessingPlugin 

21from improver.metadata.probabilistic import is_probability 

22 

23 

24class BetaRecalibrate(PostProcessingPlugin): 

25 """Recalibrate probabilities using the cumulative distribution function 

26 of the beta distribution. 

27 """ 

28 

29 def __init__(self, recalibration_dict: Dict[str, Any]): 

30 """ 

31 Args: 

32 recalibration_dict: 

33 Dictionary from which to calculate alpha and beta parameters for 

34 recalibrating blended output using the beta distribution. Dictionary 

35 format is as specified below. Weights will be interpolated over the 

36 forecast period from the values specified in the dictionary. 

37 

38 Recalibration dictionary format:: 

39 

40 { 

41 "forecast_period": [7, 12], 

42 "alpha": [1, 1.5], 

43 "beta": [1.3, 2], 

44 "units": "hours", 

45 } 

46 

47 The "units" key is optional. If it is omitted, it is assumed that the units 

48 are the same as those used in forecast_period coordinate of the input cube. 

49 """ 

50 self.recalibration_dict = recalibration_dict 

51 

52 def process(self, cube): 

53 """Recalibrate cube using the beta distribution with the alpha 

54 and beta parameters specified in self.recalibration_dict. 

55 

56 Args: 

57 cube: 

58 A cube containing a forecast_period coordinate. 

59 

60 Returns: 

61 A cube having the same dimensions as the input, with data 

62 transformed by the beta distribution cdf. 

63 

64 Raises: 

65 CoordinateNotFoundError: if cube does not contain probability data 

66 CoordinateNotFoundError: if cube does not contain forecast_period coordinate 

67 RuntimeError: if any interpolated values of alpha or beta are <= 0 

68 """ 

69 

70 if not (is_probability(cube)): 

71 raise CoordinateNotFoundError( 

72 "Input cube must be a probability forecast " 

73 "and contain a threshold coordinate." 

74 ) 

75 if len(cube.coords("forecast_period")) == 0: 

76 raise CoordinateNotFoundError( 

77 "Recalibration input must contain forecast_period coordinate." 

78 ) 

79 forecast_period = self.recalibration_dict["forecast_period"] 

80 cube_forecast_period = cube.coord("forecast_period").points 

81 if "units" in self.recalibration_dict.keys(): 

82 # convert interpolation points to cube units 

83 units = cf_units.Unit(self.recalibration_dict["units"]) 

84 forecast_period = [ 

85 units.convert(x, cube.coord("forecast_period").units) 

86 for x in forecast_period 

87 ] 

88 a = np.interp( 

89 cube_forecast_period, forecast_period, self.recalibration_dict["alpha"] 

90 ) 

91 b = np.interp( 

92 cube_forecast_period, forecast_period, self.recalibration_dict["beta"] 

93 ) 

94 if np.any(a <= 0) or np.any(b <= 0): 

95 raise RuntimeError("interpolated alpha and beta parameters must be > 0") 

96 

97 cubelist = iris.cube.CubeList([]) 

98 for i, slice in enumerate(cube.slices_over("forecast_period")): 

99 slice.data = beta(a[i], b[i]).cdf(slice.data).astype(np.float32) 

100 cubelist.append(slice) 

101 return cubelist.merge_cube()
