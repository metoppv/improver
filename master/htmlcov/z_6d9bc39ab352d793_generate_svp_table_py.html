18class SaturatedVapourPressureTable(BasePlugin): 

19 """ 

20 Plugin to create a saturated vapour pressure lookup table. 

21 """ 

22 

23 cube_name = "saturated_vapour_pressure" 

24 svp_units = "hPa" 

25 svp_si_units = "Pa" 

26 MAX_VALID_TEMPERATURE_WATER = 373.0 

27 MAX_VALID_TEMPERATURE_ICE = 273.15 

28 MIN_VALID_TEMPERATURE_WATER = 223.0 

29 MIN_VALID_TEMPERATURE_ICE = 173.0 

30 constants = { 

31 1: 10.79574, 

32 2: 5.028, 

33 3: 1.50475e-4, 

34 4: -8.2969, 

35 5: 0.42873e-3, 

36 6: 4.76955, 

37 7: 0.78614, 

38 8: -9.09685, 

39 9: 3.56654, 

40 10: 0.87682, 

41 11: 0.78614, 

42 }

44 def __init__( 

45 self, 

46 t_min: float = 183.15, 

47 t_max: float = 338.25, 

48 t_increment: float = 0.1, 

49 water_only: bool = False, 

50 ice_only: bool = False, 

51 ) -> None: 

52 """ 

53 Create a table of saturated vapour pressures that can be interpolated 

54 through to obtain an SVP value for any temperature within the range 

55 t_min --> (t_max - t_increment). 

56 

57 The default min/max values create a table that provides SVP values 

58 covering the temperature range -90C to +65.1C. Note that the last 

59 bin is not used, so the SVP value corresponding to +65C is the highest 

60 that will be used. 

61 

62 Args: 

63 t_min: 

64 The minimum temperature for the range, in Kelvin. 

65 t_max: 

66 The maximum temperature for the range, in Kelvin. 

67 t_increment: 

68 The temperature increment at which to create values for the 

69 saturated vapour pressure between t_min and t_max. 

70 water_only: 

71 The table will only create values for the saturated vapour 

72 pressure with respect to water. Values for temperatures 

73 outside the range 223 K < T < 373 K have not been 

74 experimentally validated. 

75 ice_only: 

76 The table will only create values for the saturated vapour 

77 pressure with respect to ice. Values for temperatures 

78 outside the range 173 K < T < 273.15 K have not been 

79 experimentally validated. 

80 

81 """ 

82 self.t_min = t_min 

83 self.t_max = t_max 

84 self.t_increment = t_increment 

85 self.water_only = water_only 

86 self.ice_only = ice_only 

87 

88 if self.water_only and self.ice_only: 

89 raise ValueError( 

90 "'water_only' and 'ice_only' flags cannot both be set to True" 

91 ) 

92 

93 def __repr__(self) -> str: 

94 """Represent the configured plugin instance as a string.""" 

95 result = ( 

96 f"<{self.__class__.__name__}: t_min: {self.t_min}; t_max: {self.t_max}; " 

97 f"t_increment: {self.t_increment}>" 

98 ) 

99 return result

101 def saturation_vapour_pressure_goff_gratch(self, temperature: ndarray) -> ndarray: 

102 """ 

103 Saturation Vapour pressure in a water vapour system calculated using 

104 the Goff-Gratch Equation (WMO standard method). 

105 

106 Args: 

107 temperature: 

108 Temperature values in Kelvin. Valid from 173 K to 373 K 

109 

110 Returns: 

111 Corresponding values of saturation vapour pressure for a pure 

112 water vapour system, in hPa. 

113 

114 References: 

115 Numerical data and functional relationships in science and 

116 technology. New series. Group V. Volume 4. Meteorology. 

117 Subvolume b. Physical and chemical properties of the air, P35. 

118 """ 

119 self._check_temperature_limits(temperature) 

120 

121 # Iterate over the temperature array, updating the temperature values 

122 # with newly calculated saturation vapour pressure values. 

123 svp = temperature.copy() 

124 with np.nditer(svp, op_flags=["readwrite"]) as it: 

125 for cell in it: 

126 if (cell > TRIPLE_PT_WATER or self.water_only) and not self.ice_only: 

127 n0 = self.constants[1] * (1.0 - TRIPLE_PT_WATER / cell) 

128 n1 = self.constants[2] * np.log10(cell / TRIPLE_PT_WATER) 

129 n2 = self.constants[3] * ( 

130 1.0 

131 - np.power( 

132 10.0, (self.constants[4] * (cell / TRIPLE_PT_WATER - 1.0)) 

133 ) 

134 ) 

135 n3 = self.constants[5] * ( 

136 np.power( 

137 10.0, (self.constants[6] * (1.0 - TRIPLE_PT_WATER / cell)) 

138 ) 

139 - 1.0 

140 ) 

141 log_es = n0 - n1 + n2 + n3 + self.constants[7] 

142 cell[...] = np.power(10.0, log_es) 

143 else: 

144 n0 = self.constants[8] * ((TRIPLE_PT_WATER / cell) - 1.0) 

145 n1 = self.constants[9] * np.log10(TRIPLE_PT_WATER / cell) 

146 n2 = self.constants[10] * (1.0 - (cell / TRIPLE_PT_WATER)) 

147 log_es = n0 - n1 + n2 + self.constants[11] 

148 cell[...] = np.power(10.0, log_es) 

149 

150 return svp

152 def _check_temperature_limits(self, temperature: ndarray): 

153 """ 

154 Raise exception if temperature values fall outside the range for which the 

155 method is considered valid (see reference). 

156 

157 Args: 

158 temperature (ndarray): 

159 Array of temperature values to be validated. 

160 

161 Raises: 

162 UserWarning: 

163 If any temperature value is outside the valid range defined by 

164 self.MIN_VALID_TEMPERATURE_ICE and self.MAX_VALID_TEMPERATURE_WATER, 

165 a warning is issued. 

166 

167 If either self.water_only or self.ice_only has been set to True, the 

168 warning will use the corresponding minimum and maximum temperature 

169 values. 

170 

171 Returns: 

172 None 

173 """ 

174 if ( 

175 temperature.max() > self.MAX_VALID_TEMPERATURE_WATER 

176 or temperature.min() < self.MIN_VALID_TEMPERATURE_ICE 

177 ) and not (self.water_only or self.ice_only): 

178 msg = ( 

179 f"Temperatures out of SVP table range: min {temperature.min()}, max {temperature.max()} " 

180 f"(valid in range {self.MIN_VALID_TEMPERATURE_ICE:.0f} K < T < {self.MAX_VALID_TEMPERATURE_WATER:.0f} K)" 

181 ) 

182 warnings.warn(msg) 

183 elif ( 

184 temperature.max() > self.MAX_VALID_TEMPERATURE_WATER 

185 or temperature.min() < self.MIN_VALID_TEMPERATURE_WATER 

186 ) and self.water_only: 

187 msg = ( 

188 f"Temperatures out of SVP table range for water: min {temperature.min()}, max {temperature.max()} " 

189 f"(valid in range {self.MIN_VALID_TEMPERATURE_WATER:.0f} K < T < {self.MAX_VALID_TEMPERATURE_WATER:.0f} K)" 

190 ) 

191 warnings.warn(msg) 

192 elif ( 

193 temperature.max() > self.MAX_VALID_TEMPERATURE_ICE 

194 or temperature.min() < self.MIN_VALID_TEMPERATURE_ICE 

195 ) and self.ice_only: 

196 msg = ( 

197 f"Temperatures out of SVP table range for ice: min {temperature.min()}, max {temperature.max()} " 

198 f"(valid in range {self.MIN_VALID_TEMPERATURE_ICE:.0f} K < T < {self.MAX_VALID_TEMPERATURE_ICE:.2f} K)" 

199 ) 

200 warnings.warn(msg)

202 def as_cube(self, svp_data: np.ndarray, temperatures: np.ndarray) -> Cube: 

203 """ 

204 Converts saturation vapor pressure data and corresponding temperatures into an Iris Cube. 

205 

206 Args: 

207 svp_data (np.ndarray): 

208 Array containing saturation vapor pressure values. 

209 temperatures (np.ndarray): 

210 Array of temperature values (in Kelvin) corresponding to the svp_data. 

211 

212 Returns 

213 Cube: 

214 An Iris Cube containing the saturation vapor pressure data with temperature as a dimension coordinate. 

215 The cube is converted to SI units for vapor pressure and includes attributes for the minimum temperature, 

216 maximum temperature, and temperature increment used in the data. 

217 """ 

218 temperature_coord = iris.coords.DimCoord( 

219 temperatures, "air_temperature", units="K" 

220 ) 

221 # Output of the Goff-Gratch is in hPa, but we want to return in Pa. 

222 svp = iris.cube.Cube( 

223 svp_data, 

224 long_name=self.cube_name, 

225 units=self.svp_units, 

226 dim_coords_and_dims=[(temperature_coord, 0)], 

227 ) 

228 svp.convert_units(self.svp_si_units) 

229 svp.attributes["minimum_temperature"] = self.t_min 

230 svp.attributes["maximum_temperature"] = self.t_max 

231 svp.attributes["temperature_increment"] = self.t_increment 

232 return svp

234 def process(self) -> Cube: 

235 """ 

236 Creates a lookup table of saturation vapour pressure in a pure water vapour system for a specified temperature range. 

237 

238 Args: 

239 self.t_min (float): 

240 Minimum temperature (inclusive) for the lookup table. 

241 self.t_max (float): 

242 Maximum temperature (inclusive) for the lookup table. 

243 self.t_increment (float): 

244 Temperature increment for the lookup table. 

245 self.saturation_vapour_pressure_goff_gratch (callable): 

246 Method to calculate saturation vapour pressure for given temperatures. 

247 self.as_cube (callable): 

248 Method to convert data and temperatures into a Cube object. 

249 

250 Returns: 

251 Cube: 

252 A cube of saturated vapour pressure values at temperature points defined by t_min, t_max, and t_increment. 

253 """ 

254 temperatures = np.arange( 

255 self.t_min, self.t_max + 0.5 * self.t_increment, self.t_increment 

256 ) 

257 svp_data = self.saturation_vapour_pressure_goff_gratch(temperatures) 

258 

259 svp = self.as_cube(svp_data, temperatures) 

260 

261 return svp
