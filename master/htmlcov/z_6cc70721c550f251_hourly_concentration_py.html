5"""Calculations to produce Pollen Hourly Concentration values.""" 

6 

7import numpy as np 

8from iris.cube import Cube 

9 

10from improver import PostProcessingPlugin 

11from improver.metadata.constants import FLOAT_DTYPE 

12 

13from .utilities import build_output_cube_with_new_units 

14 

15 

16class PollenHourlyConcentration(PostProcessingPlugin): 

17 """Plugin to calculate the Pollen Hourly Concentration. 

18 

19 The input cube for this plugin comes from the output of the 

20 Numerical Atmospheric dispersion Modelling Environment (NAME). 

21 It is 2D gridded data of pollen concentrations given as g/m3 and 

22 the plugin converts this to concentrations in grains/m3 using 

23 pollen diameter and density. 

24 """ 

25 

26 # The names of pollen types that are expected by this class 

27 _POLLEN_NAMES = [ 

28 "grass", 

29 "birch", 

30 "oak", 

31 "hazel", 

32 "alder", 

33 "nettle", 

34 "ash", 

35 "plane", 

36 "weed", 

37 ] 

38 # Diameter of pollen grains for each pollen type in metres 

39 _POLLEN_DIAMETER = { 

40 "grass": 35e-6, 

41 "birch": 22e-6, 

42 "oak": 29e-6, 

43 "hazel": 28e-6, 

44 "alder": 25e-6, 

45 "nettle": 13e-6, 

46 "ash": 23e-6, 

47 "plane": 19e-6, 

48 "weed": 13e-6, 

49 } # meters 

50 # Density of each pollen type in kg per cubic metre 

51 _POLLEN_DENSITY = { 

52 "grass": 1000.0, 

53 "birch": 800.0, 

54 "oak": 800.0, 

55 "hazel": 800.0, 

56 "alder": 800.0, 

57 "nettle": 1000.0, 

58 "ash": 800.0, 

59 "plane": 920.0, 

60 "weed": 1000.0, 

61 } # kg/m3 

62 _POLLEN_SHORTNAME_2_LATIN = { 

63 "grass": "Poaceae", 

64 # Trees 

65 "birch": "Betula", 

66 "oak": "Quercus", 

67 "hazel": "Corylus", 

68 "alder": "Alnus", 

69 "ash": "Fraxinus", 

70 "plane": "Platanus", 

71 # Weeds 

72 "nettle": "Urticaceae", 

73 "weed": "Urticaceae", 

74 } 

75 _POLLEN_SHORTNAME_2_LONGNAME = { 

76 k: "number_concentration_of_" + v.lower() + "_pollen_grains_in_air" 

77 for k, v in _POLLEN_SHORTNAME_2_LATIN.items() 

78 } 

79 

80 # Scaling factors can change, so need to be passed in 

81 _scaling_factors_dict = None 

82 

83 # The output cube is a deepcopy of the input cube (to keep metadata) and is then manipulated in place 

84 _output_cube = None 

85 

86 def __init__(self, scaling_factors_dict: dict = None) -> None: 

87 """Initialise class. 

88 

89 Args: 

90 scaling_factors_dict: 

91 Optional scaling factors to use per pollen type 

92 """ 

93 self._scaling_factors_dict = scaling_factors_dict 

94 

95 def _calculate(self, taxa: str): 

96 """Perform calculations on input cube. 

97 

98 Applies the scaling factor to the raw data for the relevant pollen taxa, 

99 and converts from g/m3 to grains/m3 using pollen diameter and density. 

100 

101 Args: 

102 taxa: 

103 The pollen taxa being processed, used to update the cube name and metadata 

104 """ 

105 if self._scaling_factors_dict is not None: 

106 self._scaling_factor = self._scaling_factors_dict[taxa] 

107 else: 

108 self._scaling_factor = 1.0 

109 diameter = self._POLLEN_DIAMETER[taxa] 

110 density = self._POLLEN_DENSITY[taxa] 

111 volume = (4 / 3) * np.pi * (diameter / 2) ** 3 

112 mass_per_grain = volume * density 

113 

114 # Data is in g/m3, so convert to kg/m3 by dividing by 1000, 

115 # (then apply scaling factor) 

116 # and convert to grains/m3 

117 new_data = ( 

118 (self._output_cube.data / 1000.0) * self._scaling_factor 

119 ) / mass_per_grain 

120 self._output_cube.data = new_data.astype(FLOAT_DTYPE) 

121 

122 def _metadata(self, taxa: str): 

123 """Change the cube name and other metadata. 

124 Args: 

125 taxa: 

126 The pollen taxa being processed, used to update the cube name and metadata 

127 """ 

128 self._output_cube.attributes["biological_taxon_name"] = ( 

129 self._POLLEN_SHORTNAME_2_LATIN[taxa] 

130 ) 

131 self._output_cube.attributes["forecast_period"] = np.int32(3600) 

132 self._output_cube.attributes["scaling_factor"] = self._scaling_factor 

133 self._output_cube.rename(self._POLLEN_SHORTNAME_2_LONGNAME[taxa]) 

134 

135 def process( 

136 self, 

137 cube: Cube, 

138 ) -> Cube: 

139 """Calculate the Pollen Concentrations. 

140 

141 Args: 

142 cube: 

143 Input cube for any pollen type handled by the class 

144 scaling_factors_dict: 

145 Optional scaling factors to use per pollen type 

146 

147 Returns: 

148 The calculated output cube. 

149 """ 

150 self._output_cube = build_output_cube_with_new_units(self, cube, "m-3") 

151 

152 # Check that the pollen taxa is one that is handled by the class 

153 taxa = self._output_cube.attributes.get("taxa").lower() 

154 # Remove "_pollen" from the taxa name if it is present, to match the keys in the dictionaries 

155 taxa = taxa.lower().replace("_pollen", "") 

156 if taxa not in self._POLLEN_NAMES: 

157 raise ValueError(f"Pollen taxa {taxa} not handled") 

158 

159 self._calculate(taxa) 

160 self._metadata(taxa) 

161 return self._output_cube
