44@functools.lru_cache() 

45def _svp_table(phase: Optional[str] = None) -> ndarray: 

46 """ 

47 Calculate a saturated vapour pressure (SVP) lookup table. 

48 The lru_cache decorator caches this table on first call to this function, 

49 so that the table does not need to be re-calculated if used multiple times. 

50 

51 A value of SVP for any temperature between T_MIN and T_MAX (inclusive) can be 

52 obtained by interpolating through the table, as is done in the _svp_from_lookup 

53 function. 

54 

55 Args: 

56 phase: 

57 If set to 'water' or 'ice', will create a table with respect to that 

58 phase only. 

59 

60 Returns: 

61 Array of saturated vapour pressures (Pa). 

62 """ 

63 if str(phase).lower() == "water": 

64 svp = SaturatedVapourPressureTable( 

65 t_min=SVP_T_MIN, 

66 t_max=SVP_T_MAX, 

67 t_increment=SVP_T_INCREMENT, 

68 water_only=True, 

69 ) 

70 elif str(phase).lower() == "ice": 

71 svp = SaturatedVapourPressureTable( 

72 t_min=SVP_T_MIN, t_max=SVP_T_MAX, t_increment=SVP_T_INCREMENT, ice_only=True 

73 ) 

74 else: 

75 svp = SaturatedVapourPressureTable( 

76 t_min=SVP_T_MIN, t_max=SVP_T_MAX, t_increment=SVP_T_INCREMENT 

77 ) 

78 return svp.process().data

81@functools.lru_cache() 

82def _svp_derivative_table() -> ndarray: 

83 """ 

84 Calculate a saturated vapour pressure (SVP) derivative lookup table. 

85 The lru_cache decorator caches this table on first call to this function, 

86 so that the table does not need to be re-calculated if used multiple times. 

87 

88 A value of SVP derivative for any temperature between T_MIN and T_MAX (inclusive) can be 

89 obtained by interpolating through the table, as is done in the _svp_derivative_from_lookup 

90 function. 

91 

92 Returns: 

93 Array of first derivative saturated vapour pressures (Pa). 

94 """ 

95 svp_derivative_data = SaturatedVapourPressureDerivativeTable( 

96 t_min=SVP_T_MIN, t_max=SVP_T_MAX, t_increment=SVP_T_INCREMENT 

97 ).process() 

98 return svp_derivative_data.data

101def _svp_from_lookup(temperature: ndarray, phase: Optional[str] = None) -> ndarray: 

102 """ 

103 Gets value for saturation vapour pressure in a pure water vapour system 

104 from a pre-calculated lookup table. Interpolates linearly between points in 

105 the table to the temperatures required. 

106 

107 Args: 

108 temperature: 

109 Array of air temperatures (K). 

110 phase: 

111 If set to 'water' or 'ice', will use a lookup table containing 

112 values with respect to that phase only. 

113 

114 Returns: 

115 Array of saturated vapour pressures (Pa). 

116 """ 

117 # where temperatures are outside the SVP table range, clip data to 

118 # within the available range 

119 t_clipped = np.clip(temperature, SVP_T_MIN, SVP_T_MAX - SVP_T_INCREMENT) 

120 

121 # interpolate between bracketing values 

122 table_position = (t_clipped - SVP_T_MIN) / SVP_T_INCREMENT 

123 table_index = table_position.astype(int) 

124 interpolation_factor = table_position - table_index 

125 svp_table_data = _svp_table(phase) 

126 return (1.0 - interpolation_factor) * svp_table_data[ 

127 table_index 

128 ] + interpolation_factor * svp_table_data[table_index + 1] 

129

131def _svp_derivative_from_lookup(temperature: ndarray) -> ndarray: 

132 """ 

133 Gets value for saturation vapour pressure derivative in a pure water vapour system 

134 from a pre-calculated lookup table. Interpolates linearly between points in 

135 the table to the temperatures required. 

136 

137 Args: 

138 temperature: 

139 Array of air temperatures (K). 

140 

141 Returns: 

142 Array of first derivative saturated vapour pressures (Pa). 

143 """ 

144 # where temperatures are outside the SVP derivative table range, clip data to 

145 # within the available range 

146 t_clipped = np.clip(temperature, SVP_T_MIN, SVP_T_MAX - SVP_T_INCREMENT) 

147 

148 # interpolate between bracketing values 

149 table_position = (t_clipped - SVP_T_MIN) / SVP_T_INCREMENT 

150 table_index = table_position.astype(int) 

151 interpolation_factor = table_position - table_index 

152 svp_derivative_table_data = _svp_derivative_table() 

153 return (1.0 - interpolation_factor) * svp_derivative_table_data[ 

154 table_index 

155 ] + interpolation_factor * svp_derivative_table_data[table_index + 1]

158def calculate_svp_in_air( 

159 temperature: ndarray, pressure: ndarray, phase: Optional[str] = None 

160) -> ndarray: 

161 """ 

162 Calculates the saturation vapour pressure in air. Looks up the saturation 

163 vapour pressure (SVP) in a pure water vapour system, and pressure-corrects 

164 the result to obtain the saturation vapour pressure in air. 

165 

166 Args: 

167 temperature: 

168 Array of air temperatures (K). 

169 pressure: 

170 Array of pressure (Pa). 

171 phase: 

172 If set to 'water' or 'ice', will use a SVP lookup table containing 

173 values with respect to that phase only. 

174 

175 Returns: 

176 Saturation vapour pressure in air (Pa). 

177 

178 References: 

179 Atmosphere-Ocean Dynamics, Adrian E. Gill, International Geophysics 

180 Series, Vol. 30; Equation A4.7. 

181 """ 

182 svp = _svp_from_lookup(temperature, phase) 

183 temp_C = temperature + consts.ABSOLUTE_ZERO 

184 correction = 1.0 + 1.0e-8 * pressure * (4.5 + 6.0e-4 * temp_C * temp_C) 

185 return svp * correction.astype(np.float32)

188def calculate_svp_derivative_in_air(temperature: ndarray, pressure: ndarray) -> ndarray: 

189 """ 

190 Calculates the saturation vapour pressure derivative in air. Looks up the saturation 

191 vapour pressure derivative in a pure water vapour system, and pressure-corrects the 

192 result to obtain the saturation vapour pressure derivative in air. 

193 

194 Args: 

195 temperature: 

196 Array of air temperatures (K). 

197 pressure: 

198 Array of pressure (Pa). 

199 

200 Returns: 

201 Saturation vapour pressure derivative in air (Pa). 

202 

203 References: 

204 Atmosphere-Ocean Dynamics, Adrian E. Gill, International Geophysics 

205 Series, Vol. 30; Equation A4.7. 

206 """ 

207 svp = _svp_from_lookup(temperature) 

208 svp_derivative = _svp_derivative_from_lookup(temperature) 

209 temp_C = temperature + consts.ABSOLUTE_ZERO 

210 correction = 1.0 + 1.0e-8 * pressure * (4.5 + 6.0e-4 * temp_C * temp_C) 

211 derivative_correction_term = correction * svp_derivative + ( 

212 2 * 1.0e-8 * 6.0e-4 * pressure * temp_C * svp 

213 ) 

214 return svp_derivative * derivative_correction_term.astype(np.float32)

271def saturated_humidity(temperature: ndarray, pressure: ndarray) -> ndarray: 

272 """ 

273 Calculate specific humidity mixing ratio of saturated air of given temperature and pressure. 

274 

275 Invalid and masked values are filtered and result in NaN outputs. 

276 

277 Args: 

278 temperature: 

279 Air temperature (K) 

280 pressure: 

281 Air pressure (Pa) 

282 

283 Returns: 

284 Array of specific humidity values (kg kg-1) representing saturated air. NaN is returned 

285 if either input is NaN. 

286 

287 Method from referenced documentation. Note that EARTH_REPSILON is 

288 simply given as an unnamed constant in the reference (0.62198). 

289 

290 References: 

291 ASHRAE Fundamentals handbook (2005) Equation 22, 24, p6.8 

292 """ 

293 mask, pressure_allvalid, temperature_allvalid = _to_valid_values( 

294 pressure, temperature 

295 ) 

296 

297 # Calculate saturated humidity 

298 svp = calculate_svp_in_air(temperature_allvalid, pressure_allvalid) 

299 numerator = consts.EARTH_REPSILON * svp 

300 denominator = np.maximum(svp, pressure_allvalid) - ( 

301 (1.0 - consts.EARTH_REPSILON) * svp 

302 ) 

303 result = (numerator / denominator).astype(temperature.dtype) 

304 

305 result_is_masked = isinstance(temperature, np.ma.MaskedArray) or isinstance( 

306 pressure, np.ma.MaskedArray 

307 ) 

308 result = _reapply_mask(mask, result, result_is_masked).astype(temperature.dtype) 

309 return result

312def _reapply_mask( 

313 mask: np.ndarray, array_to_mask: np.ndarray, as_masked_array: bool 

314) -> np.ndarray: 

315 """Reapply mask to data array 

316 

317 Args: 

318 mask: 

319 Boolean array where True indicates invalid data points. 

320 array_to_mask: 

321 Data array to which the mask will be applied. 

322 as_masked_array: 

323 If True, return a numpy MaskedArray. If False, return a regular ndarray 

324 with NaNs in place of masked values. 

325 Returns: 

326 Masked array or ndarray with NaNs where mask is True. 

327 """ 

328 array_to_mask[mask] = np.nan 

329 if as_masked_array: 

330 array_to_mask = np.ma.MaskedArray(data=array_to_mask, mask=mask) 

331 return array_to_mask

334def _to_valid_values( 

335 pressure: np.ndarray, temperature: np.ndarray 

336) -> Tuple[np.ndarray, np.ndarray, np.ndarray]: 

337 """Identify and replace any invalid values with dummy values for calculation purposes. 

338 Invalid data points are where either the temperature or the pressure array have masked 

339 data points, or non-finite values (NaN or Inf). 

340 The safe values are 1000 hPa and 0 degC (273.15 K). 

341 

342 Args: 

343 pressure: 

344 Array of air pressures (Pa). 

345 temperature: 

346 Array of air temperatures (K). 

347 Returns: 

348 Tuple of: 

349 - mask: Boolean array where True indicates invalid data points. 

350 - pressure: Array of air pressures (Pa) with invalid values replaced. 

351 - temperature: Array of air temperatures (K) with invalid values replaced. 

352 """ 

353 mask = np.full_like(temperature, False) 

354 if isinstance(temperature, np.ma.MaskedArray): 

355 mask = np.logical_or(mask, temperature.mask) 

356 if isinstance(pressure, np.ma.MaskedArray): 

357 mask = np.logical_or(mask, pressure.mask) 

358 mask = np.logical_or(mask, np.logical_not(np.isfinite(temperature))) 

359 mask = np.logical_or(mask, np.logical_not(np.isfinite(pressure))) 

360 temperature_allvalid = np.where(mask, 273.15, temperature) 

361 pressure_allvalid = np.where(mask, 100000.0, pressure) 

362 return mask, pressure_allvalid, temperature_allvalid

365def _calculate_latent_heat(temperature: ndarray) -> ndarray: 

366 """ 

367 Calculate a temperature adjusted latent heat of condensation for water 

368 vapour using the relationship employed by the UM. 

369 

370 Args: 

371 temperature: 

372 Array of air temperatures (K). 

373 

374 Returns: 

375 Temperature adjusted latent heat of condensation (J kg-1). 

376 """ 

377 temp_Celsius = temperature + consts.ABSOLUTE_ZERO 

378 latent_heat = ( 

379 -1.0 * consts.LATENT_HEAT_T_DEPENDENCE * temp_Celsius 

380 + consts.LH_CONDENSATION_WATER 

381 ) 

382 return latent_heat

385def _latent_heat_release(q1: ndarray, q2: ndarray, temperature: ndarray) -> ndarray: 

386 """Returns the latent heat released (K) when condensing water vapour from specific humidity 

387 value q1 to q2, both in kg kg-1 when the temperature is approximately t. 

388 Returns negative values when initial condition is subsaturated as this method assumes there 

389 is always liquid water present which can be evaporated. 

390 

391 Args: 

392 temperature: 

393 Array of air temperatures (K). Ideally, the average air temperature between q1 and q2 

394 q1: 

395 Specific humidity before latent heat release (kg kg-1) 

396 q2: 

397 Specific humidity after latent heat release (kg kg-1) 

398 

399 Returns: 

400 Temperature adjustment to apply to account for latent heat release (K). 

401 """ 

402 return (_calculate_latent_heat(temperature) / consts.CP_DRY_AIR) * (q1 - q2)

405def adjust_for_latent_heat( 

406 temperature_in: ndarray, humidity_in: ndarray, pressure: ndarray 

407) -> Tuple[ndarray, ndarray]: 

408 """ 

409 Increases temperature and reduces humidity via latent heat release from condensation until 

410 values represent 100% relative humidity. 

411 

412 Subsaturated values will be returned unaltered. 

413 

414 This method uses the scipy newton solver with a limit of 6 iterations. 

415 The deepest convection needs more iterations to converge. This is only important 

416 if we reach the position that all points in an array fail to converge at the same 

417 pressure level, because the solver raises an exception (although docs say it shouldn't). 

418 I haven't seen this be true except when the array contains only one point, but increasing 

419 the maximum number of iterations for small arrays is a very small price to pay for 

420 stability. 

421 

422 

423 .. See the documentation for a more detailed discussion of the maths. 

424 .. include:: extended_documentation/psychrometric_calculations/ 

425 psychrometric_calculations/adjust_for_latent_heat.rst 

426 

427 Args: 

428 temperature_in: 

429 The parcel temperature following a dry adiabatic cooling (K) 

430 humidity_in: 

431 The atmosphere specific humidity at the same points (kg kg-1) 

432 pressure: 

433 The atmospheric pressure at the same points (Pa) 

434 

435 Returns: 

436 tuple of temperature (K) and humidity (kg kg-1) after saturated latent heat release 

437 """ 

438 

439 def qsat_differential(qs, t, q, p): 

440 """For a given set of temperature (t), specific humidity (q) and pressure (p), 

441 and a saturated humidity guess (qs), calculate an adjusted temperature after 

442 latent heat release and return the difference between the saturated humidity 

443 at that adjusted temperature and the guess.""" 

444 adj_t = t + _latent_heat_release(q, qs, t) 

445 return saturated_humidity(adj_t, p) - qs 

446 

447 humidity = newton( 

448 qsat_differential, 

449 humidity_in.copy(), 

450 args=(temperature_in, humidity_in, pressure), 

451 tol=1e-6, 

452 maxiter=6 if humidity_in.size > 100 else 10, 

453 disp=True, 

454 ).astype(np.float32) 

455 temperature = temperature_in + _latent_heat_release( 

456 humidity_in, humidity, temperature_in 

457 ) 

458 sub_saturated = np.where(temperature < temperature_in) 

459 temperature[sub_saturated] = temperature_in[sub_saturated] 

460 humidity[sub_saturated] = humidity_in[sub_saturated] 

461 return temperature, humidity

505 def generate_pressure_cube(self, temperature_cube) -> Cube: 

506 """ 

507 Generate a pressure cube from the pressure coordinate on the temperature cube. 

508 

509 If there is a pressure coordinate in the temperature and relative humidity cubes, and 

510 no pressure cube has been provided (as is the case for calculating virtual temperature 

511 on pressure levels, for example) the pressure cube is generated from the from the 

512 pressure coordinate on the temperature cube. The temperature cube has a status flag 

513 that indicates where the data were derived by StaGE for data points that fell below 

514 the model orography, the flag meaning is above_surface_pressure below_surface_pressure. 

515 These values are expanded in the process of concatenating the cube and if they are to 

516 be retained, should be copied back in to the output of the final calculation from the 

517 input cubes. 

518 

519 See https://scitools-iris.readthedocs.io/en/stable/further_topics/controlling_merge.html 

520 for more information 

521 """ 

522 

523 coord_list = [coord.name() for coord in temperature_cube.coords()] 

524 pressure_list = self._make_pressure_list(temperature_cube) 

525 

526 try: 

527 expanded_pressure_list = CubeList( 

528 iris.util.new_axis(cube, "pressure", expand_extras=["status_flag"]) 

529 for cube in pressure_list 

530 ) 

531 except KeyError: 

532 expanded_pressure_list = CubeList( 

533 iris.util.new_axis(cube, "pressure") for cube in pressure_list 

534 ) 

535 

536 try: 

537 pressure_cube = expanded_pressure_list.concatenate_cube() 

538 except iris.exceptions.ConcatenateError as error: 

539 raise RuntimeError( 

540 "Unable to concatenate pressure cubelist with input ", 

541 expanded_pressure_list, 

542 error, 

543 ) 

544 

545 enforce_coordinate_ordering(pressure_cube, coord_list) 

546 pressure_cube.rename("surface_air_pressure") 

547 pressure_cube.units = "Pa" 

548 return pressure_cube

550 def _handle_zero_humidity(self): 

551 """Sets the minimum humidity value to half the least significant value 

552 to avoid issues with zero humidity inputs that can result in unphysical values.""" 

553 min_humidity = 0.5 * minimum_increment(self.rel_humidity, default=0.001) 

554 self.rel_humidity.data = np.where( 

555 self.rel_humidity.data < min_humidity, min_humidity, self.rel_humidity.data 

556 )

558 def process(self, *cubes: Union[Cube, CubeList]) -> Cube: 

559 """ 

560 Calculates the humidity mixing ratio from the inputs. The inputs can be on height levels 

561 or pressure levels, and the output will be on the same levels. If the input cubes are on 

562 pressure levels then a pressure cube doesn't need to be provided and the pressure levels 

563 will be inferred from the pressure coordinate on the temperature cube. 

564 

565 Args: 

566 cubes: 

567 Cubes of temperature (K) and relative humidity (1). A cube of pressure (Pa) must also 

568 be provided unless there is a pressure coordinate in the temperature and relative 

569 humidity cubes. 

570 Returns: 

571 Cube of humidity mixing ratio on same levels as input cubes 

572 """ 

573 cubes = as_cubelist(*cubes) 

574 self.rel_humidity = cubes.extract_cube( 

575 iris.Constraint(name="relative_humidity") 

576 ) 

577 self._handle_zero_humidity() 

578 

579 try: 

580 # Test if there is a cube with air_temperature in the name 

581 def test_temperature(cube): 

582 return True if "air_temperature" in cube.name() else False 

583 

584 self.temperature = cubes.extract_cube( 

585 iris.Constraint(cube_func=test_temperature) 

586 ) 

587 except ConstraintMismatchError as err: 

588 raise ValueError("No cube with name 'air_temperature' found") from err 

589 

590 try: 

591 # Test if there is one, and only one, cube with pressure in the name 

592 def test_pressure(cube): 

593 return True if "pressure" in cube.name() else False 

594 

595 self.pressure = cubes.extract_cube(iris.Constraint(cube_func=test_pressure)) 

596 

597 except ConstraintMismatchError as err: 

598 # If more than one pressure cube is provided, raise an error explaining this 

599 import re 

600 

601 more_than_one = re.search(r"Got\s([2-9]|\d\d\d*)\scubes", str(err)) 

602 if more_than_one: 

603 raise ValueError(f"{more_than_one.group()} with 'pressure' in name.") 

604 

605 # If no pressure cube is provided, check if pressure is a coordinate in the temperature and relative humidity cubes 

606 temp_cube_has_pressure_coord = any( 

607 coord.name() == "pressure" for coord in self.temperature.coords() 

608 ) 

609 rh_cube_has_pressure_coord = any( 

610 coord.name() == "pressure" for coord in self.rel_humidity.coords() 

611 ) 

612 if temp_cube_has_pressure_coord & rh_cube_has_pressure_coord: 

613 self.pressure = self.generate_pressure_cube(self.temperature) 

614 else: 

615 raise ValueError( 

616 "No pressure cube with name 'pressure' found and no pressure coordinate found in temperature or relative humidity cubes"

690 def find_falling_level( 

691 self, wb_int_data: ndarray, orog_data: ndarray, height_points: ndarray 

692 ) -> ndarray: 

693 """ 

694 Find the phase change level by finding the level of the wet-bulb 

695 integral data at the required threshold. Wet-bulb integral data 

696 is only available above ground level and there may be an insufficient 

697 number of levels in the input data, in which case the required 

698 threshold may lie outside the Wet-bulb integral data and the value 

699 at that point will be set to np.nan. 

700 

701 Args: 

702 wb_int_data: 

703 Wet bulb integral data on heights 

704 orog_data: 

705 Orographic data 

706 height_points: 

707 heights agl 

708 

709 Returns: 

710 Phase change level data asl. 

711 """ 

712 from stratify import interpolate 

713 

714 # Create cube of heights above sea level for each height in 

715 # the wet bulb integral cube. 

716 asl = wb_int_data.copy() 

717 for i, height in enumerate(height_points): 

718 asl[i, ::] = orog_data + height 

719 

720 # Calculate phase change level above sea level by 

721 # finding the level corresponding to the falling_level_threshold. 

722 # Interpolate returns an array with height indices 

723 # for falling_level_threshold so we take the 0 index 

724 phase_change_level_data = interpolate( 

725 np.array([self.falling_level_threshold]), wb_int_data, asl, axis=0 

726 )[0] 

727 

728 return phase_change_level_data

733 orog_data: ndarray, 

734 highest_wb_int_data: ndarray, 

735 highest_height: float, 

736 ) -> None: 

737 """ 

738 Fill in any data in the phase change level where the whole wet bulb

740 Set these points to the highest height level + orography. 

741 

742 Args:

744 Phase change level data (m).

746 Orographic data (m)

760 def find_extrapolated_falling_level( 

761 self, 

762 max_wb_integral: ndarray, 

763 gradient: ndarray, 

764 intercept: ndarray, 

765 phase_change_level_data: ndarray, 

766 sea_points: ndarray, 

767 ) -> None: 

768 r""" 

769 Find the phase change level below sea level using the linear 

770 extrapolation of the wet bulb temperature integral and update the 

771 phase change level array with these values. 

772 

773 

774 The phase change level is calculated from finding the point where the 

775 integral of wet bulb temperature crosses the falling level threshold. 

776 

777 In cases where the wet bulb temperature integral has not reached the 

778 threshold by the time we reach sea level, we can find a fit to the wet 

779 bulb temperature profile near the surface, and use this to estimate 

780 where the phase change level would be below sea level. 

781 

782 The difference between the wet bulb temperature integral at the 

783 threshold and the wet bulb integral at the surface is equal to the 

784 integral of the wet bulb temperature between sea level and 

785 the negative height corresponding to the phase change level. As we are 

786 using a simple linear fit, we can integrate this to find an expression 

787 for the extrapolated phase change level. 

788 

789 The form of this expression depends on whether the linear fit of wet 

790 bulb temperature crosses the height axis above or below zero altitude. 

791 

792 If we have our linear fit of the form: 

793 

794 .. math:: 

795 {{wet\:bulb\:temperature} = m \times height + c} 

796 

797 and let :math:`I` be the wet bulb temperature integral we have found 

798 above sea level. 

799 

800 If it crosses above zero, then the limits on the integral 

801 are the phase change level and zero and we find the following 

802 expression for the phase change level: 

803 

804 .. math:: 

805 {{phase\:change\:level} = \frac{c \pm \sqrt{ 

806 c^2-2 m (threshold-I)}}{-m}} 

807 

808 If the linear fit crosses below zero the limits on our integral are 

809 the phase change level and the point where the linear fit crosses the 

810 height axis, as only positive wet bulb temperatures count towards the 

811 integral. In this case our expression for the phase change level is: 

812 

813 .. math:: 

814 {{phase\:change\:level} = \frac{c \pm \sqrt{ 

815 2 m (I-threshold)}}{-m}} 

816 

817 Args: 

818 max_wb_integral: 

819 The wet bulb temperature integral at sea level. 

820 gradient: 

821 The gradient of the line of best fit we are using in the 

822 extrapolation. 

823 intercept: 

824 The intercept of the line of best fit we are using in the 

825 extrapolation. 

826 phase_change_level_data: 

827 The phase change level array with values filled in with phase 

828 change levels calculated through extrapolation. 

829 sea_points: 

830 A boolean array with True where the points are sea points. 

831 """ 

832 

833 # Make sure we only try to extrapolate points with a valid gradient. 

834 index = (gradient < 0.0) & sea_points 

835 gradient = gradient[index] 

836 intercept = intercept[index] 

837 max_wb_int = max_wb_integral[index] 

838 phase_cl = phase_change_level_data[index] 

839 

840 # For points where -intercept/gradient is greater than zero: 

841 index2 = -intercept / gradient >= 0.0 

842 intercept2 = intercept[index2] 

843 gradient2 = gradient[index2] 

844 inside_sqrt = intercept2 * intercept2 - 2 * gradient2 * ( 

845 self.falling_level_threshold - max_wb_int[index2] 

846 ) 

847 phase_cl[index2] = (intercept2 - np.sqrt(inside_sqrt)) / -gradient2 

848 

849 # For points where -intercept/gradient is less than zero: 

850 index2 = -intercept / gradient < 0.0 

851 intercept2 = intercept[index2] 

852 gradient2 = gradient[index2] 

853 inside_sqrt = ( 

854 2 * gradient2 * (max_wb_int[index2] - self.falling_level_threshold) 

855 ) 

856 phase_cl[index2] = (intercept2 - np.sqrt(inside_sqrt)) / -gradient2 

857 # Update the phase change level. Clip to ignore extremely negative 

858 # phase change levels. 

859 phase_cl = np.clip(phase_cl, -2000, np.inf) 

860 phase_change_level_data[index] = phase_cl 

861 

862 @staticmethod 

863 def linear_wet_bulb_fit( 

864 wet_bulb_temperature: ndarray, 

865 heights: ndarray, 

866 sea_points: ndarray, 

867 start_point: int = 0, 

868 end_point: int = 5, 

869 ) -> Tuple[ndarray, ndarray]: 

870 """ 

871 Calculates a linear fit to the wet bulb temperature profile close 

872 to the surface to use when we extrapolate the wet bulb temperature 

873 below sea level for sea points. 

874 

875 We only use a set number of points close to the surface for this fit, 

876 specified by a start_point and end_point. 

877 

878 Args: 

879 wet_bulb_temperature: 

880 The wet bulb temperature profile at each grid point, with 

881 height as the leading dimension. 

882 heights: 

883 The vertical height levels above orography, matching the 

884 leading dimension of the wet_bulb_temperature. 

885 sea_points: 

886 A boolean array with True where the points are sea points. 

887 start_point: 

888 The index of the the starting height we want to use in our 

889 linear fit. 

890 end_point: 

891 The index of the the end height we want to use in our 

892 linear fit. 

893 

894 Returns: 

895 - An array, the same shape as a 

896 2D slice of the wet_bulb_temperature input, containing the 

897 gradients of the fitted straight line at each point where it 

898 could be found, filled with zeros elsewhere. 

899 - An array, the same shape as a 

900 2D slice of the wet_bulb_temperature input, containing the 

901 intercepts of the fitted straight line at each point where it 

902 could be found, filled with zeros elsewhere. 

903 """ 

904 # Set up empty arrays for gradient and intercept 

905 result_shape = wet_bulb_temperature.shape[1:] 

906 gradient = np.zeros(result_shape) 

907 intercept = np.zeros(result_shape) 

908 if np.any(sea_points): 

909 # Use only subset of heights. 

910 wbt = wet_bulb_temperature[start_point:end_point, sea_points] 

911 hgt = heights[start_point:end_point].reshape(-1, 1) 

912 gradient_values, intercept_values = fast_linear_fit(hgt, wbt, axis=0) 

913 gradient[sea_points] = gradient_values 

914 intercept[sea_points] = intercept_values 

915 return gradient, intercept

917 def fill_in_sea_points( 

918 self, 

919 phase_change_level_data: ndarray, 

920 land_sea_data: ndarray, 

921 max_wb_integral: ndarray, 

922 wet_bulb_temperature: ndarray, 

923 heights: ndarray, 

924 orography: ndarray, 

925 ) -> None: 

926 """ 

927 Fill in any sea points where we have not found a phase change level 

928 by the time we get to sea level, i.e. where the whole wet bulb 

929 temperature integral is below the threshold. 

930 

931 This function finds a linear fit to the wet bulb temperature close to 

932 sea level and uses this to find where an extrapolated wet bulb 

933 temperature integral would cross the threshold. This results in 

934 phase change levels below sea level for points where we have applied 

935 the extrapolation. 

936 

937 The linear fit assumes that all sea points have an orography altitude of zero, 

938 however this is not always the case. The orography altitude is added to phase 

939 change levels calculated from the linear fit to account for this. 

940 

941 Assumes that height is the first axis in the wet_bulb_integral array. 

942 

943 Args: 

944 phase_change_level_data: 

945 The phase change level array, filled with values for points 

946 whose wet bulb temperature integral crossed the theshold. 

947 land_sea_data: 

948 The binary land-sea mask 

949 max_wb_integral: 

950 The wet bulb temperature integral at the final height level 

951 used in the integration. This has the maximum values for the 

952 wet bulb temperature integral at any level. 

953 wet_bulb_temperature: 

954 The wet bulb temperature profile at each grid point, with 

955 height as the leading dimension. 

956 heights: 

957 The vertical height levels above orography, matching the 

958 leading dimension of the wet_bulb_temperature. 

959 orography: 

960 Orography heights 

961 """ 

962 sea_points = ( 

963 np.isnan(phase_change_level_data) 

964 & (land_sea_data < 1.0) 

965 & (max_wb_integral < self.falling_level_threshold) 

966 ) 

967 if np.all(sea_points is False): 

968 return 

969 

970 gradient, intercept = self.linear_wet_bulb_fit( 

971 wet_bulb_temperature, heights, sea_points 

972 ) 

973 

974 self.find_extrapolated_falling_level( 

975 max_wb_integral, gradient, intercept, phase_change_level_data, sea_points 

976 ) 

977 

978 phase_change_level_data[sea_points] += orography[sea_points]

982 Find the maximum value of the orography in the neighbourhood around 

983 each grid point. If self.grid_point_radius is zero, the orography is used 

984 without neighbourhooding. 

985 

986 Args: 

987 orography_cube: 

988 The cube containing a single 2 dimensional array of orography 

989 data 

990 

991 Returns: 

992 The cube containing the maximum in the grid_point_radius neighbourhood 

993 of the orography data or the orography data itself if the radius is zero 

994 """ 

995 if self.grid_point_radius >= 1: 

996 max_in_nbhood_orog = OccurrenceWithinVicinity( 

997 grid_point_radii=[self.grid_point_radius] 

998 )(orography_cube) 

999 return max_in_nbhood_orog 

1000 else: 

1001 return orography_cube.copy()

1003 def _calculate_phase_change_level( 

1004 self, 

1005 wet_bulb_temp: ndarray, 

1006 wb_integral: ndarray, 

1007 orography: ndarray, 

1008 max_nbhood_orog: ndarray, 

1009 land_sea_data: ndarray, 

1010 heights: ndarray, 

1011 height_points: ndarray, 

1012 highest_height: float, 

1013 ) -> ndarray: 

1014 """ 

1015 Calculate phase change level and fill in missing points 

1016 

1017 .. See the documentation for a more detailed discussion of the steps. 

1018 .. include:: extended_documentation/psychrometric_calculations/ 

1019 psychrometric_calculations/_calculate_phase_change_level.rst 

1020 

1021 Args: 

1022 wet_bulb_temp: 

1023 Wet bulb temperature data 

1024 wb_integral: 

1025 Wet bulb temperature integral 

1026 orography: 

1027 Orography heights 

1028 max_nbhood_orog: 

1029 Maximum orography height in neighbourhood (used to determine points that 

1030 can be used for interpolation) 

1031 land_sea_data: 

1032 Mask of binary land / sea data 

1033 heights: 

1034 All heights of wet bulb temperature input 

1035 height_points: 

1036 Heights on wet bulb temperature integral slice 

1037 highest_height: 

1038 Height of the highest level to which the wet bulb 

1039 temperature has been integrated 

1040 

1041 Returns: 

1042 Level at which phase changes 

1043 """ 

1044 phase_change_data = self.find_falling_level( 

1045 wb_integral, orography, height_points 

1046 ) 

1047 

1048 # Fill in missing data 

1049 self.fill_in_high_phase_change_falling_levels( 

1050 phase_change_data, orography, wb_integral.max(axis=0), highest_height 

1051 ) 

1052 self.fill_in_sea_points( 

1053 phase_change_data, 

1054 land_sea_data, 

1055 wb_integral.max(axis=0), 

1056 wet_bulb_temp, 

1057 heights, 

1058 orography, 

1059 ) 

1060 

1061 # Any unset points at this stage are set to np.nan; these will be 

1062 # lands points where the phase-change-level is below the orography. 

1063 # These can be filled by optional horizontal interpolation. 

1064 if self.horizontal_interpolation: 

1065 phase_change_data = self._horizontally_interpolate_phase( 

1066 phase_change_data, orography, max_nbhood_orog 

1067 ) 

1068 

1069 # Mask any points that are still set to np.nan; this should be no 

1070 # points if horizontal interpolation has been used. 

1071 phase_change_data = np.ma.masked_invalid(phase_change_data) 

1072 

1073 return phase_change_data

1075 def _horizontally_interpolate_phase( 

1076 self, phase_change_data: ndarray, orography: ndarray, max_nbhood_orog: ndarray 

1077 ) -> ndarray: 

1078 """ 

1079 Fill in missing points via horizontal interpolation. 

1080 

1081 Args: 

1082 phase_change_data: 

1083 Level (height) at which the phase changes. 

1084 orography: 

1085 Orography heights 

1086 max_nbhood_orog: 

1087 Maximum orography height in neighbourhood (used to determine points that 

1088 can be used for interpolation) 

1089 

1090 Returns: 

1091 Level at which phase changes, with missing data filled in 

1092 """ 

1093 

1094 with np.errstate(invalid="ignore"): 

1095 max_nbhood_mask = phase_change_data <= max_nbhood_orog 

1096 updated_phase_cl = interpolate_missing_data( 

1097 phase_change_data, limit=orography, valid_points=max_nbhood_mask 

1098 ) 

1099 

1100 with np.errstate(invalid="ignore"): 

1101 max_nbhood_mask = updated_phase_cl <= max_nbhood_orog 

1102 phase_change_data = interpolate_missing_data( 

1103 updated_phase_cl, 

1104 method="nearest", 

1105 limit=orography, 

1106 valid_points=max_nbhood_mask, 

1107 ) 

1108 

1109 if np.isnan(phase_change_data).any(): 

1110 # This should be rare. 

1111 phase_change_data = interpolate_missing_data( 

1112 phase_change_data, method="nearest", limit=orography 

1113 ) 

1114 return phase_change_data

1141 def process(self, *cubes: Union[CubeList, List[Cube]]) -> Cube: 

1142 """ 

1143 Use the wet bulb temperature integral to find the altitude at which a 

1144 phase change occurs (e.g. snow to sleet). This is achieved by finding 

1145 the height above sea level at which the integral matches an empirical 

1146 threshold that is expected to correspond with the phase change. This 

1147 empirical threshold is the falling_level_threshold. Fill in missing 

1148 data appropriately. 

1149 

1150 Args: 

1151 cubes containing: 

1152 wet_bulb_temperature: 

1153 Cube of wet bulb temperatures on height levels. 

1154 wet_bulb_integral: 

1155 Cube of wet bulb temperature integral (Kelvin-metres). 

1156 orog: 

1157 Cube of orography (m). 

1158 land_sea_mask: 

1159 Cube containing a binary land-sea mask, with land points 

1160 set to one and sea points set to zero. 

1161 

1162 Returns: 

1163 Cube of phase change level above sea level (asl). 

1164 

1165 Raises: 

1166 ValueError: Raise exception if the model_id_attr attribute does not 

1167 match on the input cubes. 

1168 """ 

1169 cubes = as_cubelist(*cubes) 

1170 names_to_extract = [ 

1171 "wet_bulb_temperature", 

1172 "wet_bulb_temperature_integral", 

1173 "surface_altitude", 

1174 "land_binary_mask", 

1175 ] 

1176 if len(cubes) != len(names_to_extract): 

1177 raise ValueError( 

1178 f"Expected {len(names_to_extract)} cubes, found {len(cubes)}" 

1179 ) 

1180 

1181 wet_bulb_temperature, wet_bulb_integral, orog, land_sea_mask = tuple( 

1182 CubeList(cubes).extract_cube(n) for n in names_to_extract 

1183 ) 

1184 

1185 wet_bulb_temperature.convert_units("celsius") 

1186 wet_bulb_integral.convert_units("K m") 

1187 

1188 if self.model_id_attr: 

1189 if ( 

1190 wet_bulb_temperature.attributes[self.model_id_attr] 

1191 != wet_bulb_integral.attributes[self.model_id_attr] 

1192 ): 

1193 raise ValueError( 

1194 f"Attribute {self.model_id_attr} does not match on input cubes. " 

1195 f"{wet_bulb_temperature.attributes[self.model_id_attr]} != " 

1196 f"{wet_bulb_integral.attributes[self.model_id_attr]}" 

1197 ) 

1198 

1199 # Ensure the wet bulb integral cube's height coordinate is in 

1200 # descending order 

1201 wet_bulb_integral = sort_coord_in_cube( 

1202 wet_bulb_integral, "height", descending=True 

1203 ) 

1204 

1205 # Find highest height from height bounds. 

1206 wbt_height_points = wet_bulb_temperature.coord("height").points 

1207 if wet_bulb_integral.coord("height").bounds is None: 

1208 highest_height = wbt_height_points[-1] 

1209 else: 

1210 highest_height = wet_bulb_integral.coord("height").bounds[0][-1] 

1211 

1212 # Firstly we need to slice over height, x and y 

1213 x_coord = wet_bulb_integral.coord(axis="x").name() 

1214 y_coord = wet_bulb_integral.coord(axis="y").name() 

1215 orography = next(orog.slices([y_coord, x_coord])) 

1216 land_sea_data = next(land_sea_mask.slices([y_coord, x_coord])).data 

1217 max_nbhood_orog = self.find_max_in_nbhood_orography(orography) 

1218 

1219 phase_change = None 

1220 slice_list = ["height", y_coord, x_coord] 

1221 for wb_integral, wet_bulb_temp in zip( 

1222 wet_bulb_integral.slices(slice_list), 

1223 wet_bulb_temperature.slices(slice_list), 

1224 ): 

1225 phase_change_data = self._calculate_phase_change_level( 

1226 wet_bulb_temp.data, 

1227 wb_integral.data, 

1228 orography.data, 

1229 max_nbhood_orog.data, 

1230 land_sea_data, 

1231 wbt_height_points, 

1232 wb_integral.coord("height").points, 

1233 highest_height, 

1234 ) 

1235 

1236 # preserve dimensionality of input cube (in case of scalar or 

1237 # length 1 dimensions) 

1238 if phase_change is None: 

1239 phase_change = phase_change_data 

1240 else: 

1241 if not isinstance(phase_change, list): 

1242 phase_change = [phase_change] 

1243 phase_change.append(phase_change_data) 

1244 

1245 phase_change_level = self.create_phase_change_level_cube( 

1246 wet_bulb_temperature, np.ma.masked_array(phase_change, dtype=np.float32) 

1247 )
