14@cli.with_output 

15def process( 

16 *cubes: cli.inputcube, 

17 distribution, 

18 truth_attribute, 

19 point_by_point=False, 

20 use_default_initial_guess=False, 

21 units=None, 

22 predictor="mean", 

23 tolerance: float = 0.02, 

24 max_iterations: int = 1000,

26 """Estimate coefficients for Ensemble Model Output Statistics. 

27 

28 Loads in arguments for estimating coefficients for Ensemble Model 

29 Output Statistics (EMOS), otherwise known as Non-homogeneous Gaussian 

30 Regression (NGR). Two sources of input data must be provided: historical 

31 forecasts and historical truth data (to use in calibration). 

32 The estimated coefficients are output as a cube. 

33 

34 Args: 

35 cubes (list of iris.cube.Cube): 

36 A list of cubes containing the historical forecasts and 

37 corresponding truth used for calibration. They must have the same 

38 cube name and will be separated based on the truth attribute. 

39 Optionally this may also contain a single land-sea mask cube on the 

40 same domain as the historic forecasts and truth (where land points 

41 are set to one and sea points are set to zero). 

42 distribution (str): 

43 The distribution that will be used for minimising the 

44 Continuous Ranked Probability Score when estimating the EMOS 

45 coefficients. This will be dependent upon the input phenomenon. 

46 truth_attribute (str): 

47 An attribute and its value in the format of "attribute=value", 

48 which must be present on historical truth cubes. 

49 point_by_point (bool): 

50 If True, coefficients are calculated independently for each point 

51 within the input cube by creating an initial guess and minimising 

52 each grid point independently. If False, a single set of 

53 coefficients is calculated using all points. 

54 Warning: This option is memory intensive and is unsuitable for 

55 gridded input. Using a default initial guess may reduce the memory 

56 overhead option. 

57 use_default_initial_guess (bool): 

58 If True, use the default initial guess. The default initial guess 

59 assumes no adjustments are required to the initial choice of 

60 predictor to generate the calibrated distribution. This means 

61 coefficients of 1 for the multiplicative coefficients and 0 for 

62 the additive coefficients. If False, the initial guess is computed. 

63 units (str): 

64 The units that calibration should be undertaken in. The historical 

65 forecast and truth will be converted as required. 

66 predictor (str): 

67 String to specify the form of the predictor used to calculate the 

68 location parameter when estimating the EMOS coefficients. 

69 Currently the ensemble mean ("mean") and the ensemble realizations 

70 ("realizations") are supported as options. 

71 tolerance (float): 

72 The tolerance for the Continuous Ranked Probability Score (CRPS) 

73 calculated by the minimisation. Once multiple iterations result in 

74 a CRPS equal to the same value within the specified tolerance, the 

75 minimisation will terminate. 

76 max_iterations (int): 

77 The maximum number of iterations allowed until the minimisation has 

78 converged to a stable solution. If the maximum number of iterations 

79 is reached but the minimisation has not yet converged to a stable 

80 solution, then the available solution is used anyway, and a warning 

81 is raised. If the predictor is "realizations", then the number of 

82 iterations may require increasing, as there will be more 

83 coefficients to solve. 

84 

85 Returns: 

86 iris.cube.CubeList: 

87 CubeList containing the coefficients estimated using EMOS. Each 

88 coefficient is stored in a separate cube. 

89 """ 

90 

91 from improver.calibration import split_forecasts_and_truth 

92 from improver.calibration.emos_calibration import ( 

93 EstimateCoefficientsForEnsembleCalibration, 

94 ) 

95 

96 forecast, truth, land_sea_mask = split_forecasts_and_truth(cubes, truth_attribute) 

97 

98 plugin = EstimateCoefficientsForEnsembleCalibration( 

99 distribution, 

100 point_by_point=point_by_point, 

101 use_default_initial_guess=use_default_initial_guess, 

102 desired_units=units, 

103 predictor=predictor, 

104 tolerance=tolerance, 

105 max_iterations=max_iterations, 

106 ) 

107 return plugin(forecast, truth, landsea_mask=land_sea_mask)
