75 def __new__( 

76 cls,

80 ): 

81 """Initialise class object based on package and model file availability. 

82 

83 Args: 

84 model_config_dict: 

85 Dictionary containing Rainforests model configuration variables. 

86 threads: 

87 Number of threads to use during prediction with tree-model objects.

89 or LightGBM, depending on which library is used.

92 if there are many data points which fall into the same bins for all threshold

95 

96 Dictionary is of format:: 

97 

98 {

101 "lightgbm_model": "<path_to_lightgbm_model_object>",

104 "0.000050": { 

105 "lightgbm_model": "<path_to_lightgbm_model_object>", 

106 "treelite_model": "<path_to_treelite_model_object>" 

107 }, 

108 "0.000100": { 

109 "lightgbm_model": "<path_to_lightgbm_model_object>", 

110 "treelite_model": "<path_to_treelite_model_object>" 

111 }, 

112 } 

113 

114 The keys specify the lead times and model threshold values, while the 

115 associated values are the path to the corresponding tree-model objects 

116 for that lead time and threshold. 

117 

118 Treelite predictors are used if treelite_runtime is an installed dependency 

119 and an associated path has been provided for all thresholds, otherwise LightGBM 

120 Boosters are used as the default tree model type. 

121 """

129 # Check that all required files have been specified. 

130 ApplyRainForestsCalibration.check_filenames( 

131 "treelite_model", model_config_dict 

132 ) 

133 return super(ApplyRainForestsCalibration, cls).__new__(cls) 

134 except ModelFileNotFoundError as e: 

135 # Treelite files not specified 

136 if not lightgbm_available: 

137 # Re-raise error if LightGBM unavailable 

138 raise (e) 

139 if lightgbm_available: 

140 if threads is not None: 

141 # Workaround to address segfault issue in LightGBM 

142 raise RuntimeError( 

143 "Manual thread specification is unsupported due to compatibility issues with LightGBM. Please remove the --threads argument, or install Treelite dependencies." 

144 ) 

145 cls = ApplyRainForestsCalibrationLightGBM 

146 # Ensure all required files have been specified. 

147 ApplyRainForestsCalibration.check_filenames( 

148 "lightgbm_model", model_config_dict 

149 ) 

150 return super(ApplyRainForestsCalibration, cls).__new__(cls)

176 def _get_feature_splits(self, model_config_dict) -> dict[np.float32, list[ndarray]]: 

177 """Get the combined feature splits (over all thresholds) for each lead time. 

178 

179 Args: 

180 model_config_dict: dictionary of the same format expected by __init__ 

181 

182 Returns: 

183 dict where keys are the lead times and the values are lists of lists. 

184 The outer list has length equal to the number of model features, and it contains 

185 the lists of feature splits for each feature. Each feature's list of splits is ordered. 

186 """ 

187 # These string patterns are defined by LightGBM and are used for finding the feature and 

188 # threshold information in the model .txt files. 

189 split_feature_string = "split_feature=" 

190 feature_threshold_string = "threshold=" 

191 combined_feature_splits = {} 

192 for lead_time in model_config_dict.keys(): 

193 all_splits = [set() for i in range(self._get_num_features())] 

194 for threshold_str in model_config_dict[lead_time].keys(): 

195 lgb_model_filename = Path( 

196 os.path.expandvars( 

197 model_config_dict[lead_time][threshold_str].get( 

198 "lightgbm_model" 

199 ) 

200 ) 

201 ).expanduser() 

202 with open(lgb_model_filename, "r") as f: 

203 for line in f: 

204 if line.startswith(split_feature_string): 

205 line = line[len(split_feature_string) : -1] 

206 if len(line) == 0: 

207 # This deals with the situation where the tree has no splits 

208 continue 

209 features = [int(x) for x in line.split(" ")] 

210 elif line.startswith(feature_threshold_string): 

211 line = line[len(feature_threshold_string) : -1] 

212 if len(line) == 0: 

213 continue 

214 splits = [float(x) for x in line.split(" ")] 

215 for feature_ind, threshold in zip(features, splits): 

216 all_splits[feature_ind].add(threshold) 

217 combined_feature_splits[np.float32(lead_time)] = [ 

218 np.sort(list(x)) for x in all_splits 

219 ] 

220 return combined_feature_splits

224 key_name: Model, model_config_dict: dict[str, dict[str, dict[str, str]]] 

225 ): 

226 """Check whether files specified by model_config_dict exist, 

227 and raise an error if any are missing. 

228 

229 Args: 

230 key_name: One of "treelite_model" or "lightgbm_model". 

231 model_config_dict: Dictionary containing Rainforests model configuration variables.

252 raise ModelFileNotFoundError(

262 def _parse_model_config( 

263 self, model_config_dict: dict[str, dict[str, dict[str, str]]] 

264 ) -> OrderedDict[np.float32, OrderedDict[np.float32, dict[str, str]]]: 

265 """Parse the model config dictionary, set self.lead_times and self.model_thresholds, 

266 and return a sorted version of the config dictionary. 

267 

268 Args: 

269 model_config_dict: Nested dictionary with string keys. Keys of outer level are 

270 lead times, and keys of inner level are thresholds. 

271 

272 Returns: 

273 Dictionary with the same nested structure as model_config_dict, but 

274 the lead time and threshold keys now have type np.float. 

275 """ 

276 

277 sorted_model_config_dict = OrderedDict() 

278 for key, lead_time_dict in model_config_dict.items(): 

279 sorted_model_config_dict[np.float32(key)] = OrderedDict( 

280 sorted({np.float32(k): v for k, v in lead_time_dict.items()}.items()) 

281 ) 

282 

283 self.lead_times = np.sort(np.array([*sorted_model_config_dict.keys()])) 

284 if len(self.lead_times) > 0: 

285 self.model_thresholds = np.sort( 

286 np.array([*sorted_model_config_dict[self.lead_times[0]].keys()]) 

287 ) 

288 else: 

289 warnings.warn( 

290 "Model config does not match the expected specification; calibration will not work" 

291 ) 

292 self.model_thresholds = np.array([]) 

293 return sorted_model_config_dict

393 def _align_feature_variables( 

394 self, feature_cubes: CubeList, forecast_cube: Cube 

395 ) -> tuple[CubeList, Cube]: 

396 """Ensure that feature cubes have consistent dimension coordinates. If realization 

397 dimension present in any cube, all cubes lacking this dimension will have realization 

398 dimension added and broadcast along this new dimension. 

399 

400 This situation occurs when derived fields (such as accumulated solar radiation) 

401 are used as predictors. As these fields do not contain a realization dimension, 

402 they must be broadcast to match the NWP fields that do contain realization, so that 

403 all features have consistent shape. 

404 

405 In the case of deterministic models (those without a realization dimension), a 

406 realization dimension is added to allow consistent behaviour between ensemble and 

407 deterministic models. 

408 

409 Args: 

410 feature_cubes: 

411 Cubelist containing feature variables to align. 

412 forecast_cube: 

413 Cube containing the forecast variable to align. 

414 

415 Returns: 

416 - feature_cubes with realization coordinate added to each cube if absent 

417 - forecast_cube with realization coordinate added if absent 

418 

419 Raises: 

420 ValueError: 

421 if feature/forecast variables have inconsistent dimension coordinates 

422 (excluding realization dimension), or if feature/forecast variables have 

423 different length realization coordinate over cubes containing a realization 

424 coordinate. 

425 """ 

426 combined_cubes = CubeList(list([*feature_cubes, forecast_cube])) 

427 

428 # Compare feature cube coordinates, raise error if dim-coords don't match 

429 compare_feature_coords = compare_coords( 

430 combined_cubes, ignored_coords=["realization"] 

431 ) 

432 for misaligned_coords in compare_feature_coords: 

433 for coord_info in misaligned_coords.values(): 

434 if coord_info["data_dims"] is not None: 

435 raise ValueError( 

436 "Mismatch between non-realization dimension coords." 

437 ) 

438 

439 # Compare realization coordinates across cubes where present; 

440 # raise error if realization coordinates don't match, otherwise set 

441 # common_realization_coord to broadcast over. 

442 realization_coords = { 

443 variable.name(): variable.coord("realization") 

444 for variable in combined_cubes 

445 if variable.coords("realization") 

446 } 

447 if not realization_coords: 

448 # Case I: realization_coords is empty. Add single realization dim to all cubes. 

449 common_realization_coord = DimCoord( 

450 [0], standard_name="realization", units=1 

451 ) 

452 else: 

453 # Case II: realization_coords is not empty. 

454 # Note: In future, another option here could be to filter to common realization 

455 # values using filter_realizations() in utilities.cube_manipulation. 

456 variables_with_realization = list(realization_coords.keys()) 

457 sample_realization = realization_coords[variables_with_realization[0]] 

458 for feature in variables_with_realization[1:]: 

459 if realization_coords[feature] != sample_realization: 

460 raise ValueError("Mismatch between realization dimension coords.") 

461 common_realization_coord = sample_realization 

462 

463 # Add realization coord to cubes where absent by broadcasting along this dimension 

464 aligned_cubes = CubeList() 

465 for cube in combined_cubes: 

466 if not cube.coords("realization"): 

467 expanded_cube = add_coordinate_to_cube( 

468 cube, new_coord=common_realization_coord 

469 ) 

470 aligned_cubes.append(expanded_cube) 

471 else: 

472 aligned_cubes.append(cube) 

473 

474 # Make data contiguous (required for numba interpolation) 

475 for cube in aligned_cubes: 

476 if not cube.data.flags["C_CONTIGUOUS"]: 

477 cube.data = np.ascontiguousarray(cube.data, dtype=cube.data.dtype) 

478 

479 return aligned_cubes[:-1], aligned_cubes[-1]

481 def _prepare_threshold_probability_cube(self, forecast_cube, thresholds): 

482 """Initialise a cube with the same dimensions as the input forecast_cube, 

483 with an additional threshold dimension added as the leading dimension. 

484 

485 Args:

487 Cube containing the forecast to be calibrated. 

488 thresholds: 

489 Points of the the threshold dimension. 

490 

491 Returns: 

492 An empty probability cube. 

493 """

495 forecast_variable = forecast_cube.name()

498 name=f"probability_of_{forecast_variable}_above_threshold",

517 def _prepare_features_array(self, feature_cubes: CubeList) -> ndarray: 

518 """Convert gridded feature cubes into a numpy array, with feature variables 

519 sorted alphabetically. 

520 

521 Note: It is expected that feature_cubes has been aligned using 

522 _align_feature_variables prior to calling this function. 

523 

524 Args: 

525 feature_cubes: 

526 Cubelist containing the independent feature variables for prediction. 

527 

528 Returns: 

529 Array containing flattened feature variables, 

530 

531 Raises: 

532 ValueError: 

533 If flattened cubes have differing length. 

534 """ 

535 # Get the names of features and sort alphabetically 

536 feature_variables = [cube.name() for cube in feature_cubes] 

537 feature_variables.sort() 

538 

539 # Unpack the cube-data into an array to feed into the tree-models. 

540 features_list = [] 

541 for feature in feature_variables: 

542 cube = feature_cubes.extract_cube(feature) 

543 features_list.append(cube.data.ravel()[:, np.newaxis]) 

544 features_arr = np.concatenate(features_list, axis=1) 

545 

546 return features_arr

548 def _make_decreasing(self, probability_data: ndarray) -> ndarray: 

549 """Enforce monotonicity on the CDF data, where threshold dimension 

550 is assumed to be the leading dimension. 

551 

552 This is achieved by identifying the minimum value progressively along 

553 the leading dimension by comparing to all preceding probability values along 

554 this dimension. The same is done for maximum values, comparing to all 

555 succeeding values along the leading dimension. Averaging these resulting 

556 arrays results in an array decreasing monotonically in the threshold dimension. 

557 

558 Args: 

559 probability_data: 

560 The probability data as exceedence probabilities. 

561 

562 Returns: 

563 The probability data, enforced to be monotonically decreasing along 

564 the leading dimension. 

565 """ 

566 lower = np.minimum.accumulate(probability_data, axis=0) 

567 upper = np.flip( 

568 np.maximum.accumulate(np.flip(probability_data, axis=0), axis=0), axis=0 

569 ) 

570 return 0.5 * (upper + lower)

572 def _evaluate_probabilities( 

573 self, input_data: ndarray, lead_time_hours: int, output_data: ndarray 

574 ) -> None: 

575 """Evaluate probability that forecast exceeds thresholds. 

576 

577 Args: 

578 input_data: 

579 2-d array of data for the feature variables of the model 

580 lead_time_hours: 

581 lead time in hours 

582 output_data: 

583 array to populate with output; will be modified in place 

584 """ 

585 

586 if np.float32(lead_time_hours) in self.lead_times: 

587 model_lead_time = np.float32(lead_time_hours) 

588 else: 

589 # find closest model lead time 

590 best_ind = np.argmin(np.abs(self.lead_times - lead_time_hours)) 

591 model_lead_time = self.lead_times[best_ind] 

592 

593 if self.bin_data: 

594 # bin by feature splits 

595 feature_splits = self.combined_feature_splits[model_lead_time] 

596 binned_data = np.empty(input_data.shape, dtype=np.int32) 

597 n_features = len(feature_splits) 

598 for i in range(n_features): 

599 binned_data[:, i] = np.digitize( 

600 input_data[:, i], bins=feature_splits[i] 

601 ) 

602 # sort so rows in the same bins are grouped 

603 sort_ind = np.lexsort(tuple([binned_data[:, i] for i in range(n_features)])) 

604 sorted_data = binned_data[sort_ind] 

605 reverse_sort_ind = np.argsort(sort_ind) 

606 # we only need to predict for rows which are different from the previous row 

607 diff = np.any(np.diff(sorted_data, axis=0) != 0, axis=1) 

608 predict_rows = np.concatenate([[0], np.nonzero(diff)[0] + 1]) 

609 data_for_prediction = input_data[sort_ind][predict_rows] 

610 full_prediction = np.empty((input_data.shape[0],)) 

611 # forward fill code inspired by this: https://stackoverflow.com/a/41191127 

612 fill_inds = np.zeros(len(full_prediction), dtype=np.int32) 

613 fill_inds[predict_rows] = predict_rows 

614 fill_inds = np.maximum.accumulate(fill_inds) 

615 dataset_for_prediction = self.model_input_converter(data_for_prediction) 

616 

617 for threshold_index, threshold in enumerate(self.model_thresholds): 

618 model = self.tree_models[model_lead_time, threshold] 

619 prediction = model.predict(dataset_for_prediction) 

620 prediction = np.clip(prediction, 0, 1) 

621 if type(self) is ApplyRainForestsCalibrationTreelite: 

622 # treelite 4.x.x changed output dimensions, so must flatten here 

623 # See https://treelite.readthedocs.io/en/latest/treelite-gtil-api.html#treelite.gtil.predict 

624 prediction = prediction.flatten() 

625 full_prediction[predict_rows] = prediction 

626 full_prediction = full_prediction[fill_inds] 

627 # restore original order 

628 full_prediction = full_prediction[reverse_sort_ind] 

629 output_data[threshold_index, :] = np.reshape( 

630 full_prediction, output_data.shape[1:] 

631 ) 

632 else: 

633 dataset_for_prediction = self.model_input_converter(input_data) 

634 for threshold_index, threshold in enumerate(self.model_thresholds): 

635 model = self.tree_models[model_lead_time, threshold] 

636 prediction = model.predict(dataset_for_prediction) 

637 prediction = np.clip(prediction, 0, 1) 

638 if type(self) is ApplyRainForestsCalibrationTreelite: 

639 # treelite 4.x.x changed output dimensions, so must flatten here 

640 # See https://treelite.readthedocs.io/en/latest/treelite-gtil-api.html#treelite.gtil.predict 

641 prediction = prediction.flatten() 

642 output_data[threshold_index, :] = np.reshape( 

643 prediction, output_data.shape[1:] 

644 )

646 def _calculate_threshold_probabilities( 

647 self, forecast_cube: Cube, feature_cubes: CubeList 

648 ) -> Cube: 

649 """Evaluate the threshold exceedence probabilities for each ensemble member in 

650 forecast_cube using the tree_models, with the associated feature_cubes taken as 

651 inputs to the tree_model predictors. 

652 

653 Args: 

654 forecast_cube: 

655 Cube containing the variable to be calibrated. 

656 feature_cubes: 

657 Cubelist containing the independent feature variables for prediction. 

658 

659 Returns: 

660 A cube containing threshold exceedence probabilities. 

661 

662 Raises: 

663 ValueError: 

664 If an unsupported model object is passed. Expects LightGBM Booster, or 

665 tl2cgen Predictor (if the latter's dependencies are available). 

666 """ 

667 

668 threshold_probability_cube = self._prepare_threshold_probability_cube( 

669 forecast_cube, self.model_thresholds 

670 ) 

671 

672 input_dataset = self._prepare_features_array(feature_cubes) 

673 

674 lead_time_hours = forecast_cube.coord("forecast_period").points[0] / ( 

675 SECONDS_IN_MINUTE * MINUTES_IN_HOUR 

676 ) 

677 

678 self._evaluate_probabilities( 

679 input_dataset, lead_time_hours, threshold_probability_cube.data 

680 ) 

681 

682 # Enforcing monotonicity 

683 threshold_probability_cube.data = self._make_decreasing( 

684 threshold_probability_cube.data 

685 ) 

686 

687 return threshold_probability_cube

689 def _get_ensemble_distributions( 

690 self, per_realization_CDF: Cube, forecast: Cube, output_thresholds: ndarray 

691 ) -> Cube: 

692 """ 

693 Interpolate probabilities calculated at model thresholds to extract probabilities 

694 at output thresholds for all realizations. 

695 

696 Args: 

697 per_realization_CDF: 

698 Cube containing the CDF probabilities for each ensemble member at model 

699 thresholds, with threshold as the first dimension. 

700 forecast: 

701 Cube containing NWP ensemble forecast. 

702 output_thresholds: 

703 Sorted array of thresholds at which to calculate the output probabilities. 

704 

705 Returns: 

706 Cube containing probabilities at output thresholds for all realizations. Dimensions 

707 are same as forecast cube with additional threshold dimension first. 

708 """ 

709 

710 input_probabilities = per_realization_CDF.data 

711 output_thresholds = np.array(output_thresholds, dtype=np.float32) 

712 bounds_data = get_bounds_of_distribution(forecast.name(), forecast.units) 

713 lower_bound = bounds_data[0].astype(np.float32) 

714 if (len(self.model_thresholds) == len(output_thresholds)) and np.allclose( 

715 self.model_thresholds, output_thresholds 

716 ): 

717 output_probabilities = np.copy(input_probabilities.data) 

718 else: 

719 # add lower bound with probability 1 

720 input_probabilities = np.concatenate( 

721 [ 

722 np.ones((1,) + input_probabilities.shape[1:], dtype=np.float32), 

723 input_probabilities, 

724 ], 

725 axis=0, 

726 ) 

727 input_thresholds = np.concatenate([[lower_bound], self.model_thresholds]) 

728 # reshape to 2 dimensions 

729 input_probabilities_2d = np.reshape( 

730 input_probabilities, (input_probabilities.shape[0], -1) 

731 ) 

732 output_probabilities_2d = interpolate_multiple_rows_same_x( 

733 output_thresholds, input_thresholds, input_probabilities_2d.transpose() 

734 ) 

735 output_probabilities = np.reshape( 

736 output_probabilities_2d.transpose(), 

737 (len(output_thresholds),) + input_probabilities.shape[1:], 

738 ) 

739 # force interpolated probabilities to be monotone (sometimes they 

740 # are not due to small floating-point errors) 

741 output_probabilities = self._make_decreasing(output_probabilities) 

742 

743 # set probability for lower bound to 1 

744 if np.isclose(output_thresholds[0], lower_bound): 

745 output_probabilities[0, :] = 1 

746 

747 # Make output cube 

748 probability_cube = self._prepare_threshold_probability_cube( 

749 forecast, output_thresholds 

750 ) 

751 probability_cube.data = output_probabilities.astype(np.float32) 

752 return probability_cube

754 def process( 

755 self, 

756 forecast_cube: Cube, 

757 feature_cubes: CubeList, 

758 output_thresholds: list, 

759 threshold_units: str | None = None, 

760 ) -> Cube: 

761 """Apply rainforests calibration to forecast cube. 

762 

763 Ensemble forecasts must be in realization representation. Deterministic forecasts 

764 can be processed to produce a pseudo-ensemble; a realization dimension will be added 

765 to deterministic forecast cubes if one is not already present. 

766 

767 The calibration is done in a situation dependent fashion using a series of 

768 decision-tree models to construct representative probability distributions for 

769 each input ensemble member which are then blended to give the calibrated 

770 distribution for the full ensemble. 

771 

772 These distributions are formed in a two-step process: 

773 

774 1. Evaluate CDF defined over the specified model thresholds for each ensemble member. 

775 Each threshold exceedance probability is evaluated using the corresponding 

776 decision-tree model. 

777 

778 2. Interpolate each ensemble member distribution to the output thresholds. 

779 

780 Args: 

781 forecast_cube: 

782 Cube containing the forecast to be calibrated; must be as realizations. 

783 feature_cubes: 

784 Cubelist containing the feature variables (physical parameters) used as inputs 

785 to the tree-models for the generation of the associated probability distributions. 

786 Feature cubes are expected to have the same dimensions as forecast_cube, with 

787 the exception of the realization dimension. Where the feature_cube contains a 

788 realization dimension this is expected to be consistent, otherwise the cube will 

789 be broadcast along the realization dimension. 

790 output_thresholds: 

791 Set of output thresholds. 

792 threshold_units: 

793 Units in which output_thresholds are specified. If None, assumed to be the same as 

794 forecast_cube. 

795 

796 Returns: 

797 The calibrated forecast cube. 

798 

799 Raises: 

800 RuntimeError: 

801 If the number of tree models is inconsistent with the number of model 

802 thresholds. 

803 """ 

804 # Check that the correct number of feature variables has been supplied. 

805 self._check_num_features(feature_cubes) 

806 

807 # Align forecast and feature datasets 

808 aligned_features, aligned_forecast = self._align_feature_variables( 

809 feature_cubes, forecast_cube 

810 ) 

811 

812 # Evaluate the CDF using tree models. 

813 per_realization_CDF = self._calculate_threshold_probabilities( 

814 aligned_forecast, aligned_features 

815 ) 

816 

817 # convert units of output thresholds 

818 if threshold_units: 

819 original_threshold_unit = Unit(threshold_units) 

820 forecast_unit = forecast_cube.units 

821 output_thresholds_in_forecast_units = np.array( 

822 [ 

823 original_threshold_unit.convert(x, forecast_unit) 

824 for x in output_thresholds 

825 ] 

826 ) 

827 else: 

828 output_thresholds_in_forecast_units = np.array(output_thresholds) 

829 

830 # Calculate probabilities at output thresholds 

831 interpolated_per_realization_CDF = self._get_ensemble_distributions( 

832 per_realization_CDF, aligned_forecast, output_thresholds_in_forecast_units 

833 ) 

834 

835 # Average over realizations 

836 calibrated_probability_cube = interpolated_per_realization_CDF.collapsed( 

837 "realization", MEAN 

838 ) 

839 calibrated_probability_cube.remove_coord("realization") 

840 

841 return calibrated_probability_cube
