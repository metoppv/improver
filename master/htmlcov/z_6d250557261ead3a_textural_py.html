27class FieldTexture(BasePlugin): 

28 """Plugin to calculate whether or not the input field texture exceeds a 

29 given threshold. 

30 

31 1) Takes a binary field that has been thresholded and looks for the transitions/edges 

32 in the field that mark out a transition. 

33 2) The transition calculation is then fed into the neighbourhooding code 

34 (_calculate_ratio) to calculate a ratio for each cell. This is the texture 

35 of the input field. 

36 3) The new cube of ratios is then thresholded and the realization coordinates 

37 are collapsed to generate a mean of the thresholded ratios. This gives a binary 

38 indication of whether a field is rough (texture values close to 1) or smooth 

39 (texture values close to zero). 

40 """ 

41 

42 def __init__( 

43 self, 

44 nbhood_radius: float, 

45 textural_threshold: float, 

46 diagnostic_threshold: float, 

47 model_id_attr: Optional[str] = None, 

48 ) -> None: 

49 """ 

50 Args: 

51 nbhood_radius: 

52 The neighbourhood radius in metres within which the number of potential 

53 transitions should be calculated. This forms the denominator in the 

54 calculation of the ratio of actual to potential transitions that indicates a 

55 field's texture. A larger radius should be used for diagnosing larger-scale 

56 textural features. 

57 

58 textural_threshold: 

59 A unit-less threshold value that defines the ratio value above which 

60 the field is considered rough and below which the field is considered 

61 smoother. 

62 

63 diagnostic_threshold: 

64 A user defined threshold value related either to cloud or precipitation, 

65 used to extract the corresponding dimensional cube with assumed units of 1. 

66 

67 model_id_attr: 

68 Name of the attribute used to identify the source model for 

69 blending. 

70 """ 

71 self.nbhood_radius = nbhood_radius 

72 self.textural_threshold = textural_threshold 

73 self.diagnostic_threshold = diagnostic_threshold 

74 self.model_id_attr = model_id_attr 

75 

76 def _calculate_ratio(self, cube: Cube, cube_name: str, radius: float) -> Cube: 

77 """ 

78 Calculates the ratio of actual to potential value transitions in a 

79 neighbourhood about each cell. 

80 

81 The process is as follows: 

82 

83 1. For each grid cell find the number of cells of value 1 in a surrounding 

84 neighbourhood of a size defined by the arg radius. The potential 

85 transitions within that neighbourhood are defined as the number of 

86 orthogonal neighbours (up, down, left, right) about cells of value 1. 

87 This is 4 times the number of cells of value 1. 

88 2. Calculate the number of actual transitions within the neighbourhood, 

89 that is the number of cells of value 0 that orthogonally abut cells 

90 of value 1. 

91 3. Calculate the ratio of actual to potential transitions. 

92 

93 Ratios approaching 1 indicate that there are many transitions, so the field 

94 is highly textured (rough). Ratios close to 0 indicate a smoother field. 

95 

96 A neighbourhood full of cells of value 1 will return ratios of 0; the 

97 diagnostic that has been thresholded to produce the binary field is found 

98 everywhere within that neighbourhood, giving a smooth field. At the other 

99 extreme, in neighbourhoods in which there are no cells of value 1 the ratio 

100 is set to 1. 

101 

102 Args: 

103 cube: 

104 Input data in cube format containing a two-dimensional field 

105 of binary data. 

106 cube_name: 

107 Name of input data cube, used for determining output texture cube name. 

108 radius: 

109 Radius for neighbourhood in metres. 

110 

111 Returns: 

112 A ratio between 0 and 1 of actual transitions over potential transitions. 

113 """ 

114 # Calculate the potential transitions within neighbourhoods. 

115 potential_transitions = NeighbourhoodProcessing( 

116 "square", radius, sum_only=True 

117 ).process(cube) 

118 potential_transitions.data = 4 * potential_transitions.data 

119 

120 # Calculate the actual transitions for each grid cell of value 1 and 

121 # store them in a cube. 

122 actual_transitions = potential_transitions.copy( 

123 data=self._calculate_transitions(cube.data) 

124 ) 

125 

126 # Sum the number of actual transitions within the neighbourhood. 

127 actual_transitions = NeighbourhoodProcessing( 

128 "square", radius, sum_only=True 

129 ).process(actual_transitions) 

130 

131 # Calculate the ratio of actual to potential transitions in areas where the 

132 # original diagnostic value was greater than zero. Where the original value 

133 # was zero, set ratio value to one. 

134 

135 ratio = np.ones_like(actual_transitions.data) 

136 ratio[cube.data > 0] = ( 

137 actual_transitions.data[cube.data > 0] 

138 / potential_transitions.data[cube.data > 0] 

139 ) 

140 

141 # Create a new cube to contain the resulting ratio data. 

142 ratio = create_new_diagnostic_cube( 

143 "texture_of_{}".format(cube_name), 

144 "1", 

145 cube, 

146 mandatory_attributes=generate_mandatory_attributes( 

147 [cube], model_id_attr=self.model_id_attr 

148 ), 

149 data=ratio, 

150 ) 

151 return ratio 

152 

153 @staticmethod 

154 def _calculate_transitions(data: ndarray) -> ndarray: 

155 """ 

156 Identifies actual transitions present in a binary field. These transitions 

157 are defined as the number of cells of value zero that directly neighbour 

158 cells of value 1. The number of transitions is calculated for all cells 

159 of value 1 whilst avoiding double-counting transitions. The number of 

160 transitions for cells of value 0 is set to 0. 

161 

162 Args: 

163 data: 

164 A NumPy array of the input cube for data manipulation. 

165 

166 Returns: 

167 A NumPy array containing the transitions for ratio calculation. 

168 """ 

169 padded_data = np.pad(data, 1, mode="edge") 

170 diff_x = np.abs(np.diff(padded_data, axis=1)) 

171 diff_y = np.abs(np.diff(padded_data, axis=0)) 

172 cell_sum_x = diff_x[:, 0:-1] + diff_x[:, 1:] 

173 cell_sum_y = diff_y[0:-1, :] + diff_y[1:, :] 

174 cell_sum = cell_sum_x[1:-1, :] + cell_sum_y[:, 1:-1] 

175 cell_sum = np.where(data > 0, cell_sum, 0) 

176 return cell_sum

178 def process(self, input_cube: Cube) -> Cube: 

179 """ 

180 Calculates a field of texture to use in differentiating solid and 

181 more scattered features. 

182 

183 Args: 

184 input_cube: 

185 Input data in cube format containing the field for which the 

186 texture is to be assessed. 

187 

188 Returns: 

189 A cube containing either the mean across realization of the 

190 thresholded ratios to give the field texture, if a realization 

191 coordinate is present, or the thresholded ratios directly, if 

192 no realization coordinate is present. 

193 """ 

194 

195 values = np.unique(input_cube.data) 

196 non_binary = np.where((values != 0) & (values != 1), True, False) 

197 if non_binary.any(): 

198 raise ValueError("Incorrect input. Cube should hold binary data only") 

199 

200 # Create new cube name for _calculate_ratio method. 

201 cube_name = find_threshold_coordinate(input_cube).name() 

202 # Extract threshold from input data to work with, taking into account floating 

203 # point comparisons. 

204 cube = input_cube.extract( 

205 iris.Constraint( 

206 coord_values={ 

207 cube_name: lambda cell: np.isclose( 

208 cell.point, self.diagnostic_threshold 

209 ) 

210 } 

211 ) 

212 ) 

213 try: 

214 cube.remove_coord(cube_name) 

215 except AttributeError: 

216 msg = "Threshold {} is not present on coordinate with values {} {}" 

217 raise ValueError( 

218 msg.format( 

219 self.diagnostic_threshold, 

220 input_cube.coord(cube_name).points, 

221 input_cube.coord(cube_name).units, 

222 ) 

223 ) 

224 ratios = iris.cube.CubeList() 

225 

226 try: 

227 cslices = cube.slices_over("realization") 

228 except CoordinateNotFoundError: 

229 cslices = [cube] 

230 

231 for cslice in cslices: 

232 ratios.append(self._calculate_ratio(cslice, cube_name, self.nbhood_radius)) 

233 

234 ratios = ratios.merge_cube() 

235 thresholded = Threshold(threshold_values=self.textural_threshold).process( 

236 ratios 

237 ) 

238 

239 # Squeeze scalar threshold coordinate. 

240 try: 

241 field_texture = iris.util.squeeze(collapse_realizations(thresholded)) 

242 except CoordinateNotFoundError: 

243 field_texture = iris.util.squeeze(thresholded) 

244 

245 return field_texture
