20class DistanceToFeature(BasePlugin): 

21 """ 

22 Plugin to calculate the distance from sites to the nearest feature in a geometry. 

23 This is a generic plugin that can be used for any type of feature (e.g., coastlines, 

24 rivers, lakes, etc.). Distances are calculated to the nearest metre. 

25 

26 Given a cube containing site locations and a GeoDataFrame, the distance to each site 

27 from the nearest point of the feature geometry is calculated. This is done by 

28 converting the feature geometry and sites to a common target projection that must be 

29 specified using a European Petroleum Survey Group (EPSG) code that identifies the 

30 projection. For the UK, EPSG code 3035 may be used to provide a Lambert Azimuthal 

31 Equal Area projection that is suitable for the region. The chosen projection should 

32 match the projection on which the ancillary will be used. The distance method from 

33 Shapely is used to find the distance from each site to every point in the feature 

34 geometry. The minimum of these distances is returned as the distance to the nearest 

35 feature in the feature geometry and this is rounded to the nearest metre. 

36 

37 If requested, the provided geometry will be clipped to the bounds of the site 

38 locations with a buffer to improve performance by reducing computation. This is 

39 useful when the geometry is large and it would be expensive to calculate the 

40 distance to all features in the geometry but information may be lost at the edges of 

41 the domain. 

42 

43 Optionally, an exclusion geometry can be provided to the process method. For sites 

44 outside the exclusion geometry, the distance to the feature will be set to 0. This 

45 is useful for cases where the target region cannot be represented as a GeoDataFrame 

46 directly (e.g. for distance to ocean, you would provide the coastline as the feature 

47 geometry and the land geometry as the exclusion geometry, so that sites outside the 

48 land geometry — i.e. in the ocean — have a distance of 0). 

49 """

51 def __init__( 

52 self, 

53 epsg_projection: int, 

54 new_name: str, 

55 buffer: float = 30000, 

56 clip_geometry: bool = False, 

57 parallel: bool = False, 

58 n_parallel_jobs: Optional[int] = len(os.sched_getaffinity(0)), 

59 ) -> None: 

60 """ 

61 Initialise the DistanceToFeature plugin. 

62 

63 Args: 

64 epsg_projection: 

65 The EPSG code of the coordinate reference system on to which latitudes 

66 and longitudes will be projected to calculate distances. This is 

67 a projected coordinate system in which distances are measured in metres, 

68 for example, EPSG code 3035, which defines a Lambert Azimuthal Equal 

69 Areas projection suitable for the UK. 

70 new_name: 

71 The name of the output cube. E.g. 'distance_to_coast'. 

72 buffer: 

73 A buffer distance in m. If the geometry is clipped, this distance will 

74 be added onto the outermost site locations to define the domain to clip 

75 the geometry to. 

76 clip_geometry: 

77 A flag to indicate whether the geometry should be clipped to the bounds 

78 of the site locations with a buffer distance added to the bounds. If set 

79 to False, the full geometry will be used to calculate the distance to 

80 the nearest feature. 

81 parallel: 

82 A flag to indicate whether to use parallel processing when calculating 

83 distances. 

84 n_parallel_jobs: 

85 The number of parallel jobs to use when calculating distances. 

86 By default, os.sched_getaffinity(0) is used to give the number of cores 

87 that the process is eligible to use. 

88 """ 

89 self.epsg_projection = epsg_projection 

90 self.new_name = new_name 

91 self.buffer = buffer 

92 self._do_geometry_clipping = clip_geometry 

93 self.parallel = parallel 

94 self.n_parallel_jobs = n_parallel_jobs

96 @staticmethod 

97 def get_clip_values(points: List[float], buffer: float) -> List[float]: 

98 """Get the coordinates to use when clipping the geometry. This is determined by 

99 finding the maximum and minimum coordinate points from a list. A buffer distance 

100 may then be added/subtracted to the max/min. 

101 

102 Args: 

103 points: 

104 A list of points to find the min and max from. 

105 buffer: 

106 The buffer distance to add/subtract to the max/min points. 

107 Returns: 

108 A list containing the maximum and minimum points with the buffer added 

109 or subtracted respectively.""" 

110 

111 ordered_points = sorted(set(points)) 

112 min_point = ordered_points[0] - buffer 

113 max_point = ordered_points[-1] + buffer 

114 

115 return [min_point, max_point]

117 def clip_geometry( 

118 self, geometry: GeoDataFrame, bounds_x: List[float], bounds_y: List[float] 

119 ) -> GeoDataFrame: 

120 """Clip the geometry to the provided bounds. 

121 

122 Args: 

123 geometry: 

124 The geometry to clip. 

125 bounds_x: 

126 A list containing the minimum and maximum x coordinates. 

127 bounds_y: 

128 A list containing the minimum and maximum y coordinates. 

129 Returns: 

130 The clipped geometry. 

131 

132 Raises: 

133 ValueError: If the clipped geometry is empty after clipping with the 

134 provided bounds.""" 

135 

136 clipped_geometry = clip( 

137 geometry, mask=[bounds_x[0], bounds_y[0], bounds_x[1], bounds_y[1]] 

138 ) 

139 if clipped_geometry.empty: 

140 raise ValueError( 

141 "Clipping the geometry with a buffer size of " 

142 f"{self.buffer}m has produced an empty geometry. Either " 

143 "increase the buffer size or set clip_geometry to " 

144 "False to use the full geometry." 

145 ) 

146 

147 return clipped_geometry

149 def check_target_crs(self, site_points: GeoSeries): 

150 """Check that the provided target projection is suitable for the sites 

151 being used. 

152 

153 Args: 

154 site_points: 

155 A GeoSeries containing the site points. 

156 

157 Raises: 

158 ValueError: If the provided target coordinate reference system is not 

159 suitable for the site points. 

160 """ 

161 x_bounds = self.get_clip_values(site_points.x, 0) 

162 y_bounds = self.get_clip_values(site_points.y, 0) 

163 

164 target_crs = pyproj.CRS.from_epsg(self.epsg_projection) 

165 x_min, y_min, x_max, y_max = target_crs.area_of_use.bounds 

166 

167 valid_target = ( 

168 x_bounds[0] > x_min 

169 and x_bounds[1] < x_max 

170 and y_bounds[0] > y_min 

171 and y_bounds[1] < y_max 

172 ) 

173 if not valid_target: 

174 raise ValueError( 

175 "The provided projection defined by EPSG code " 

176 f"{self.epsg_projection} is not suitable for the site " 

177 "locations provided. Limits of this domain are: " 

178 f"x: {x_min} to {x_max}, y: {y_min} to {y_max}, whilst " 

179 f"the site locations are bounded by x: {x_bounds[0]} to {x_bounds[1]}, " 

180 f"y: {y_bounds[0]} to {y_bounds[1]}." 

181 )

183 def project_geometry( 

184 self, geometry: GeoDataFrame, site_cube: Cube 

185 ) -> Tuple[GeoSeries, GeoDataFrame]: 

186 """Project the geometry and site cube to the target projection. 

187 

188 Args: 

189 geometry: 

190 The geometry to reproject. 

191 site_cube: 

192 The cube containing the site locations. It is assumed that the site 

193 coordinates are defined as latitude and longitude on a WGS84 

194 coordinate system (EPSG:4326). 

195 Returns: 

196 A tuple containing the projected site locations and geometry.""" 

197 

198 x_points = site_cube.coord(axis="x").points 

199 y_points = site_cube.coord(axis="y").points 

200 

201 site_points_list = [Point(x, y) for x, y in zip(x_points, y_points)] 

202 # Assumes site coordinates are defined as latitude and longitude 

203 # defaulting to an EPSG:4326 coordinate system, which is WGS84. 

204 site_points = GeoSeries(site_points_list, crs=4326) 

205 

206 # Check that the provided target projection is suitable for the site points 

207 self.check_target_crs(site_points) 

208 

209 geometry_reprojection = geometry.to_crs(self.epsg_projection) 

210 site_points = site_points.to_crs(self.epsg_projection) 

211 return site_points, geometry_reprojection

213 def _apply_exclusion( 

214 self, 

215 site_coords: GeoSeries, 

216 distance_results: List[int], 

217 exclude_outside_of: GeoDataFrame, 

218 exclusion_buffer: float, 

219 ) -> List[int]: 

220 """Apply exclusion geometry logic: set distances to 0 for sites outside of the 

221 provided geometry. 

222 

223 Args: 

224 site_coords: 

225 A GeoSeries containing the projected site coordinates. 

226 distance_results: 

227 The calculated distances to the feature. 

228 exclude_outside_of: 

229 A GeoDataFrame containing the geometry defining the valid region. Sites 

230 outside of this geometry will have their distance set to 0. 

231 exclusion_buffer: 

232 A buffer distance in m used when clipping the exclusion geometry. 

233 

234 Returns: 

235 The distance results with distances set to 0 for sites outside the exclusion 

236 geometry.""" 

237 

238 # Project the provided geometry to the target projection 

239 exclusion_projection = exclude_outside_of.to_crs(self.epsg_projection) 

240 

241 # Clip the provided geometry for performance 

242 x_bounds = self.get_clip_values(site_coords.x, exclusion_buffer) 

243 y_bounds = self.get_clip_values(site_coords.y, exclusion_buffer) 

244 

245 try: 

246 clipped_exclusion = self.clip_geometry( 

247 exclusion_projection, x_bounds, y_bounds 

248 ) 

249 except ValueError: 

250 # If clipping produces empty geometry, no sites are in exclusion area 

251 return distance_results 

252 

253 # Calculate distances to exclusion geometry to identify sites within it 

254 distance_to_exclusion = self.distance_to(site_coords, clipped_exclusion) 

255 

256 # Convert to list for modification if needed 

257 distance_results_list = list(distance_results) 

258 

259 # Set distances to 0 for sites outside the provided geometry (distance != 0) 

260 for i, dist_to_exclusion in enumerate(distance_to_exclusion): 

261 if dist_to_exclusion != 0: 

262 distance_results_list[i] = 0 

263 

264 return distance_results_list

266 def distance_to(self, site_points: GeoSeries, geometry: GeoDataFrame) -> List[int]: 

267 """Calculate the distance from each site point to the nearest feature in the 

268 geometry. 

269 

270 Args: 

271 site_points: 

272 A GeoSeries containing the site points in the target projection. 

273 geometry: 

274 A GeoDataFrame containing the geometry in the target projection. 

275 Returns: 

276 A list of distances from each site point to the nearest feature in the 

277 geometry rounded to the nearest metre.""" 

278 

279 def _distance_to_nearest(point: Point, geometry: GeoDataFrame) -> float: 

280 """Calculate the distance from a point to the nearest feature in the 

281 geometry. 

282 Args: 

283 point: 

284 A shapely Point object representing the site location. 

285 geometry: 

286 A GeoDataFrame containing the geometry in the target projection. 

287 Returns: 

288 The distance from the point to the nearest feature in the geometry 

289 rounded to the nearest metre. 

290 """ 

291 return round(min(point.distance(geometry.geometry))) 

292 

293 if self.parallel: 

294 from joblib import Parallel, delayed 

295 

296 parallel = Parallel(n_jobs=self.n_parallel_jobs, prefer="threads") 

297 output_generator = parallel( 

298 delayed(_distance_to_nearest)(point, geometry) for point in site_points 

299 ) 

300 distance_results = list(output_generator) 

301 else: 

302 distance_results = [] 

303 for point in site_points: 

304 distance_results.append(_distance_to_nearest(point, geometry)) 

305 

306 return distance_results

308 def create_output_cube(self, site_cube: Cube, data: List[int]) -> Cube: 

309 """Create an output cube that will have the same metadata as the input site 

310 cube except the units are changed to metres and, if requested, the name of the 

311 output cube will be changed. 

312 

313 Args: 

314 site_cube: 

315 The input cube containing site locations that are defined by latitude 

316 and longitude coordinates. 

317 data: 

318 A list of distances from each site point to the nearest feature in the 

319 geometry. 

320 Returns: 

321 A new cube containing the distances with the same metadata as input site 

322 cube but with updated units and name.""" 

323 

324 output_cube = site_cube.copy(data=array(data)) 

325 output_cube.rename(self.new_name) 

326 output_cube.units = "m" 

327 

328 return output_cube

330 def process( 

331 self, 

332 site_cube: Cube, 

333 geometry: GeoDataFrame, 

334 exclude_outside_of: Optional[GeoDataFrame] = None, 

335 exclusion_buffer: Optional[float] = 10, 

336 ) -> Cube: 

337 """Generate a cube of the distance from sites in site_cube to the 

338 nearest point in geometry. 

339 

340 The latitude, longitude coordinates in the site_cube are extracted 

341 to define the location of the sites. The sites and feature geometry 

342 are reprojected to the target projection. 

343 

344 If requested the feature geometry will be clipped to the smallest square 

345 possible such that all sites in site_cube are included. A buffer distance is 

346 then added to each edge of the square which defines the size the feature 

347 geometry will be clipped to. This is useful when the feature geometry size is 

348 large and it would be expensive to calculate the distance to all features in the 

349 geometry or where the domain of the feature geometry is much larger than the 

350 area containing the site locations. Information may be lost at the edges of 

351 the domain if the feature is sparsely located in the geometry. 

352 

353 The distance from each site to every point in the feature geometry is then 

354 calculated and the minimum of these distances is returned. The distances are 

355 rounded to the nearest metre. 

356 

357 If a geometry is provided, distances are set to 0 for sites outside 

358 that geometry. 

359 

360 The output cube will have the same metadata as the input site_cube except the 

361 units will be changed to meters and, if requested, the name of the output cube 

362 will be updated. 

363 

364 Args: 

365 site_cube: 

366 The input cube containing site locations. This cube must have x and y 

367 axis which contain the site coordinates in latitude and longitude. 

368 geometry: 

369 The GeoDataFrame containing the geometry to calculate distances to. 

370 exclude_outside_of: 

371 An optional GeoDataFrame containing the geometry defining the valid 

372 region. For sites outside this geometry, the distance to the feature 

373 will be set to 0. This is useful when the target region cannot be 

374 represented as a GeoDataFrame directly (e.g., providing land geometry 

375 so that ocean sites, which lie outside the land, receive a distance of 

376 0). 

377 exclusion_buffer: 

378 A buffer distance in m used when clipping the optional geometry for 

379 performance. Default is 10m. Only used if exclude_outside_of is 

380 provided. This is independent of the clip_geometry / buffer 

381 settings, which apply only to clipping the feature geometry used for 

382 the initial distance calculation. 

383 

384 Returns: 

385 A new cube containing the distances from each site to the nearest feature 

386 in the geometry rounded to the nearest metre.""" 

387 

388 # Project the geometry and site cube coordinates to the target projection. 

389 site_coords, geometry_projection = self.project_geometry(geometry, site_cube) 

390 

391 if self._do_geometry_clipping: 

392 # Clip the geometry to the bounds of the site coordinates with a buffer if 

393 # requested 

394 x_bounds = self.get_clip_values(site_coords.x, self.buffer) 

395 y_bounds = self.get_clip_values(site_coords.y, self.buffer) 

396 

397 clipped_geometry = self.clip_geometry( 

398 geometry_projection, x_bounds, y_bounds 

399 ) 

400 else: 

401 clipped_geometry = geometry_projection 

402 

403 # Calculate the distance to the nearest feature in the geometry 

404 distance_to_results = self.distance_to(site_coords, clipped_geometry) 

405 

406 # If exclusion geometry is provided, set distances to 0 where it is present 

407 if exclude_outside_of is not None: 

408 distance_to_results = self._apply_exclusion( 

409 site_coords, 

410 distance_to_results, 

411 exclude_outside_of, 

412 exclusion_buffer, 

413 ) 

414 

415 return self.create_output_cube(site_cube, distance_to_results)

418class DistanceToNearestFeature(BasePlugin): 

419 """ 

420 Plugin to calculate the distance to the closest feature from multiple distance 

421 cubes, intended to be used following the DistanceToFeature plugin. This is useful 

422 when you have distances to multiple features and want to find the minimum distance 

423 to any feature (e.g., creating a distance to water ancillary cube from distance to 

424 river, lake, and ocean) cubes. 

425 

426 This plugin does not support parallelisation unlike the DistanceToFeature plugin as 

427 the distance cubes provided to this plugin should already have been generated and 

428 therefore the computationally expensive distance calculation has already been 

429 performed. 

430 """ 

431 

432 def process( 

433 self, 

434 distance_to_features: CubeList, 

435 new_name: str, 

436 ignored_coords_for_validation: Optional[List[str]] = None, 

437 ) -> Cube: 

438 """Calculate the distance to the closest feature from a CubeList of distance 

439 cubes. The distance to closest feature is the minimum of all the provided 

440 distance to feature cubes. 

441 

442 Example use case: if we want to generate a distance to water ancillary cube, we 

443 could use this function with distance to river, distance to lake, and distance 

444 to ocean cubes as the input. The output cube would then give the distance to the 

445 closest water feature for each site. 

446 

447 The first cube in distance_to_features is used as a template for the output 

448 metadata with the name updated to the provided new_name. 

449 

450 Args: 

451 distance_to_features: 

452 A CubeList containing distance to feature cubes from sites (e.g., 

453 distance to rivers, lakes, oceans, or any other features). 

454 Each cube should have the same set of sites defined. 

455 new_name: 

456 The name to assign to the output cube. E.g. "distance_to_water" 

457 If None, then the name of the output cube will be set to the same name 

458 as the first cube in distance_to_features. 

459 ignored_coords_for_validation: 

460 A list of coordinate names to ignore when validating that the cubes in 

461 distance_to_features have matching coordinates. 

462 

463 Returns: 

464 A cube containing the distance to closest feature ancillary data. 

465 

466 Raises: 

467 ValueError: If the input CubeList is empty. 

468 ValueError: If the cubes in the input CubeList have unmatched coordinates 

469 (other than those specified in ignored_coords_for_validation). 

470 """ 

471 import numpy as np 

472 

473 # Validate that the input CubeList is not empty 

474 if not distance_to_features: 

475 raise ValueError("The input CubeList distance_to_features cannot be empty.") 

476 

477 unmatched_coords = compare_coords( 

478 cubes=distance_to_features, ignored_coords=ignored_coords_for_validation 

479 ) 

480 if any(item.keys() for item in unmatched_coords): 

481 msg = f"Input cube coordinates {unmatched_coords} are unmatched." 

482 raise ValueError(msg) 

483 

484 # Calculate the minimum distance across all features 

485 distances_to_features = np.stack([cube.data for cube in distance_to_features]) 

486 min_distance = distances_to_features.min(axis=0) 

487 

488 # Create a new cube for the distance to closest feature using the first cube in 

489 # the input CubeList as a template for the metadata and updating the name to the 

490 # provided new_name. 

491 distance_to_closest = distance_to_features[0].copy(data=min_distance) 

492 distance_to_closest.rename(new_name) 

493 

494 return distance_to_closest
