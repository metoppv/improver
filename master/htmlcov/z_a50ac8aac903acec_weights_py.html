36 def normalise_weights(weights: ndarray, axis: Optional[int] = None) -> ndarray: 

37 """Ensures all weights add up to one. 

38 

39 Args: 

40 weights: 

41 array of weights 

42 axis: 

43 The axis that we want to normalise along for a multiple 

44 dimensional array. Defaults to None, meaning the whole 

45 array is used for the normalisation. 

46 

47 Returns: 

48 array of weights where sum = 1.0 

49 

50 Raises: 

51 ValueError: any negative weights are found in input. 

52 ValueError: sum of weights in the input is 0. 

53 """ 

54 if np.any(weights.min(axis=axis) < 0.0): 

55 msg = ( 

56 "Weights must be positive. The weights have at least one " 

57 "value < 0.0: {}".format(weights) 

58 ) 

59 raise ValueError(msg) 

60 

61 sumval = np.sum(weights, axis=axis, keepdims=True) 

62 if np.any(sumval == 0): 

63 msg = "Sum of weights must be > 0.0" 

64 raise ValueError(msg) 

65 

66 normalised_weights = weights / sumval 

67 return normalised_weights

70 def build_weights_cube(cube: Cube, weights: ndarray, blending_coord: str) -> Cube: 

71 """Build a cube containing weights for use in blending. 

72 

73 Args: 

74 cube: 

75 The cube that is being blended over blending_coord. 

76 weights: 

77 Array of weights 

78 blending_coord: 

79 Name of the coordinate over which the weights will be used 

80 to blend data, e.g. across model name when grid blending. 

81 

82 Returns: 

83 A cube containing the array of weights. 

84 

85 Raises: 

86 ValueError : If weights array is not of the same length as the 

87 coordinate being blended over on cube. 

88 """ 

89 

90 if len(weights) != len(cube.coord(blending_coord).points): 

91 msg = ( 

92 "Weights array provided is not the same size as the " 

93 "blending coordinate; weights shape: {}, blending " 

94 "coordinate shape: {}".format( 

95 len(weights), len(cube.coord(blending_coord).points) 

96 ) 

97 ) 

98 raise ValueError(msg) 

99 

100 try: 

101 weights_cube = next(cube.slices(blending_coord)) 

102 except ValueError: 

103 weights_cube = iris.util.new_axis(cube, blending_coord) 

104 weights_cube = next(weights_cube.slices(blending_coord)) 

105 weights_cube.attributes = None 

106 # Find dim associated with blending_coord and don't remove any coords 

107 # associated with this dimension. 

108 blending_dim = cube.coord_dims(blending_coord) 

109 defunct_coords = [ 

110 crd.name() 

111 for crd in cube.coords(dim_coords=True) 

112 if not cube.coord_dims(crd) == blending_dim 

113 ] 

114 for crd in defunct_coords: 

115 weights_cube.remove_coord(crd) 

116 weights_cube.data = weights 

117 weights_cube.rename("weights") 

118 weights_cube.units = 1 

119 

120 return weights_cube

123class ChooseWeightsLinear(BasePlugin): 

124 """Plugin to interpolate weights linearly to the required points, where 

125 original weights are provided as a configuration dictionary""" 

126 

127 def __init__( 

128 self, 

129 weighting_coord_name: str, 

130 config_dict: Dict[str, Dict[str, Any]], 

131 config_coord_name: str = MODEL_NAME_COORD, 

132 ) -> None: 

133 """ 

134 Set up for calculating linear weights from a dictionary or input cube 

135 

136 Args: 

137 weighting_coord_name: 

138 Standard name of the coordinate along which the weights will be 

139 interpolated. For example, if the intention is to provide 

140 weights varying with forecast period, then this argument would 

141 be "forecast_period". This coordinate must be included within 

142 the configuration dictionary. 

143 config_dict: 

144 Dictionary containing the configuration information, namely 

145 an initial set of weights and information regarding the 

146 points along the specified coordinate at which the weights are 

147 valid. An example dictionary is shown below. 

148 config_coord_name: 

149 Name of the coordinate used to select the configuration. 

150 For example, if the intention is to create weights that scale 

151 differently with the weighting_coord for different models, then 

152 MODEL_NAME_COORD would be the config_coord. 

153 

154 Dictionary of format:: 

155 

156 { 

157 "uk_det": { 

158 "forecast_period": [7, 12], 

159 "weights": [1, 0], 

160 "units": "hours" 

161 } 

162 "uk_ens": { 

163 "forecast_period": [7, 12, 48, 54], 

164 "weights": [0, 1, 1, 0], 

165 "units": "hours" 

166 } 

167 } 

168 

169 

170 To assign a different constant weight to each model, choose any coordinate 

171 for "weighting_coord_name" and choose any two points for its value in the 

172 dictionary. Set the value of "weights" to be the same for both points in 

173 each model's dictionary. For example, we can assign weights of 0.3 and 0.7 

174 to models uk_det and uk_ens as follows:: 

175 

176 { 

177 "uk_det": { 

178 "forecast_period": [0, 48], 

179 "weights": [0.3, 0.3], 

180 "units": "hours" 

181 } 

182 "uk_ens": { 

183 "forecast_period": [0, 48], 

184 "weights": [0.7, 0.7], 

185 "units": "hours" 

186 } 

187 } 

188 """ 

189 self.weighting_coord_name = weighting_coord_name 

190 self.config_coord_name = config_coord_name 

191 self.config_dict = config_dict 

192 self.weights_key_name = "weights" 

193 self._check_config_dict()

206 """

235 def _get_interpolation_inputs_from_dict( 

236 self, cube: Cube 

237 ) -> Tuple[ndarray, ndarray, ndarray, Tuple[int, int]]: 

238 """ 

239 Generate inputs required for linear interpolation. 

240 

241 Args: 

242 cube: 

243 Cube containing the coordinate information that will be used 

244 for setting up the interpolation inputs. 

245 

246 Returns: 

247 - Points within the configuration dictionary that will 

248 be used as the input to the interpolation. 

249 - Points within the cube that will be the target points 

250 for the interpolation. 

251 - Weights from the configuration dictionary that will be 

252 used as the input to the interpolation. 

253 - Values that be used if extrapolation is required. The 

254 fill values will be used as constants that are extrapolated 

255 if the target_points are outside the source_points 

256 provided. These are equal to the first and last values 

257 provided by the source weights. 

258 """ 

259 (config_point,) = cube.coord(self.config_coord_name).points 

260 source_points = self.config_dict[config_point][self.weighting_coord_name] 

261 source_points = np.array(source_points) 

262 if "units" in self.config_dict[config_point].keys(): 

263 units = cf_units.Unit(self.config_dict[config_point]["units"]) 

264 source_points = units.convert( 

265 source_points, cube.coord(self.weighting_coord_name).units 

266 ) 

267 

268 target_points = cube.coord(self.weighting_coord_name).points 

269 source_weights = self.config_dict[config_point][self.weights_key_name] 

270 

271 fill_value = (source_weights[0], source_weights[-1]) 

272 return source_points, target_points, source_weights, fill_value

274 @staticmethod 

275 def _interpolate_to_find_weights( 

276 source_points: ndarray, 

277 target_points: ndarray, 

278 source_weights: ndarray, 

279 fill_value: Tuple[int, int], 

280 axis: int = 0, 

281 ) -> ndarray: 

282 """ 

283 Use of scipy.interpolate.interp1d to interpolate source_weights 

284 (valid at source_points) onto target_points grid. This allows 

285 the specification of an axis for the interpolation, so that the 

286 source_weights can be a multi-dimensional numpy array. 

287 

288 Args: 

289 source_points: 

290 Points within the configuration dictionary that will 

291 be used as the input to the interpolation. 

292 target_points: 

293 Points within the cube that will be the target points 

294 for the interpolation. 

295 source_weights: 

296 Weights from the configuration dictionary that will be 

297 used as the input to the interpolation. 

298 fill_value: 

299 Values to be used if extrapolation is required. The 

300 fill values are used for target_points that are outside 

301 the source_points grid. 

302 axis: 

303 Axis along which the interpolation will occur. 

304 

305 Returns: 

306 Weights corresponding to target_points following interpolation. 

307 """ 

308 f_out = interp1d( 

309 source_points, 

310 source_weights, 

311 axis=axis, 

312 fill_value=fill_value, 

313 bounds_error=False, 

314 ) 

315 weights = f_out(target_points) 

316 return weights

318 def _create_new_weights_cube(self, cube: Cube, weights: ndarray) -> Cube: 

319 """Create a cube to contain the output of the interpolation. 

320 It is currently assumed that the output weights matches the size 

321 of the input cube. 

322 

323 Args: 

324 cube: 

325 Cube containing the coordinate information that will be used 

326 for setting up the new_weights_cube. 

327 weights: 

328 Weights calculated following interpolation. 

329 

330 Returns: 

331 Cube containing the output from the interpolation. This has 

332 the same shape as "cube", without the x and y dimensions. 

333 """ 

334 spatial = [cube.coord(axis="y"), cube.coord(axis="x")] 

335 

336 cubelist = iris.cube.CubeList([]) 

337 for cube_slice, weight in zip( 

338 cube.slices_over(self.weighting_coord_name), weights 

339 ): 

340 sub_slice = next(cube_slice.slices_over(spatial)) 

341 sub_slice.data = np.ones(sub_slice.data.shape) * weight 

342 cubelist.append(sub_slice) 

343 

344 # re-order dimension coordinates to match input cube 

345 new_weights_cube = check_cube_coordinates( 

346 next(cube.slices_over(spatial)), cubelist.merge_cube() 

347 ) 

348 

349 # remove all scalar coordinates that are not time-, model- or 

350 # blend-related 

351 dim_coords = new_weights_cube.coords(dim_coords=True) 

352 keep_coords = [ 

353 "time", 

354 "forecast_period", 

355 "forecast_reference_time", 

356 MODEL_BLEND_COORD, 

357 MODEL_NAME_COORD, 

358 self.weighting_coord_name, 

359 self.config_coord_name, 

360 ] 

361 for coord in new_weights_cube.coords(): 

362 if coord not in dim_coords and coord.name() not in keep_coords: 

363 new_weights_cube.remove_coord(coord) 

364 

365 # remove attributes 

366 new_weights_cube.attributes = {} 

367 

368 # rename cube 

369 new_weights_cube.rename(self.weights_key_name) 

370 new_weights_cube.units = cf_units.Unit("1") 

371 

372 return new_weights_cube

374 def _calculate_weights(self, cube: Cube) -> Cube: 

375 """Method to wrap the calls to other methods to support calculation 

376 of the weights by interpolation. 

377 

378 Args: 

379 cube: 

380 Cube containing the coordinate information that will be used 

381 for setting up the interpolation and create the new weights 

382 cube. 

383 

384 Returns: 

385 Cube containing the output from the interpolation. This 

386 has been renamed using the self.weights_key_name but 

387 otherwise matches the input cube. 

388 """ 

389 (source_points, target_points, source_weights, fill_value) = ( 

390 self._get_interpolation_inputs_from_dict(cube) 

391 ) 

392 axis = 0 

393 

394 weights = self._interpolate_to_find_weights( 

395 source_points, target_points, source_weights, fill_value, axis=axis 

396 ) 

397 

398 new_weights_cube = self._create_new_weights_cube(cube, weights) 

399 

400 return new_weights_cube

402 def _define_slice(self, cube: Cube) -> List[Coord]: 

403 """ 

404 Returns a list of coordinates over which to slice the input cube to 

405 create a list of cubes for blending. 

406 

407 Args: 

408 cube: 

409 Cube input to plugin 

410 

411 Returns: 

412 List of coordinates defining the slice to iterate over 

413 """ 

414 if cube.coord_dims(self.weighting_coord_name): 

415 slice_list = [ 

416 cube.coord(self.weighting_coord_name), 

417 cube.coord(axis="y"), 

418 cube.coord(axis="x"), 

419 ] 

420 else: 

421 slice_list = [cube.coord(axis="y"), cube.coord(axis="x")] 

422 

423 # To handle non-orthogonal spatial coordinates, i.e. multiple coordinates 

424 # that share the same dimension, as in a spot-forecast. 

425 unique_slice_list = [] 

426 for dim in set([cube.coord_dims(crd) for crd in slice_list]): 

427 unique_slice_list.append(cube.coords(dimensions=dim)[0]) 

428 

429 return unique_slice_list

431 def _slice_input_cubes(self, cubes: Union[Cube, CubeList]) -> CubeList: 

432 """ 

433 From input iris.cube.Cube or iris.cube.CubeList, create a list of 

434 cubes with different values of the config coordinate (over which to 

435 blend), with irrelevant dimensions sliced out. 

436 

437 Args: 

438 cubes: 

439 Cubes passed into the plugin. 

440 

441 Returns: 

442 List of cubes (from which to calculate weights) with 

443 dimensions (y, x) if weighting_coord is scalar on the input 

444 cube, or (weighting_coord, y, x) if weighting_coord is 

445 non-scalar 

446 """ 

447 if isinstance(cubes, iris.cube.Cube): 

448 # check how many points there are in the config coordinate 

449 if len(cubes.coord(self.config_coord_name).points) == 1: 

450 cubelist = [next(cubes.slices(self._define_slice(cubes)))] 

451 else: 

452 # if passed a merged cube, split this up into a cube list 

453 cubelist = [] 

454 for cube in cubes.slices_over(cubes.coord(self.config_coord_name)): 

455 cubelist.append(next(cube.slices(self._define_slice(cube)))) 

456 else: 

457 cubelist = [] 

458 for cube in cubes: 

459 cubelist.append(next(cube.slices(self._define_slice(cube)))) 

460 

461 return iris.cube.CubeList(cubelist)

463 def process(self, cubes: Union[Cube, CubeList]) -> Cube: 

464 """Calculation of linear weights based on an input dictionary. 

465 

466 Args: 

467 cubes: 

468 Cubes containing the coordinate (source point) information 

469 that will be used for setting up the interpolation. Each cube 

470 should have "self.config_coord_name" as a scalar dimension; if 

471 a merged cube is passed in, the plugin will split this into a 

472 list cubes. 

473 

474 Returns: 

475 Cube containing the output from the interpolation. 

476 DimCoords (such as model_id) will be in sorted-ascending order. 

477 """ 

478 # create 2D cube lists with relevant dimensions only for dict 

479 # processing 

480 cubes = self._slice_input_cubes(cubes) 

481 

482 # calculate weights 

483 cube_slices = iris.cube.CubeList([]) 

484 for cube in cubes: 

485 new_weights_cube = self._calculate_weights(cube) 

486 cube_slices.append(new_weights_cube) 

487 

488 # normalise weights 

489 new_weights_cube = cube_slices.merge_cube() 

490 axis = new_weights_cube.coord_dims(self.config_coord_name) 

491 new_weights_cube.data = WeightsUtilities.normalise_weights( 

492 new_weights_cube.data, axis=axis 

493 ) 

494 

495 return new_weights_cube

498class ChooseDefaultWeightsLinear(BasePlugin): 

499 """Calculate Default Weights using Linear Function.""" 

500 

501 def __init__(self, y0val: float, ynval: float) -> None: 

502 """ 

503 Set up for calculating default weights using linear function. 

504 

505 Args: 

506 y0val: 

507 Relative weight of first point. Must be positive. 

508 ynval: 

509 Relative weight of last point. 

510 """ 

511 if y0val is None or ynval is None: 

512 raise ValueError( 

513 "y0val and ynval are required arguments to the " 

514 "ChooseDefaultWeightsLinear plugin" 

515 ) 

516 

517 if y0val < 0.0: 

518 msg = "y0val must be a float >= 0.0, y0val = {0:s}".format(str(y0val)) 

519 raise ValueError(msg) 

520 

521 self.y0val = float(y0val) 

522 self.ynval = float(ynval) 

523 

524 def linear_weights(self, num_of_weights: int) -> ndarray: 

525 """Create linear weights 

526 

527 Args: 

528 num_of_weights: 

529 Number of weights to create. 

530 

531 Returns: 

532 array of weights, sum of all weights = 1.0 

533 """ 

534 # Special case num_of_weights == 1 i.e. Scalar coordinate. 

535 if num_of_weights == 1: 

536 weights = np.array([1.0], dtype=np.float32) 

537 return weights 

538 

539 slope = (self.ynval - self.y0val) / (num_of_weights - 1.0) 

540 

541 weights_list = [] 

542 for tval in range(0, num_of_weights): 

543 weights_list.append(slope * tval + self.y0val) 

544 

545 weights = WeightsUtilities.normalise_weights( 

546 np.array(weights_list, dtype=np.float32) 

547 ) 

548 

549 return weights

551 def process(self, cube: Cube, coord_name: str) -> Cube: 

552 """ 

553 Calculated weights for a given cube and coord. Weights scale linearly 

554 between self.y0val and self.ynval for the cube provided in ascending 

555 order of blend coordinate. self.y0val = self.ynval gives equal 

556 weightings across all input fields. 

557 

558 Args: 

559 cube: 

560 Cube to blend across the coord. 

561 coord_name: 

562 Name of coordinate in the cube to be blended. 

563 

564 Returns: 

565 1D cube of normalised (sum = 1.0) weights matching length 

566 of input dimension to be blended 

567 

568 Raises: 

569 TypeError : input is not a cube 

570 """ 

571 if not isinstance(cube, iris.cube.Cube): 

572 msg = ( 

573 "The first argument must be an instance of " 

574 "iris.cube.Cube but is" 

575 " {0:s}".format(str(type(cube))) 

576 ) 

577 raise TypeError(msg) 

578 

579 weights = self.linear_weights(len(cube.coord(coord_name).points)) 

580 weights_cube = WeightsUtilities.build_weights_cube(cube, weights, coord_name) 

581 return weights_cube

591class ChooseDefaultWeightsNonLinear(BasePlugin): 

592 """Calculate Default Weights using NonLinear Function.""" 

593 

594 def __init__(self, cval: float) -> None: 

595 """ 

596 Set up for calculating default weights using non-linear function. 

597 

598 Args: 

599 cval: 

600 Value greater than 0, less than equal 1.0. Weights are 

601 calculated for input cubes in order such that the first has 

602 weight cval**0, then cval**1, cval**2, etc. The weights are 

603 then re-normalised. Thus a value of 1 gives equal weighting 

604 across all input fields. 

605 

606 Raises: 

607 ValueError: an inappropriate value of cval is input. 

608 """ 

609 if cval is None: 

610 raise ValueError( 

611 "cval is a required argument to the " 

612 "ChooseDefaultWeightsNonLinear plugin" 

613 ) 

614 

615 if cval <= 0.0 or cval > 1.0: 

616 msg = ( 

617 "cval must be greater than 0.0 and less " 

618 "than or equal to 1.0 cval = {}".format(cval) 

619 ) 

620 raise ValueError(msg) 

621 self.cval = cval 

622 

623 def nonlinear_weights(self, num_of_weights: int) -> ndarray: 

624 """ 

625 Create nonlinear weights. 

626 

627 Args: 

628 num_of_weights: 

629 Number of weights to create 

630 

631 Returns: 

632 Normalised array of weights 

633 """ 

634 weights_list = [] 

635 for tval_minus1 in range(0, num_of_weights): 

636 weights_list.append(self.cval ** (tval_minus1)) 

637 

638 weights = WeightsUtilities.normalise_weights( 

639 np.array(weights_list, dtype=np.float32) 

640 ) 

641 

642 return weights

644 def process( 

645 self, cube: Cube, coord_name: str, inverse_ordering: bool = False 

646 ) -> Cube: 

647 """ 

648 Calculate nonlinear weights for a given cube and coord. 

649 

650 Args: 

651 cube: 

652 Cube to be blended across the coord. 

653 coord_name: 

654 Name of coordinate in the cube to be blended. 

655 inverse_ordering: 

656 The input cube blend coordinate will be in ascending order, 

657 so that calculated blend weights decrease with increasing 

658 value. For eg cycle blending by forecast reference time, we 

659 wish to weight more recent cubes more highly. This flag gives 

660 the option to reverse the blend coordinate order so as to have 

661 higher weights for the higher values. 

662 

663 Returns: 

664 1D cube of normalised (sum = 1.0) weights matching input 

665 dimension to be blended 

666 

667 Raises: 

668 TypeError : input is not a cube 

669 """ 

670 if not isinstance(cube, iris.cube.Cube): 

671 msg = ( 

672 "The first argument must be an instance of " 

673 "iris.cube.Cube but is" 

674 " {0:s}".format(str(type(cube))) 

675 ) 

676 raise TypeError(msg) 

677 

678 if inverse_ordering: 

679 # make a copy of the input cube from which to calculate weights 

680 inverted_cube = cube.copy() 

681 inverted_cube = sort_coord_in_cube( 

682 inverted_cube, coord_name, descending=True 

683 ) 

684 cube = inverted_cube 

685 

686 weights = self.nonlinear_weights(len(cube.coord(coord_name).points)) 

687 weights_cube = WeightsUtilities.build_weights_cube(cube, weights, coord_name) 

688 

689 if inverse_ordering: 

690 # re-sort the weights cube so that it is in ascending order of 

691 # blend coordinate (and hence matches the input cube) 

692 weights_cube = sort_coord_in_cube(weights_cube, coord_name) 

693 

694 return weights_cube

702class ChooseDefaultWeightsTriangular(BasePlugin): 

703 """Calculate Default Weights using a Triangular Function.""" 

704 

705 def __init__(self, width: float, units: Union[Unit, str] = "no_unit") -> None: 

706 """Set up for calculating default weights using triangular function. 

707 

708 Args: 

709 width: 

710 The width of the triangular function from the centre point. 

711 units: 

712 The cf units of the width and midpoint. 

713 """ 

714 self.width = width 

715 if not isinstance(units, cf_units.Unit): 

716 units = cf_units.Unit(units) 

717 self.parameters_units = units 

718 

719 def __repr__(self) -> str: 

720 """Represent the configured plugin instance as a string.""" 

721 msg = "<ChooseDefaultTriangularWeights width={}, parameters_units={}>" 

722 desc = msg.format(self.width, self.parameters_units) 

723 return desc 

724 

725 @staticmethod 

726 def triangular_weights( 

727 coord_vals: ndarray, midpoint: float, width: float 

728 ) -> ndarray: 

729 """Calculate triangular weights. 

730 

731 Args: 

732 coord_vals: 

733 An array of coordinate values that we want to calculate 

734 weights for. 

735 midpoint: 

736 The centre point of the triangular function. 

737 width: 

738 The width from the triangle’s centre point, in units of the plugin's 

739 units argument, which will determine the triangular weighting function 

740 used to blend that specified point with its adjacent points. Beyond 

741 this width the weighting drops to zero. 

742 

743 Returns: 

744 An array of weights, the sum of which should equal 1.0. 

745 """ 

746 

747 def calculate_weight(point: float, slope: float) -> float: 

748 """ 

749 A helper function to calculate the weights for each point using a 

750 piecewise function to build up the triangular function. 

751 Args: 

752 point: 

753 The point in the coordinate from the cube for 

754 which we want to calculate a weight for. 

755 slope: 

756 The gradient of the triangle, calculated from 

757 1/(width of triangle). 

758 

759 Returns: 

760 The individual weight calculated by the function. 

761 """ 

762 if point == midpoint: 

763 weight = 1 

764 else: 

765 weight = 1 - abs(point - midpoint) * slope 

766 return weight 

767 

768 slope = 1.0 / width 

769 weights = np.zeros(coord_vals.shape, dtype=np.float32) 

770 # Find the indices of the points where there will be non-zero weights. 

771 condition = (coord_vals >= (midpoint - width)) & ( 

772 coord_vals <= (midpoint + width) 

773 ) 

774 points_with_weights = np.where(condition)[0] 

775 # Calculate for weights for points where we want a non-zero weight. 

776 for index in points_with_weights: 

777 weights[index] = calculate_weight(coord_vals[index], slope) 

778 # Normalise the weights. 

779 weights = WeightsUtilities.normalise_weights(weights) 

780 

781 return weights 

782 

783 def process(self, cube: Cube, coord_name: str, midpoint: float) -> Cube: 

784 """Calculate triangular weights for a given cube and coord. 

785 

786 Args: 

787 cube: 

788 Cube to blend across the coord. 

789 coord_name: 

790 Name of coordinate in the cube to be blended. 

791 midpoint: 

792 The centre point of the triangular function. This is 

793 assumed to be provided in the same units as "self.width", 

794 ie "self.parameter_units" as initialised. 

795 

796 Returns: 

797 1D cube of normalised (sum = 1.0) weights matching length 

798 of input dimension to be blended. 

799 

800 Raises: 

801 TypeError : input is not a cube 

802 """ 

803 if not isinstance(cube, iris.cube.Cube): 

804 msg = ( 

805 "The first argument must be an instance of " 

806 "iris.cube.Cube but is" 

807 " {0:s}".format(str(type(cube))) 

808 ) 

809 raise TypeError(msg) 

810 

811 cube_coord = cube.coord(coord_name) 

812 coord_vals = cube_coord.points 

813 coord_units = cube_coord.units 

814 

815 # Rescale width and midpoint if in different units to the coordinate 

816 if coord_units != self.parameters_units: 

817 width_in_coord_units = self.parameters_units.convert( 

818 self.width, coord_units 

819 ) 

820 midpoint = self.parameters_units.convert(midpoint, coord_units) 

821 else: 

822 width_in_coord_units = copy.deepcopy(self.width) 

823 

824 weights = self.triangular_weights(coord_vals, midpoint, width_in_coord_units) 

825 

826 weights_cube = WeightsUtilities.build_weights_cube(cube, weights, coord_name) 

827 return weights_cube
