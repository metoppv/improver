12def as_iterable(obj): 

13 """Return obj as an iterable, or an empty list if obj is None.""" 

14 if obj is None: 

15 return [] 

16 elif not isinstance(obj, Iterable) or isinstance(obj, (str, bytes)): 

17 obj = [obj] 

18 return obj

21def as_cubelist(*cubes: Union[Cube, CubeList]): 

22 """ 

23 Standardise input handling of cube/cubelist arguments. 

24 

25 The role of this function is to flatten the provided inputs and thereby 

26 return a single CubeList object. 

27 

28 Args: 

29 cubes: 

30 Input data provided in the form of one or more cubes or cubelists 

31 (or mixture thereof). 

32 Any iterable is supported. 

33 

34 Returns: 

35 CubeList: 

36 A CubeList containing all the cubes provided as input. 

37 """ 

38 cubes = CubeList(flatten(cubes)) 

39 if len(cubes) == 0: 

40 raise ValueError("One or more cubes should be provided.") 

41 return cubes

44def as_cube(*cube: Union[Cube, CubeList]): 

45 """ 

46 Standardise input handling of cube arguments. 

47 

48 The role of this function is to return a single cube object. 

49 Where more than one cube is provided, a cubelist merge is attempted. 

50 

51 Args: 

52 cube: 

53 Input data provided in the form of a cube or cubelists. 

54 Any iterable is supported. 

55 

56 Returns: 

57 Cube: 

58 A single cube. 

59 """ 

60 cubelist = as_cubelist(*cube) 

61 if len(cubelist) != 1: 

62 # cube merge changes the object ID so this is conditional 

63 try: 

64 cubelist = [cubelist.merge_cube()] 

65 except Exception as err: 

66 err_msg = "Unable to return a single cube." 

67 raise ValueError(err_msg) from err 

68 return cubelist[0]
