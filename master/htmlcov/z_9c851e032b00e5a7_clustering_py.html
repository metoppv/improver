5"""Plugins to perform clustering on DataFrames using scikit-learn or kmedoids.""" 

6 

7from typing import Any 

8 

9import pandas as pd 

10 

11from improver import BasePlugin 

12 

13 

14class FitClustering(BasePlugin): 

15 """Class to perform clustering on DataFrames using scikit-learn or kmedoids. 

16 

17 This plugin provides a unified interface for applying various clustering algorithms 

18 to pandas DataFrames. It supports clustering methods from scikit-learn's cluster 

19 module as well as the KMedoids algorithm from the kmedoids package. 

20 The plugin automatically selects the appropriate package based on the specified 

21 clustering method: 

22 - "KMedoids": Uses the kmedoids package 

23 - All other methods: Uses sklearn.cluster 

24 """ 

25 

26 def __init__(self, clustering_method: str, **kwargs: Any) -> None: 

27 """Initialise the clustering plugin. 

28 

29 Args: 

30 clustering_method: The name of the clustering method to use. 

31 Must be either "KMedoids" (from kmedoids package) or a valid 

32 clustering class name from sklearn.cluster (e.g., "KMeans", 

33 "DBSCAN", "AgglomerativeClustering"). 

34 **kwargs: Additional keyword arguments to pass to the clustering 

35 algorithm. These are method-specific parameters. Common examples: 

36 - n_clusters (int): Number of clusters (for KMeans, 

37 AgglomerativeClustering) 

38 - random_state (int): Random seed for reproducibility 

39 Refer to the scikit-learn or kmedoids documentation for the complete 

40 list of parameters for each clustering method. 

41 Raises: 

42 ValueError: If the specified clustering method is not found in 

43 sklearn.cluster or kmedoids packages. 

44 """ 

45 self.clustering_method = clustering_method 

46 self.kwargs = kwargs 

47 

48 def process(self, df: pd.DataFrame) -> Any: 

49 """Apply the clustering method to the DataFrame. Fits the specified clustering 

50 algorithm to the input DataFrame and returns the fitted clustering model. 

51 

52 Args: 

53 df: The input DataFrame to cluster. Each row represents 

54 a sample and each column represents a feature. The DataFrame should 

55 contain numeric data suitable for the chosen clustering algorithm. 

56 Returns: 

57 A fitted clustering model object from either sklearn.cluster or kmedoids. 

58 The returned object will have at minimum a `labels_` attribute containing 

59 the cluster assignment for each sample. Additional attributes depend on 

60 the specific clustering method used (e.g., `cluster_centers_` for KMeans, 

61 `core_sample_indices_` for DBSCAN). 

62 Raises: 

63 ValueError: If the specified clustering method is not found in 

64 sklearn.cluster or is not "KMedoids". 

65 """ 

66 # Use kmedoids directly if requested 

67 if self.clustering_method == "KMedoids": 

68 import kmedoids 

69 

70 # Set default metric to euclidean if not specified 

71 kwargs = self.kwargs.copy() 

72 if "metric" not in kwargs: 

73 kwargs["metric"] = "euclidean" 

74 

75 clustering_class = getattr(kmedoids, self.clustering_method) 

76 # Convert DataFrame to numpy array for kmedoids 

77 return clustering_class(**kwargs).fit(df.values) 

78 

79 # Otherwise, use sklearn 

80 from sklearn import cluster 

81 

82 if hasattr(cluster, self.clustering_method): 

83 clustering_class = getattr(cluster, self.clustering_method) 

84 return clustering_class(**self.kwargs).fit(df) 

85 else: 

86 msg = ( 

87 f"The clustering method '{self.clustering_method}' is not supported. " 

88 "Please check sklearn.cluster documentation for available methods." 

89 ) 

90 raise ValueError(msg)
