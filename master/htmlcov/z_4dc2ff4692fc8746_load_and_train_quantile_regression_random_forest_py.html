39class LoadForTrainQRF(PostProcessingPlugin): 

40 """Plugin to load input files for training a Quantile Regression Random Forest

43 def __init__( 

44 self, 

45 feature_config: dict[str, list[str]], 

46 parquet_diagnostic_names: Union[list[str], str], 

47 cf_names: Union[list[str], str], 

48 forecast_periods: str, 

49 cycletime: str, 

50 training_length: int, 

51 experiments: list[str], 

52 unique_site_id_keys: Union[list[str], str] = "wmo_id", 

53 ): 

54 """Initialise the LoadForTrainQRF plugin. 

55 

56 Args: 

57 feature_config: Feature configuration defining the features to be used for 

58 Quantile Regression Random Forests. 

59 parquet_diagnostic_names: A list containing the diagnostic names that will 

60 be used for filtering the forecast and truth DataFrames read in from 

61 the parquet files. The target diagnostic name is expected to be the 

62 first item in the list. These names could be different from the 

63 CF name e.g. 'temperature_at_screen_level'. This is expected to be the 

64 same length as the cf_names and experiments lists. 

65 cf_names: A list containing the CF names of the diagnostics. The CF names should 

66 match the order of the parquet_diagnostic_names. The target diagnostic to be 

67 calibrated is expected to be the first item in the list. These names 

68 could be different from the diagnostic name used to identify in the 

69 parquet files. For example, the diagnostic name could be 

70 'temperature_at_screen_level' and the corresponding CF name could be 

71 'air_temperature'. This is expected to be the same length as the 

72 parquet_diagnostic_names and experiments lists. 

73 forecast_periods: Range of forecast periods to be calibrated in hours in 

74 the form: "start:end:interval" e.g. "6:18:6" or a single forecast period 

75 e.g. "6". 

76 cycletime: The time at which the forecast is valid in the form: 

77 YYYYMMDDTHHMMZ. 

78 training_length: The number of days of training data to use. 

79 experiment: The name of the experiment (step) that calibration is applied to. 

80 experiments: The names of the experiment (step) that calibration is 

81 applied to. This is used to filter the forecast DataFrame on load. 

82 This is expected to be the same length as the parquet_diagnostic_names 

83 and cf_names lists. 

84 unique_site_id_key: The names of the coordinates that uniquely identify 

85 each site, e.g. "wmo_id" or "latitude,longitude". 

86 """ 

87 self.quantile_forest_installed = quantile_forest_package_available() 

88 self.feature_config = feature_config 

89 self.parquet_diagnostic_names = parquet_diagnostic_names 

90 self.cf_names = cf_names 

91 self.forecast_periods = forecast_periods 

92 self.cycletime = cycletime 

93 self.training_length = training_length 

94 self.experiments = experiments 

95 if isinstance(unique_site_id_keys, str): 

96 unique_site_id_keys = [unique_site_id_keys] 

97 self.unique_site_id_keys = unique_site_id_keys 

98 

99 if ( 

100 len(set([len(parquet_diagnostic_names), len(cf_names), len(experiments)])) 

101 > 1 

102 ): 

103 msg = ( 

104 "The length of the parquet_diagnostic_names list must match the length " 

105 "of the cf_names list and the length of the experiments list. " 

106 "The lengths provided are " 

107 f"{len(parquet_diagnostic_names)}, {len(cf_names)} and " 

108 f"{len(experiments)}, respectively." 

109 ) 

110 raise ValueError(msg)

112 def _parse_forecast_periods(self) -> list[int]: 

113 """Parse the forecast periods argument to produce a list of forecast periods 

114 in seconds. 

115 

116 Returns: 

117 List of forecast periods in seconds. 

118 Raises: 

119 ValueError: If the forecast_periods argument is not a single integer or 

120 a range in the form 'start:end:interval'. 

121 """ 

122 if ":" in self.forecast_periods: 

123 forecast_periods = list(range(*map(int, self.forecast_periods.split(":")))) 

124 forecast_periods = [fp * 3600 for fp in forecast_periods] 

125 else: 

126 try: 

127 forecast_periods = [int(self.forecast_periods) * 3600] 

128 except ValueError: 

129 msg = ( 

130 "The forecast_periods argument must be a single integer after " 

131 "extraction from the string input, or a range in the form " 

132 "'start:end:interval'. The forecast period provided was: " 

133 f"{self.forecast_periods}." 

134 ) 

135 raise ValueError(msg) 

136 return forecast_periods

138 def _read_parquet_files( 

139 self, 

140 forecast_table_path: pathlib.Path | str, 

141 truth_table_path: pathlib.Path | str, 

142 forecast_periods: list[int], 

143 ) -> tuple[pd.DataFrame, pd.DataFrame]: 

144 """Read the forecast and truth data from parquet files. 

145 self.quantile_forest_installed = quantile_forest_package_available() 

146 Args: 

147 forecast_table_path: Path to the forecast parquet file. 

148 truth_table_path: Path to the truth parquet file. 

149 forecast_periods: List of forecast periods in seconds. 

150 

151 Returns: 

152 Tuple containing: 

153 - DataFrame containing the forecast data. 

154 - DataFrame containing the truth data. 

155 

156 Raises: 

157 ValueError: If the forecast parquet file does not contain the expected 

158 fields. Either "percentile" or "realization". 

159 ValueError: If the forecast parquet file does not contain the expected 

160 features. 

161 ValueError: If the truth parquet file does not contain the expected 

162 fields. 

163 """ 

164 import pyarrow as pa 

165 import pyarrow.parquet as pq 

166 

167 cycletimes = [] 

168 for forecast_period in forecast_periods: 

169 # Load forecasts from parquet file filtering by diagnostic and blend_time. 

170 cycletimes.extend( 

171 get_training_period_cycles( 

172 self.cycletime, forecast_period, self.training_length 

173 ) 

174 ) 

175 cycletimes = list(set(cycletimes)) 

176 

177 example_file_path = next(Path(forecast_table_path).glob("**/*.parquet")) 

178 if pq.read_schema(example_file_path).get_all_field_indices("percentile"): 

179 altered_schema = CalibrationSchemas().FORECAST_SCHEMA 

180 elif pq.read_schema(example_file_path).get_all_field_indices("realization"): 

181 altered_schema = CalibrationSchemas().FORECAST_SCHEMA.remove( 

182 CalibrationSchemas().FORECAST_SCHEMA.get_field_index("percentile") 

183 ) 

184 altered_schema = altered_schema.append(pa.field("realization", pa.int64())) 

185 else: 

186 msg = ( 

187 "The forecast parquet file is expected to contain either a " 

188 "'percentile' or 'realization' field. Neither was found." 

189 ) 

190 raise ValueError(msg) 

191 

192 forecast_df = None 

193 for parquet_diagnostic_name, cf_name, experiment in zip( 

194 self.parquet_diagnostic_names, self.cf_names, self.experiments 

195 ): 

196 filters = [ 

197 [ 

198 ("diagnostic", "==", parquet_diagnostic_name), 

199 ("blend_time", "in", cycletimes), 

200 ("experiment", "==", experiment), 

201 ] 

202 ] 

203 

204 additional_df = pd.read_parquet( 

205 forecast_table_path, 

206 filters=filters, 

207 schema=altered_schema, 

208 engine="pyarrow", 

209 ) 

210 

211 if forecast_df is None: 

212 # If processing the first diagnostic, use it to create the base 

213 # DataFrame. 

214 if additional_df.empty: 

215 return None, None 

216 additional_df.rename(columns={"forecast": cf_name}, inplace=True) 

217 forecast_df = additional_df 

218 continue 

219 

220 # Convert additional features from rows to columns. 

221 representation = ( 

222 "percentile" if "percentile" in additional_df.columns else "realization" 

223 ) 

224 

225 if additional_df.empty: 

226 msg = ( 

227 "The requested parquet diagnostic name is not present in the " 

228 f"forecast parquet file: {parquet_diagnostic_name}." 

229 ) 

230 raise ValueError(msg) 

231 

232 merge_on = [ 

233 *self.unique_site_id_keys, 

234 "forecast_reference_time", 

235 "forecast_period", 

236 representation, 

237 ] 

238 

239 # If e.g. percentile is all NaN as this is a deterministic diagnostic, remove this column. 

240 if additional_df[representation].isna().all(): 

241 merge_on.remove(representation) 

242 

243 forecast_df = add_feature_from_df_to_df( 

244 forecast_df, 

245 additional_df.rename(columns={"forecast": cf_name}), 

246 cf_name, 

247 merge_on, 

248 ) 

249 

250 seconds_to_ns = 1e9 

251 forecast_df = forecast_df[ 

252 forecast_df["forecast_period"].isin( 

253 np.array(forecast_periods) * seconds_to_ns 

254 ) 

255 ].reset_index(drop=True) 

256 

257 # Convert df columns from ns to pandas timestamp object. 

258 for column in ["time", "forecast_reference_time", "blend_time"]: 

259 forecast_df[column] = pd.to_datetime( 

260 forecast_df[column], unit="ns", utc=True 

261 ) 

262 forecast_df[column] = forecast_df[column].astype("datetime64[ns, UTC]") 

263 for column in ["forecast_period", "period"]: 

264 forecast_df[column] = pd.to_timedelta(forecast_df[column], unit="ns") 

265 forecast_df[column] = forecast_df[column].astype("timedelta64[ns]") 

266 

267 # Load truths from parquet file filtering by diagnostic. 

268 filters = [[("diagnostic", "==", self.parquet_diagnostic_names[0])]] 

269 truth_df = pd.read_parquet( 

270 truth_table_path, 

271 filters=filters, 

272 schema=CalibrationSchemas().TRUTH_SCHEMA, 

273 engine="pyarrow", 

274 ) 

275 

276 truth_df["time"] = pd.to_datetime(truth_df["time"], unit="ns", utc=True) 

277 truth_df["time"] = truth_df["time"].astype("datetime64[ns, UTC]") 

278 

279 if truth_df.empty: 

280 msg = ( 

281 f"The requested filepath {truth_table_path} does not contain the " 

282 f"requested contents: {filters}" 

283 ) 

284 raise IOError(msg) 

285 return forecast_df, truth_df

287 def process( 

288 self, 

289 file_paths: list[pathlib.Path | str], 

290 ) -> Optional[tuple[iris.cube.CubeList, pathlib.Path | str, pathlib.Path | str]]: 

291 """Load input files for training a Quantile Regression Random Forest (QRF) 

292 model. Two sources of input data must be provided: historical forecasts and 

293 historical truth data (to use in calibration). 

294 

295 Args: 

296 file_paths (cli.inputpaths): 

297 A list of input paths containing: 

298 - The path to a Parquet file containing the truths to be used 

299 for calibration. The expected columns within the 

300 Parquet file are: ob_value, time, wmo_id, diagnostic, latitude, 

301 longitude and altitude. 

302 - The path to a Parquet file containing the forecasts to be used 

303 for calibration. The expected columns within the Parquet file are: 

304 forecast, blend_time, forecast_period, forecast_reference_time, time, 

305 wmo_id, percentile, diagnostic, latitude, longitude, period, height, 

306 cf_name, units. Please note that the presence of a forecast_period 

307 column is used to separate the forecast parquet file from the truth 

308 parquet file. 

309 - Optionally, paths to NetCDF files containing additional predictors. 

310 

311 Returns: 

312 Tuple containing: 

313 - DataFrame containing the forecast data. 

314 - DataFrame containing the truth data. 

315 - List of cubes containing additional features. 

316 

317 A tuple of (None, None, None) is returned if: 

318 - The quantile_forest package is not installed. 

319 - No parquet files are provided. 

320 - Either the forecast or truth parquet files are missing. 

321 

322 """ 

323 if not self.quantile_forest_installed: 

324 return None, None, None 

325 cube_inputs, parquets, _ = split_netcdf_parquet_pickle(file_paths) 

326 

327 # If there are no parquet files, return None. 

328 if not parquets: 

329 msg = "No parquet files have been provided." 

330 warnings.warn(msg) 

331 return None, None, None 

332 

333 forecast_table_path, truth_table_path = identify_parquet_type(parquets) 

334 

335 # If either the forecast or truth parquet files are missing, return None. 

336 if not forecast_table_path or not truth_table_path: 

337 msg = "Both forecast and truth parquet files must be provided." 

338 warnings.warn(msg) 

339 return None, None, None 

340 

341 forecast_periods = self._parse_forecast_periods() 

342 forecast_df, truth_df = self._read_parquet_files( 

343 forecast_table_path, truth_table_path, forecast_periods 

344 ) 

345 if forecast_df is None: 

346 msg = ( 

347 "The forecast parquet files are empty after filtering. " 

348 "Unable to train the Quantile Regression Random Forest model." 

349 ) 

350 warnings.warn(msg) 

351 return None, None, None 

352 

353 if cube_inputs is None: 

354 cube_inputs = iris.cube.CubeList() 

355 missing_features = ( 

356 set(self.feature_config.keys()) 

357 - set(forecast_df.columns) 

358 - set([c.name() for c in cube_inputs]) 

359 ) 

360 

361 if len(missing_features) > 0: 

362 msg = ( 

363 "The features requested in the feature_config are absent from " 

364 "the forecast parquet file and the input cubes. " 

365 f"The missing fields are: {missing_features}." 

366 ) 

367 raise ValueError(msg) 

368 return forecast_df, truth_df, cube_inputs

420 @staticmethod 

421 def _check_matching_times( 

422 forecast_df: pd.DataFrame, truth_df: pd.DataFrame 

423 ) -> list[pd.Timestamp]: 

424 """Find the intersecting times available within the forecast and truth 

425 DataFrames. 

426 

427 Args: 

428 forecast_df: DataFrame containing the forecast data. 

429 truth_df: DataFrame containing the truth data. 

430 Returns: 

431 List of intersecting times as pandas Timestamp objects. 

432 """ 

433 # Calling unique() on the time column is quicker than relying upon set() to 

434 # find the unique times. 

435 return list( 

436 set(forecast_df["time"].unique()).intersection( 

437 set(truth_df["time"].unique()) 

438 ) 

439 )

441 def _add_static_features_from_cubes_to_df( 

442 self, forecast_df: pd.DataFrame, cube_inputs: iris.cube.CubeList 

443 ) -> pd.DataFrame: 

444 """Add features to the forecast DataFrame from cubes based on the feature 

445 configuration. Other features are expected to already be present in the 

446 forecast DataFrame. 

447 

448 Args: 

449 forecast_df: DataFrame containing the forecast data. 

450 cube_inputs: List of cubes containing additional features. 

451 

452 Returns: 

453 DataFrame with additional features added from the input cubes. 

454 """ 

455 if cube_inputs is None or len(cube_inputs) == 0: 

456 return forecast_df 

457 

458 for feature_name, feature_list in self.feature_config.items(): 

459 for feature in feature_list: 

460 if feature == "static": 

461 # Use the cube's data directly as a feature. 

462 constr = iris.Constraint(name=feature_name) 

463 # Static features can be provided either as a cube or as a column 

464 # in the forecast DataFrame. 

465 try: 

466 feature_cube = cube_inputs.extract_cube(constr) 

467 except iris.exceptions.ConstraintMismatchError: 

468 feature_cube = None 

469 if not feature_cube: 

470 continue 

471 forecast_df = add_static_feature_from_cube_to_df( 

472 forecast_df, 

473 feature_cube, 

474 feature_name, 

475 [*self.unique_site_id_keys], 

476 float_decimals=self.float_decimals, 

477 ) 

478 return forecast_df

480 def filter_bad_sites( 

481 self, 

482 forecast_df: pd.DataFrame, 

483 truth_df: pd.DataFrame, 

484 ) -> tuple[pd.DataFrame, pd.DataFrame]: 

485 """Remove sites that have NaNs in the data. 

486 

487 Args: 

488 feature_df: DataFrame containing the forecast data with features. 

489 truth_df: DataFrame containing the truth data. 

490 

491 Returns: 

492 Tuple containing: 

493 - DataFrame containing the forecast data with bad sites removed. 

494 - DataFrame containing the truth data with bad sites removed. 

495 

496 Raises: 

497 ValueError: If the truth DataFrame is empty after removing NaNs. 

498 ValueError: If there are no matching sites and times between the 

499 forecast and truth DataFrames after removing NaNs. 

500 """ 

501 truth_df.dropna(subset=["ob_value"] + [*self.unique_site_id_keys], inplace=True) 

502 

503 if truth_df.empty: 

504 msg = "Empty truth DataFrame after removing NaNs." 

505 raise ValueError(msg) 

506 

507 # Include time in the index, so that forecasts will be dropped if they 

508 # correspond to a site and time that is not in the truth data. 

509 forecast_index = forecast_df.set_index( 

510 [*self.unique_site_id_keys] + ["time"] 

511 ).index 

512 truth_index = truth_df.set_index([*self.unique_site_id_keys] + ["time"]).index 

513 forecast_df = forecast_df[forecast_index.isin(truth_index)] 

514 truth_df = truth_df[truth_index.isin(forecast_index)] 

515 

516 if truth_df.empty: 

517 msg = ( 

518 "Empty truth DataFrame after finding the intersection of sites " 

519 "and times between the truth DataFrame and the forecast DataFrame." 

520 ) 

521 raise ValueError(msg) 

522 

523 return forecast_df, truth_df

525 def process( 

526 self, 

527 forecast_df: pd.DataFrame, 

528 truth_df: pd.DataFrame, 

529 cube_inputs: Optional[iris.cube.CubeList] = None, 

530 ) -> Optional[tuple[RandomForestQuantileRegressor, str, float]]: 

531 """Load input files and train a Quantile Regression Random Forest (QRF) 

532 model. This model can be applied later to calibrate the forecast. Two sources 

533 of input data must be provided: historical forecasts and historical truth data 

534 (to use in calibration). The model is output as a pickle file. 

535 

536 Args: 

537 forecast_df: DataFrame containing the forecast data. 

538 truth_df: DataFrame containing the truth data. 

539 cube_inputs: List of cubes containing additional features. 

540 

541 Returns: A tuple containing: 

542 - The trained RandomForestQuantileRegressor model. 

543 - The transformation applied to the data before fitting. 

544 - The value added before transformation. 

545 

546 Raises: 

547 ValueError: If there are no matching times between the forecast and truth 

548 data. 

549 """ 

550 if not self.quantile_forest_installed: 

551 return None, None, None 

552 

553 intersecting_times = self._check_matching_times(forecast_df, truth_df) 

554 if len(intersecting_times) == 0: 

555 msg = ( 

556 "No matching times between the forecast and truth data. " 

557 "Unable to train the Quantile Regression Random Forest model." 

558 ) 

559 warnings.warn(msg) 

560 return None, None, None 

561 

562 forecast_df = self._add_static_features_from_cubes_to_df( 

563 forecast_df, cube_inputs 

564 ) 

565 forecast_df, truth_df = self.filter_bad_sites(forecast_df, truth_df) 

566 

567 result = TrainQuantileRegressionRandomForests( 

568 target_name=self.target_cf_name, 

569 feature_config=self.feature_config, 

570 n_estimators=self.n_estimators, 

571 max_depth=self.max_depth, 

572 max_samples=self.max_samples, 

573 random_state=self.random_state, 

574 transformation=self.transformation, 

575 pre_transform_addition=self.pre_transform_addition, 

576 unique_site_id_keys=self.unique_site_id_keys, 

577 **self.kwargs, 

578 )(forecast_df, truth_df) 

579 

580 # Create a tuple that returns the model, transformation and 

581 # pre_transform_addition to allow these to be saved together. 

582 return (result, self.transformation, self.pre_transform_addition)
