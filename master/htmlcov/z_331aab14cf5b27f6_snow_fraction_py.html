22class SnowFraction(PostProcessingPlugin): 

23 """ 

24 Calculates a snow-fraction field from fields of snow and rain (rate or 

25 accumulation). Where no precipitation is present, the data are filled in from 

26 the nearest precipitating point. 

27 

28 snow_fraction = snow / (snow + rain) 

29 """ 

30 

31 def __init__(self, model_id_attr: Optional[str] = None) -> None: 

32 """ 

33 Initialise the class 

34 

35 Args: 

36 model_id_attr: 

37 Name of the attribute used to identify the source model for 

38 blending. 

39 """ 

40 self.model_id_attr = model_id_attr

42 def _get_input_cubes(self, input_cubes: CubeList) -> None: 

43 """ 

44 Separates out the rain and snow cubes from the input list and checks that 

45 * No other cubes are present 

46 * Cubes represent the same time quantity (instantaneous or accumulation length) 

47 * Cubes have compatible units 

48 * Cubes have same dimensions 

49 * Cubes are not masked (or are masked with an all-False mask) 

50 

51 Args: 

52 input_cubes: 

53 Contains exactly two cubes, one of rain and one of snow. Both must be 

54 either rates or accumulations of the same length and of compatible units. 

55 

56 Raises: 

57 ValueError: 

58 If any of the criteria above are not met. 

59 """ 

60 if len(input_cubes) != 2: 

61 raise ValueError( 

62 f"Expected exactly 2 input cubes, found {len(input_cubes)}" 

63 ) 

64 rain_name, snow_name = self._get_input_cube_names(input_cubes) 

65 self.rain = input_cubes.extract(rain_name).merge_cube() 

66 self.snow = input_cubes.extract(snow_name).merge_cube() 

67 self.snow.convert_units(self.rain.units) 

68 if not spatial_coords_match([self.rain, self.snow]): 

69 raise ValueError("Rain and snow cubes are not on the same grid") 

70 if not self.rain.coord("time") == self.snow.coord("time"): 

71 raise ValueError("Rain and snow cubes do not have the same time coord") 

72 if np.ma.is_masked(self.rain.data) or np.ma.is_masked(self.snow.data): 

73 raise ValueError("Unexpected masked data in input cube(s)") 

74 if isinstance(self.rain.data, np.ma.masked_array): 

75 self.rain.data = self.rain.data.data 

76 if isinstance(self.snow.data, np.ma.masked_array): 

77 self.snow.data = self.snow.data.data 

78 

79 @staticmethod 

80 def _get_input_cube_names(input_cubes: CubeList) -> Tuple[str, str]: 

81 """ 

82 Identifies the rain and snow cubes from the presence of "rain" or "snow" in 

83 the cube names. 

84 

85 Args: 

86 input_cubes: 

87 The unsorted rain and snow cubes. 

88 

89 Returns: 

90 rain_name and snow_name, in that order. 

91 """ 

92 cube_names = [cube.name() for cube in input_cubes] 

93 try: 

94 (rain_name,) = [x for x in cube_names if "rain" in x] 

95 (snow_name,) = [x for x in cube_names if "snow" in x] 

96 except ValueError: 

97 raise ValueError(f"Could not find both rain and snow in {cube_names}") 

98 if rain_name == snow_name: 

99 raise ValueError( 

100 f"Failed to find unique rain and snow cubes from {cube_names}" 

101 ) 

102 return rain_name, snow_name

104 def _calculate_snow_fraction(self) -> Cube: 

105 """ 

106 Calculates the snow fraction data and interpolates to fill in the missing points. 

107 

108 Returns: 

109 Snow fraction cube. 

110 """ 

111 with np.errstate(divide="ignore", invalid="ignore"): 

112 snow_fraction = self.snow.data / (self.rain.data + self.snow.data) 

113 snow_fraction_cube = create_new_diagnostic_cube( 

114 "snow_fraction", 

115 "1", 

116 template_cube=self.rain, 

117 mandatory_attributes=generate_mandatory_attributes( 

118 iris.cube.CubeList([self.rain, self.snow]), 

119 model_id_attr=self.model_id_attr, 

120 ), 

121 data=snow_fraction, 

122 ) 

123 

124 spatial_dims = [snow_fraction_cube.coord(axis=n).name() for n in ["y", "x"]] 

125 snow_fraction_interpolated = iris.cube.CubeList() 

126 for snow_fraction_slice in snow_fraction_cube.slices(spatial_dims): 

127 snow_fraction_interpolated.append( 

128 snow_fraction_slice.copy( 

129 interpolate_missing_data(snow_fraction_slice.data, method="nearest") 

130 ) 

131 ) 

132 return snow_fraction_interpolated.merge_cube() 

133 

134 def process(self, input_cubes: CubeList) -> Cube: 

135 """Check input cubes, then calculate and interpolate a snow fraction cube. 

136 

137 Args: 

138 input_cubes: 

139 Contains cubes of rain and snow, both must be either rates or accumulations. 

140 

141 Returns: 

142 Cube of snow-fraction. The data within this 

143 cube will contain values between 0 and 1. Points where no precipitation 

144 is present will be filled using a nearest-neighbour interpolation. 

145 

146 The cube meta-data will contain: 

147 * Input_cube name "snow_fraction" 

148 * Cube units set to (1). 

149 

150 Raises: 

151 ValueError: if input cubes fail any comparison tests. 

152 """ 

153 self._get_input_cubes(input_cubes) 

154 

155 return self._calculate_snow_fraction()
