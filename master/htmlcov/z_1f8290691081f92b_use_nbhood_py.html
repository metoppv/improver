23class ApplyNeighbourhoodProcessingWithAMask(PostProcessingPlugin): 

24 r"""Class for applying neighbourhood processing when passing in a mask 

25 cube that is iterated over. 

26 

27 Example: 

28 

29 This plugin is designed to work with a set of masks which help you 

30 select points which are similar and only use these in your 

31 neighbourhood. The most obvious example of this is to divide the 

32 points in your cube into bands of similar orographic height. 

33 :: 

34 

35 .............................. 

36 Band 2 --- 

37 ............./...\.../\....... 

38 Band 1 / --- \ 

39 .........../............\..... 

40 Band 0 / -- 

41 ........--.................\.. 

42 

43 In this case the mask cube that comes in has a "topographic_zone" 

44 coordinate and each slice along this dimension has a 2D mask, 

45 masking out any points which are outside the topographic band that is 

46 described by the "topographic_zone" coordinate. 

47 

48 The result from this plugin is a cube which has applied 

49 neighbourhooding to the plugin *n* times for the *n* bands in the mask 

50 cube. Each topography mask has been applied to the input cube in turn, 

51 resulting in a cube with a "topographic_zone" coordinate which is 

52 returned from this plugin. 

53 

54 However, if weights are provided then you can weight between 

55 adjacent bands when you collapse the new "topographic_zone" coordinate. 

56 This takes into account the result from the neighbourhood processing to 

57 adjust the weights for points in a "topographic_zone" that don't have 

58 a valid result. 

59 

60 For example below we have two points A and B. Say point A was halfway 

61 between the midpoint and top of the lower band. We would want to 

62 generate a final result by weighting 0.75 times to neighbourhooded 

63 value from the bottom band and 0.25 times the neighbourhooded value in 

64 the upper band. For point B we would take equal weightings between the 

65 bands. There is a plugin to generate these weights: 

66 :class:`~improver.generate_ancillaries.\ 

67 generate_topographic_zone_weights.GenerateTopographicZoneWeights` 

68 :: 

69 

70 A B 

71 .......................... 

72 band 2 

73 

74 ..................x....... 

75 x 

76 band 1 

77 .......................... 

78 

79 We may need to adjust the weights if there is missing data in the 

80 adjacent band. If we look at the diagram with labelled bands, points 

81 that are near the top of band 2 could be weighted with band 3, except 

82 there are no nearby points in band 3. In this case the neighbourhood 

83 code puts NaNs in band 3 and we want to take 100% of band 2. This can 

84 be easily done by renormalization of the weights, which happens 

85 automatically within the numpy functions called within the 

86 iris collapse method. 

87 

88 After collapsing the "topographic_zone" coordinate we end up with a 

89 cube with the same dimensions as the original cube, but the neighbourhood 

90 processing has been applied using masks so that only similar points 

91 are used in the neighbourhood. 

92 

93 See also :class:`~improver.generate_ancillaries.generate_ancillary.\ 

94 GenerateOrographyBandAncils` 

95 for a plugin for generating topographic band masks. 

96 

97 """ 

98 

99 def __init__( 

100 self, 

101 coord_for_masking: str, 

102 neighbourhood_method: str, 

103 radii: Union[float, List[float]], 

104 lead_times: Optional[List[float]] = None, 

105 collapse_weights: Optional[Cube] = None, 

106 weighted_mode: bool = False, 

107 sum_only: bool = False, 

108 ) -> None: 

109 """ 

110 Initialise the class. 

111 

112 Args: 

113 coord_for_masking: 

114 String matching the name of the coordinate that will be used 

115 for masking. 

116 neighbourhood_method: 

117 Name of the neighbourhood method to use. Options: 'circular', 

118 'square'. 

119 radii: 

120 The radii in metres of the neighbourhood to apply. 

121 Rounded up to convert into integer number of grid 

122 points east and north, based on the characteristic spacing 

123 at the zero indices of the cube projection-x and y coords. 

124 collapse_weights: 

125 A cube from an ancillary file containing the weights for each 

126 point in the 'coord_for_masking' at each grid point. If given, 

127 the coord_for_masking coordinate will be collapsed using these 

128 weights resulting in an output cube with the same dimensions 

129 as the input cube. Otherwise the output cube will have the 

130 coord_for_masking from the supplied mask_cube as an additional 

131 dimension. The data in this cube may be an instance of 

132 numpy.ma.masked_array, for example if sea points have been 

133 set to np.nan and masked in order for them to be discounted 

134 when calculating the result. In this case the result returned 

135 from the process method of this plugin will have the same 

136 points masked. 

137 lead_times: 

138 List of lead times or forecast periods, at which the radii 

139 within 'radii' are defined. The lead times are expected 

140 in hours. 

141 weighted_mode: 

142 If True, use a circle for neighbourhood kernel with 

143 weighting decreasing with radius. 

144 If False, use a circle with constant weighting. 

145 sum_only: 

146 If true, return neighbourhood sum instead of mean. 

147 

148 """ 

149 self.coord_for_masking = coord_for_masking 

150 self.neighbourhood_method = neighbourhood_method 

151 self.radii = radii 

152 self.lead_times = lead_times 

153 self.collapse_weights = collapse_weights 

154 self.weighted_mode = weighted_mode 

155 self.sum_only = sum_only 

156 self.re_mask = False 

157 

158 def collapse_mask_coord(self, cube: Cube) -> Cube: 

159 """ 

160 Collapse the chosen coordinate with the available weights. The result 

161 of the neighbourhood processing is taken into account to renormalize 

162 any weights corresponding to a NaN in the result from neighbourhooding. 

163 In this case the weights are re-normalized so that we do not lose 

164 probability. 

165 

166 Args: 

167 cube: 

168 Cube containing the array to which the square neighbourhood 

169 with a mask has been applied. 

170 Dimensions self.coord_for_masking, y and x. 

171 

172 Returns: 

173 Cube containing the weighted mean from neighbourhood after 

174 collapsing the chosen coordinate. 

175 """ 

176 # Mask out any NaNs in the neighbourhood data so that Iris ignores 

177 # them when calculating the weighted mean. 

178 cube.data = ma.masked_invalid(cube.data, copy=False) 

179 # Collapse the coord_for_masking. Renormalization of the weights happen 

180 # within the underlying call to a numpy function within the Iris method. 

181 result = collapsed( 

182 cube, 

183 self.coord_for_masking, 

184 iris.analysis.MEAN, 

185 weights=self.collapse_weights.data, 

186 ) 

187 # Set masked invalid data points back to np.nans 

188 if np.ma.is_masked(result.data): 

189 result.data.data[result.data.mask] = np.nan 

190 # Remove references to self.coord_masked in the result cube. 

191 result.remove_coord(self.coord_for_masking) 

192 return result 

193 

194 def process(self, cube: Cube, mask_cube: Cube) -> Cube: 

195 """ 

196 Apply neighbourhood processing with a mask to the input cube, 

197 collapsing the coord_for_masking if collapse_weights have been provided. 

198 

199 Args: 

200 cube: 

201 Cube containing the array to which the square neighbourhood 

202 will be applied. 

203 mask_cube: 

204 Cube containing the array to be used as a mask. The data in 

205 this array is not an instance of numpy.ma.MaskedArray. Any sea 

206 points that should be ignored are set to zeros in every layer 

207 of the mask_cube. 

208 

209 Returns: 

210 Cube containing the smoothed field after the square 

211 neighbourhood method has been applied when applying masking 

212 for each point along the coord_for_masking coordinate. 

213 The resulting cube is concatenated so that the dimension 

214 coordinates match the input cube. 

215 """ 

216 plugin = NeighbourhoodProcessing( 

217 self.neighbourhood_method, 

218 self.radii, 

219 lead_times=self.lead_times, 

220 weighted_mode=self.weighted_mode, 

221 sum_only=self.sum_only, 

222 re_mask=self.re_mask, 

223 ) 

224 yname = cube.coord(axis="y").name() 

225 xname = cube.coord(axis="x").name() 

226 result_slices = iris.cube.CubeList([]) 

227 # Take 2D slices of the input cube for memory issues. 

228 prev_x_y_slice = None 

229 for x_y_slice in cube.slices([yname, xname]): 

230 if prev_x_y_slice is not None and np.array_equal( 

231 prev_x_y_slice.data, x_y_slice.data 

232 ): 

233 # Use same result as last time! 

234 prev_result = result_slices[-1].copy() 

235 for coord in x_y_slice.coords(dim_coords=False): 

236 prev_result.coord(coord).points = coord.points.copy() 

237 result_slices.append(prev_result) 

238 continue 

239 prev_x_y_slice = x_y_slice 

240 

241 cube_slices = iris.cube.CubeList([]) 

242 # Apply each mask in mask_cube to the 2D input slice. 

243 for mask_slice in mask_cube.slices_over(self.coord_for_masking): 

244 output_cube = plugin(x_y_slice, mask_cube=mask_slice) 

245 coord_object = mask_slice.coord(self.coord_for_masking).copy() 

246 output_cube.add_aux_coord(coord_object) 

247 output_cube = iris.util.new_axis(output_cube, self.coord_for_masking) 

248 cube_slices.append(output_cube) 

249 concatenated_cube = cube_slices.concatenate_cube() 

250 if self.collapse_weights is not None: 

251 concatenated_cube = self.collapse_mask_coord(concatenated_cube) 

252 result_slices.append(concatenated_cube) 

253 result = result_slices.merge_cube() 

254 # Promote any single value dimension coordinates if they were 

255 # dimension on the input cube. 

256 exception_coordinates = find_dimension_coordinate_mismatch( 

257 cube, result, two_way_mismatch=False 

258 ) 

259 result = check_cube_coordinates( 

260 cube, result, exception_coordinates=exception_coordinates 

261 ) 

262 return result
