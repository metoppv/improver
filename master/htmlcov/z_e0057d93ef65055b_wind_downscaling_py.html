34class FrictionVelocity(BasePlugin): 

35 """Class to calculate the friction velocity. 

36 

37 This holds the function to calculate the friction velocity u_star, 

38 given a reference height h_ref, the velocity at the reference 

39 height u_href and the surface roughness z_0. 

40 

41 """ 

42 

43 def __init__( 

44 self, u_href: ndarray, h_ref: ndarray, z_0: ndarray, mask: ndarray 

45 ) -> None: 

46 """Initialise the class. 

47 

48 Args: 

49 u_href: 

50 A 2D array of float32 for the wind speed at h_ref 

51 h_ref: 

52 A 2D array of float32 for the reference heights 

53 z_0: 

54 A 2D array of float32 for the vegetative roughness lengths 

55 mask: 

56 A 2D array of booleans where True indicates calculate u* 

57 

58 Notes: 

59 * z_0 and h_ref need to have identical units. 

60 * the calculated friction velocity will have the units of the 

61 supplied velocity u_href. 

62 """ 

63 self.u_href = u_href 

64 self.h_ref = h_ref 

65 self.z_0 = z_0 

66 self.mask = mask 

67 

68 # Check that input cubes are the same size 

69 array_sizes = [np.size(u_href), np.size(h_ref), np.size(z_0), np.size(mask)] 

70 

71 if not all(x == array_sizes[0] for x in array_sizes): 

72 raise ValueError("Different size input arrays u_href, h_ref, z_0, mask") 

73 

74 def process(self) -> ndarray: 

75 """Function to calculate the friction velocity. 

76 

77 ustar = K * (u_href / ln(h_ref / z_0)) 

78 

79 where ustar is the friction velocity, K is Von Karman's 

80 constant, u_ref is the wind speed at the reference height, 

81 h_ref is the reference height and z_0 is the vegetative 

82 roughness length. 

83 

84 Returns: 

85 A 2D array of float32 friction velocities 

86 """ 

87 ustar = np.full(self.u_href.shape, RMDI, dtype=np.float32) 

88 numerator = self.u_href[self.mask] 

89 with np.errstate(invalid="ignore"): 

90 denominator = np.log(self.h_ref[self.mask] / self.z_0[self.mask]) 

91 ustar[self.mask] = VONKARMAN * (numerator / denominator) 

92 return ustar

95class RoughnessCorrectionUtilities: 

96 """Class to calculate the height and roughness wind corrections. 

97 

98 This holds functions to calculate the roughness and height 

99 corrections given the ancil files: 

100 

101 * standard deviation of height in grid cell as sigma (model grid on 

102 pp grid) 

103 * Silhouette roughness as a_over_s (model grid on pp grid) 

104 * vegetative roughness length z_0 (model grid on pp grid) 

105 * post-processing grid orography pporo 

106 * model grid orography interpolated on post-processing grid modoro 

107 * height level 3D/ 1D grid 

108 * windspeed 3D field on height level 3D grid (from above). 

109 """ 

110 

111 def __init__( 

112 self, 

113 a_over_s: ndarray, 

114 sigma: ndarray, 

115 z_0: ndarray, 

116 pporo: ndarray, 

117 modoro: ndarray, 

118 ppres: float, 

119 modres: float, 

120 ) -> None: 

121 """Set up roughness and height correction. 

122 

123 This sets up the parameters used for roughness and height 

124 correction given the ancillary file inputs: 

125 

126 Args: 

127 a_over_s: 

128 2D array float32 - Silhouette roughness field, dimensionless 

129 ancillary data, calculated according to Robinson, D. (2008) 

130 - Ancillary file creation for the UM, Unified Model 

131 Documentation Paper 73. 

132 sigma: 

133 2D array float32 - Standard deviation field of height in the 

134 grid cell, units of length 

135 z_0: 

136 2D array float32 - Vegetative roughness height field, 

137 units of length 

138 pporo: 

139 2D array float32 - Post processing grid orography field 

140 modoro: 

141 2D array float32 - Model orography field interpolated to post 

142 processing grid 

143 ppres: 

144 Float - Grid cell length of post processing grid 

145 modres: 

146 Float - Grid cell length of model grid 

147 """ 

148 self.a_over_s = a_over_s 

149 self.z_0 = z_0 

150 self.pporo = pporo 

151 self.modoro = modoro 

152 self.h_over_2 = self.sigma2hover2(sigma) # half peak to trough height 

153 self.hcmask, self.rcmask = self._setmask() # HC mask, RC mask 

154 if self.z_0 is not None: 

155 self.z_0[z_0 <= 0] = Z0M_SEA 

156 self.dx_min = ppres / 2.0 # scales smaller than this not resolved in pp 

157 # the original code had hardcoded 500 

158 self.dx_max = 3.0 * modres # scales larger than this resolved in model 

159 # the original code had hardcoded 4000 

160 self.wavenum = self._calc_wav() # k = 2*pi / L 

161 self.h_ref = self._calc_h_ref() 

162 self._refinemask() # HC mask needs to be updated for missing orography 

163 self.h_at0 = self._delta_height() # pp orography - model orography

165 def _refinemask(self) -> None: 

166 """Remask over RMDI and NaN orography. 

167 

168 The mask for HC needs to be False where either of the 

169 orographies (model or pp) has an invalid number. This cannot be 

170 done before because the mask is used to calculate the 

171 wavenumber which can and should be calculated for all points 

172 where h_over_2 and a_over_s is a valid number. 

173 """ 

174 self.hcmask[self.pporo == RMDI] = False 

175 self.hcmask[self.modoro == RMDI] = False 

176 self.hcmask[np.isnan(self.pporo)] = False 

177 self.hcmask[np.isnan(self.modoro)] = False

179 def _setmask(self) -> Tuple[ndarray, ndarray]: 

180 """Create a ~land-sea mask. 

181 

182 Create a mask that is basically a land-sea mask: 

183 Both, the standard deviation and the silouette roughness, are 0 

184 over the sea. A standard deviation of 0 results in a RMDI for 

185 h_over_2. 

186 

187 Returns: 

188 - 2D array of booleans- True for land-points, 

189 false for Sea (HC) 

190 - 2D array of booleans- additionally False for 

191 invalid z_0 (RC) 

192 """ 

193 hcmask = np.full(self.h_over_2.shape, True, dtype=bool) 

194 hcmask[self.h_over_2 <= 0] = False 

195 hcmask[self.a_over_s <= 0] = False 

196 hcmask[np.isnan(self.h_over_2)] = False 

197 hcmask[np.isnan(self.a_over_s)] = False 

198 rcmask = np.copy(hcmask) 

199 if self.z_0 is not None: 

200 rcmask[self.z_0 <= 0] = False 

201 rcmask[np.isnan(self.z_0)] = False 

202 return hcmask, rcmask

204 @staticmethod 

205 def sigma2hover2(sigma: ndarray) -> ndarray: 

206 """Calculate the half peak-to-trough height. 

207 

208 The ancillary data used to estimate the peak to trough height 

209 contains the standard deviation of height in a cell. For 

210 sine-waves, this relates to the amplitude of the wave as: 

211 

212 Amplitude = sigma * sqrt(2) 

213 

214 The amplitude would correspond to half the peak-to-trough 

215 height (h_o_2). 

216 

217 Args: 

218 sigma: 

219 2D array of float32 - standard deviation of height in 

220 grid cell. 

221 

222 Returns: 

223 2D array of float32 - half peak-to-trough height. 

224 

225 Comments: 

226 Points that had sigma = 0 (i.e. sea points) are set to 

227 RMDI. 

228 """ 

229 h_o_2 = np.full(sigma.shape, RMDI, dtype=np.float32) 

230 h_o_2[sigma > 0] = sigma[sigma > 0] * np.sqrt(2.0) 

231 return h_o_2

233 def _calc_wav(self) -> ndarray: 

234 """Calculate wavenumber k of typical orographic lengthscale. 

235 

236 Function to calculate wavenumber k of typical orographic 

237 lengthscale L: 

238 

239 .. math:: 

240 :label: 

241 

242 k = 2 * \\pi / L 

243 

244 L is approximated from half the peak-to-trough height h_over_2 

245 and the silhoutte roughness a_over_s (average of up-slopes per 

246 unit length over several cross-sections through a grid cell) 

247 as: 

248 

249 .. math:: 

250 :label: 

251 

252 L = 2 * \\rm{h\\_over\\_2} / \\rm{a\\_over\\_s} 

253 

254 a_over_s is dimensionless since it is the sum of up-slopes 

255 measured in the same unit lengths as it is calculated over. 

256 

257 h_over_2 is calculated from the standard deviation of height in 

258 a grid cell, sigma, as: 

259 

260 .. math:: 

261 :label: 

262 

263 \\rm{h\\_over\\_2} = \\sqrt{2} * \\rm{sigma} 

264 

265 which is based on the assumptions of sine waves, see 

266 sigma2hover2. 

267 

268 From eq. (1) and (2) it follows that: 

269 

270 .. math:: 

271 :label: 

272 

273 k = 2*\\pi / (2 * \\rm{h\\_over\\_2} / \\rm{a\\_over\\_s)} 

274 = \\rm{a\\_over\\_s} * \\pi / \\rm{h\\_over\\_2} 

275 

276 Returns: 

277 2D array float32 - wavenumber in units of inverse units of 

278 supplied h_over_2. 

279 """ 

280 wavn = np.full(self.a_over_s.shape, RMDI, dtype=np.float32) 

281 wavn[self.hcmask] = (self.a_over_s[self.hcmask] * np.pi) / self.h_over_2[ 

282 self.hcmask 

283 ] 

284 wavn[wavn > np.pi / self.dx_min] = np.pi / self.dx_min 

285 wavn[self.h_over_2 == 0] = RMDI 

286 wavn[abs(wavn) < np.pi / self.dx_max] = np.pi / self.dx_max 

287 return wavn

289 def _calc_h_ref(self) -> ndarray: 

290 """Calculate the reference height for roughness correction. 

291 

292 The reference height below which the flow is in equilibrium 

293 with the vegetative roughness is proportional to 1/wavenum 

294 (Howard & Clark, 2007). 

295 

296 Vosper (2009) and Clark (2009) argue that at the reference 

297 height, the perturbation should have decayed to a fraction 

298 epsilon (ABSOLUTE_CORRECTION_TOL). The factor alpha 

299 implements eq. 1.3 in Clark (2009): UK Climatology - Wind 

300 Screening Tool. See also Vosper (2009) for a motivation. 

301 For a freely available external reference, see the Virtual Met 

302 Mast Version 1 Methodology and Verification paper under 

303 www.thecrownestate.co.uk. 

304 

305 alpha is the log of scale parameter to determine reference 

306 height which is currently set to 0.04 (this corresponds to 

307 epsilon in both Vosper and Clark) 

308 

309 Returns: 

310 2D array float32 - reference height for roughness correction 

311 """ 

312 alpha = -np.log(ABSOLUTE_CORRECTION_TOL) 

313 tunable_param = np.full(self.wavenum.shape, RMDI, dtype=np.float32) 

314 h_ref = np.full(self.wavenum.shape, RMDI, dtype=np.float32) 

315 tunable_param[self.hcmask] = alpha + np.log( 

316 self.wavenum[self.hcmask] * self.h_over_2[self.hcmask] 

317 ) 

318 tunable_param[tunable_param > 1.0] = 1.0 

319 tunable_param[tunable_param < 0.0] = 0.0 

320 h_ref[self.hcmask] = tunable_param[self.hcmask] / self.wavenum[self.hcmask] 

321 h_ref[h_ref < 1.0] = 1.0 

322 h_ref = np.minimum(h_ref, HREF_SCALE * self.h_over_2) 

323 h_ref[h_ref < 1.0] = 1.0 

324 h_ref[~self.hcmask] = 0.0 

325 return h_ref

327 def calc_roughness_correction( 

328 self, hgrid: ndarray, uold: ndarray, mask: ndarray 

329 ) -> ndarray: 

330 """Function to perform the roughness correction. 

331 

332 Args: 

333 hgrid: 

334 3D or 1D array float32 - height above orography 

335 uold: 

336 3D array float32 - original velocities at hgrid. 

337 mask: 

338 2D array of bools that is True for land-points, False for Sea 

339 and False for invalid z_0. 

340 

341 Returns: 

342 3D np.array float32 - Corrected wind speed on hgrid. Above 

343 href, this is equal to uold. 

344 

345 Comments: 

346 Replace the windspeed profile below the reference height with one 

347 that increases logarithmically with height, bound by the original 

348 velocity uhref at the reference height h_ref and by a 0 velocity at 

349 the vegetative roughness height z_0 

350 """ 

351 uhref = self._calc_u_at_h(uold, hgrid, self.h_ref, mask) 

352 if hgrid.ndim == 1: 

353 hgrid = hgrid[np.newaxis, np.newaxis, :] 

354 ustar = FrictionVelocity(uhref, self.h_ref, self.z_0, mask)() 

355 unew = np.copy(uold) 

356 mhref = self.h_ref 

357 mhref[~mask] = RMDI 

358 cond = hgrid < self.h_ref[:, :, np.newaxis] 

359 

360 # Create array of ones. 

361 arr_ones = np.ones(unew.shape, dtype=np.float32) 

362 

363 first_arg = (ustar[:, :, np.newaxis] * arr_ones)[cond] 

364 sec_arg = np.log( 

365 hgrid / (np.reshape(self.z_0, self.z_0.shape + (1,)) * arr_ones) 

366 )[cond] 

367 

368 unew[cond] = (first_arg * sec_arg) / VONKARMAN 

369 

370 return unew

372 def _calc_u_at_h( 

373 self, 

374 u_in: ndarray, 

375 h_in: ndarray, 

376 hhere: ndarray, 

377 mask: ndarray, 

378 dolog: bool = False, 

379 ) -> ndarray: 

380 """Function to interpolate u_in on h_in at hhere. 

381 

382 Args: 

383 u_in: 

384 3D array float32 - velocity on h_in layer, last dim is height 

385 h_in: 

386 3D or 1D array float32 - height layer array 

387 hhere: 

388 2D array float32 - height grid to interpolate at 

389 mask: 

390 2D array of bools - mask the final result for uath 

391 dolog: 

392 if True, log interpolation, default False 

393 

394 Returns: 

395 2D array float32 - velocity interpolated at h 

396 """ 

397 u_in = np.ma.masked_less(u_in, 0.0) 

398 h_in = np.ma.masked_less(h_in, 0.0) 

399 # h_in.mask = u_in.mask 

400 # If I allow 1D height grids, I think I cannot do the hop over. 

401 

402 # Ignores the height at the position where u_in is RMDI,"hops over" 

403 hhere = np.ma.masked_less(hhere, 0.0) 

404 upidx = np.argmax(h_in > hhere[:, :, np.newaxis], axis=2) 

405 # loidx = np.maximum(upidx-1, 0) #if RMDI, need below 

406 loidx = np.argmin( 

407 np.ma.masked_less(hhere[:, :, np.newaxis] - h_in, 0.0), axis=2 

408 ) 

409 

410 if h_in.ndim == 3: 

411 hup = h_in.take( 

412 upidx.flatten() 

413 + np.arange(0, upidx.size * h_in.shape[2], h_in.shape[2]) 

414 ) 

415 hlow = h_in.take( 

416 loidx.flatten() 

417 + np.arange(0, loidx.size * h_in.shape[2], h_in.shape[2]) 

418 ) 

419 elif h_in.ndim == 1: 

420 hup = h_in[upidx].flatten() 

421 hlow = h_in[loidx].flatten() 

422 uup = u_in.take( 

423 upidx.flatten() + np.arange(0, upidx.size * u_in.shape[2], u_in.shape[2]) 

424 ) 

425 ulow = u_in.take( 

426 loidx.flatten() + np.arange(0, loidx.size * u_in.shape[2], u_in.shape[2]) 

427 ) 

428 mask = mask.flatten() 

429 uath = np.full(mask.shape, RMDI, dtype=np.float32) 

430 if dolog: 

431 uath[mask] = self._interpolate_log( 

432 hup[mask], hlow[mask], hhere.flatten()[mask], uup[mask], ulow[mask] 

433 ) 

434 else: 

435 uath[mask] = self._interpolate_1d( 

436 hup[mask], hlow[mask], hhere.flatten()[mask], uup[mask], ulow[mask] 

437 ) 

438 uath = np.reshape(uath, hhere.shape) 

439 return uath

441 @staticmethod 

442 def _interpolate_1d( 

443 xup: ndarray, xlow: ndarray, at_x: ndarray, yup: ndarray, ylow: ndarray 

444 ) -> ndarray: 

445 """Simple 1D linear interpolation for 2D grid inputs level. 

446 

447 Args: 

448 xup: 

449 2D array float32 - upper x-bins 

450 xlow: 

451 2D array float32 - lower x-bins 

452 at_x: 

453 2D array float32 - x values to interpolate y at 

454 yup: 

455 2D array float32 - y(xup) 

456 ylow: 

457 2D array float32 - y(xlow) 

458 

459 Returns: 

460 2D array float32 - y(at_x) assuming a lin function 

461 between xlow and xup 

462 """ 

463 interp = np.full(xup.shape, RMDI, dtype=np.float32) 

464 diffs = xup - xlow 

465 interp[diffs != 0] = ylow[diffs != 0] + ( 

466 (at_x[diffs != 0] - xlow[diffs != 0]) 

467 / diffs[diffs != 0] 

468 * (yup[diffs != 0] - ylow[diffs != 0]) 

469 ) 

470 interp[diffs == 0] = at_x[diffs == 0] / xup[diffs == 0] * (yup[diffs == 0]) 

471 return interp 

472 

473 @staticmethod 

474 def _interpolate_log( 

475 xup: ndarray, xlow: ndarray, at_x: ndarray, yup: ndarray, ylow: ndarray 

476 ) -> ndarray: 

477 """Simple 1D log interpolation y(x), except if lowest layer is 

478 ground level. 

479 

480 Args: 

481 xup: 

482 2D array float32 - upper x-bins 

483 xlow: 

484 2D array float32 - lower x-bins 

485 at_x: 

486 2D array float32 - x values to interpolate y at 

487 yup: 

488 2D array float32 - y(xup) 

489 ylow: 

490 2D array float32 -y(xlow) 

491 

492 Returns: 

493 2D array float32 - y(at_x) assuming a log function 

494 between xlow and xup 

495 """ 

496 ain = np.full(xup.shape, RMDI, dtype=np.float32) 

497 loginterp = np.full(xup.shape, RMDI, dtype=np.float32) 

498 mfrac = xup / xlow 

499 mtest = (xup / xlow != 1) & (at_x != xup) 

500 ain[mtest] = (yup[mtest] - ylow[mtest]) / np.log(mfrac[mtest]) 

501 loginterp[mtest] = ain[mtest] * np.log(at_x[mtest] / xup[mtest]) + yup[mtest] 

502 mtest = xup / xlow == 1 # below lowest layer, make lin interp 

503 loginterp[mtest] = at_x[mtest] / xup[mtest] * (yup[mtest]) 

504 mtest = at_x == xup # just use yup 

505 loginterp[mtest] = yup[mtest] 

506 return loginterp

508 def _calc_height_corr( 

509 self, 

510 u_a: ndarray, 

511 heightg: ndarray, 

512 mask: ndarray, 

513 onemfrac: Union[float, ndarray], 

514 ) -> ndarray: 

515 """Function to calculate the additive height correction. 

516 

517 Args: 

518 u_a: 

519 2D array float32 - outer velocity, e.g. velocity at h_ref_orig 

520 heightg: 

521 1D or 3D array float32 - heights above orography 

522 mask: 

523 3D array of bools - Masks the hc_add result 

524 onemfrac: 

525 Currently, scalar = 1. But can be a function of position and 

526 height, e.g. a 3D array (float32) 

527 

528 Returns: 

529 3D array float32 - additive height correction to wind speed 

530 

531 Comments: 

532 The height correction is a disturbance of the flow that 

533 decays exponentially with height. The larger the vertical 

534 offset h_at0 (the higher the unresolved hill), the larger 

535 is the disturbance. 

536 

537 The more smooth the disturbance (the larger the horizontal 

538 scale of the disturbance), the smaller the height 

539 correction (hence, a larger wavenumber results in a larger 

540 disturbance). 

541 hc_add = exp(-height*wavenumber)*u(href)*h_at_0*wavenumber 

542 

543 A final factor of 1 is assumed and omitted for the Bessel 

544 function term. 

545 """ 

546 (xdim, ydim) = u_a.shape 

547 if heightg.ndim == 1: 

548 zdim = heightg.shape[0] 

549 heightg = heightg[np.newaxis, np.newaxis, :] 

550 elif heightg.ndim == 3: 

551 zdim = heightg.shape[2] 

552 ml2 = self.h_at0 * self.wavenum 

553 expon = np.ones([xdim, ydim, zdim], dtype=np.float32) 

554 mult = self.wavenum[:, :, np.newaxis] * heightg 

555 expon[mult > 0.0001] = np.exp(-mult[mult > 0.0001]) 

556 hc_add = expon * u_a[:, :, np.newaxis] * ml2[:, :, np.newaxis] * onemfrac 

557 hc_add[~mask, :] = 0 

558 return hc_add

573 def do_rc_hc_all(self, hgrid: ndarray, uorig: ndarray) -> ndarray: 

574 """Function to call HC and RC (height and roughness corrections). 

575 

576 Args: 

577 hgrid: 

578 1D or 3D array float32 - height grid of wind input 

579 uorig: 

580 3D array float32 - wind speed on these levels 

581 

582 Returns: 

583 - RC corrected windspeed 

584 - HC additional part 

585 

586 Friedrich, M. M., 2016 

587 Wind Downscaling Program (Internal Met Office Report) 

588 """ 

589 if hgrid.ndim == 3: 

590 condition1 = (hgrid == RMDI).any(axis=2) 

591 self.hcmask[condition1] = False 

592 self.rcmask[condition1] = False 

593 mask_rc = np.copy(self.rcmask) 

594 mask_rc[(uorig == RMDI).any(axis=2)] = False 

595 mask_hc = np.copy(self.hcmask) 

596 mask_hc[(uorig == RMDI).any(axis=2)] = False 

597 if self.z_0 is not None: 

598 unew = self.calc_roughness_correction(hgrid, uorig, mask_rc) 

599 else: 

600 unew = uorig 

601 uhref_orig = self._calc_u_at_h(uorig, hgrid, 1.0 / self.wavenum, mask_hc) 

602 mask_hc[uhref_orig <= 0] = False 

603 # Setting this value to 1, is equivalent to setting the 

604 # Bessel function to 1. (Friedrich, 2016) 

605 # Example usage if the Bessel function was not set to 1 is: 

606 # onemfrac = 1.0 - BfuncFrac(nx,ny,nz,heightvec,z_0,waveno, Ustar, UI) 

607 onemfrac = 1.0 

608 hc_add = self._calc_height_corr(uhref_orig, hgrid, mask_hc, onemfrac) 

609 result = unew + hc_add 

610 result[result < 0.0] = 0 # HC can be negative if pporo<modeloro 

611 return result.astype(np.float32)

681 def find_coord_names(self, cube: Cube) -> Tuple[str, str, str, str]: 

682 """Extract x, y, z, and time coordinate names. 

683 

684 Args: 

685 cube: 

686 some iris cube to find coordinate names from 

687 

688 Returns: 

689 - name of the axis name in x-direction 

690 - name of the axis name in y-direction 

691 - name of the axis name in z-direction 

692 - name of the axis name in t-direction 

693 """ 

694 clist = {cube.coords()[i].name() for i in range(len(cube.coords()))} 

695 try: 

696 xname = cube.coord(axis="x").name() 

697 except CoordinateNotFoundError as exc: 

698 print("'{0}' while xname setting. Args: {1}.".format(exc, exc.args)) 

699 try: 

700 yname = cube.coord(axis="y").name() 

701 except CoordinateNotFoundError as exc: 

702 print("'{0}' while yname setting. Args: {1}.".format(exc, exc.args)) 

703 if clist.intersection(self.zcoordnames): 

704 zname = list(clist.intersection(self.zcoordnames))[0] 

705 else: 

706 zname = None 

707 

708 if clist.intersection(self.tcoordnames): 

709 tname = list(clist.intersection(self.tcoordnames))[0] 

710 else: 

711 tname = None 

712 return xname, yname, zname, tname

714 def calc_av_ppgrid_res(self, a_cube: Cube) -> float: 

715 """Calculate average grid resolution from a cube. 

716 

717 Args: 

718 a_cube: 

719 Cube to calculate average resolution of 

720 

721 Returns: 

722 Average grid resolution. 

723 """ 

724 x_name, y_name, _, _ = self.find_coord_names(a_cube) 

725 [exp_xname, exp_yname] = ["projection_x_coordinate", "projection_y_coordinate"] 

726 exp_unit = Unit("m") 

727 if (x_name != exp_xname) or (y_name != exp_yname): 

728 raise ValueError("cannot currently calculate resolution") 

729 

730 if a_cube.coord(x_name).bounds is None and a_cube.coord(y_name).bounds is None: 

731 xres = (np.diff(a_cube.coord(x_name).points)).mean() 

732 yres = (np.diff(a_cube.coord(y_name).points)).mean() 

733 else: 

734 xres = (np.diff(a_cube.coord(x_name).bounds)).mean() 

735 yres = (np.diff(a_cube.coord(y_name).bounds)).mean() 

736 if (a_cube.coord(x_name).units != exp_unit) or ( 

737 a_cube.coord(y_name).units != exp_unit 

738 ): 

739 raise ValueError("cube axis have units different from m.") 

740 return (abs(xres) + abs(yres)) / 2.0

743 def check_ancils( 

744 a_over_s_cube: Cube, 

745 sigma_cube: Cube, 

746 z0_cube: Optional[Cube], 

747 pp_oro_cube: Cube, 

748 model_oro_cube: Cube, 

749 ) -> ndarray: 

750 """Check ancils grid and units. 

751 

752 Check if ancil cubes are on the same grid and if they have the 

753 expected units. The testing for "same grid" might be replaced 

754 if there is a general utils function made for it or so. 

755 

756 Args: 

757 a_over_s_cube: 

758 holding the silhouette roughness field 

759 sigma_cube: 

760 holding the standard deviation of height in a grid cell 

761 z0_cube: 

762 holding the vegetative roughness field 

763 pp_oro_cube: 

764 holding the post processing grid orography 

765 model_oro_cube: 

766 holding the model orography on post processing grid 

767 

768 Returns: 

769 Containing bools describing whether or not the tests passed 

770 """ 

771 ancil_list = [a_over_s_cube, sigma_cube, pp_oro_cube, model_oro_cube] 

772 unwanted_coord_list = [ 

773 "time", 

774 "height", 

775 "model_level_number", 

776 "forecast_time", 

777 "forecast_reference_time", 

778 "forecast_period", 

779 ] 

780 for field, exp_unit in zip(ancil_list, [1, Unit("m"), Unit("m"), Unit("m")]): 

781 for unwanted_coord in unwanted_coord_list: 

782 try: 

783 field.remove_coord(unwanted_coord) 

784 except CoordinateNotFoundError: 

785 pass 

786 if field.units != exp_unit: 

787 msg = ( 

788 "{} ancil field has unexpected unit:" 

789 " {} (expected) vs. {} (actual)" 

790 ) 

791 raise ValueError(msg.format(field.name(), exp_unit, field.units)) 

792 if z0_cube is not None: 

793 ancil_list.append(z0_cube) 

794 for unwanted_coord in unwanted_coord_list: 

795 try: 

796 z0_cube.remove_coord(unwanted_coord) 

797 except CoordinateNotFoundError: 

798 pass 

799 if z0_cube.units != Unit("m"): 

800 msg = "z0 ancil has unexpected unit: should be {} is {}" 

801 raise ValueError(msg.format(Unit("m"), z0_cube.units)) 

802 permutated_ancil_list = list(itertools.permutations(ancil_list, 2)) 

803 oklist = [] 

804 for entry in permutated_ancil_list: 

805 x_axis_flag = entry[0].coord(axis="y") == entry[1].coord(axis="y") 

806 y_axis_flag = entry[0].coord(axis="x") == entry[1].coord(axis="x") 

807 oklist.append(x_axis_flag & y_axis_flag) 

808 # HybridHeightToPhenomOnPressure._cube_compatibility_check(entr[0], 

809 # entr[1]) 

810 return np.array(oklist).all() # replace by a return value of True

812 def find_coord_order(self, mcube: Cube) -> Tuple[int, int, int, int]: 

813 """Extract coordinate ordering within a cube. 

814 

815 Use coord_dims to assess the dimension associated with a particular 

816 dimension coordinate. If a coordinate is not a dimension coordinate, 

817 then a NaN value will be returned for that coordinate. 

818 

819 Args: 

820 mcube: 

821 cube to check the order of coordinate axis 

822 

823 Returns: 

824 - position of x axis. 

825 - position of y axis. 

826 - position of z axis. 

827 - position of t axis. 

828 """ 

829 coord_names = [self.x_name, self.y_name, self.z_name, self.t_name] 

830 positions = len(coord_names) * [np.nan] 

831 for coord_index, coord_name in enumerate(coord_names): 

832 if mcube.coords(coord_name, dim_coords=True): 

833 (positions[coord_index],) = mcube.coord_dims(coord_name) 

834 return positions

836 def find_heightgrid(self, wind: Cube) -> ndarray: 

837 """Setup the height grid. 

838 

839 Setup the height grid either from the 1D or 3D height grid 

840 that was supplied to the plugin or from the z-axis information 

841 from the wind grid. 

842 

843 Args: 

844 wind: 

845 3D or 4D - representing the wind data. 

846 

847 Returns: 

848 1D or 3D array - representing the height grid. 

849 """ 

850 if self.height_levels is None: 

851 hld = wind.coord(self.z_name).points 

852 else: 

853 hld = iris.util.squeeze(self.height_levels) 

854 if np.isnan(hld.data).any() or (hld.data == RMDI).any(): 

855 raise ValueError("height grid contains invalid points") 

856 if hld.ndim == 3: 

857 try: 

858 xap, yap, zap, _ = self.find_coord_order(hld) 

859 hld.transpose([yap, xap, zap]) 

860 except KeyError: 

861 raise ValueError("height grid different from wind grid") 

862 elif hld.ndim == 1: 

863 try: 

864 hld = next(hld.slices([self.z_name])) 

865 except CoordinateNotFoundError: 

866 raise ValueError("height z coordinate differs from wind z") 

867 else: 

868 raise ValueError( 

869 "hld must have a dimension length of " 

870 "either 3 or 1" 

871 "hld.ndim is {}".format(hld.ndim) 

872 ) 

873 hld = hld.data 

874 return hld

894 def process(self, input_cube: Cube) -> Cube: 

895 """Adjust the 4d wind field - cube - (x, y, z including times). 

896 

897 Args: 

898 input_cube: 

899 The wind cube to be operated upon. Should be wind speed on 

900 height_levels for all desired forecast times. 

901 

902 Returns: 

903 The 4d wind field with roughness and height correction 

904 applied in the same order as the input cube. 

905 

906 Raises: 

907 TypeError: If input_cube is not a cube. 

908 """ 

909 if not isinstance(input_cube, iris.cube.Cube): 

910 msg = "wind input is not a cube, but {}" 

911 raise TypeError(msg.format(type(input_cube))) 

912 (self.x_name, self.y_name, self.z_name, self.t_name) = self.find_coord_names( 

913 input_cube 

914 ) 

915 xwp, ywp, zwp, twp = self.find_coord_order(input_cube) 

916 if np.isnan(twp): 

917 input_cube.transpose([ywp, xwp, zwp]) 

918 else: 

919 input_cube.transpose([ywp, xwp, zwp, twp]) # problems with slices 

920 rchc_list = iris.cube.CubeList() 

921 if self.z_0 is None: 

922 z0_data = None 

923 else: 

924 z0_data = self.z_0.data 

925 roughness_correction = RoughnessCorrectionUtilities( 

926 self.a_over_s.data, 

927 self.sigma.data, 

928 z0_data, 

929 self.pp_oro.data, 

930 self.model_oro.data, 

931 self.ppres, 

932 self.modres, 

933 ) 

934 self.check_wind_ancil(xwp, ywp) 

935 hld = self.find_heightgrid(input_cube) 

936 for time_slice in input_cube.slices_over("time"): 

937 if np.isnan(time_slice.data).any() or (time_slice.data < 0.0).any(): 

938 msg = "{} has invalid wind data" 

939 raise ValueError(msg.format(time_slice.coord(self.t_name))) 

940 rc_hc = copy.deepcopy(time_slice) 

941 rc_hc.data = roughness_correction.do_rc_hc_all(hld, time_slice.data) 

942 rchc_list.append(rc_hc) 

943 output_cube = rchc_list.merge_cube() 

944 # reorder input_cube and output_cube as original 

945 if np.isnan(twp): 

946 input_cube.transpose(np.argsort([ywp, xwp, zwp])) 

947 output_cube.transpose(np.argsort([ywp, xwp, zwp])) 

948 else: 

949 input_cube.transpose(np.argsort([ywp, xwp, zwp, twp])) 

950 output_cube.transpose(np.argsort([twp, ywp, xwp, zwp])) 

951 

952 return output_cube
