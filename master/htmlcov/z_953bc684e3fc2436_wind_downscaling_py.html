8from improver import cli 

9 

10 

11@cli.clizefy 

12@cli.with_output 

13def process( 

14 wind_speed: cli.inputcube, 

15 sigma: cli.inputcube, 

16 target_orography: cli.inputcube, 

17 standard_orography: cli.inputcube, 

18 silhouette_roughness: cli.inputcube, 

19 vegetative_roughness: cli.inputcube = None, 

20 *, 

21 model_resolution: float, 

22 output_height_level: float = None, 

23 output_height_level_units="m", 

24): 

25 """Wind downscaling. 

26 

27 Run wind downscaling to apply roughness correction and height correction 

28 to wind fields as described in Howard and Clark (2007). All inputs must 

29 be on the same standard grid. 

30 

31 Args: 

32 wind_speed (iris.cube.Cube): 

33 Cube of wind speed on standard grid. 

34 Any units can be supplied. 

35 sigma (iris.cube.Cube): 

36 Cube of standard deviation of model orography height. 

37 Units of field: m. 

38 target_orography (iris.cube.Cube): 

39 Cube of orography to downscale fields to. 

40 Units of field: m. 

41 standard_orography (iris.cube.Cube): 

42 Cube of orography on standard grid. (interpolated model orography). 

43 Units of field: m. 

44 silhouette_roughness (iris.cube.Cube): 

45 Cube of model silhouette roughness. 

46 Units of field: dimensionless. 

47 vegetative_roughness (iris.cube.Cube): 

48 Cube of vegetative roughness length. 

49 Units of field: m. 

50 model_resolution (float): 

51 Original resolution of model orography (before interpolation to 

52 standard grid) 

53 Units of field: m. 

54 output_height_level (float): 

55 If only a single height level is desired as output from 

56 wind-downscaling, this option can be used to select the height 

57 level. If no units are provided with 'output_height_level_units', 

58 metres are assumed. 

59 output_height_level_units (str): 

60 If a single height level is selected as output using 

61 'output_height_level', this additional argument may be used to 

62 specify the units of the value entered to select the level. 

63 e.g hPa. 

64 

65 Returns: 

66 iris.cube.Cube: 

67 The processed Cube. 

68 

69 Rises: 

70 ValueError: 

71 If the requested height value is not found. 

72 

73 """ 

74 import warnings 

75 

76 import iris 

77 from iris.exceptions import CoordinateNotFoundError 

78 

79 from improver.utilities.cube_extraction import apply_extraction 

80 from improver.wind_calculations import wind_downscaling 

81 

82 if output_height_level_units and output_height_level is None: 

83 warnings.warn( 

84 "output_height_level_units has been set but no " 

85 "associated height level has been provided. These units " 

86 "will have no effect." 

87 ) 

88 try: 

89 wind_speed_iterator = wind_speed.slices_over("realization") 

90 except CoordinateNotFoundError: 

91 wind_speed_iterator = [wind_speed] 

92 wind_speed_list = iris.cube.CubeList() 

93 for wind_speed_slice in wind_speed_iterator: 

94 result = wind_downscaling.RoughnessCorrection( 

95 silhouette_roughness, 

96 sigma, 

97 target_orography, 

98 standard_orography, 

99 model_resolution, 

100 z0_cube=vegetative_roughness, 

101 height_levels_cube=None, 

102 )(wind_speed_slice) 

103 wind_speed_list.append(result) 

104 

105 wind_speed = wind_speed_list.merge_cube() 

106 non_dim_coords = [x.name() for x in wind_speed.coords(dim_coords=False)] 

107 if "realization" in non_dim_coords: 

108 wind_speed = iris.util.new_axis(wind_speed, "realization") 

109 if output_height_level is not None: 

110 constraints = {"height": output_height_level} 

111 units = {"height": output_height_level_units} 

112 single_level = apply_extraction( 

113 wind_speed, iris.Constraint(**constraints), units 

114 ) 

115 if not single_level: 

116 raise ValueError( 

117 "Requested height level not found, no cube " 

118 "returned. Available height levels are:\n" 

119 "{0:}\nin units of {1:}".format( 

120 wind_speed.coord("height").points, wind_speed.coord("height").units 

121 ) 

122 ) 

123 wind_speed = single_level 

124 return wind_speed
