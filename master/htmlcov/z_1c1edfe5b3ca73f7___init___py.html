24class BasePlugin(ABC): 

25 """An abstract class for IMPROVER plugins. 

26 Subclasses must be callable. We preserve the process 

27 method by redirecting to __call__. 

28 """ 

29 

30 def __call__(self, *args, **kwargs) -> Any: 

31 """Makes subclasses callable to use process 

32 Args: 

33 *args: 

34 Positional arguments. 

35 **kwargs: 

36 Keyword arguments. 

37 Returns: 

38 Output of self.process() 

39 """ 

40 return self.process(*args, **kwargs)

42 @abstractmethod 

43 def process(self, *args, **kwargs) -> Any: 

44 """Abstract class for rest to implement.""" 

45 pass

48class PostProcessingPlugin(BasePlugin): 

49 """An abstract class for IMPROVER post-processing plugins. 

50 Makes generalised changes to metadata relating to post-processing. 

51 """ 

52 

53 def __call__(self, *args, **kwargs) -> Any: 

54 """Makes subclasses callable to use process 

55 Args: 

56 *args: 

57 Positional arguments. 

58 **kwargs: 

59 Keyword arguments. 

60 

61 Returns: 

62 Output of self.process() with updated title attribute 

63 """ 

64 from iris.cube import Cube 

65 

66 result = super().__call__(*args, **kwargs) 

67 if isinstance(result, Cube): 

68 self.post_processed_title(result) 

69 elif isinstance(result, Iterable) and not isinstance(result, str): 

70 for item in result: 

71 if isinstance(item, Cube): 

72 self.post_processed_title(item) 

73 return result 

74 

75 @staticmethod 

76 def post_processed_title(cube): 

77 """Updates title attribute on output cube to include 

78 "Post-Processed" 

79 """ 

80 from improver.metadata.constants.attributes import MANDATORY_ATTRIBUTE_DEFAULTS 

81 

82 default_title = MANDATORY_ATTRIBUTE_DEFAULTS["title"] 

83 if ( 

84 "title" in cube.attributes.keys() 

85 and cube.attributes["title"] != default_title 

86 and "Post-Processed" not in cube.attributes["title"] 

87 ): 

88 title = cube.attributes["title"] 

89 cube.attributes["title"] = f"Post-Processed {title}"

110 return target
