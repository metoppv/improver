31class SpotManipulation(BasePlugin): 

32 """ 

33 A wrapper to the spot-extraction plugin that handles additional options 

34 for manipulating the spot-data. This allows for the extraction of 

35 percentiles, including the conversion to percentiles of probabilities. 

36 Lapse rate adjustment of spot temperatures is also possible. 

37 """ 

38 

39 def __init__( 

40 self, 

41 apply_lapse_rate_correction: bool = False, 

42 fixed_lapse_rate: Optional[float] = None, 

43 land_constraint: bool = False, 

44 similar_altitude: bool = False, 

45 extract_percentiles: Optional[Union[float, List[float]]] = None, 

46 ignore_ecc_bounds_exceedance: bool = False, 

47 skip_ecc_bounds: bool = False, 

48 new_title: Optional[str] = None, 

49 suppress_warnings: bool = False, 

50 realization_collapse: bool = False, 

51 subset_coord: str = None, 

52 ) -> None: 

53 """ 

54 Initialise the wrapper plugin using the selected options. 

55 

56 Args: 

57 apply_lapse_rate_correction (bool): 

58 Use to apply a lapse-rate correction to screen temperature 

59 forecasts so that they better represent the altitude of the 

60 spot site for which they have been extracted. This lapse rate 

61 will be applied for a fixed orographic difference between the 

62 site and grid point altitude. Differences in orography in 

63 excess of this fixed limit will use the Environmental Lapse 

64 Rate (also known as the Standard Atmosphere Lapse Rate). 

65 Lapse rate adjustment cannot be applied to existing spot 

66 forecasts that are passed in for subsetting. 

67 fixed_lapse_rate (float): 

68 If provided, use this fixed value as a lapse-rate for adjusting 

69 the forecast values if apply_lapse_rate_correction is True. This 

70 can be used instead of providing a lapse rate cube. Value is 

71 given in Kelvin / metre of temperature change with ascent. For 

72 example a dry adiabatic lapse rate would be given as -0.0098. 

73 This lapse rate will be applied for a fixed orographic difference 

74 between the site and gridpoint altitude. Differences in orography 

75 in excess of this fixed limit will use the Environmental Lapse 

76 Rate (also known as the Standard Atmosphere Lapse Rate). 

77 land_constraint (bool): 

78 Use to select the nearest-with-land-constraint neighbour-selection 

79 method from the neighbour_cube. This means that the grid points 

80 should be land points except for sites where none were found within 

81 the search radius when the neighbour cube was created. May be used 

82 with similar_altitude. 

83 similar_altitude (bool): 

84 Use to select the nearest-with-height-constraint 

85 neighbour-selection method from the neighbour_cube. These are grid 

86 points that were found to be the closest in altitude to the spot 

87 site within the search radius defined when the neighbour cube was 

88 created. May be used with land_constraint. 

89 extract_percentiles (list or int): 

90 If set to a percentile value or a list of percentile values, 

91 data corresponding to those percentiles will be returned. For 

92 example "25, 50, 75" will result in the 25th, 50th and 75th 

93 percentiles being returned from a cube of probabilities, 

94 percentiles or realizations. Deterministic input data will raise 

95 a warning message. 

96 Note that for percentile inputs, if the desired percentile(s) do 

97 not exist in the input cube the available percentiles will be 

98 resampled to produce those requested. 

99 ignore_ecc_bounds_exceedance (bool): 

100 Demotes exceptions where calculated percentiles are outside the ECC 

101 bounds range to warnings. 

102 skip_ecc_bounds (bool): 

103 If True, ECC bounds are not included when converting probabilities to 

104 percentiles or from one set of percentiles to another. This has the 

105 effect that percentiles outside of the range given by the input 

106 percentiles will be computed by nearest neighbour interpolation from 

107 the nearest available percentile, rather than using linear 

108 interpolation between the nearest available percentile and the ECC 

109 bound. 

110 new_title (str): 

111 New title for the spot-extracted data. If None, this attribute is 

112 removed from the output cube since it has no prescribed standard 

113 and may therefore contain grid information that is no longer 

114 correct after spot-extraction. 

115 suppress_warnings (bool): 

116 Suppress warning output. This option should only be used if it 

117 is known that warnings will be generated but they are not required. 

118 realization_collapse (bool): 

119 Triggers equal-weighting blending of the realization coord if required. 

120 Use this if a threshold coord is also present on the input cube. 

121 subset_coord (str): 

122 If a spot cube is provided as input this plugin can return a subset of 

123 the sites based on the sites specified in the neighbour cube. To 

124 achieve this the plugin needs the name of the site ID coordinate to be 

125 used for matching, e.g. wmo_id. If subset_coord is not provided, and a 

126 spot forecast is passed in, the entire spot cube will be processed and 

127 returned. The neighbour selection method options have no impact if a 

128 spot cube is passed in. 

129 """ 

130 self.neighbour_selection_method = get_neighbour_finding_method_name( 

131 land_constraint, similar_altitude 

132 ) 

133 self.apply_lapse_rate_correction = apply_lapse_rate_correction 

134 self.fixed_lapse_rate = fixed_lapse_rate 

135 self.extract_percentiles = extract_percentiles 

136 self.ignore_ecc_bounds_exceedance = ignore_ecc_bounds_exceedance 

137 self.skip_ecc_bounds = skip_ecc_bounds 

138 self.new_title = new_title 

139 self.suppress_warnings = suppress_warnings 

140 self.realization_collapse = realization_collapse 

141 self.subset_coord = subset_coord 

142 

143 def process(self, cubes: CubeList) -> Cube: 

144 """ 

145 Call spot-extraction and other plugins to manipulate the resulting 

146 spot forecasts. 

147 

148 Args: 

149 cubes: 

150 A list of cubes containing the diagnostic data to be extracted, 

151 a temperature lapse rate (optional) and the neighbour cube. 

152 

153 Returns: 

154 Spot-extracted forecast data following any optional manipulations. 

155 

156 Warns: 

157 If diagnostic cube is not a known probabilistic type. 

158 If a lapse rate cube was not provided, but the option to apply 

159 the lapse rate correction was enabled. 

160 """ 

161 neighbour_cube = cubes[-1] 

162 cube = cubes[0] 

163 

164 # If a spot forecast cube is passed in, constrain the sites to 

165 # those that are found in the neighbour cube if an ID coordinate on 

166 # which to constrain is provided, e.g. wmo_id. Otherwise pass the 

167 # spot forecast cube forwards unchanged. 

168 if cube.coords("spot_index"): 

169 if ( 

170 self.apply_lapse_rate_correction is not False 

171 or self.fixed_lapse_rate is not None 

172 ): 

173 raise NotImplementedError( 

174 "Lapse rate adjustment when subsetting an existing spot " 

175 "forecast cube has not been implemented." 

176 ) 

177 if self.subset_coord is None: 

178 result = cube 

179 else: 

180 try: 

181 sites = neighbour_cube.coord(self.subset_coord).points 

182 except CoordinateNotFoundError as err: 

183 raise ValueError( 

184 "Subset_coord not found in neighbour cube." 

185 ) from err 

186 # Exclude unset site IDs as this value is non-unique. 

187 sites = [item for item in sites if item != "None"] 

188 site_constraint = iris.Constraint( 

189 coord_values={self.subset_coord: sites} 

190 ) 

191 result = cube.extract(site_constraint) 

192 if not result: 

193 raise ValueError("No spot sites retained after subsetting.") 

194 else: 

195 result = SpotExtraction( 

196 neighbour_selection_method=self.neighbour_selection_method 

197 )(neighbour_cube, cube, new_title=self.new_title) 

198 

199 if self.realization_collapse: 

200 result = collapse_realizations(result) 

201 

202 # If a probability or percentile diagnostic cube is provided, extract 

203 # the given percentile if available. This is done after the spot-extraction 

204 # to minimise processing time; usually there are far fewer spot sites than 

205 # grid points. 

206 if self.extract_percentiles: 

207 extract_percentiles = [float(x) for x in self.extract_percentiles] 

208 try: 

209 perc_coordinate = find_percentile_coordinate(result) 

210 except CoordinateNotFoundError: 

211 if "probability_of_" in result.name(): 

212 result = ConvertProbabilitiesToPercentiles( 

213 ecc_bounds_warning=self.ignore_ecc_bounds_exceedance, 

214 skip_ecc_bounds=self.skip_ecc_bounds, 

215 )(result, percentiles=extract_percentiles) 

216 result = iris.util.squeeze(result) 

217 elif result.coords("realization", dim_coords=True): 

218 fast_percentile_method = not np.ma.is_masked(result.data) 

219 result = PercentileConverter( 

220 "realization", 

221 percentiles=extract_percentiles, 

222 fast_percentile_method=fast_percentile_method, 

223 )(result) 

224 else: 

225 if not self.suppress_warnings: 

226 msg = ( 

227 "Diagnostic cube is not a known probabilistic type. " 

228 "The {} percentile(s) could not be extracted. The " 

229 "spot-extracted outputs will be returned without " 

230 "percentile extraction.".format(extract_percentiles) 

231 ) 

232 warnings.warn(msg) 

233 else: 

234 if set(extract_percentiles).issubset(perc_coordinate.points): 

235 constraint = [ 

236 "{}={}".format(perc_coordinate.name(), extract_percentiles) 

237 ] 

238 result = extract_subcube(result, constraint) 

239 else: 

240 result = ResamplePercentiles(skip_ecc_bounds=self.skip_ecc_bounds)( 

241 result, percentiles=extract_percentiles 

242 ) 

243 

244 # Check whether a lapse rate cube has been provided 

245 if self.apply_lapse_rate_correction: 

246 if len(cubes) == 3: 

247 plugin = SpotLapseRateAdjust( 

248 neighbour_selection_method=self.neighbour_selection_method 

249 ) 

250 result = plugin(result, neighbour_cube, cubes[-2]) 

251 elif self.fixed_lapse_rate is not None: 

252 plugin = SpotLapseRateAdjust( 

253 neighbour_selection_method=self.neighbour_selection_method, 

254 fixed_lapse_rate=self.fixed_lapse_rate, 

255 ) 

256 result = plugin(result, neighbour_cube) 

257 elif not self.suppress_warnings: 

258 warnings.warn( 

259 "A lapse rate cube or fixed lapse rate was not provided, but the " 

260 "option to apply the lapse rate correction was enabled. No lapse rate " 

261 "correction could be applied." 

262 ) 

263 

264 # Remove the internal model_grid_hash attribute if present. 

265 result.attributes.pop("model_grid_hash", None) 

266 

267 return result
