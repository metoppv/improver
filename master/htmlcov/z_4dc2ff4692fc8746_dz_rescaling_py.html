25class EstimateDzRescaling(PostProcessingPlugin): 

26 """Estimate a rescaling of the input forecasts based on the difference in 

27 altitude between the grid point and the site.""" 

28 

29 def __init__( 

30 self, 

31 forecast_period: float, 

32 dz_lower_bound: Union[str, float] = None, 

33 dz_upper_bound: Union[str, float] = None, 

34 land_constraint: bool = False, 

35 similar_altitude: bool = False, 

36 site_id_coord: str = "wmo_id", 

37 ): 

38 """Initialise class. 

39 

40 Args: 

41 forecast_period: The forecast period in hours that is considered 

42 representative of the input forecasts. This is required as the input 

43 forecasts could contain multiple forecast periods. 

44 dz_lower_bound: The lowest acceptable value for the difference in 

45 altitude between the grid point and the site. Sites with a lower 

46 (or more negative) difference in altitude will be excluded. 

47 Defaults to None. 

48 dz_upper_bound: The highest acceptable value for the difference in 

49 altitude between the grid point and the site. Sites with a larger 

50 positive difference in altitude will be excluded. Defaults to None. 

51 land_constraint: 

52 If True, this will return a cube containing the nearest grid point 

53 neighbours to spot sites that are also land points. May be used 

54 with the similar_altitude option. 

55 similar_altitude: 

56 If True, this will return a cube containing the nearest grid point 

57 neighbour to each spot site that is found, within a given search 

58 radius, to minimise the height difference between the two. May be 

59 used with the land_constraint option. 

60 site_id_coord: 

61 The name of the site ID coordinate. This defaults to 'wmo_id'. 

62 """ 

63 self.forecast_period = forecast_period 

64 if dz_lower_bound is None: 

65 self.dz_lower_bound = -np.inf 

66 else: 

67 self.dz_lower_bound = np.float32(dz_lower_bound) 

68 if dz_upper_bound is None: 

69 self.dz_upper_bound = np.inf 

70 else: 

71 self.dz_upper_bound = np.float32(dz_upper_bound) 

72 # The degree chosen for the fitting polynomial. This is set to 1. 

73 # Please see numpy.polynomial.polynomial.Polynomial.fit for further information. 

74 self.polyfit_deg = 1 

75 

76 self.neighbour_selection_method = get_neighbour_finding_method_name( 

77 land_constraint=land_constraint, minimum_dz=similar_altitude 

78 ) 

79 self.site_id_coord = site_id_coord

81 def _fit_polynomial(self, forecasts: Cube, truths: Cube, dz: Cube) -> float: 

82 """Create a polynomial fit between the log of the ratio of forecasts and truths, 

83 and the difference in altitude between the grid point and the site. 

84 

85 Args: 

86 forecasts: Forecast cube. 

87 truths: Truth cube. 

88 dz: Difference in altitude between the grid point and the site location. 

89 

90 Returns: 

91 A scale factor deduced from a polynomial fit. This is a single value 

92 deduced from the fit between the forecasts and the truths. 

93 """ 

94 truths_data = np.reshape(truths.data, forecasts.shape) 

95 

96 data_filter = ( 

97 (forecasts.data != 0) 

98 & (truths_data != 0) 

99 & (dz.data >= self.dz_lower_bound) 

100 & (dz.data <= self.dz_upper_bound) 

101 ) 

102 

103 forecasts_data = forecasts.data.flatten() 

104 truths_data = truths_data.flatten() 

105 dz_data = np.broadcast_to(dz.data, forecasts.shape).flatten() 

106 data_filter = data_filter.flatten() 

107 

108 log_error_ratio = np.log(forecasts_data[data_filter] / truths_data[data_filter]) 

109 

110 scale_factor = poly1d( 

111 polyfit(dz_data[data_filter], log_error_ratio, self.polyfit_deg) 

112 ).coef 

113 

114 # Only retain the multiplicative coefficient as the scale factor. 

115 # This helps conceptually with the difference in altitude rescaling 

116 # where if the dz of the grid point and the site are the same, then no 

117 # adjustment will be made. 

118 return scale_factor[1]

120 def _compute_scaled_dz(self, scale_factor: float, dz: np.ndarray) -> np.ndarray: 

121 """Compute the scaled difference in altitude. 

122 

123 Args: 

124 scale_factor: A scale factor deduced from a polynomial fit. 

125 dz: The difference in altitude between the grid point and the site. 

126 

127 Returns: 

128 Scaled difference in altitude at each site. 

129 """ 

130 # Multiplication by -1 using negative exponent rule, so that this term can 

131 # be multiplied by the forecast during the application step. 

132 scaled_dz = np.exp(-1.0 * scale_factor * dz) 

133 

134 # Compute lower and upper bounds for the scaled dz. 

135 # dz_lower_bound may not result in the lower bound for the scaled dz depending 

136 # upon the sign of the scale_factor term. 

137 scaled_dz_a = np.exp(-1.0 * scale_factor * self.dz_lower_bound) 

138 scaled_dz_b = np.exp(-1.0 * scale_factor * self.dz_upper_bound) 

139 scaled_dz_lower = np.amin([scaled_dz_a, scaled_dz_b]) 

140 scaled_dz_upper = np.amax([scaled_dz_a, scaled_dz_b]) 

141 

142 return np.clip(scaled_dz.data, scaled_dz_lower, scaled_dz_upper)

144 def _compute_scaled_dz_cube( 

145 self, forecast: Cube, dz: Cube, scale_factor: float 

146 ) -> Cube: 

147 """Compute the scaled difference in altitude and ensure that the output cube 

148 has the correct metadata. 

149 

150 Args: 

151 forecast: Forecast cube. 

152 dz: The difference in altitude between the grid point and the site. 

153 scale_factor: A scale factor deduced from a polynomial fit. 

154 

155 Returns: 

156 Scaled difference in altitude cube with appropriate metadata. 

157 """ 

158 scaled_dz = dz.copy() 

159 scaled_dz.rename("scaled_vertical_displacement") 

160 scaled_dz.units = "1" 

161 for coord_name in [ 

162 "grid_attributes", 

163 "grid_attributes_key", 

164 "neighbour_selection_method", 

165 ]: 

166 scaled_dz.remove_coord(coord_name) 

167 scaled_dz.attributes.pop("model_grid_hash", None) 

168 

169 scaled_dz.data = self._compute_scaled_dz(scale_factor, scaled_dz.data) 

170 

171 fp_forecast_slice = next(forecast.slices_over("forecast_period")) 

172 

173 fp_forecast_slice.coord("forecast_period").points = np.array( 

174 self.forecast_period * SECONDS_IN_HOUR, 

175 dtype=TIME_COORDS["forecast_period"].dtype, 

176 ) 

177 scaled_dz.add_aux_coord(fp_forecast_slice.coord("forecast_period")) 

178 self._create_hour_coord(forecast, scaled_dz) 

179 return scaled_dz

181 def _create_hour_coord(self, source_cube: Cube, target_cube: Cube): 

182 """Create a coordinate exclusively containing the hour of the forecast 

183 reference time. This is required as the date of the forecast reference time 

184 is not relevant when using a training dataset with the aim that the resulting 

185 scaling factor is applicable to future forecasts with the same hour for the 

186 forecast reference time. The auxiliary coordinate will be added to the 

187 target_cube. 

188 

189 Args: 

190 source_cube: Cube containing the forecast reference time from which 

191 the hour will be extracted. 

192 target_cube: Cube to which an auxiliary coordinate will be added. 

193 """ 

194 # Create forecast_reference_time_hour coordinate. Use the time coordinate and 

195 # the forecast_period argument provided in case the forecast_reference_time 

196 # coordinate is not always the same within all input forecasts. 

197 frt_hour = ( 

198 source_cube.coord("time").cell(0).point 

199 - pd.Timedelta(hours=self.forecast_period) 

200 ).hour 

201 hour_coord = AuxCoord( 

202 np.array(frt_hour, np.int32), 

203 long_name="forecast_reference_time_hour", 

204 units="hours", 

205 ) 

206 hour_coord.convert_units("seconds") 

207 hour_coord.points = hour_coord.points.astype(np.int32) 

208 target_cube.add_aux_coord(hour_coord)

210 def process(self, forecasts: Cube, truths: Cube, neighbour_cube: Cube) -> Cube: 

211 """Fit a polynomial using the forecasts and truths to compute a scaled 

212 version of the difference of altitude between the grid point and the 

213 site location. There is expected to be overlap between the sites provided by 

214 the forecasts, truths and neighbour_cube. The polynomial will be fitted using 

215 the forecasts and truths and the resulting scale factor will be applied to 

216 all sites within the neighbour cube. The output scaled dz will contain sites 

217 matching the neighbour cube. 

218 

219 A mathematical summary of the steps within this plugin are: 

220 

221 1. Estimate a scale factor for the relationship between the difference in 

222 altitude between the grid point and the site, and the natural log of the 

223 forecast divided by the truth, where s is the scale factor in the equations 

224 below. 

225 

226 .. math:: 

227 

228 dz = \\ln(forecast / truth) \\times s 

229 

230 2. Rearranging this equation gives: 

231 

232 .. math:: 

233 

234 truth = forecast / \\exp(s \\times dz) 

235 

236 or alternatively: 

237 

238 .. math:: 

239 

240 truth = forecast \\times \\exp(-s \\times dz) 

241 

242 This plugin is aiming to estimate the :math:`\\exp(-s \\times dz)` component, 

243 which can later be used for multiplying by a forecast to estimate the truth. 

244 

245 Args: 

246 forecasts: Forecast cube. 

247 truths: Truth cube. 

248 neighbour_cube: A neighbour cube containing the difference in altitude 

249 between the grid point and the site location. Note that the output 

250 will have the same sites as found within the neighbour cube. 

251 

252 Returns: 

253 A scaled difference of altitude between the grid point and the 

254 site location. 

255 """ 

256 method = iris.Constraint( 

257 neighbour_selection_method_name=self.neighbour_selection_method 

258 ) 

259 index_constraint = iris.Constraint( 

260 grid_attributes_key=["vertical_displacement"] 

261 ) 

262 dz_cube = neighbour_cube.extract(method & index_constraint) 

263 

264 sites = list( 

265 set(forecasts.coord(self.site_id_coord).points) 

266 & set(truths.coord(self.site_id_coord).points) 

267 & set(dz_cube.coord(self.site_id_coord).points) 

268 ) 

269 

270 constr = iris.Constraint(coord_values={self.site_id_coord: sites}) 

271 training_forecasts = forecasts.extract(constr) 

272 training_truths = truths.extract(constr) 

273 dz_training_cube = dz_cube.extract(constr) 

274 

275 forecast_cube, truth_cube = filter_non_matching_cubes( 

276 training_forecasts, training_truths 

277 ) 

278 

279 constr = iris.Constraint(percentile=50.0) 

280 forecast_cube = forecast_cube.extract(constr) 

281 

282 scale_factor = self._fit_polynomial(forecast_cube, truth_cube, dz_training_cube) 

283 scaled_dz_cube = self._compute_scaled_dz_cube( 

284 forecast_cube, dz_cube, scale_factor 

285 ) 

286 

287 return scaled_dz_cube

290class ApplyDzRescaling(PostProcessingPlugin): 

291 """Apply rescaling of the forecast using the difference in altitude between the 

292 grid point and the site.""" 

293 

294 def __init__(self, site_id_coord: str = "wmo_id", frt_hour_leniency=1): 

295 """Initialise class. 

296 

297 Args: 

298 site_id_coord: 

299 The name of the site ID coordinate. This defaults to 'wmo_id'. 

300 frt_hour_leniency: 

301 The forecast reference time hour for the forecast and the scaled_dz 

302 are expected to match. If no match is found and a leniency greater 

303 than zero is specified, the forecast reference time hour will be 

304 compared with e.g. a +/-1 hour leniency. 

305 """ 

306 self.site_id_coord = site_id_coord 

307 self.frt_hour_leniency = frt_hour_leniency

309 def _check_mismatched_sites(self, forecast: Cube, scaled_dz: Cube) -> None: 

310 """Check that the sites match for the forecast and the scaled_dz inputs. 

311 

312 Args: 

313 forecast: Forecast to be adjusted using dz rescaling. 

314 scaled_dz: A scaled version of the difference in altitude between the 

315 grid point and the site. 

316 

317 Raises: 

318 ValueError: Sites do not match between the forecast and the scaled dz. 

319 """ 

320 if len(forecast.coord(self.site_id_coord).points) != len( 

321 scaled_dz.coord(self.site_id_coord).points 

322 ) or np.any( 

323 forecast.coord(self.site_id_coord).points 

324 != scaled_dz.coord(self.site_id_coord).points 

325 ): 

326 mismatched_sites = set( 

327 forecast.coord(self.site_id_coord).points 

328 ).symmetric_difference(scaled_dz.coord(self.site_id_coord).points) 

329 mismatched_sites = set(map(str, mismatched_sites)) 

330 n_forecast_sites = len(forecast.coord(self.site_id_coord).points) 

331 n_scaled_dz_sites = len(scaled_dz.coord(self.site_id_coord).points) 

332 msg = ( 

333 "The sites do not match between the forecast and the scaled " 

334 "version of the difference in altitude. " 

335 f"Forecast number of sites: {n_forecast_sites}. " 

336 f"Scaled_dz number of sites: {n_scaled_dz_sites}. " 

337 f"The mismatched sites are: {mismatched_sites}." 

338 ) 

339 raise ValueError(msg)

341 @staticmethod 

342 def _create_forecast_period_constraint( 

343 forecast: Cube, scaled_dz: Cube 

344 ) -> iris.Constraint: 

345 """Create a forecast period constraint to identify the most appropriate 

346 forecast period from the scaled_dz to extract. The most appropriate scaled dz 

347 is selected by choosing the nearest forecast period that is greater than or 

348 equal to the forecast period of the forecast. If no forecast periods in the 

349 scaled dz cube are greater than the forecast period of the forecast, the 

350 longest available forecast period is used. 

351 

352 Args: 

353 forecast: Forecast to be adjusted using dz rescaling. 

354 scaled_dz: A scaled version of the difference in altitude between the 

355 grid point and the site. 

356 

357 Returns: 

358 Forecast period constraint. 

359 

360 Raises: 

361 ValueError: No scaled dz could be found for the forecast period provided. 

362 """ 

363 fp_diff = ( 

364 scaled_dz.coord("forecast_period").points 

365 - forecast.coord("forecast_period").points 

366 ) 

367 

368 if any(fp_diff >= 0): 

369 fp_index = np.argmax(fp_diff >= 0) 

370 chosen_fp = scaled_dz.coord("forecast_period").points[fp_index] 

371 else: 

372 chosen_fp = scaled_dz.coord("forecast_period").points[-1] 

373 

374 return iris.Constraint(forecast_period=chosen_fp)

376 @staticmethod 

377 def _create_forecast_reference_time_constraint( 

378 forecast: Cube, leniency: int 

379 ) -> iris.Constraint: 

380 """Create a forecast reference time constraint based on the hour within the 

381 forecast reference time. 

382 

383 Args: 

384 forecast: Forecast to be adjusted using dz rescaling. 

385 leniency: The leniency in hours to adjust the forecast reference time hour 

386 when looking for a match. 

387 

388 Returns: 

389 Forecast reference time hour constraint. 

390 """ 

391 # Define forecast_reference_time constraint 

392 frt_hour_in_seconds = ( 

393 (forecast.coord("forecast_reference_time").cell(0).point.hour + leniency) 

394 % 24 

395 ) * SECONDS_IN_HOUR 

396 return iris.Constraint(forecast_reference_time_hour=frt_hour_in_seconds)

398 def process(self, forecast: Cube, scaled_dz: Cube) -> Cube: 

399 """Apply rescaling of the forecast to account for differences in the altitude 

400 between the grid point and the site, as assessed using a training dataset. 

401 The most appropriate scaled dz is selected by choosing the nearest forecast 

402 period that is greater than or equal to the forecast period of the forecast. 

403 

404 Args: 

405 forecast: Forecast to be adjusted using dz rescaling. 

406 scaled_dz: A scaled version of the difference in altitude between the 

407 grid point and the site. 

408 

409 Returns: 

410 Altitude-corrected forecast. 

411 

412 Raises: 

413 ValueError: No scaled dz could be found for the forecast period and 

414 forecast reference time hour provided. 

415 

416 """ 

417 self._check_mismatched_sites(forecast, scaled_dz) 

418 fp_constr = self._create_forecast_period_constraint(forecast, scaled_dz) 

419 frt_hour_leniency_range = sorted( 

420 list(range(-self.frt_hour_leniency, self.frt_hour_leniency + 1)), key=abs 

421 ) 

422 for leniency in frt_hour_leniency_range: 

423 frt_constr = self._create_forecast_reference_time_constraint( 

424 forecast, leniency 

425 ) 

426 scaled_dz_extracted = scaled_dz.extract(fp_constr & frt_constr) 

427 if scaled_dz_extracted is not None: 

428 break 

429 else: 

430 frt_hour = forecast.coord("forecast_reference_time").cell(0).point.hour 

431 (fp_hour,) = forecast.coord("forecast_period").points / SECONDS_IN_HOUR 

432 msg = ( 

433 "There is no scaled version of the difference in altitude for " 

434 f"a forecast period greater than or equal to {fp_hour} and " 

435 f"a forecast reference time hour equal to {frt_hour} or within " 

436 f"the specified leniency {self.frt_hour_leniency}." 

437 ) 

438 raise ValueError(msg) 

439 

440 forecast.data = forecast.data * scaled_dz_extracted.data 

441 return forecast
