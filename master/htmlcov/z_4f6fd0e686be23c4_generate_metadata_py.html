23DEFAULT_GRID_SPACING = {"latlon": 0.02, "equalarea": 2000}

29def _get_units(name: str) -> str: 

30 """Get output variable units from iris.std_names.STD_NAMES""" 

31 try: 

32 units = STD_NAMES[name]["canonical_units"] 

33 except KeyError: 

34 raise ValueError("Units of {} are not known.".format(name)) 

35 

36 return units

39def _create_time_bounds(time: datetime, time_period: int) -> Tuple[datetime, datetime]: 

40 """Create time bounds using time - time_period as the lower bound and time as the 

41 upper bound""" 

42 lower_bound = time - timedelta(minutes=time_period) 

43 upper_bound = time 

44 

45 return (lower_bound, upper_bound)

48def _create_data_array( 

49 ensemble_members: int, 

50 leading_dimension: Optional[List[float]], 

51 npoints: int, 

52 vertical_levels: Optional[List[float]], 

53) -> ndarray: 

54 """Create data array of specified shape filled with zeros""" 

55 if leading_dimension is not None: 

56 nleading_dimension = len(leading_dimension) 

57 else: 

58 nleading_dimension = ensemble_members 

59 

60 if vertical_levels is not None: 

61 nvertical_levels = len(vertical_levels) 

62 else: 

63 nvertical_levels = None 

64 

65 data_shape = [] 

66 

67 if nleading_dimension > 1: 

68 data_shape.append(nleading_dimension) 

69 if nvertical_levels is not None: 

70 data_shape.append(nvertical_levels) 

71 

72 data_shape.append(npoints) 

73 data_shape.append(npoints) 

74 

75 return np.zeros(data_shape, dtype=np.float32)

78def generate_metadata( 

79 mandatory_attributes: dict, 

80 name: str = "air_pressure_at_sea_level", 

81 units: Optional[str] = None, 

82 time_period: Optional[int] = None, 

83 ensemble_members: int = 8, 

84 leading_dimension: Optional[List[float]] = None, 

85 cube_type: str = "variable", 

86 spp__relative_to_threshold: str = "greater_than", 

87 npoints: int = 71, 

88 **kwargs: Any, 

89) -> Cube: 

90 """Generate a cube with metadata only. 

91 

92 Args: 

93 mandatory_attributes: 

94 Specifies the values of the mandatory attributes, title, institution and 

95 source. 

96 name: 

97 Output variable name, or if creating a probability cube the name of the 

98 underlying variable to which the probability field applies. 

99 units: 

100 Output variable units, or if creating a probability cube the units of the 

101 underlying variable / threshold. 

102 time_period: 

103 The period in minutes between the time bounds. This is used to calculate 

104 the lower time bound. If unset the diagnostic will be instantaneous, i.e. 

105 without time bounds. 

106 ensemble_members: 

107 Number of ensemble members. Default 8, unless percentile or probability set 

108 to True. 

109 leading_dimension: 

110 List of realizations, percentiles or thresholds. 

111 cube_type: 

112 The type of cube to be generated. Permitted values are "variable", 

113 "percentile" or "probability". 

114 spp__relative_to_threshold: 

115 Value of the attribute "spp__relative_to_threshold" which is required for 

116 IMPROVER probability cubes. 

117 npoints: 

118 Number of points along each of the y and x spatial axes. 

119 **kwargs: 

120 Additional keyword arguments to pass to the required cube setup function. 

121 

122 Raises: 

123 ValueError: 

124 If any options are not supported 

125 KeyError: 

126 If mandatory_attributes does not contain all the required keys 

127 

128 Returns: 

129 Output of set_up_variable_cube(), set_up_percentile_cube() or 

130 set_up_probability_cube() 

131 """ 

132 if cube_type not in CUBE_TYPES: 

133 raise ValueError( 

134 ( 

135 "Cube type {} not supported. " 

136 'Specify one of "variable", "percentile" or "probability".' 

137 ).format(cube_type) 

138 ) 

139 

140 if "spatial_grid" in kwargs and kwargs["spatial_grid"] not in ( 

141 "latlon", 

142 "equalarea", 

143 ): 

144 raise ValueError( 

145 "Spatial grid {} not supported. Specify either latlon or equalarea.".format( 

146 kwargs["spatial_grid"] 

147 ) 

148 ) 

149 

150 if ( 

151 "domain_corner" in kwargs 

152 and kwargs["domain_corner"] is not None 

153 and len(kwargs["domain_corner"]) != 2 

154 ): 

155 raise ValueError("Domain corner must be a list or tuple of length 2.") 

156 

157 if units is None: 

158 units = _get_units(name) 

159 

160 # If time_period specified, create time bounds using time as upper bound 

161 if time_period is not None: 

162 if "time" not in kwargs: 

163 kwargs["time"] = DEFAULT_TIME 

164 

165 time_bounds = _create_time_bounds(kwargs["time"], time_period) 

166 kwargs["time_bounds"] = time_bounds 

167 

168 # If grid_spacing not specified, use default for requested spatial grid 

169 for spacing_axis in ["x_grid_spacing", "y_grid_spacing"]: 

170 if spacing_axis not in kwargs or kwargs[spacing_axis] is None: 

171 if "spatial_grid" not in kwargs: 

172 kwargs["spatial_grid"] = DEFAULT_SPATIAL_GRID 

173 

174 kwargs[spacing_axis] = DEFAULT_GRID_SPACING[kwargs["spatial_grid"]] 

175 

176 # Create ndimensional array of zeros 

177 if "vertical_levels" not in kwargs: 

178 kwargs["vertical_levels"] = None 

179 

180 data = _create_data_array( 

181 ensemble_members, leading_dimension, npoints, kwargs["vertical_levels"] 

182 ) 

183 missing_mandatory_attributes = MANDATORY_ATTRIBUTES - mandatory_attributes.keys() 

184 if missing_mandatory_attributes: 

185 raise KeyError( 

186 f"No values for these mandatory attributes: {missing_mandatory_attributes}" 

187 ) 

188 if "attributes" in kwargs: 

189 kwargs["attributes"] = kwargs["attributes"].copy() 

190 else: 

191 kwargs["attributes"] = {} 

192 kwargs["attributes"].update(mandatory_attributes) 

193 

194 # Set up requested cube 

195 if cube_type == "percentile": 

196 metadata_cube = set_up_percentile_cube( 

197 data, percentiles=leading_dimension, name=name, units=units, **kwargs 

198 ) 

199 elif cube_type == "probability": 

200 metadata_cube = set_up_probability_cube( 

201 data, 

202 leading_dimension, 

203 variable_name=name, 

204 threshold_units=units, 

205 spp__relative_to_threshold=spp__relative_to_threshold, 

206 **kwargs, 

207 ) 

208 else: 

209 metadata_cube = set_up_variable_cube( 

210 data, name=name, units=units, realizations=leading_dimension, **kwargs 

211 ) 

212 

213 metadata_cube = squeeze(metadata_cube) 

214 

215 return metadata_cube
