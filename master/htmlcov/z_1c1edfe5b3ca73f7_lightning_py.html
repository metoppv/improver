28class LightningFromCapePrecip(PostProcessingPlugin): 

29 """ 

30 Apply latitude-dependent thresholds to CAPE and precipitation rate to derive a 

31 presence-of-lightning cube. 

32 

33 Lightning is based on the presence of both CAPE and precipitation rate with 

34 thresholds varying linearly between 

35 

36 +---------------------------+------------+---------------------+ 

37 | | CAPE | Precipitation rate | 

38 | | (J kg-1) | (mm h-1) | 

39 +===========================+============+=====================+ 

40 | Mid-latitudes | 350 | 1 | 

41 | (above 50 degrees N or S) | | | 

42 +---------------------------+------------+---------------------+ 

43 | Tropics | 500 | 4 | 

44 | (below 10 degrees N or S) | | | 

45 +---------------------------+------------+---------------------+ 

46 

47 """

50 """ 

51 Initialise the plugin with the model_id_attr. 

52 

53 Args: 

54 model_id_attr: 

55 The name of the dataset attribute to be used to identify the source 

56 model when blending data from different models. 

57 """ 

58 self._model_id_attr = model_id_attr

60 @staticmethod 

61 def _get_inputs(cubes: CubeList) -> Tuple[Cube, Cube]: 

62 """ 

63 Separates CAPE and precipitation rate cubes and checks that the following 

64 match: forecast_reference_time, spatial coords, time-bound interval and 

65 that CAPE time is at the lower bound of precipitation rate time. 

66 The precipitation rate data must represent a period of 1 or 3 hours. 

67 """ 

68 cape = cubes.extract( 

69 iris.Constraint( 

70 cube_func=lambda cube: "atmosphere_convective_available_potential_energy" 

71 in cube.name() 

72 ) 

73 ) 

74 if cape: 

75 cape = cape.merge_cube() 

76 else: 

77 raise ValueError( 

78 f"No cube named atmosphere_convective_available_potential_energy found " 

79 f"in {cubes}" 

80 ) 

81 precip = cubes.extract( 

82 iris.Constraint( 

83 cube_func=lambda cube: "precipitation_rate_max" in cube.name() 

84 ) 

85 ) 

86 if precip: 

87 precip = precip.merge_cube() 

88 else: 

89 raise ValueError(f"No cube named precipitation_rate_max found in {cubes}") 

90 (cape_time,) = list(cape.coord("time").cells()) 

91 (precip_time,) = list(precip.coord("time").cells()) 

92 if cape_time.point != precip_time.bound[0]: 

93 raise ValueError( 

94 f"CAPE cube time ({cape_time.point}) should be valid at the " 

95 f"precipitation_rate_max cube lower bound ({precip_time.bound[0]})." 

96 ) 

97 if np.diff(precip_time.bound) not in [timedelta(hours=1), timedelta(hours=3)]: 

98 raise ValueError( 

99 f"Precipitation_rate_max cube time window must be one or three hours, " 

100 f"not {np.diff(precip_time.bound)}." 

101 ) 

102 if cape.coord("forecast_reference_time") != precip.coord( 

103 "forecast_reference_time" 

104 ): 

105 raise ValueError( 

106 "Supplied cubes must have the same forecast reference times" 

107 ) 

108 if not spatial_coords_match([cape, precip]): 

109 raise ValueError("Supplied cubes do not have the same spatial coordinates") 

110 return cape, precip

112 def process(self, *cubes: Union[Cube, CubeList]) -> Cube: 

113 """ 

114 From the supplied CAPE and precipitation-rate cubes, calculate a probability 

115 of lightning cube. 

116 

117 Args: 

118 cubes: 

119 Cubes of CAPE and Precipitation rate. 

120 

121 Returns: 

122 Cube of lightning data 

123 

124 Raises: 

125 ValueError: 

126 If one of the cubes is not found or doesn't match the other 

127 """ 

128 cubes = as_cubelist(*cubes) 

129 cape, precip = self._get_inputs(cubes) 

130 

131 cape_true = LatitudeDependentThreshold( 

132 lambda lat: latitude_to_threshold(lat, midlatitude=350.0, tropics=500.0), 

133 threshold_units="J kg-1", 

134 comparison_operator=">", 

135 )(cape) 

136 

137 precip_true = LatitudeDependentThreshold( 

138 lambda lat: latitude_to_threshold(lat, midlatitude=1.0, tropics=4.0), 

139 threshold_units="mm h-1", 

140 comparison_operator=">", 

141 )(precip) 

142 

143 data = cape_true.data * precip_true.data 

144 

145 cube = create_new_diagnostic_cube( 

146 name="probability_of_number_of_lightning_flashes_per_unit_area_above_threshold", 

147 units="1", 

148 template_cube=precip, 

149 data=data.astype(FLOAT_DTYPE), 

150 mandatory_attributes=generate_mandatory_attributes( 

151 cubes, model_id_attr=self._model_id_attr 

152 ), 

153 ) 

154 

155 coord = DimCoord( 

156 np.array([0], dtype=FLOAT_DTYPE), 

157 units="m-2", 

158 long_name="number_of_lightning_flashes_per_unit_area", 

159 var_name="threshold", 

160 attributes={"spp__relative_to_threshold": "greater_than"}, 

161 ) 

162 cube.add_aux_coord(coord) 

163 

164 cell_method = CellMethod( 

165 "sum", 

166 coords="time", 

167 comments="of number_of_lightning_flashes_per_unit_area", 

168 ) 

169 cube.add_cell_method(cell_method) 

170 

171 return cube

174def latitude_to_threshold( 

175 latitude: np.ndarray, midlatitude: float, tropics: float 

176) -> np.ndarray: 

177 """ 

178 Rescale a latitude range into a range of threshold values suitable for 

179 thresholding a different diagnostic. This is based on the value provided 

180 for that diagnostic at midlatitude (more than 50 degrees from the equator) 

181 and in the tropics (closer than 10 degrees from the equator). Varies 

182 linearly in between. 

183 

184 Args: 

185 latitude: 

186 An array of latitude points (e.g. cube.coord("latitude").points) 

187 midlatitude: 

188 The threshold value to return above 50N or below 50S. 

189 tropics: 

190 The threshold value to return below 10N or above 10S. 

191 

192 Returns: 

193 An array of thresholds, one for each latitude point 

194 """ 

195 return np.where( 

196 latitude > 0, 

197 rescale(latitude, (50.0, 10), (midlatitude, tropics), clip=True), 

198 rescale(latitude, (-50.0, -10), (midlatitude, tropics), clip=True), 

199 )

202class LightningMultivariateProbability_USAF2024(PostProcessingPlugin): 

203 """ 

204 The algorithm outputs the probability of at least one lightning strike within 

205 20 km of the location in a three hour period. The probabilities are calculated 

206 for each individual ensemble member, and then they are averaged for the ensemble 

207 probability forecast. 

208 

209 Inputs: 

210 Convective Available Potential Energy (CAPE in J/kg. Most Unstable CAPE is ideal and was used 

211 to determine the regression statistics, but surface-based CAPE is supported and yields similar 

212 results but will not forecast elevated convection.) 

213 Lifted Index (liftind in K) 

214 Precipitable Water (pwat in kg m-2 or mm. This is used as mm in the regression equations) 

215 Convective Inhibition (CIN in J/kg) 

216 3-hour Accumulated Precipitation (apcp in kg m-2 or millimetres) 

217 Outputs: 

218 20 km lightning probability over the valid time of the accumulated precipitation 

219 

220 

221 Description of the algorithm: 

222 

223 Regression equation when CAPE and APCP are greater than zero: 

224 lprob= cape*APCP 

225 lprob=0.13*ln(lprob+0.7)+0.05 

226 

227 If APCP is very low, a separate regression equation is used to predict lightning probability. 

228 The definition of “very low” is raised slightly when PWAT values are high. This is because 

229 the model often produces showery precipitation that doesn’t access the actual instability in 

230 very moist environments: 

231 lprob_noprecip=CAPE/(CIN+100.0) 

232 lprob_noprecip=0.025*ln(lprob_noprecip+0.31)+0.03 

233 APCP=APCP-(PWAT/1000) 

234 

235 IF APCP is less than 0.01 inches THEN lprob=lprob_noprecip 

236 If there is no CAPE but the atmosphere is “close” to unstable, lightning does sometimes occur, 

237 especially when heavy precipitation may have stabilized the atmosphere in the model. Unstable 

238 values of lifted index are positive here: 

239 LIFTIDX=LIFTIDX+4.0 

240 IF LIFTIDX is less than 0 K THEN LIFTIDX=0.0 

241 IF CAPE<=0 J/kg THEN lprob=0.2*(LIFTIDX*APCP)^0.5 

242 

243 Finally, the probability of lightning is reduced when there is not much PW, because graupel 

244 cannot form and start the whole charging process. Therefore we reduce the probability: 

245 IF PWAT is less than 20 mm THEN lprob=lprob*(PWAT/20.0) 

246 

247 Limit final probabilities to 95% as that is as skillful as the regression equations could get: 

248 IF lprob is greater than 0.95 THEN lprob=0.95 

249 """ 

250

253 """Extract the relevant cube based on the cube name. 

254 

255 Args: 

256 cubes: Cubes from which to extract required input. 

257 cube_name: Name of cube to extract. 

258 

259 Returns: 

260 The extracted cube. 

261 """

264 except iris.exceptions.ConstraintMismatchError: 

265 raise ValueError(f"No cube named {cube_name} found in {cubes}") 

266 return cube

268 def _get_inputs(self, cubes: CubeList) -> Tuple[Cube, Cube]: 

269 """ 

270 Separates CAPE, LI, PW, CIN, and APCP cubes and checks that the following 

271 match: forecast_reference_time, spatial coords, time-bound interval and 

272 that CAPE time is at the lower bound of precipitation accumulation time. 

273 """ 

274 

275 output_cubes = iris.cube.CubeList() 

276 input_names = { 

277 "atmosphere_specific_convective_available_potential_energy": ["J kg-1"], 

278 "temperature_difference_between_ambient_air_and_air_lifted_adiabatically": [ 

279 "K" 

280 ], 

281 "precipitable_water": ["kg m-2", "mm"], 

282 "atmosphere_specific_convective_inhibition": ["J kg-1"], 

283 "precipitation_amount": ["kg m-2", "mm"], 

284 } 

285 

286 for input_name, units in input_names.items(): 

287 output_cubes.append(self._extract_input(cubes, input_name)) 

288 if output_cubes[-1].units not in units: 

289 expected_unit_string = " or ".join(map(str, units)) 

290 received_unit_string = str(output_cubes[-1].units) 

291 raise ValueError( 

292 f"The {output_cubes[-1].name()} units are incorrect, expected " 

293 f"units as {expected_unit_string} but received {received_unit_string})." 

294 ) 

295 

296 cape, liftidx, pwat, cin, apcp = output_cubes 

297 

298 (apcp_time,) = list(apcp.coord("time").cells()) 

299 for cube in [cape, liftidx, pwat, cin]: 

300 (cube_time,) = list(cube.coord("time").cells()) 

301 if cube_time.point != apcp_time.bound[0]: 

302 raise ValueError( 

303 f"The {cube.name()} time point ({cube_time.point}) should be valid at the " 

304 f"precipitation_accumulation cube lower bound ({apcp_time.bound[0]})." 

305 ) 

306 if not np.diff(apcp_time.bound) == timedelta(hours=3): 

307 raise ValueError( 

308 f"Precipitation_accumulation cube time window must be three hours, " 

309 f"not {np.diff(apcp_time.bound)}." 

310 ) 

311 # Following time/space checks should be made for all diagnostics 

312 for cube in [cape, liftidx, pwat, cin]: 

313 if cube.coord("forecast_reference_time") != apcp.coord( 

314 "forecast_reference_time" 

315 ): 

316 raise ValueError( 

317 f"{cube.name()} and {apcp.name()} do not have the same forecast reference time" 

318 ) 

319 for cube in [cape, liftidx, pwat, cin]: 

320 if not spatial_coords_match([cube, apcp]): 

321 raise ValueError( 

322 f"{cube.name()} and {apcp.name()} do not have the same spatial " 

323 f"coordinates" 

324 ) 

325 

326 return cape, liftidx, pwat, cin, apcp

328 def process(self, cubes: CubeList, model_id_attr: str = None) -> Cube: 

329 """ 

330 From the supplied CAPE, Lifted Index, Precipitable Water, CIN, and 3-hr Accumulated 

331 Precipitation cubes, calculate a probability of lightning cube. 

332 

333 Args: 

334 cubes: 

335 Cubes of CAPE, Lifted Index, Precipitable Water, CIN, and 3-hr Accumulated 

336 Precipitation cubes. 

337 model_id_attr: 

338 The name of the dataset attribute to be used to identify the source 

339 model when blending data from different models. 

340 

341 Returns: 

342 Cube of lightning data 

343 

344 Raises: 

345 ValueError: 

346 If one of the cubes is not found, doesn't match the others, or has incorrect units 

347 """ 

348 cape, liftidx, pwat, cin, apcp = self._get_inputs(cubes) 

349 

350 # Regression equations require math on cubes with incompatible units, so strip data 

351 template = apcp.copy() 

352 cape = cape.data 

353 liftidx = liftidx.data 

354 pwat = pwat.data 

355 cin = ( 

356 -1 * cin.data 

357 ) # use inverse convention where CIN is positive instead of negative 

358 apcp = apcp.data / 1000 * 39.3701 # convert kg m-2 to inches 

359 

360 # Regression equation when CAPE and APCP are greater than zero: 

361 lprob = cape * apcp 

362 lprob = 0.13 * np.log(lprob + 0.7) + 0.05 

363 

364 # If APCP is very low, a separate regression equation is used to predict lightning 

365 # probability. The definition of “very low” is raised slightly when PWAT values are 

366 # high. This is because the model often produces showery precipitation that doesn’t 

367 # access the actual instability in very moist environments: 

368 lprob_noprecip = cape / (cin + 100.0) 

369 lprob_noprecip = 0.025 * np.log(lprob_noprecip + 0.31) + 0.03 

370 apcp_temp = apcp - (pwat / 1000) 

371 

372 lprob[np.where(apcp_temp < 0.01)] = lprob_noprecip[np.where(apcp_temp < 0.01)] 

373 

374 # If there is no CAPE but the atmosphere is “close” to unstable, lightning does sometimes 

375 # occur, especially when heavy precipitation may have stabilized the atmosphere in the 

376 # model. Unstable values of lifted index are positive here: 

377 liftidx = liftidx + 4.0 

378 

379 liftidx[liftidx < 0] = 0 

380 

381 lprob[np.where(cape <= 0)] = ( 

382 0.2 * (liftidx[np.where(cape <= 0)] * apcp[np.where(cape <= 0)]) ** 0.5 

383 ) 

384 

385 # Finally, the probability of lightning is reduced when there is not much PWAT, since 

386 # graupel cannot form required for the charging process. The probability is reduced as: 

387 low_pwat = np.where(pwat < 20) 

388 lprob[low_pwat] = lprob[low_pwat] * (pwat[low_pwat] / 20.0) 

389 

390 # Limit probabilities to 95% as that is as skillful as the regression equations could get: 

391 lprob[lprob > 0.95] = 0.95 

392 

393 cube = create_new_diagnostic_cube( 

394 name=( 

395 "probability_of_number_of_lightning_flashes_per_unit_area_in_vicinity_above_" 

396 "threshold" 

397 ), 

398 units="1", 

399 template_cube=template, 

400 data=lprob.astype(FLOAT_DTYPE), 

401 mandatory_attributes=generate_mandatory_attributes( 

402 cubes, model_id_attr=model_id_attr 

403 ), 

404 ) 

405 

406 # Add auxiliary coordinate for threshold of any lightning occurring (above 0 flashes) 

407 coord = DimCoord( 

408 np.array([0], dtype=FLOAT_DTYPE), 

409 units="m-2", 

410 long_name="number_of_lightning_flashes_per_unit_area", 

411 var_name="threshold", 

412 attributes={"spp__relative_to_threshold": "greater_than"}, 

413 ) 

414 cube.add_aux_coord(coord) 

415 

416 # Add auxiliary coordinate for vicinity of 20 km 

417 vic_coord = create_vicinity_coord(20000) 

418 cube.add_aux_coord(vic_coord) 

419 

420 return cube
