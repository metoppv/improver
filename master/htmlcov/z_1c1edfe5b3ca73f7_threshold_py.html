37class Threshold(PostProcessingPlugin): 

38 """Apply a threshold truth criterion to a cube. 

39 

40 Calculate the threshold truth values based on a linear membership function 

41 around the threshold values provided. A cube will be returned with a new 

42 threshold dimension coordinate. 

43 

44 Can operate on multiple time sequences within a cube. 

45 """ 

46 

47 def __init__( 

48 self, 

49 threshold_values: Optional[Union[float, List[float]]] = None, 

50 threshold_config: Optional[Dict[str, Union[List[float], str]]] = None, 

51 fuzzy_factor: Optional[float] = None, 

52 threshold_units: Optional[str] = None, 

53 comparison_operator: str = ">", 

54 collapse_coord: Optional[Union[str, List[str]]] = None, 

55 collapse_cell_methods: Optional[dict] = None, 

56 vicinity: Optional[Union[float, List[float]]] = None, 

57 fill_masked: Optional[float] = None, 

58 ) -> None: 

59 """ 

60 Set up for processing an in-or-out of threshold field, including the 

61 generation of fuzzy_bounds which are required to threshold an input 

62 cube (through self.process(cube)). If fuzzy_factor is not None, fuzzy 

63 bounds are calculated using the threshold value in the units in which 

64 it is provided. 

65 

66 The usage of fuzzy_factor is exemplified as follows: 

67 

68 For a 6 mm/hr threshold with a 0.75 fuzzy factor, a range of 25% 

69 around this threshold (between (6*0.75=) 4.5 and (6*(2-0.75)=) 7.5) 

70 would be generated. The probabilities of exceeding values within this 

71 range are scaled linearly, so that 4.5 mm/hr yields a thresholded value 

72 of 0 and 7.5 mm/hr yields a thresholded value of 1. Therefore, in this 

73 case, the thresholded exceedance probabilities between 4.5 mm/hr and 

74 7.5 mm/hr would follow the pattern: 

75 

76 :: 

77 

78 Data value | Probability 

79 ------------|------------- 

80 4.5 | 0 

81 5.0 | 0.167 

82 5.5 | 0.333 

83 6.0 | 0.5 

84 6.5 | 0.667 

85 7.0 | 0.833 

86 7.5 | 1.0 

87 

88 Args: 

89 threshold_values: 

90 Threshold value or values (e.g. 270K, 300K) to use when calculating 

91 the probability of the input relative to the threshold value(s). 

92 The units of these values, e.g. K in the example can be defined 

93 using the threshold_units argument or are otherwise assumed to 

94 match the units of the diagnostic being thresholded. 

95 threshold_values and and threshold_config are mutually exclusive 

96 arguments, defining both will lead to an exception. 

97 threshold_config (dict): 

98 Threshold configuration containing threshold values and 

99 (optionally) fuzzy bounds. Best used in combination with 

100 'threshold_units'. It should contain a dictionary of strings that 

101 can be interpreted as floats with the structure: 

102 "THRESHOLD_VALUE": [LOWER_BOUND, UPPER_BOUND] 

103 e.g: {"280.0": [278.0, 282.0], "290.0": [288.0, 292.0]}, 

104 or with structure "THRESHOLD_VALUE": "None" (no fuzzy bounds). 

105 Repeated thresholds with different bounds are ignored; only the 

106 last duplicate will be used. 

107 threshold_values and and threshold_config are mutually exclusive 

108 arguments, defining both will lead to an exception. 

109 fuzzy_factor: 

110 Optional: specifies lower bound for fuzzy membership value when 

111 multiplied by each threshold. Upper bound is equivalent linear 

112 distance above threshold. 

113 threshold_units: 

114 Units of the threshold values. If not provided the units are 

115 assumed to be the same as those of the input cube. 

116 comparison_operator: 

117 Indicates the comparison_operator to use with the threshold. 

118 e.g. 'ge' or '>=' to evaluate 'data >= threshold' or '<' to 

119 evaluate 'data < threshold'. When using fuzzy thresholds, there 

120 is no difference between < and <= or > and >=. 

121 Valid choices: > >= < <= gt ge lt le. 

122 collapse_coord: 

123 A coordinate or list of coordinates over which an average is 

124 calculated by collapsing these coordinates. The only supported 

125 options are combinations of "realization" and "percentile" with 

126 "time"; realization and percentile cannot both be collapsed. 

127 If "percentile" is included the percentile coordinate will be 

128 rebadged as a realization coordinate prior to collapse. 

129 The percentile coordinate needs to be evenly spaced around the 

130 50th percentile to allow successful conversion from percentiles 

131 to realizations and subsequent collapsing over the realization 

132 coordinate. 

133 collapse_cell_methods: 

134 An optional dictionary that describes cell methods to apply in 

135 relation to the collapsed coordinates. By default the threshold 

136 method does not return a cell method for collapsed coordinates. 

137 The dictionary should take the form: {"coord_name": "method"} 

138 with an entry for every coordinate for which a method is 

139 desired. 

140 fill_masked: 

141 If provided all masked points in cube will be replaced with the 

142 provided value. 

143 vicinity: 

144 A list of vicinity radii to use to calculate maximum in vicinity 

145 thresholded values. This must be done prior to realization 

146 collapse. 

147 

148 Raises: 

149 ValueError: If threshold_config and threshold_values are both set 

150 ValueError: If neither threshold_config or threshold_values are set 

151 ValueError: If both fuzzy_factor and bounds within the threshold_config are set. 

152 ValueError: If the fuzzy_factor is not strictly between 0 and 1. 

153 ValueError: If using a fuzzy factor with a threshold of 0.0. 

154 ValueError: Can only collapse over one or a combination of realization 

155 and percentile coordinates with the time coordinate. 

156 ValueError: If collapse_coord contains both percentile and realization. 

157 """ 

158 if threshold_config and threshold_values: 

159 raise ValueError( 

160 "threshold_config and threshold_values are mutually exclusive " 

161 "arguments - please provide one or the other, not both" 

162 ) 

163 if threshold_config is None and threshold_values is None: 

164 raise ValueError( 

165 "One of threshold_config or threshold_values must be provided." 

166 ) 

167 thresholds, fuzzy_bounds = self._set_thresholds( 

168 threshold_values, threshold_config 

169 ) 

170 self.thresholds = [thresholds] if np.isscalar(thresholds) else thresholds 

171 self.threshold_units = ( 

172 None if threshold_units is None else Unit(threshold_units) 

173 ) 

174 self.threshold_coord_name = None 

175 

176 # read fuzzy factor or set to 1 (basic thresholding) 

177 fuzzy_factor_loc = 1.0 

178 if fuzzy_factor is not None: 

179 if fuzzy_bounds is not None: 

180 raise ValueError( 

181 "Invalid combination of keywords. Cannot specify " 

182 "both a fuzzy_factor and use a threshold_config that " 

183 "specifies bounds." 

184 ) 

185 if not 0 < fuzzy_factor < 1: 

186 raise ValueError( 

187 "Invalid fuzzy_factor: must be >0 and <1: {}".format(fuzzy_factor) 

188 ) 

189 if 0 in self.thresholds: 

190 raise ValueError( 

191 "Invalid threshold with fuzzy factor: cannot use a " 

192 "multiplicative fuzzy factor with threshold == 0, use " 

193 "the threshold_config approach instead." 

194 ) 

195 fuzzy_factor_loc = fuzzy_factor 

196 

197 if fuzzy_bounds is None: 

198 self.fuzzy_bounds = self._generate_fuzzy_bounds(fuzzy_factor_loc) 

199 else: 

200 self.fuzzy_bounds = ( 

201 [fuzzy_bounds] if isinstance(fuzzy_bounds, tuple) else fuzzy_bounds 

202 ) 

203 self._check_fuzzy_bounds() 

204 

205 self.original_units = None 

206 self.comparison_operator_dict = comparison_operator_dict() 

207 self.comparison_operator_string = comparison_operator 

208 self._decode_comparison_operator_string() 

209 

210 if collapse_coord and not isinstance(collapse_coord, list): 

211 collapse_coord = [collapse_coord] 

212 

213 if collapse_coord and not set(collapse_coord).issubset( 

214 ["percentile", "realization", "time"] 

215 ): 

216 raise ValueError( 

217 "Can only collapse over one or a combination of realization, " 

218 "percentile, and time coordinates." 

219 ) 

220 if ( 

221 collapse_coord 

222 and "percentile" in collapse_coord 

223 and "realization" in collapse_coord 

224 ): 

225 raise ValueError( 

226 "Cannot collapse over both percentile and realization coordinates." 

227 ) 

228 self.collapse_coord = collapse_coord 

229 

230 if collapse_cell_methods: 

231 if not self.collapse_coord: 

232 raise ValueError( 

233 "Cannot apply cell methods without collapsing a coordinate." 

234 ) 

235 if not set(collapse_cell_methods.keys()).issubset(self.collapse_coord): 

236 raise ValueError( 

237 "Cell methods can only be defined for coordinates that " 

238 "are being collapsed." 

239 ) 

240 

241 self.collapse_cell_methods = collapse_cell_methods 

242 

243 self.vicinity = None 

244 if vicinity: 

245 if isinstance(vicinity, Iterable): 

246 self.vicinity = [float(x) for x in vicinity] 

247 else: 

248 self.vicinity = [float(vicinity)] 

249 

250 if fill_masked is not None: 

251 fill_masked = float(fill_masked) 

252 

253 self.fill_masked = fill_masked

255 @staticmethod 

256 def _set_thresholds( 

257 threshold_values: Optional[Union[float, List[float]]], 

258 threshold_config: Optional[dict], 

259 ) -> Tuple[List[float], Optional[List[float]]]: 

260 """ 

261 Interprets a threshold_config dictionary if provided, or ensures that 

262 a list of thresholds has suitable precision. 

263 

264 Args: 

265 threshold_values: 

266 A list of threshold values or a single threshold value. 

267 threshold_config: 

268 A dictionary defining threshold values and optionally upper 

269 and lower bounds for those values to apply fuzzy thresholding. 

270 

271 Returns: 

272 A tuple containing: 

273 thresholds: 

274 A list of threshold values as float64 type. 

275 fuzzy_bounds: 

276 A list of tuples that define the upper and lower bounds associated 

277 with each threshold value, these also as float64 type. If these 

278 are not set, None is returned instead. 

279 """ 

280 if threshold_config: 

281 thresholds = [] 

282 fuzzy_bounds = [] 

283 for key in threshold_config.keys(): 

284 # Ensure thresholds are float64 to avoid rounding errors during 

285 # possible unit conversion. 

286 thresholds.append(float(key)) 

287 # If the first threshold has no bounds, fuzzy_bounds is 

288 # set to None and subsequent bounds checks are skipped 

289 if threshold_config[key] == "None": 

290 fuzzy_bounds = None 

291 continue 

292 fuzzy_bounds.append(tuple(threshold_config[key])) 

293 else: 

294 # Ensure thresholds are float64 to avoid rounding errors during possible 

295 # unit conversion. 

296 if isinstance(threshold_values, numbers.Number): 

297 threshold_values = [threshold_values] 

298 thresholds = [float(x) for x in threshold_values] 

299 fuzzy_bounds = None 

300 return thresholds, fuzzy_bounds

302 def _generate_fuzzy_bounds( 

303 self, fuzzy_factor_loc: float 

304 ) -> List[Tuple[float, float]]: 

305 """Construct fuzzy bounds from a fuzzy factor. If the fuzzy factor is 1, 

306 the fuzzy bounds match the threshold values for basic thresholding. 

307 """ 

308 fuzzy_bounds = [] 

309 for thr in self.thresholds: 

310 lower_thr = thr * fuzzy_factor_loc 

311 upper_thr = thr * (2.0 - fuzzy_factor_loc) 

312 if thr < 0: 

313 lower_thr, upper_thr = upper_thr, lower_thr 

314 fuzzy_bounds.append((lower_thr, upper_thr)) 

315 return fuzzy_bounds

317 def _check_fuzzy_bounds(self) -> None: 

318 """If fuzzy bounds have been set from the command line, check they 

319 are consistent with the required thresholds 

320 """ 

321 for thr, bounds in zip(self.thresholds, self.fuzzy_bounds): 

322 if len(bounds) != 2: 

323 raise ValueError( 

324 "Invalid bounds for one threshold: {}." 

325 " Expected 2 floats.".format(bounds) 

326 ) 

327 if bounds[0] > thr or bounds[1] < thr: 

328 bounds_msg = ( 

329 "Threshold must be within bounds: " "!( {} <= {} <= {} )".format( 

330 bounds[0], thr, bounds[1] 

331 ) 

332 ) 

333 raise ValueError(bounds_msg)

407 def _calculate_truth_value( 

408 self, cube: Cube, threshold: float, bounds: Tuple[float, float] 

409 ) -> np.ndarray: 

410 """ 

411 Compares the diagnostic values to the threshold value, converting units 

412 and applying fuzzy bounds as required. Returns the truth value. 

413 

414 Args: 

415 cube: 

416 A cube containing the diagnostic values. The cube rather than array 

417 is passed in to allow for unit conversion. 

418 threshold: 

419 A single threshold value against which to compare the diagnostic 

420 values. 

421 bounds: 

422 The fuzzy bounds used for applying fuzzy thresholding. 

423 

424 Returns: 

425 truth_value: 

426 An array of truth values given by the comparison of the diagnostic 

427 data to the threshold value. This is returned at the default float 

428 precision. 

429 """ 

430 if self.threshold_units is not None: 

431 cube.convert_units(self.threshold_units) 

432 # Ensure that the threshold value is the same datatype as the cube data to 

433 # avoid numerical precision affecting equality comparisons. 

434 threshold = np.float64(threshold).astype(cube.dtype) 

435 # if upper and lower bounds are equal, set a deterministic 0/1 

436 # probability based on exceedance of the threshold 

437 if bounds[0] == bounds[1]: 

438 truth_value = self.comparison_operator.function(cube.data, threshold) 

439 # otherwise, scale exceedance probabilities linearly between 0/1 

440 # at the min/max fuzzy bounds and 0.5 at the threshold value 

441 else: 

442 truth_value = np.where( 

443 cube.data < threshold, 

444 rescale( 

445 cube.data, 

446 data_range=(bounds[0], threshold), 

447 scale_range=(0.0, 0.5), 

448 clip=True, 

449 ), 

450 rescale( 

451 cube.data, 

452 data_range=(threshold, bounds[1]), 

453 scale_range=(0.5, 1.0), 

454 clip=True, 

455 ), 

456 ) 

457 # Rescaling exposes masked points so use the rescaled data only in 

458 # the unmasked locations, set masked locations to 0 and reapply 

459 # masking. 

460 if np.ma.is_masked(cube.data): 

461 truth_value = np.where(cube.data.mask, 0, truth_value) 

462 truth_value = np.ma.masked_array(truth_value, mask=cube.data.mask) 

463 

464 # if requirement is for probabilities less_than or 

465 # less_than_or_equal_to the threshold (rather than 

466 # greater_than or greater_than_or_equal_to), invert 

467 # the exceedance probability 

468 if "less_than" in self.comparison_operator.spp_string: 

469 truth_value = 1.0 - truth_value 

470 

471 return truth_value.astype(FLOAT_DTYPE)

473 def _vicinity_processing( 

474 self, 

475 thresholded_cube: Cube, 

476 truth_value: np.ndarray, 

477 unmasked: np.ndarray, 

478 landmask: np.ndarray, 

479 grid_point_radii: List[int], 

480 index: int, 

481 ): 

482 """ 

483 Apply max in vicinity processing to the thresholded values. The 

484 resulting modified threshold values are changed in place in 

485 thresholded_cube. 

486 

487 Args: 

488 thresholded_cube: 

489 The cube into which the resulting values are added. 

490 truth_value: 

491 An array of thresholded values prior to the application of 

492 vicinity processing. 

493 unmasked: 

494 Array identifying unmasked data points that should be updated. 

495 landmask: 

496 A binary grid of the same size as truth_value that 

497 differentiates between land and sea points to allow the 

498 different surface types to be processed independently. 

499 grid_point_radii: 

500 The vicinity radius to apply expressed as a number of grid 

501 cells. 

502 index: 

503 Index corresponding to the threshold coordinate to identify 

504 which array we are summing the contribution into. 

505 """ 

506 for ivic, vicinity in enumerate(grid_point_radii): 

507 if truth_value.ndim > 2: 

508 maxes = np.zeros(truth_value.shape, dtype=FLOAT_DTYPE) 

509 for yxindex in np.ndindex(*truth_value.shape[:-2]): 

510 slice_max = maximum_within_vicinity( 

511 truth_value[yxindex + (slice(None), slice(None))], 

512 vicinity, 

513 landmask, 

514 ) 

515 maxes[yxindex] = slice_max 

516 else: 

517 maxes = maximum_within_vicinity(truth_value, vicinity, landmask) 

518 thresholded_cube.data[ivic][index][unmasked] += maxes[unmasked]

520 def _create_threshold_cube(self, cube: Cube) -> Cube: 

521 """ 

522 Create a cube with suitable metadata and zeroed data array for 

523 storing the thresholded diagnostic data. 

524 

525 Args: 

526 cube: 

527 A template cube from which to take the data shape. 

528 

529 Returns: 

530 thresholded_cube: 

531 A cube of a suitable form for storing and describing 

532 the thresholded data. 

533 """ 

534 template = cube.copy(data=np.zeros(cube.shape, dtype=FLOAT_DTYPE)) 

535 

536 if self.threshold_units is not None: 

537 template.units = self.threshold_units 

538 

539 thresholded_cube = iris.cube.CubeList() 

540 for threshold in self.thresholds: 

541 thresholded = template.copy() 

542 self._add_threshold_coord(thresholded, threshold) 

543 thresholded.units = 1 

544 thresholded_cube.append(thresholded) 

545 thresholded_cube = thresholded_cube.merge_cube() 

546 

547 # Promote the threshold coordinate to be dimensional if it is not already. 

548 if not thresholded_cube.coord_dims(self.threshold_coord_name): 

549 thresholded_cube = iris.util.new_axis( 

550 thresholded_cube, self.threshold_coord_name 

551 ) 

552 

553 if self.vicinity is not None: 

554 vicinity_coord = create_vicinity_coord(self.vicinity) 

555 vicinity_expanded = iris.cube.CubeList() 

556 for vicinity_coord_slice in vicinity_coord: 

557 thresholded_copy = thresholded_cube.copy() 

558 thresholded_copy.add_aux_coord(vicinity_coord_slice) 

559 vicinity_expanded.append(thresholded_copy) 

560 del thresholded_copy 

561 thresholded_cube = vicinity_expanded.merge_cube() 

562 if not thresholded_cube.coords(vicinity_coord.name(), dim_coords=True): 

563 thresholded_cube = iris.util.new_axis( 

564 thresholded_cube, vicinity_coord.name() 

565 ) 

566 

567 # Ensure the threshold cube has suitable metadata for a probabilistic output. 

568 self._update_metadata(thresholded_cube) 

569 return thresholded_cube

571 def process(self, input_cube: Cube, landmask: Cube = None) -> Cube: 

572 """Convert each point to a truth value based on provided threshold 

573 values. The truth value may or may not be fuzzy depending upon if 

574 fuzzy_bounds are supplied. If the plugin has a "threshold_units" 

575 member, this is used to convert both thresholds and fuzzy bounds into 

576 the units of the input cube. 

577 

578 Args: 

579 input_cube: 

580 Cube to threshold. The code is dimension-agnostic. 

581 landmask: 

582 Cube containing a landmask. Used with vicinity processing 

583 only. 

584 

585 Returns: 

586 Cube after a threshold has been applied, possibly using fuzzy 

587 bounds, and / or with vicinity processing applied to return a 

588 maximum in vicinity value. The data within this cube will 

589 contain values between 0 and 1 to indicate whether a given 

590 threshold has been exceeded or not. 

591 

592 The cube meta-data will contain: 

593 * Input_cube name prepended with 

594 probability_of_X_above(or below)_threshold (where X is 

595 the diagnostic under consideration) 

596 * The cube name will be suffixed with _in_vicinity if 

597 vicinity processing has been applied. 

598 * Threshold dimension coordinate with same units as input_cube 

599 * Threshold attribute ("greater_than", 

600 "greater_than_or_equal_to", "less_than", or 

601 less_than_or_equal_to" depending on the operator) 

602 * Cube units set to (1). 

603 

604 Raises: 

605 ValueError: if collapse coordinates are not found in input_cube. 

606 ValueError: Cannot apply land-mask cube without in-vicinity processing. 

607 ValueError: if a np.nan value is detected within the input cube. 

608 """ 

609 input_cube_crds = [crd.name() for crd in input_cube.coords()] 

610 if self.collapse_coord and not set(self.collapse_coord).issubset( 

611 input_cube_crds 

612 ): 

613 raise ValueError( 

614 "Cannot collapse over coordinates not present in the input_cube." 

615 f"collapse_coords: {self.collapse_coord}\n" 

616 f"input_cube_coords: {input_cube_crds}" 

617 ) 

618 

619 if self.vicinity is None and landmask is not None: 

620 raise ValueError( 

621 "Cannot apply land-mask cube without in-vicinity processing" 

622 ) 

623 

624 if self.fill_masked is not None: 

625 input_cube.data = np.ma.filled(input_cube.data, self.fill_masked) 

626 

627 if self.collapse_coord and "percentile" in self.collapse_coord: 

628 input_cube = RebadgePercentilesAsRealizations()(input_cube) 

629 self.collapse_coord.remove("percentile") 

630 self.collapse_coord.append("realization") 

631 

632 self.original_units = input_cube.units 

633 self.threshold_coord_name = input_cube.name() 

634 # Retain only the landmask array as bools. 

635 if landmask is not None: 

636 landmask = landmask.data.astype(bool) 

637 if self.vicinity is not None: 

638 grid_point_radii = [ 

639 distance_to_number_of_grid_cells(input_cube, radius) 

640 for radius in self.vicinity 

641 ] 

642 

643 # Slice over collapse coords if required and create an empty threshold 

644 # cube to store the resulting thresholded data. 

645 if self.collapse_coord is not None: 

646 input_slices = input_cube.slices_over(self.collapse_coord) 

647 thresholded_cube = self._create_threshold_cube( 

648 next(input_cube.slices_over(self.collapse_coord)) 

649 ) 

650 else: 

651 input_slices = [input_cube] 

652 thresholded_cube = self._create_threshold_cube(input_cube) 

653 

654 # Create a zeroed array for storing contributions (i.e. number of 

655 # unmasked realization values contributing to calculation). 

656 contribution_total = np.zeros( 

657 next(thresholded_cube.slices_over(self.threshold_coord_name)).shape, 

658 dtype=int, 

659 ) 

660 

661 for cube in input_slices: 

662 # Tests performed on each slice rather than whole cube to avoid 

663 # realising all of the data. 

664 if np.isnan(cube.data).any(): 

665 raise ValueError("Error: NaN detected in input cube data") 

666 

667 if np.ma.is_masked(cube.data): 

668 mask = cube.data.mask 

669 unmasked = ~mask 

670 else: 

671 unmasked = np.ones(cube.shape, dtype=bool) 

672 

673 # All unmasked points contribute 1 to the numerator for calculating 

674 # a coordinate collapsed truth value. Note that if input_slices 

675 # above includes the realization/time coordinate (i.e. we are not 

676 # collapsing that coordinate) then contribution_total will include 

677 # this extra dimension and our denominator will be 1 at all unmasked 

678 # points. 

679 contribution_total += unmasked 

680 

681 for index, (threshold, bounds) in enumerate( 

682 zip(self.thresholds, self.fuzzy_bounds) 

683 ): 

684 truth_value = self._calculate_truth_value(cube, threshold, bounds) 

685 if self.vicinity is not None: 

686 self._vicinity_processing( 

687 thresholded_cube, 

688 truth_value, 

689 unmasked, 

690 landmask, 

691 grid_point_radii, 

692 index, 

693 ) 

694 else: 

695 thresholded_cube.data[index][unmasked] += truth_value[unmasked] 

696 

697 # Any x-y position for which there are no valid contributions must be 

698 # a masked point in every realization, so we can use this array to 

699 # modify only unmasked points and reapply a mask to the final result. 

700 valid = contribution_total.astype(bool) 

701 

702 # Slice over the array to avoid ballooning the memory required for the 

703 # denominators through broadcasting. 

704 enforce_coordinate_ordering(thresholded_cube, self.threshold_coord_name) 

705 for i, dslice in enumerate(thresholded_cube.data): 

706 result = np.divide(dslice[valid], contribution_total[valid]) 

707 thresholded_cube.data[i, valid] = result 

708 

709 if not valid.all(): 

710 thresholded_cube.data = np.ma.masked_array( 

711 thresholded_cube.data, 

712 mask=np.broadcast_to(~valid, thresholded_cube.shape), 

713 ) 

714 

715 # Revert threshold coordinate units to those of the input cube. 

716 thresholded_cube.coord(self.threshold_coord_name).convert_units( 

717 self.original_units 

718 ) 

719 

720 # If we've collapsed a time coordinate we need to add back suitable 

721 # time coordinates that capture the range of the times collapsed. 

722 if self.collapse_coord and "time" in self.collapse_coord: 

723 for crd in ["time", "forecast_period"]: 

724 time_crd = input_cube.coord(crd).copy().collapsed() 

725 thresholded_cube.remove_coord(crd) 

726 thresholded_cube.add_aux_coord(time_crd) 

727 

728 enforce_time_point_standard(thresholded_cube) 

729 

730 # Squeeze any single value dimension coordinates to make them scalar. 

731 thresholded_cube = iris.util.squeeze(thresholded_cube) 

732 

733 if self.collapse_coord is not None and "realization" in self.collapse_coord: 

734 thresholded_cube.remove_coord("realization") 

735 

736 # Re-cast to 32bit now that any unit conversion has already taken place. 

737 thresholded_cube.coord(var_name="threshold").points = thresholded_cube.coord( 

738 var_name="threshold" 

739 ).points.astype(FLOAT_DTYPE) 

740 

741 if self.vicinity is not None: 

742 rename_vicinity_cube(thresholded_cube) 

743 

744 enforce_coordinate_ordering( 

745 thresholded_cube, 

746 [ 

747 "time", 

748 "realization", 

749 "percentile", 

750 self.threshold_coord_name, 

751 "radius_of_vicinity", 

752 ], 

753 ) 

754 

755 return thresholded_cube

758class LatitudeDependentThreshold(Threshold): 

759 """Apply a latitude-dependent threshold truth criterion to a cube. 

760 

761 Calculates the threshold truth values based on the threshold function provided. 

762 A cube will be returned with a new threshold dimension auxillary coordinate on 

763 the latitude axis. 

764 

765 Can operate on multiple time sequences within a cube. 

766 """ 

767 

768 def __init__( 

769 self, 

770 threshold_function: callable, 

771 threshold_units: Optional[str] = None, 

772 comparison_operator: str = ">", 

773 ) -> None: 

774 """ 

775 Sets up latitude-dependent threshold class 

776 

777 Args: 

778 threshold_function: 

779 A function which takes a latitude value (in degrees) and returns 

780 the desired threshold. 

781 threshold_units: 

782 Units of the threshold values. If not provided the units are 

783 assumed to be the same as those of the input cube. 

784 comparison_operator: 

785 Indicates the comparison_operator to use with the threshold. 

786 e.g. 'ge' or '>=' to evaluate 'data >= threshold' or '<' to 

787 evaluate 'data < threshold'. When using fuzzy thresholds, there 

788 is no difference between < and <= or > and >=. 

789 Valid choices: > >= < <= gt ge lt le. 

790 """ 

791 super().__init__( 

792 threshold_values=[1], 

793 threshold_units=threshold_units, 

794 comparison_operator=comparison_operator, 

795 ) 

796 if not callable(threshold_function): 

797 raise TypeError("Threshold must be callable") 

798 self.threshold_function = threshold_function 

799 

800 def _add_latitude_threshold_coord(self, cube: Cube, threshold: np.ndarray) -> None: 

801 """ 

802 Add a 1D threshold-type coordinate with correct name and units 

803 to a 2D slice containing thresholded data. 

804 Assumes latitude coordinate is always the penultimate one (which standardise 

805 will have enforced) 

806 

807 Args: 

808 cube: 

809 Cube containing thresholded data (1s and 0s) 

810 threshold: 

811 Values at which the data has been thresholded (matches cube's y-axis) 

812 """ 

813 coord = iris.coords.AuxCoord(threshold.astype(FLOAT_DTYPE), units=cube.units) 

814 coord.rename(self.threshold_coord_name) 

815 coord.var_name = "threshold" 

816 cube.add_aux_coord(coord, data_dims=len(cube.shape) - 2) 

817 

818 def process(self, input_cube: Cube) -> Cube: 

819 """Convert each point to a truth value based on provided threshold, 

820 fuzzy bound, and vicinity values. If the plugin has a "threshold_units" 

821 member, this is used to convert a copy of the input_cube into 

822 the units specified. 

823 

824 Args: 

825 input_cube: 

826 Cube to threshold. Must have a latitude coordinate. 

827 

828 Returns: 

829 Cube after a threshold has been applied. The data within this 

830 cube will contain values between 0 and 1 to indicate whether 

831 a given threshold has been exceeded or not. 

832 

833 The cube meta-data will contain: 

834 * Input_cube name prepended with 

835 probability_of_X_above(or below)_threshold (where X is 

836 the diagnostic under consideration) 

837 * Threshold dimension coordinate with same units as input_cube 

838 * Threshold attribute ("greater_than", 

839 "greater_than_or_equal_to", "less_than", or 

840 less_than_or_equal_to" depending on the operator) 

841 * Cube units set to (1). 

842 

843 Raises: 

844 ValueError: if a np.nan value is detected within the input cube. 

845 """ 

846 if np.isnan(input_cube.data).any(): 

847 raise ValueError("Error: NaN detected in input cube data") 

848 

849 self.threshold_coord_name = input_cube.name() 

850 

851 cube = input_cube.copy() 

852 if self.threshold_units is not None: 

853 cube.convert_units(self.threshold_units) 

854 

855 cube.coord("latitude").convert_units("degrees") 

856 threshold_variant = cube.coord("latitude").points 

857 threshold_over_latitude = np.array(self.threshold_function(threshold_variant)) 

858 

859 # Add a scalar axis for the longitude axis so that numpy's array- 

860 # broadcasting knows what we want to do 

861 truth_value = self.comparison_operator.function( 

862 cube.data, np.expand_dims(threshold_over_latitude, 1) 

863 ) 

864 

865 truth_value = truth_value.astype(FLOAT_DTYPE) 

866 

867 if np.ma.is_masked(cube.data): 

868 # update unmasked points only 

869 cube.data[~input_cube.data.mask] = truth_value[~input_cube.data.mask] 

870 else: 

871 cube.data = truth_value 

872 

873 self._add_latitude_threshold_coord(cube, threshold_over_latitude) 

874 cube.coord(var_name="threshold").convert_units(input_cube.units) 

875 

876 self._update_metadata(cube) 

877 enforce_coordinate_ordering(cube, ["time", "realization", "percentile"]) 

878 

879 return cube
