5"""CLI to calculate a gradient between two vertical levels.""" 

6 

7from improver import cli 

8 

9 

10@cli.clizefy 

11@cli.with_output 

12def process(*cubes: cli.inputcubelist): 

13 """ 

14 Calculate the gradient between two vertical levels. The gradient is calculated as the 

15 difference between the input cubes divided by the difference in height. 

16 

17 Input cubes can be provided at height or pressure levels. If the cubes are provided 

18 at a pressure level, the height above sea level is extracted from height_of_pressure_levels 

19 cube. If exactly one of the cubes are provided at height levels this is assumed to be a 

20 height above ground level and the height above sea level is calculated by adding the height 

21 of the orography to the "height" coordinate of the cube. If both cubes have height coordinates 

22 no additional cubes are required. 

23 It is possible for one cube to be defined at height levels and the other at pressure levels. 

24 

25 Args: 

26 cubes (iris.cube.CubeList): 

27 Contains two cubes of a diagnostic at two different vertical levels. The cubes must 

28 either have a height coordinate or a pressure coordinate. If only one cube is defined at 

29 height levels, an orography cube must also be provided. If either cube is defined at 

30 pressure levels, a geopotential_height cube must also be provided. 

31 

32 Returns: 

33 iris.cube.Cube: 

34 A single cube containing the gradient between the two height levels. This cube will be 

35 renamed to "gradient_of" the cube name and will have a units attribute of the input 

36 cube units per metre. 

37 

38 """ 

39 from improver.utilities.gradient_between_vertical_levels import ( 

40 GradientBetweenVerticalLevels, 

41 ) 

42 

43 return GradientBetweenVerticalLevels()(*cubes)
