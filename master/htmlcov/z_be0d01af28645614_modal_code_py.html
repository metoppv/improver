147class ModalCategory(BaseModalCategory): 

148 """Plugin that returns the modal category over the period spanned by the 

149 input data. In cases of a tie in the mode values, scipy returns the smaller 

150 value. The opposite is desirable in this case as the significance / 

151 importance of the weather code categories generally increases with the value. To 

152 achieve this the categories are subtracted from an arbitrarily larger 

153 number prior to calculating the mode, and this operation is reversed before the 

154 final output is returned. 

155 

156 If there are many different categories for a single point over the time 

157 spanned by the input cubes it may be that the returned mode is not robust. 

158 Given the preference to return more significant categories explained above, 

159 a 12 hour period with 12 different categories, one of which is severe, will 

160 return that severe category to describe the whole period. This is likely not a 

161 good representation. In these cases grouping is used to try and select 

162 a suitable category (e.g. a rain shower if the codes include a mix of 

163 rain showers and dynamic rain) by providing a more robust mode. The lowest 

164 number (least significant) member of the group is returned as the code. 

165 Use of the least significant member reflects the lower certainty in the 

166 forecasts. 

167 

168 Where there are different categories available for night and day, the 

169 modal code returned is always a day code, regardless of the times 

170 covered by the input files. 

171 """ 

172 

173 def __init__( 

174 self, 

175 decision_tree: Dict, 

176 model_id_attr: Optional[str] = None, 

177 record_run_attr: Optional[str] = None, 

178 ): 

179 """ 

180 Set up plugin and create an aggregator instance for reuse 

181 

182 Args: 

183 decision_tree: 

184 The decision tree used to generate the categories and which contains the 

185 mapping of day and night categories and of category groupings. 

186 model_id_attr: 

187 Name of attribute recording source models that should be 

188 inherited by the output cube. The source models are expected as 

189 a space-separated string. 

190 record_run_attr: 

191 Name of attribute used to record models and cycles used in 

192 constructing the categories. 

193 """ 

194 super().__init__(decision_tree) 

195 self.aggregator_instance = Aggregator("mode", self.mode_aggregator) 

196 self.model_id_attr = model_id_attr 

197 self.record_run_attr = record_run_attr 

198 

199 codes = [ 

200 node["leaf"] 

201 for node in self.decision_tree.values() 

202 if "leaf" in node.keys() 

203 ] 

204 self.code_max = max(codes) + 1 

205 self.unset_code_indicator = min(codes) - 100 

206 self.min_dtype = np.result_type( 

207 np.min_scalar_type(self.unset_code_indicator), 

208 np.min_scalar_type(self.code_max), 

209 ) 

210 self.code_groups = self._code_groups() 

211 

212 def _code_groups(self) -> Dict: 

213 """Determines code groupings from the decision tree""" 

214 groups = {} 

215 for key, node in self.decision_tree.items(): 

216 if "group" not in node.keys(): 

217 continue 

218 groups[node["group"]] = groups.get(node["group"], []) + [node["leaf"]] 

219 return groups

221 def _group_codes(self, modal: Cube, cube: Cube): 

222 """In instances where the mode returned is not significant, i.e. the 

223 category chosen occurs infrequently in the period, the codes can be 

224 grouped to yield a more definitive period code. Given the uncertainty, 

225 the least significant category (lowest number in a group that is 

226 found in the data) is used to replace the other data values that belong 

227 to that group prior to recalculating the modal code. 

228 

229 The modal cube is modified in place. 

230 

231 Args: 

232 modal: 

233 The modal categorical cube which contains UNSET_CODE_INDICATOR 

234 values that need to be replaced with a more definitive period 

235 code. 

236 cube: 

237 The original input data. Data relating to unset points will be 

238 grouped and the mode recalculated.""" 

239 

240 undecided_points = np.argwhere(modal.data == self.unset_code_indicator) 

241 

242 for point in undecided_points: 

243 data = cube.data[(..., *point)].copy() 

244 

245 for _, codes in self.code_groups.items(): 

246 default_code = sorted([code for code in data if code in codes]) 

247 if default_code: 

248 data[np.isin(data, codes)] = default_code[0] 

249 mode_result, counts = stats.mode(self.code_max - data) 

250 modal.data[tuple(point)] = self.code_max - mode_result

252 def mode_aggregator(self, data: ndarray, axis: int) -> ndarray: 

253 """An aggregator for use with iris to calculate the mode along the 

254 specified axis. If the modal value selected comprises less than 30% 

255 of data along the dimension being collapsed, the value is set to the 

256 UNSET_CODE_INDICATOR to indicate that the uncertainty was too high to 

257 return a mode. 

258 

259 Args: 

260 data: 

261 The data for which a mode is to be calculated. 

262 axis: 

263 The axis / dimension over which to calculate the mode. 

264 

265 Returns: 

266 The data array collapsed over axis, containing the calculated modes. 

267 """ 

268 # Iris aggregators support indexing from the end of the array. 

269 if axis < 0: 

270 axis += data.ndim 

271 # Aggregation coordinate is moved to the -1 position in initialisation; 

272 # move this back to the leading coordinate 

273 data = np.moveaxis(data, [axis], [0]) 

274 data = data.astype(self.min_dtype) 

275 minimum_significant_count = 0.3 * data.shape[0] 

276 mode_result, counts = stats.mode(self.code_max - data, axis=0) 

277 mode_result[counts < minimum_significant_count] = ( 

278 self.code_max - self.unset_code_indicator 

279 ) 

280 return (self.code_max - np.squeeze(mode_result)).astype(data.dtype)

282 @staticmethod 

283 def _set_blended_times(cube: Cube) -> None: 

284 """Updates time coordinates so that time point is at the end of the time bounds, 

285 blend_time and forecast_reference_time (if present) are set to the end of the 

286 bound period and bounds are removed, and forecast_period is updated to match.""" 

287 cube.coord("time").points = cube.coord("time").bounds[0][-1] 

288 

289 for coord_name in ["blend_time", "forecast_reference_time"]: 

290 if coord_name in [c.name() for c in cube.coords()]: 

291 coord = cube.coord(coord_name) 

292 if coord.has_bounds(): 

293 coord = coord.copy(coord.bounds[0][-1]) 

294 cube.replace_coord(coord) 

295 

296 if "forecast_period" in [c.name() for c in cube.coords()]: 

297 calculated_coord = forecast_period_coord( 

298 cube, force_lead_time_calculation=True 

299 ) 

300 new_coord = cube.coord("forecast_period").copy( 

301 points=calculated_coord.points, bounds=calculated_coord.bounds 

302 ) 

303 cube.replace_coord(new_coord)

305 def process(self, cubes: CubeList) -> Cube: 

306 """Calculate the modal categorical code, with handling for edge cases. 

307 

308 Args: 

309 cubes: 

310 A list of categorical cubes at different times. A modal 

311 code will be calculated over the time coordinate to return 

312 the most common code, which is taken to be the best 

313 representation of the whole period. 

314 

315 Returns: 

316 A single categorical cube with time bounds that span those of 

317 the input categorical cubes. 

318 """ 

319 cube = self._prepare_input_cubes( 

320 cubes, self.record_run_attr, self.model_id_attr 

321 ) 

322 # dtype may be escalated in modal calculation to accommodate the 

323 # unset_code_indicator and code_max. The modes must exist within 

324 # the range of the original data type, so we keep this to reset 

325 # the data type at the end of the process. 

326 original_dtype = cube.data.dtype 

327 

328 self._unify_day_and_night(cube) 

329 

330 # Handle case in which a single time is provided. 

331 if len(cube.coord("time").points) == 1: 

332 result = cube 

333 else: 

334 result = cube.collapsed("time", self.aggregator_instance) 

335 self._set_blended_times(result) 

336 

337 result = self._prepare_result_cube( 

338 cube, cubes, result, self.record_run_attr, self.model_id_attr 

339 ) 

340 

341 # Handle any unset points where it was hard to determine a suitable mode 

342 if (result.data == self.unset_code_indicator).any(): 

343 self._group_codes(result, cube) 

344 

345 result.data = result.data.astype(original_dtype) 

346 return result

349class ModalFromGroupings(BaseModalCategory): 

350 """Plugin that creates a modal weather code over a period using a grouping 

351 approach. Firstly, a wet and dry grouping is computed. Secondly, for the 

352 wet grouping, groupings can be provided, such as, "extreme", "frozen" and "liquid", 

353 so that wet weather codes can be grouped further. These groupings can be controlled 

354 as follows. Firstly, a day weighting functionality is provided so that daytime 

355 hours can be weighted more heavily. A wet bias can also be provided, so that 

356 wet codes are given a larger weight as they are considered more impactful. 

357 A second categorisation is then available for the wet codes. This is useful when 

358 e.g. a period is represented using a variety of frozen precipitation weather codes, 

359 so that a frozen precipitation weather code can be diagnosed as an appropriate 

360 summary. The ignore intensity option allows light and heavy weather types to be 

361 considered together when ascertaining the most common weather type. The final 

362 daily symbol will be the most common of the light and heavy input codes of 

363 the chosen type. 

364 

365 The ordering of the codes within the category dictionaries guides which 

366 category is selected in the event of the tie with preference given to the lowest 

367 index. Incrementing the codes within the category dictionaries from most significant 

368 code to least significant code helps to ensure that the most significant code is 

369 returned in the event of a tie, if desired. 

370 

371 Where there are different categories available for night and day, the 

372 modal code returned is always a day code, regardless of the times 

373 covered by the input files. 

374 

375 If a location is to return a dry code after consideration of the various 

376 weightings, the wet codes for that location are converted into the best 

377 matching dry cloud code and these are included in determining the resulting 

378 dry code. The wet bias has no impact on the weight of these converted wet 

379 codes, but the day weighting still applies. 

380 """ 

381

508 def _emphasise_day_period(self, cube: Cube) -> Cube: 

509 """A day weighting can be set which biases the forecasts towards the hours of 

510 e.g. 6am-6pm. This is achieved by counting the number of input times available 

511 e.g. hourly and taking those that are 18 times from the end up to those 

512 that are 6 from the end and duplicating these symbols by the integer weighting. 

513 This approach is taken to accommodate different timezones without the need for 

514 any timezone awareness. Inputs are always provided from midnight to midnight, or 

515 ending at midnight if a partial day is provided. The middle of the set of input 

516 times therefore corresponds to the local middle of the day. The count back 

517 from the end of the period is done to accommodate partial periods (same day 

518 updates). The index counted backwards is clipped to 0, meaning if there are 

519 only 12 files being passed in (because we're around midday when we perform 

520 the update), the first index will be 0, rather than -6, and only symbols from 

521 6 periods will be multiplied up by the day_weighting. 

522 

523 Metadata is not used to select the day period as the times recorded 

524 within the cubes are all UTC, rather than local time, so the local day period 

525 can not be identified. The time and forecast_period coordinates are 

526 incremented by the the minimum arbitrary amount (1 second) to ensure 

527 non-duplicate coordinates. 

528 

529 Args: 

530 cube: Weather codes cube. 

531 

532 Returns: 

533 Cube with more times during the daytime period, so that daytime hours 

534 are emphasised, depending upon the day_weighting chosen. 

535 """ 

536 day_cubes = iris.cube.CubeList() 

537 for cube_slice in cube.slices_over("time"): 

538 cube_slice = self._promote_time_coords(cube_slice, cube) 

539 day_cubes.append(cube_slice) 

540 

541 time_coord = cube.coord("time").copy() 

542 time_coord.convert_units("hours since 1970-01-01 00:00:00") 

543 interval = time_coord.bounds[0][1] - time_coord.bounds[0][0] 

544 

545 n_times = len(day_cubes) 

546 start_file = np.clip( 

547 (n_times - int((self.DAY_LENGTH - self.day_start) / interval)), 0, None 

548 ) 

549 end_file = np.clip( 

550 (n_times - int((self.DAY_LENGTH - self.day_end) / interval)), 0, None 

551 ) 

552 

553 for increment in range(1, self.day_weighting): 

554 for day_slice in day_cubes[start_file:end_file]: 

555 day_slice = day_slice.copy() 

556 for coord in ["time", "forecast_period"]: 

557 if len(cube.coord_dims(coord)) > 0: 

558 day_slice.coord(coord).points = ( 

559 day_slice.coord(coord).points + increment 

560 ) 

561 bounds = day_slice.coord(coord).bounds.copy() 

562 bounds[0] = bounds[0] + increment 

563 day_slice.coord(coord).bounds = bounds 

564 day_cubes.append(day_slice) 

565 return day_cubes.concatenate_cube()

567 def _find_wet_indices(self, cube: Cube, time_axis: int) -> np.ndarray: 

568 """Identify the points at which a wet weather code should be selected. 

569 This can include a wet bias if supplied. 

570 

571 Args: 

572 cube: Weather codes cube. 

573 time_axis: The time coordinate dimension. 

574 

575 Returns: 

576 Boolean array that is true if the weather codes are wet or False otherwise. 

577 """ 

578 # Find indices corresponding to dry weather codes inclusive of a wet bias. 

579 dry_counts = np.sum( 

580 np.isin(cube.data, self.broad_categories["dry"]), axis=time_axis 

581 ) 

582 wet_counts = np.sum( 

583 np.isin(cube.data, self.broad_categories["wet"]), axis=time_axis 

584 ) 

585 return (self.wet_bias * wet_counts) >= dry_counts

587 @staticmethod 

588 def counts_per_category(data: np.ndarray, bin_max: int) -> np.ndarray: 

589 """Implemented following https://stackoverflow.com/questions/46256279/bin-elements-per-row-vectorized-2d-bincount-for-numpy/46256361#46256361 # noqa: E501 

590 Use np.bincount to count the number of occurrences within each category, so that 

591 the most common occurrence can then be found. 

592 

593 Args: 

594 data: Array where occurrences of each possible integer value between 0 

595 and data.max() will be counted. 

596 bin_max: Integer defining the number of categories expected. 

597 

598 Returns: 

599 An array of counts for the occurrence of each category within each row. 

600 """ 

601 n_cat = bin_max + 1 

602 a_offs = data + np.arange(data.shape[0])[:, None] * n_cat 

603 # Use compressed() to avoid counting masked values. 

604 return np.bincount( 

605 a_offs.ravel().compressed(), minlength=data.shape[0] * n_cat 

606 ).reshape(-1, n_cat)

608 def _find_most_significant_dry_code( 

609 self, cube: Cube, result: Cube, dry_indices: np.ndarray

611 """Find the most significant dry weather code at each point. 

612 

613 Args: 

614 cube: Weather code cube. 

615 result: Cube into which to put the result. 

616 dry_indices: Boolean, which is true if the weather codes at that point, 

617 are dry. 

618 

619 Returns: 

620 Cube where points that are dry are filled with the most common dry 

621 code present at that point. If there is a tie, the most significant dry 

622 weather code is used, assuming higher values for the weather code indicates 

623 more significant weather. 

624 """ 

625 data = cube.data.copy() 

626 data = np.ma.masked_where( 

627 ~np.isin(cube.data.copy(), self.broad_categories["dry"]), data 

628 ) 

629 

630 # If required, reshape a 3D dataset e.g. a gridded dataset to 2D. 

631 # 2D datasets e.g. site datasets will be unchanged. 

632 reshaped_data = data.T.reshape(-1, len(cube.coord("time").points), order="F") 

633 bins = [i for v in self.broad_categories.values() for i in v] 

634 bin_max = np.amax(bins) 

635 

636 counts = self.counts_per_category(reshaped_data, bin_max=bin_max) 

637 

638 # Flip counts with the aim that the counts for the higher index weather codes 

639 # are on the left, and will therefore be selected by argmax. 

640 counts = np.fliplr(counts) 

641 

642 # Reshape the 2D counts back to a 3D dataset, if required. If the input 

643 # data is 2D, this reshape will have no impact. 

644 reshaped_counts = counts.reshape(*data.shape[1:], -1) 

645 

646 result.data[dry_indices] = (bin_max - np.argmax(reshaped_counts, axis=-1))[ 

647 dry_indices 

648 ] 

649 return result

651 def _find_intensity_indices(self, cube: Cube) -> np.ndarray: 

652 """Find which points / sites include any weather code predictions that fall 

653 within the intensity categories. 

654 

655 Args: 

656 cube: Weather code cube. 

657 

658 Returns: 

659 Boolean that is True if any weather code from the intensity categories 

660 are found at a given point, otherwise False. 

661 """ 

662 values = np.isin( 

663 cube.data, [x for v in self.intensity_categories.values() for x in v] 

664 ) 

665 return values.astype(bool)

667 def _get_most_likely_following_grouping( 

668 self, 

669 cube: Cube, 

670 result: Cube, 

671 categories: Dict, 

672 required_indices: np.ndarray, 

673 time_axis: int, 

674 categorise_using_modal: bool, 

675 ): 

676 """Determine the most common category and subcategory using a dictionary 

677 defining the categorisation. The category could be a group of weather codes 

678 representing frozen precipitation, where the subcategory would be the individual 

679 weather codes, so that this method is able to identify the most likely weather 

680 code within the most likely weather code category. If a category or subcategory 

681 is tied, then the first as defined within the categories dictionary is taken. 

682 As the categories and subcategories within the dictionary are expected to be 

683 in descending priority order, this will ensure that the highest priority item 

684 is chosen in the event of a tie. 

685 

686 Args: 

687 cube: Weather codes cube. 

688 result: Cube in which to put the result. 

689 categories: Dictionary defining the categories (keys) and 

690 subcategories (values). The most likely category and then the most 

691 likely value for the subcategory is put into the result cube. 

692 required_indices: Boolean indicating which indices within the result cube 

693 to fill. 

694 time_axis: The time coordinate dimension. 

695 categorise_using_modal: Boolean defining whether the top level 

696 categorisation should use the input cube or the processed result time. 

697 The input cube will have a time dimension, whereas the result cube 

698 will not have a time dimension. 

699 

700 Returns: 

701 A result cube containing the most appropriate weather code following 

702 categorisation. 

703 """ 

704 # Identify the most likely weather code within each of the subcategory. 

705 category_counter = [] 

706 most_likely_subcategory = {} 

707 for key in categories.keys(): 

708 if categorise_using_modal: 

709 category_counter.append(np.isin(result.data, categories[key])) 

710 else: 

711 category_counter.append( 

712 np.sum(np.isin(cube.data, categories[key]), axis=time_axis) 

713 ) 

714 

715 subcategory_counter = [] 

716 for value in categories[key]: 

717 subcategory_counter.append(np.sum(cube.data == value, axis=time_axis)) 

718 most_likely_subcategory[key] = np.array(categories[key])[ 

719 np.argmax(subcategory_counter, axis=time_axis) 

720 ] 

721 

722 # Identify which category is most likely. 

723 most_likely_category = np.argmax(category_counter, axis=time_axis) 

724 # Use the most likely subcategory from the most likely category. 

725 for index, key in enumerate(categories.keys()): 

726 category_index = np.logical_and( 

727 required_indices, most_likely_category == index 

728 ) 

729 result.data[category_index] = most_likely_subcategory[key][category_index] 

730 return result

733 def _set_blended_times(cube: Cube, result: Cube) -> None: 

734 """Updates time coordinates so that time point is at the end of the time bounds, 

735 blend_time and forecast_reference_time (if present) are set to the end of the 

736 bound period and bounds are removed, and forecast_period is updated to match. 

737 The result cube is modified in-place. 

738 

739 Args: 

740 cube: Cube containing metadata on the temporal coordinates that will be 

741 used to add the relevant metadata to the result cube. 

742 result: Cube containing the computed modal weather code. This cube will be 

743 updated in-place. 

744 """ 

745 result.coord("time").points = cube.coord("time").points[-1] 

746 result.coord("time").bounds = [ 

747 cube.coord("time").bounds[0][0], 

748 cube.coord("time").bounds[-1][-1], 

749 ] 

750 

751 for coord_name in ["blend_time", "forecast_reference_time"]: 

752 if coord_name in [c.name() for c in result.coords()] and coord_name in [ 

753 c.name() for c in cube.coords() 

754 ]: 

755 coord = cube.coord(coord_name) 

756 coord = coord.copy(coord.points[-1]) 

757 result.replace_coord(coord) 

758 

759 if "forecast_period" in [c.name() for c in result.coords()]: 

760 calculated_coord = forecast_period_coord( 

761 result, force_lead_time_calculation=True 

762 ) 

763 new_coord = result.coord("forecast_period").copy( 

764 points=calculated_coord.points, bounds=calculated_coord.bounds 

765 ) 

766 result.replace_coord(new_coord)

768 def _get_dry_equivalents( 

769 self, cube: Cube, dry_indices: np.ndarray, time_axis: int 

770 ) -> Cube: 

771 """ 

772 Returns a cube with only dry codes in which all wet codes have 

773 been replaced by their nearest dry cloud equivalent. For example a 

774 shower code is replaced with a partly cloudy code, a light rain code 

775 is replaced with a cloud code, and a heavy rain code is replaced with 

776 an overcast cloud code. 

777 

778 Args: 

779 cube: Weather code cube. 

780 dry_indices: An array of bools which are true for locations where 

781 the summary weather code will be dry. 

782 time_axis: The time coordinate dimension. 

783 

784 Returns: 

785 cube: Wet codes converted to their dry equivalent for those points 

786 that will receive a dry summary weather code. 

787 """ 

788 dry_cube = cube.copy() 

789 for value, target in self.dry_map.items(): 

790 dry_cube.data = np.where(cube.data == value, target, dry_cube.data) 

791 

792 # Note that np.rollaxis returns a new view of the input data. As such 

793 # changes to `original` here are also changes to cube.data. 

794 original = np.rollaxis(cube.data, time_axis) 

795 dried = np.rollaxis(dry_cube.data, time_axis) 

796 original[..., dry_indices] = dried[..., dry_indices] 

797 

798 return cube

800 def process(self, cubes: CubeList) -> Cube: 

801 """Calculate the modal categorical code by grouping weather codes. 

802 

803 Args: 

804 cubes: 

805 A list of categorical cubes at different times. A modal 

806 code will be calculated over the time coordinate to return 

807 the most common code, which is taken to be the best 

808 representation of the whole period. 

809 

810 Returns: 

811 A single categorical cube with time bounds that span those of 

812 the input categorical cubes. 

813 """ 

814 cube = self._prepare_input_cubes( 

815 cubes, self.record_run_attr, self.model_id_attr 

816 ) 

817 

818 self._unify_day_and_night(cube) 

819 

820 if len(cube.coord("time").points) == 1: 

821 result = cube 

822 else: 

823 cube = self._emphasise_day_period(cube) 

824 

825 result = cube[0].copy() 

826 (time_axis,) = cube.coord_dims("time") 

827 

828 wet_indices = self._find_wet_indices(cube, time_axis) 

829 

830 # For dry locations convert the wet codes to their equivalent dry 

831 # codes for use in determining the summary symbol. 

832 cube = self._get_dry_equivalents(cube, ~wet_indices, time_axis) 

833 

834 original_cube = cube.copy() 

835 cube = self._consolidate_intensity_categories(cube) 

836 result = self._find_most_significant_dry_code(cube, result, ~wet_indices) 

837 

838 result = self._get_most_likely_following_grouping( 

839 cube, 

840 result, 

841 self.wet_categories, 

842 wet_indices, 

843 time_axis, 

844 categorise_using_modal=False, 

845 ) 

846 

847 if self.intensity_categories: 

848 intensity_indices = self._find_intensity_indices(result) 

849 result = self._get_most_likely_following_grouping( 

850 original_cube, 

851 result, 

852 self.intensity_categories, 

853 intensity_indices, 

854 time_axis, 

855 categorise_using_modal=True, 

856 ) 

857 

858 self._set_blended_times(cube, result) 

859 

860 result = self._prepare_result_cube( 

861 cube, cubes, result, self.record_run_attr, self.model_id_attr 

862 ) 

863 

864 return result
