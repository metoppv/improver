5"""Plugin to calculate air density from virtual temperature.""" 

6 

7from typing import Union 

8 

9import iris 

10from iris.cube import Cube, CubeList 

11 

12from improver import BasePlugin 

13from improver.constants import R_DRY_AIR 

14 

15 

16class AirDensity(BasePlugin): 

17 """Calculate air density from virtual temperature.""" 

18 

19 def _get_pressure_cube(self, cubes: CubeList) -> Cube: 

20 """Extract pressure cube if present.""" 

21 for cube in cubes: 

22 if cube.standard_name == "air_pressure": 

23 return cube 

24 return None 

25 

26 def _get_temperature_cube(self, cubes: CubeList) -> Cube: 

27 """Extract virtual temperature cube.""" 

28 for cube in cubes: 

29 if cube.standard_name == "virtual_temperature": 

30 return cube 

31 raise ValueError("No virtual_temperature cube provided.") 

32 

33 def process(self, inputs: Union[Cube, CubeList]) -> Cube: 

34 """ 

35 Calculate air density from virtual temperature. 

36 

37 If an explicit pressure cube is provided the pressure is taken from here. 

38 Otherwise the virtual temperature cube needs to be on pressure levels. 

39 The cube arguments are not checked explicitly for conformant dimensions 

40 but must have the same shape. 

41 

42 Args: 

43 inputs: 

44 Either: 

45 - A single Cube (virtual temperature), or 

46 - A CubeList containing virtual temperature and optionally pressure. 

47 

48 Returns: 

49 Cube containing air density (kg m-3) on the same grid/levels as virtual temmperature 

50 """ 

51 

52 # --- Normalize input to CubeList --- 

53 if isinstance(inputs, Cube): 

54 cubes = CubeList([inputs]) 

55 else: 

56 cubes = inputs 

57 

58 # --- Get temperature cube --- 

59 Tv_cube = self._get_temperature_cube(cubes) 

60 

61 # --- Try to get explicit pressure cube --- 

62 pressure_cube = self._get_pressure_cube(cubes) 

63 

64 # --- Determine pressure source --- 

65 if pressure_cube is not None: 

66 # Use explicit pressure cube 

67 pressure = pressure_cube.copy() 

68 pressure.convert_units("Pa") 

69 

70 pressure_data = pressure.data 

71 

72 else: 

73 # Try to infer from temperature cube (pressure levels case) 

74 try: 

75 pressure_coord = Tv_cube.coord("air_pressure") 

76 except iris.exceptions.CoordinateNotFoundError: 

77 raise ValueError( 

78 "No pressure information supplied: " 

79 "provide an air_pressure cube or use a temperature cube " 

80 "on pressure levels." 

81 ) 

82 

83 pressure = pressure_coord.copy() 

84 pressure.convert_units("Pa") 

85 

86 pressure_data = iris.util.broadcast_to_shape( 

87 pressure.points, 

88 Tv_cube.shape, 

89 Tv_cube.coord_dims("air_pressure"), 

90 ) 

91 

92 # --- Ensure temperature is in Kelvin --- 

93 Tv = Tv_cube.copy() 

94 Tv.convert_units("K") 

95 

96 # --- Compute density --- 

97 density_data = pressure_data / (R_DRY_AIR * Tv.data) 

98 

99 # --- Create output cube --- 

100 density = Tv_cube.copy(data=density_data) 

101 density.rename("air_density") 

102 density.units = "kg m-3" 

103 

104 return density
