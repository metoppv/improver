6"""Script to apply coefficients for Ensemble Model Output 

7Statistics (EMOS), otherwise known as Non-homogeneous Gaussian 

8Regression (NGR).""" 

9 

10from improver import cli 

11 

12 

13@cli.clizefy 

14@cli.with_output 

15def process( 

16 *cubes: cli.inputcubelist, 

17 validity_times: cli.comma_separated_list = None, 

18 realizations_count: int = None, 

19 randomise=False, 

20 random_seed: int = None, 

21 ignore_ecc_bounds_exceedance=False, 

22 tolerate_time_mismatch=False, 

23 predictor="mean", 

24 land_sea_mask_name: str = None, 

25 percentiles: cli.comma_separated_list = None, 

26): 

27 """Applying coefficients for Ensemble Model Output Statistics. 

28 

29 Load in arguments for applying coefficients for Ensemble Model Output 

30 Statistics (EMOS), otherwise known as Non-homogeneous Gaussian 

31 Regression (NGR). The coefficients are applied to the forecast 

32 that is supplied, so as to calibrate the forecast. The calibrated 

33 forecast is written to a cube. If no coefficients are provided the input 

34 forecast is returned unchanged. 

35 

36 Args: 

37 input_cubes (iris.cube.CubeList): 

38 A list of cubes containing: 

39 - A Cube containing the forecast to be calibrated. The input format 

40 could be either realizations, probabilities or percentiles. 

41 - A cubelist containing the coefficients used for calibration or None. 

42 If none then the input, or probability template if provided, 

43 is returned unchanged. 

44 - Optionally, cubes representing static additional predictors. 

45 These static additional predictors are expected not to have a 

46 time coordinate. 

47 - Optionally, a cube containing the land-sea mask on the same domain 

48 as the forecast that is to be calibrated. Land points are 

49 specified by ones and sea points are specified by zeros. 

50 The presence of a land-sea mask will enable land-only calibration, in 

51 which sea points are returned without the application of 

52 calibration. If a land-sea mask is provided, the land_sea_mask_name 

53 must also be provided, in order to identify the land-sea mask. 

54 - Optionally, a cube containing a probability forecast that will be 

55 used as a template when generating probability output when the input 

56 format of the forecast cube is not probabilities i.e. realizations 

57 or percentiles. If no coefficients are provided and a probability 

58 template is provided, the probability template forecast will be 

59 returned as the uncalibrated probability forecast. 

60 validity_times (List[str]): 

61 Times at which the forecast must be valid. This must be provided 

62 as a four digit string (HHMM) where the first two digits represent the hour 

63 and the last two digits represent the minutes e.g. 0300 or 0315. If the 

64 forecast provided is at a different validity time then no coefficients 

65 will be applied. 

66 realizations_count (int): 

67 Option to specify the number of ensemble realizations that will be 

68 created from probabilities or percentiles when applying the EMOS 

69 coefficients. 

70 randomise (bool): 

71 Option to reorder the post-processed forecasts randomly. If not 

72 set, the ordering of the raw ensemble is used. This option is 

73 only valid when the input format is realizations. 

74 random_seed (int): 

75 Option to specify a value for the random seed for testing 

76 purposes, otherwise the default random seen behaviour is utilised. 

77 The random seed is used in the generation of the random numbers 

78 used for either the randomise option to order the input 

79 percentiles randomly, rather than use the ordering from the raw 

80 ensemble, or for splitting tied values within the raw ensemble, 

81 so that the values from the input percentiles can be ordered to 

82 match the raw ensemble. 

83 ignore_ecc_bounds_exceedance (bool): 

84 If True, where the percentiles exceed the ECC bounds range, 

85 raises a warning rather than an exception. This occurs when the 

86 current forecasts is in the form of probabilities and is 

87 converted to percentiles, as part of converting the input 

88 probabilities into realizations. 

89 tolerate_time_mismatch (bool): 

90 If True, tolerate a mismatch in validity time and forecast period 

91 for coefficients vs forecasts. Use with caution! 

92 predictor (str): 

93 String to specify the form of the predictor used to calculate 

94 the location parameter when estimating the EMOS coefficients. 

95 Currently the ensemble mean ("mean") and the ensemble 

96 realizations ("realizations") are supported as the predictors. 

97 land_sea_mask_name (str): 

98 Name of the land-sea mask cube. This must be provided if a 

99 land-sea mask is provided within the list of input cubes, in 

100 order to identify the land-sea mask. Providing the land-sea mask 

101 ensures that only land points will be calibrated. 

102 percentiles (List[float]): 

103 The set of percentiles used to create the calibrated forecast. 

104 

105 Returns: 

106 iris.cube.Cube: 

107 The calibrated forecast cube. 

108 

109 """ 

110 

111 from improver.calibration import ( 

112 split_forecasts_and_coeffs, 

113 ) 

114 from improver.calibration.emos_calibration import ApplyEMOS 

115 from improver.calibration.utilities import prepare_cube_no_calibration 

116 

117 (forecast, coefficients, additional_predictors, land_sea_mask, prob_template) = ( 

118 split_forecasts_and_coeffs(cubes, land_sea_mask_name) 

119 ) 

120 

121 uncalibrated_forecast = prepare_cube_no_calibration( 

122 forecast, 

123 coefficients, 

124 ignore_ecc_bounds_exceedance=ignore_ecc_bounds_exceedance, 

125 validity_times=validity_times, 

126 percentiles=percentiles, 

127 prob_template=prob_template, 

128 ) 

129 if uncalibrated_forecast is not None: 

130 return uncalibrated_forecast 

131 

132 calibration_plugin = ApplyEMOS(percentiles=percentiles) 

133 result = calibration_plugin( 

134 forecast, 

135 coefficients, 

136 additional_fields=additional_predictors, 

137 land_sea_mask=land_sea_mask, 

138 prob_template=prob_template, 

139 realizations_count=realizations_count, 

140 ignore_ecc_bounds=ignore_ecc_bounds_exceedance, 

141 tolerate_time_mismatch=tolerate_time_mismatch, 

142 predictor=predictor, 

143 randomise=randomise, 

144 random_seed=random_seed, 

145 ) 

146 return result
