21def load_cubelist( 

22 filepath: Union[str, List[str]], 

23 constraints: Optional[Union[Constraint, str]] = None, 

24 no_lazy_load: bool = False, 

25) -> CubeList: 

26 """Load cubes from filepath(s) into a cubelist. Strips off all 

27 var names except for "threshold"-type coordinates, where this is different 

28 from the standard or long name. 

29 

30 Args: 

31 filepath: 

32 Filepath(s) that will be loaded. 

33 constraints: 

34 Constraint to be applied when loading from the input filepath. 

35 This can be in the form of an iris.Constraint or could be a string 

36 that is intended to match the name of the cube. 

37 The default is None. 

38 no_lazy_load: 

39 If True, bypass cube deferred (lazy) loading and load the whole 

40 cube into memory. This can increase performance at the cost of 

41 memory. If False (default) then lazy load. 

42 

43 Returns: 

44 CubeList that has been created from the input filepath given the 

45 constraints provided. 

46 """ 

47 # Remove legacy metadata prefix cube if present 

48 constraints = ( 

49 iris.Constraint(cube_func=lambda cube: cube.long_name != "prefixes") 

50 & constraints 

51 ) 

52 

53 # Load each file individually to avoid partial merging (not used 

54 # iris.load_raw() due to issues with time representation) 

55 if isinstance(filepath, str): 

56 cubes = iris.load(filepath, constraints=constraints) 

57 else: 

58 cubes = iris.cube.CubeList([]) 

59 for item in filepath: 

60 cubes.extend(iris.load(item, constraints=constraints)) 

61 

62 if not cubes: 

63 message = "No cubes found using constraints {}".format(constraints) 

64 raise ValueError(message) 

65 

66 # Remove var_name from cubes and coordinates (except where needed to 

67 # describe probabilistic data) 

68 cubes = strip_var_names(cubes) 

69 

70 for cube in cubes: 

71 # Remove metadata attributes pointing to legacy prefix cube 

72 cube.attributes.pop("bald__isPrefixedBy", None) 

73 

74 # Ensure the probabilistic coordinates are the first coordinates within 

75 # a cube and are in the specified order. 

76 enforce_coordinate_ordering(cube, ["realization", "percentile", "threshold"]) 

77 # Ensure the y and x dimensions are the last within the cube. 

78 y_name = cube.coord(axis="y").name() 

79 x_name = cube.coord(axis="x").name() 

80 enforce_coordinate_ordering(cube, [y_name, x_name], anchor_start=False) 

81 if no_lazy_load: 

82 # Force cube's data into memory by touching the .data attribute. 

83 cube.data 

84 

85 return cubes

88def load_cube( 

89 filepath: Union[str, List[str]], 

90 constraints: Optional[Union[Constraint, str]] = None, 

91 no_lazy_load: bool = False, 

92) -> Cube: 

93 """Load the filepath provided using Iris into a cube. Strips off all 

94 var names except for "threshold"-type coordinates, where this is different 

95 from the standard or long name. 

96 

97 Args: 

98 filepath: 

99 Filepath that will be loaded or list of filepaths that can be 

100 merged into a single cube. 

101 constraints: 

102 Constraint to be applied when loading from the input filepath. 

103 This can be in the form of an iris.Constraint or could be a string 

104 that is intended to match the name of the cube. 

105 The default is None. 

106 no_lazy_load: 

107 If True, bypass cube deferred (lazy) loading and load the whole 

108 cube into memory. This can increase performance at the cost of 

109 memory. If False (default) then lazy load. 

110 

111 Returns: 

112 Cube that has been loaded from the input filepath given the 

113 constraints provided. 

114 """ 

115 cubes = load_cubelist(filepath, constraints, no_lazy_load) 

116 # Merge loaded cubes 

117 if len(cubes) == 1: 

118 cube = cubes[0] 

119 else: 

120 cube = MergeCubes()(cubes) 

121 return cube

124def load_baseline_cube( 

125 reference_cube: Cube, value: float, name: str, units: str 

126) -> Cube: 

127 """Create a baseline cube with the structure of the reference cube but filled with a constant value. 

128 

129 Args: 

130 reference_cube: 

131 A cube to define the shape and data type of the baseline cube 

132 value: 

133 The value that will be populated as a constant across the baseline cube 

134 name: 

135 The name of the baseline cube 

136 units: 

137 The unit of measurement for the cube data 

138 

139 """ 

140 data = np.full(reference_cube.shape, value, dtype=reference_cube.core_data().dtype) 

141 

142 cube = reference_cube.copy(data=data) 

143 cube.rename(name) 

144 cube.units = units 

145 

146 return cube
