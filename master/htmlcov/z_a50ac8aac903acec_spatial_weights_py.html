22class SpatiallyVaryingWeightsFromMask(BasePlugin): 

23 """ 

24 Plugin for adjusting weights spatially based on masked data in the input 

25 cube. It takes in an initial one dimensional cube of weights which would 

26 be used to collapse a dimension on the input cube and outputs weights 

27 which have been adjusted based on the presence of missing data in x-y 

28 slices of input data. The resulting weights cube has a x and y dimensions 

29 in addition to the one dimension in the initial cube of weights. 

30 """ 

31 

32 def __init__(self, blend_coord: str, fuzzy_length: Union[int, float] = 10) -> None: 

33 """ 

34 Initialise class. 

35 

36 Args: 

37 blend_coord: 

38 Coordinate over which the input 1D weights will vary 

39 fuzzy_length: 

40 Distance, in grid squares, over which the weights from the input 

41 data mask are smoothed. This is used to calculate a fuzzy 

42 scaling factor for the input weights based on how far away each 

43 grid point is from a masked point. The distance is taken as 

44 the euclidean distance from the masked point, so the fuzzy 

45 length can be a non-integer value. Any points that are at least 

46 this distance away from a masked point keep a weight of one, 

47 and any points closer than this distance to a masked point have 

48 a weight of less than one based on how close to the masked 

49 point they are. 

50 """ 

51 self.fuzzy_length = fuzzy_length 

52 self.blend_coord = blend_coord 

53 self.blend_axis = None 

54 

55 def __repr__(self) -> str: 

56 """Represent the configured plugin instance as a string.""" 

57 result = "<SpatiallyVaryingWeightsFromMask: fuzzy_length: {}>".format( 

58 self.fuzzy_length 

59 ) 

60 return result

62 def _create_template_slice(self, cube_to_collapse: Cube) -> Cube: 

63 """ 

64 Create a template cube from a slice of the cube we are collapsing. 

65 The slice will be over blend_coord, y and x and will remove any other 

66 dimensions. This means that the resulting spatial weights won't vary in 

67 any other dimension other than the blend_coord. If the mask does 

68 vary in another dimension an error is raised. 

69 

70 Args: 

71 cube_to_collapse: 

72 The cube that will be collapsed along the blend_coord 

73 using the spatial weights generated using this plugin. Must 

74 be masked where there is invalid data. 

75 

76 Returns: 

77 A cube with dimensions blend_coord, y, x, on which to shape the 

78 output weights cube. 

79 

80 Raises: 

81 ValueError: if the blend coordinate is associated with more than 

82 one dimension on the cube to collapse, or no dimension 

83 ValueError: if the mask on cube_to_collapse varies along a 

84 dimension other than the dimension associated with 

85 blend_coord. 

86 """ 

87 self.blend_coord = find_blend_dim_coord(cube_to_collapse, self.blend_coord) 

88 

89 # Find original dim coords in input cube 

90 original_dim_coords = get_dim_coord_names(cube_to_collapse) 

91 

92 # Slice over required coords 

93 x_coord = cube_to_collapse.coord(axis="x").name() 

94 y_coord = cube_to_collapse.coord(axis="y").name() 

95 coords_to_slice_over = [self.blend_coord, y_coord, x_coord] 

96 

97 if original_dim_coords == coords_to_slice_over: 

98 return cube_to_collapse 

99 

100 # Check mask does not vary over additional dimensions 

101 slices = cube_to_collapse.slices(coords_to_slice_over) 

102 first_slice = next(slices) 

103 if np.ma.is_masked(first_slice.data): 

104 first_mask = first_slice.data.mask 

105 for cube_slice in slices: 

106 if not np.all(cube_slice.data.mask == first_mask): 

107 message = ( 

108 "The mask on the input cube can only vary along the " 

109 "blend_coord, differences in the mask were found " 

110 "along another dimension" 

111 ) 

112 raise ValueError(message) 

113 # Remove non-spatial non-blend dimensions, returning a 3D template cube without 

114 # additional scalar coordinates (eg realization) 

115 for coord in original_dim_coords: 

116 if coord not in coords_to_slice_over: 

117 first_slice.remove_coord(coord) 

118 # Return slice template 

119 return first_slice

121 def _normalise_initial_weights(self, weights: Cube) -> None: 

122 """Normalise weights so that they add up to 1 along the blend dimension 

123 at each spatial point. This is different from the normalisation that 

124 happens after the application of fuzzy smoothing near mask boundaries. 

125 Modifies weights cube in place. Array broadcasting relies on blend_coord 

126 being the leading dimension, as enforced in self._create_template_slice. 

127 

128 Args: 

129 weights: 

130 3D weights containing zeros for masked points, but before 

131 fuzzy smoothing 

132 """ 

133 weights_sum = np.sum(weights.data, axis=self.blend_axis) 

134 weights.data = np.where( 

135 weights_sum > 0, np.divide(weights.data, weights_sum), 0 

136 ).astype(FLOAT_DTYPE) 

137 

138 def _rescale_masked_weights(self, weights: Cube) -> Tuple[Cube, Cube]: 

139 """Apply fuzzy smoothing to weights at the edge of masked areas 

140 

141 Args: 

142 weights: 

143 Pre-normalised weights where the weights of masked data points 

144 have been set to 0 

145 

146 Returns: 

147 - Weights where MASKED slices have been rescaled, but UNMASKED 

148 slices have not 

149 - Binary (0/1) map showing which weights have been rescaled 

150 """ 

151 is_rescaled = iris.cube.CubeList() 

152 rescaled_weights = iris.cube.CubeList() 

153 for weights_slice in weights.slices_over(self.blend_coord): 

154 weights_nonzero = np.where(weights_slice.data > 0, True, False) 

155 if np.all(weights_nonzero): 

156 # if there are no masked points in this slice, keep current weights 

157 # and mark as unchanged (not rescaled) 

158 rescaled_weights.append(weights_slice) 

159 is_rescaled.append(weights_slice.copy(data=~weights_nonzero)) 

160 else: 

161 weights_orig = weights_slice.data.copy() 

162 

163 # calculate the distance to the nearest invalid point, in grid squares, 

164 # for each point on the grid 

165 distance = distance_transform_edt(weights_nonzero) 

166 

167 # calculate a 0-1 scaling factor based on the distance from the 

168 # nearest invalid data point, which scales between 1 at the fuzzy length 

169 # towards 0 for points closest to the edge of the mask 

170 fuzzy_factor = rescale( 

171 distance, data_range=[0.0, self.fuzzy_length], clip=True 

172 ) 

173 

174 # multiply existing weights by fuzzy scaling factor 

175 rescaled_weights_data = np.multiply( 

176 weights_slice.data, fuzzy_factor 

177 ).astype(FLOAT_DTYPE) 

178 rescaled_weights.append(weights_slice.copy(data=rescaled_weights_data)) 

179 

180 # identify spatial points where weights have been rescaled 

181 is_rescaled_data = np.where( 

182 rescaled_weights_data != weights_orig, True, False 

183 ) 

184 is_rescaled.append(weights_slice.copy(data=is_rescaled_data)) 

185 

186 weights = rescaled_weights.merge_cube() 

187 rescaled = is_rescaled.merge_cube() 

188 

189 return weights, rescaled

191 def _rescale_unmasked_weights(self, weights: Cube, is_rescaled: Cube) -> None: 

192 """Increase weights of unmasked slices at locations where masked slices 

193 have been smoothed, so that the sum of weights over self.blend_coord is 

194 re-normalised (sums to 1) at each point and the relative weightings of 

195 multiple unmasked slices are preserved. Modifies weights cube in place. 

196 

197 Args: 

198 weights: 

199 Cube of weights to which fuzzy smoothing has been applied to any 

200 masked slices 

201 is_rescaled: 

202 Cube matching weights.shape, with value of 1 where masked weights 

203 have been rescaled, and 0 where they are unchanged. 

204 """ 

205 rescaled_data = np.multiply(weights.data, is_rescaled.data) 

206 unscaled_data = np.multiply(weights.data, ~is_rescaled.data) 

207 unscaled_sum = np.sum(unscaled_data, axis=self.blend_axis) 

208 required_sum = 1.0 - np.sum(rescaled_data, axis=self.blend_axis) 

209 normalisation_factor = np.where( 

210 unscaled_sum > 0, np.divide(required_sum, unscaled_sum), 0 

211 ) 

212 normalised_weights = ( 

213 np.multiply(unscaled_data, normalisation_factor) + rescaled_data 

214 ) 

215 weights.data = normalised_weights.astype(FLOAT_DTYPE)

217 def process( 

218 self, cube_to_collapse: Cube, one_dimensional_weights_cube: Cube 

219 ) -> Cube: 

220 """ 

221 Create fuzzy spatial weights based on missing data in the cube we 

222 are going to collapse and combine these with 1D weights along the 

223 blend_coord. The method is as follows: 

224 

225 1. Broadcast 1D weights to the 3D shape of the input cube 

226 2. Set masked weights to zero. If there is no mask, return original 

227 unnormalised 1D weights with a warning message. 

228 3. Normalise 3D weights along the blend axis. This is needed so that a 

229 smooth spatial transition between layers can be achieved near mask boundaries. 

230 4. Reduce weights of masked layers near the mask boundary. 

231 5. Increase weights of unmasked layers near the mask boundary. 

232 

233 Args: 

234 cube_to_collapse: 

235 The cube that will be collapsed along self.blend_coord 

236 using the spatial weights generated using this plugin. Must 

237 be masked where there is invalid data. The mask may only 

238 vary along the blend and spatial coordinates, and not along 

239 any other dimensions on the cube. 

240 one_dimensional_weights_cube: 

241 A cube containing a single dimension coordinate with the same 

242 name given blend_coord. This cube contains 1D weights 

243 that will be applied along the blend_coord but need 

244 adjusting spatially based on missing data. 

245 

246 Returns: 

247 A cube containing normalised 3D spatial weights based on the 

248 cube_to_collapse mask and the one_dimensional weights supplied. 

249 Has dimensions: self.blend_coord, y, x. 

250 """ 

251 template_cube = self._create_template_slice(cube_to_collapse) 

252 (self.blend_axis,) = template_cube.coord_dims(self.blend_coord) 

253 

254 weights_data = iris.util.broadcast_to_shape( 

255 one_dimensional_weights_cube.data, template_cube.shape, (self.blend_axis,) 

256 ) 

257 weights = template_cube.copy(data=weights_data) 

258 

259 if np.ma.is_masked(template_cube.data): 

260 # Set masked weights to zero 

261 weights.data = np.where(template_cube.data.mask, 0, weights.data) 

262 else: 

263 message = "Expected masked input to SpatiallyVaryingWeightsFromMask" 

264 warnings.warn(message) 

265 return weights 

266 

267 self._normalise_initial_weights(weights) 

268 weights, rescaled = self._rescale_masked_weights(weights) 

269 self._rescale_unmasked_weights(weights, rescaled) 

270 

271 return weights
