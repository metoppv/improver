19class ExtractValueFromTable(BasePlugin): 

20 """Plugin to extract values from a table using the provided inputs cubes to select which 

21 rows and columns to extract. 

22 

23 The table is expected to contain numerical labels for every row and column. These labels will be 

24 sorted into ascending order. 

25 

26 The input cubes are used to select the row and column labels from the table 

27 with one cube used to select the rows and the other cube used to select the columns. The values 

28 in the table are then extracted based on the row and column labels selected from the cubes such 

29 that the nearest lower label from the table is used. 

30 

31 For Example: 

32 If the table has the following row labels of [0,1,2,3] and the row cube had a values of 1.9 then 

33 the row with label 1 would be selected. If the row cube had a value of 2 then the row with label 

34 2 would be selected. If the row cube had a value of -4 then the row with label 0 would be 

35 selected. 

36 """ 

37 

38 def __init__(self, row_name: str, new_name: str = None) -> None: 

39 """Initialise the plugin 

40 

41 Args: 

42 row_name: 

43 Name of the cube used for indexing rows. 

44 new_name: 

45 Optional new name for the resulting cube. 

46 """ 

47 self.row_name = row_name 

48 self.new_name = new_name 

49 

50 @staticmethod 

51 def nearest_lower_index( 

52 values: np.array, sorted_table_labels: np.array 

53 ) -> np.array: 

54 """Returns the index of the nearest lower label for every element of values. 

55 Args: 

56 values: 

57 Array of values to extract from table 

58 table_label: 

59 An array of the labels along an axis of the table. 

60 Returns: 

61 An array of indices of the nearest lower label for each value in values. 

62 """ 

63 

64 sorted_index = np.searchsorted(sorted_table_labels, values, side="right") - 1 

65 

66 # If floating points are being used in the table_labels and values numpy may not 

67 # recognise that the values are equal. This can lead to the wrong index being selected so 

68 # we need to check if the values are within a tolerance of the table_labels and correct the 

69 # index if necessary. 

70 tol = 1e-5 

71 condition = np.abs(np.array(sorted_table_labels)[:, None] - values) < tol 

72 true_index, index = np.where(condition) 

73 

74 sorted_index[index] = true_index 

75 

76 # If value is below lowest table_label then index will be -1. We need to correct this to 0. 

77 sorted_index = np.clip(sorted_index, 0, None) 

78 

79 return sorted_index 

80 

81 def extract_table_values( 

82 self, table: DataFrame, columns_cube: iris.cube.Cube, row_cube: iris.cube.Cube 

83 ) -> np.array: 

84 """Extract values from the table based on the provided row and column cubes. 

85 Args: 

86 table: 

87 DataFrame representing the table from which values are extracted. 

88 columns_cube: 

89 Cube used to index the columns of the table. 

90 row_cube: 

91 Cube used to index the rows of the table. 

92 Returns: 

93 Array of values extracted from the table. 

94 """ 

95 shape = columns_cube.shape 

96 

97 columns_data = columns_cube.data.flatten() 

98 row_data = row_cube.data.flatten() 

99 column_index = self.nearest_lower_index( 

100 values=columns_data, sorted_table_labels=table.columns 

101 ) 

102 row_index = self.nearest_lower_index( 

103 values=row_data, sorted_table_labels=table.index 

104 ) 

105 

106 result = table.to_numpy()[row_index, column_index] 

107 

108 nan_condition = np.logical_or(np.isnan(row_data), np.isnan(columns_data)) 

109 if nan_condition.any(): 

110 result = result.astype(np.float32) 

111 result[nan_condition] = np.nan 

112 

113 result = result.reshape(shape) 

114 return result

116 def convert_dict_to_dataframe(self, table: dict) -> DataFrame: 

117 """Converts a dictionary to a pandas DataFrame""" 

118 

119 table = DataFrame.from_dict(table) 

120 table.columns = table.columns.astype(float) 

121 table.index = table.index.astype(float) 

122 

123 table = table.reindex(sorted(table.columns), axis=1) 

124 table = table.reindex(sorted(table.index), axis=0) 

125 

126 return table

128 def process(self, *cubes: List[Cube], table: dict): 

129 """ 

130 Process the input cubes and extract values from a table based on the provided row and 

131 column indices. The row name is used to identify the cube used for indexing the rows 

132 of the table. The new name is an optional argument that can be used to rename the 

133 resulting cube. 

134 

135 Args: 

136 cubes: 

137 Input cubes for indexing columns and rows of the table. Exactly 2 cubes should 

138 be provided one which contains the values to extract from the rows and one for 

139 the columns. 

140 table: 

141 A dictionary representing the table from which values are extracted. Dictionary 

142 should be in the form: 

143 {"data":{column_name_1:{row_name_1:value, row_name_2:value},...}, 

144 "metadata":{"units":table_units}} 

145 Other metadata can be included in the metadata dictionary such as a title for 

146 the table but this will be ignored. 

147 

148 Returns: 

149 Cube of the same shape and metadata as the row input cube with values extracted 

150 from the table based on the row and column input cubes. The cube will be re-named 

151 if new_name is provided and the units updated to match the units of the table. 

152 

153 Raises: 

154 ValueError: 

155 - If exactly 2 cubes are not provided. 

156 - If the shapes of the column and row cubes do not match. 

157 """ 

158 cubes = as_cubelist(*cubes) 

159 

160 if len(cubes) != 2: 

161 raise ValueError( 

162 f"""Exactly 2 cubes should be provided, one for indexing columns and one for 

163 indexing rows. Provided cubes are {[cube.name() for cube in cubes]}""" 

164 ) 

165 

166 row_cube = cubes.extract_cube(self.row_name) 

167 cubes.remove(row_cube) 

168 column_cube = cubes[0] 

169 coord_order = [coord.name() for coord in column_cube.coords()] 

170 enforce_coordinate_ordering(row_cube, coord_order) 

171 

172 if column_cube.shape != row_cube.shape: 

173 raise ValueError( 

174 f"""Shapes of cubes do not match. Column cube shape: 

175 {column_cube.shape}, row cube shape: {row_cube.shape}""" 

176 ) 

177 

178 table_df = self.convert_dict_to_dataframe(table["data"]) 

179 result = self.extract_table_values(table_df, column_cube, row_cube) 

180 

181 if result.dtype == np.float64: 

182 result = result.astype(np.float32) 

183 result_cube = row_cube.copy(data=result) 

184 if self.new_name: 

185 result_cube.rename(self.new_name) 

186 result_cube.units = table["metadata"]["units"] 

187 return result_cube
