92def iris_time_to_datetime( 

93 time_coord: Coord, point_or_bound: str = "point" 

94) -> List[datetime]: 

95 """ 

96 Convert iris time to python datetime object. Working in UTC. 

97 

98 Args: 

99 time_coord: 

100 Iris time coordinate element(s). 

101 

102 Returns: 

103 The time element(s) recast as a python datetime object. 

104 """ 

105 coord = time_coord.copy() 

106 coord.convert_units("seconds since 1970-01-01 00:00:00") 

107 

108 if point_or_bound == "point": 

109 datetime_list = [value.point._to_real_datetime() for value in coord.cells()] 

110 elif point_or_bound == "bound": 

111 datetime_list = [ 

112 [c.bound[0]._to_real_datetime(), c.bound[1]._to_real_datetime()] 

113 for c in coord.cells() 

114 ] 

115 

116 return datetime_list

119def datetime_to_iris_time(dt_in: Union[datetime, DatetimeGregorian]) -> int64: 

120 """ 

121 Convert python datetime.datetime or cftime.DatetimeGregorian object into 

122 seconds since 1970-01-01 00Z. 

123 

124 Args: 

125 dt_in: 

126 Time to be converted into seconds since 1970-01-01 00Z. 

127 

128 Returns: 

129 Time since epoch in the seconds. 

130 """ 

131 if isinstance(dt_in, DatetimeGregorian): 

132 dt_in = datetime( 

133 dt_in.year, dt_in.month, dt_in.day, dt_in.hour, dt_in.minute, dt_in.second 

134 ) 

135 result = dt_in.replace(tzinfo=timezone.utc).timestamp() 

136 return np.int64(result)

139def datetime_constraint( 

140 time_in: datetime, time_max: Optional[datetime] = None 

141) -> Constraint: 

142 """ 

143 Constructs an iris equivalence constraint from a python datetime object. 

144 

145 Args: 

146 time_in: 

147 The time to be used to build an iris constraint. 

148 time_max: 

149 Optional max time, which if provided leads to a range constraint 

150 being returned up to < time_max. 

151 

152 Returns: 

153 An iris constraint to be used in extracting data at the given time 

154 from a cube. 

155 """ 

156 time_start = PartialDateTime(time_in.year, time_in.month, time_in.day, time_in.hour) 

157 

158 if time_max is None: 

159 time_extract = Constraint(time=lambda cell: cell.point == time_start) 

160 else: 

161 time_limit = PartialDateTime( 

162 time_max.year, time_max.month, time_max.day, time_max.hour 

163 ) 

164 time_extract = Constraint(time=lambda cell: time_start <= cell < time_limit) 

165 return time_extract

168def extract_cube_at_time( 

169 cubes: CubeList, time: datetime, time_extract: Constraint 

170) -> Cube: 

171 """ 

172 Extract a single cube at a given time from a cubelist. 

173 

174 Args: 

175 cubes: 

176 CubeList of a given diagnostic over several times. 

177 time: 

178 Time at which forecast data is needed. 

179 time_extract: 

180 Iris constraint for the desired time. 

181 

182 Returns: 

183 Cube of data at the desired time. 

184 

185 Raises: 

186 ValueError if the desired time is not available within the cubelist. 

187 """ 

188 try: 

189 (cube_in,) = cubes.extract(time_extract) 

190 return cube_in 

191 except ValueError: 

192 msg = "Forecast time {} not found within data cubes.".format( 

193 time.strftime("%Y-%m-%d:%H:%M") 

194 ) 

195 warnings.warn(msg) 

196 return None

199def extract_nearest_time_point( 

200 cube: Cube, dt: datetime, time_name: str = "time", allowed_dt_difference: int = 0 

201) -> Cube: 

202 """Find the nearest time point to the time point provided. 

203 

204 Args: 

205 cube: 

206 Cube or CubeList that will be extracted from using the supplied 

207 time_point 

208 dt: 

209 Datetime representation of a time that will be used within the 

210 extraction from the cube supplied. 

211 time_name: 

212 Name of the "time" coordinate that will be extracted. This must be 

213 "time" or "forecast_reference_time". 

214 allowed_dt_difference: 

215 An int in seconds to define a limit to the maximum difference 

216 between the datetime provided and the time points available within 

217 the cube. If this limit is exceeded, then an error is raised. 

218 This must be defined in seconds. 

219 Default is 0. 

220 

221 Returns: 

222 Cube following extraction to return the cube that is nearest 

223 to the time point supplied. 

224 

225 Raises: 

226 ValueError: The requested datetime is not available within the 

227 allowed difference. 

228 """ 

229 if time_name not in ["time", "forecast_reference_time"]: 

230 msg = ( 

231 "{} is not a valid time_name. " 

232 "The time_name must be either " 

233 "'time' or 'forecast_reference_time'." 

234 ) 

235 raise ValueError(msg) 

236 

237 time_point = datetime_to_iris_time(dt) 

238 time_point_index = cube.coord(time_name).nearest_neighbour_index(time_point) 

239 (nearest_dt,) = iris_time_to_datetime( 

240 cube.coord(time_name).copy()[time_point_index] 

241 ) 

242 if abs((dt - nearest_dt).total_seconds()) > allowed_dt_difference: 

243 msg = ( 

244 "The datetime {} is not available within the input " 

245 "cube within the allowed difference {} seconds. " 

246 "The nearest datetime available was {}".format( 

247 dt, allowed_dt_difference, nearest_dt 

248 ) 

249 ) 

250 raise ValueError(msg) 

251 constr = iris.Constraint(coord_values={time_name: nearest_dt}) 

252 cube = cube.extract(constr) 

253 return cube

256def relabel_to_period(cube: Cube, period: Optional[int] = None): 

257 """Add or replace bounds for the forecast period and time coordinates 

258 on a cube. 

259 

260 Args: 

261 cube: 

262 The cube for a diagnostic that will be modified to represent the 

263 required period. 

264 period: 

265 The period in hours. 

266 

267 Returns: 

268 Cube with metadata updated to represent the specified period. 

269 """ 

270 if period is None: 

271 msg = ( 

272 "A period must be specified when relabelling a diagnostic " 

273 "to have a particular period." 

274 ) 

275 raise ValueError(msg) 

276 elif period < 1: 

277 msg = ( 

278 "Only periods of one hour or greater are supported. " 

279 f"The period supplied was {period} hours." 

280 ) 

281 raise ValueError(msg) 

282 

283 for coord in ["forecast_period", "time"]: 

284 cube.coord(coord).bounds = np.array( 

285 [cube.coord(coord).points[0] - period * 3600, cube.coord(coord).points[0]], 

286 dtype=TIME_COORDS[coord].dtype, 

287 ) 

288 return cube

291def integrate_time(cube: Cube, new_name: str = None) -> Cube: 

292 """ 

293 Multiply a frequency or rate cube by the time period given by the 

294 time bounds over which it is defined to return a count or accumulation. 

295 The frequency or rate must be defined with time bounds, e.g. an average 

296 frequency across the period. This function will handle a cube with a 

297 non-scalar time coordinate, multiplying each time in the coordinate by the 

298 related bounds. 

299 

300 The returned cube has units equivalent to the input cube multiplied by 

301 seconds. 

302 

303 Any time related cell methods are removed from the output cube and a new 

304 "sum" over time cell method is added. 

305 

306 Args: 

307 Cube: 

308 A cube of average frequency or rate within a defined period. 

309 new_name: 

310 A new name for the resulting diagnostic. 

311 

312 Returns: 

313 The cube with the data multiplied by the period in seconds defined 

314 by the bounds on the time coordinate. 

315 

316 Raises: 

317 ValueError: If the input cube time coordinate does not have time 

318 bounds. 

319 """ 

320 # Ensure cube has a time coordinate with bounds 

321 if not cube.coord("time").has_bounds(): 

322 raise ValueError( 

323 "time coordinate must have bounds to apply this time-bounds integration" 

324 ) 

325 

326 # For each grid of data associated with a time, multiply the rate / frequency 

327 # by the associated time interval to get an accumulation / count over the 

328 # period. 

329 integrated_cube = iris.cube.CubeList() 

330 for cslice in cube.slices_over("time"): 

331 (multiplier,) = np.diff(cslice.coord("time").cell(0).bound) 

332 multiplier = multiplier.total_seconds() 

333 cslice.data *= multiplier 

334 integrated_cube.append(cslice) 

335 

336 integrated_cube = integrated_cube.merge_cube() 

337 

338 # Modify the cube units to reflect the multiplication by time. 

339 integrated_cube.units *= cf_units.Unit("s") 

340 if new_name is not None: 

341 integrated_cube.rename(new_name) 

342 

343 # Add a suitable cell method to describe what has been done and remove 

344 # former cell methods associated with the time coordinate which are now 

345 # out of date. 

346 new_cell_method = CellMethod("sum", coords=["time"]) 

347 new_cell_methods = [new_cell_method] 

348 for cm in integrated_cube.cell_methods: 

349 if "time" not in cm.coord_names: 

350 new_cell_methods.append(cm) 

351 

352 integrated_cube.cell_methods = new_cell_methods 

353 

354 return integrated_cube
