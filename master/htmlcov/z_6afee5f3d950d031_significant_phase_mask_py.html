5"""Module for calculating the significant phase mask.""" 

6 

7from typing import Optional 

8 

9import numpy as np 

10from iris.cube import Cube 

11from numpy import ndarray 

12 

13from improver import BasePlugin 

14from improver.metadata.utilities import ( 

15 create_new_diagnostic_cube, 

16 generate_mandatory_attributes, 

17) 

18 

19 

20class SignificantPhaseMask(BasePlugin): 

21 """ 

22 Derives a categorical field for the specified precipitation phase indicating whether 

23 that phase is the dominant phase at each point (1 where true, else 0) based on input 

24 snow-fraction data. 

25 The decision is: snow-fraction <= 0.01: Rain; snow-fraction >= 0.99: Snow; Sleet 

26 in between. 

27 """ 

28 

29 def __init__(self, model_id_attr: Optional[str] = None) -> None: 

30 """ 

31 Initialise the class 

32 

33 Args: 

34 model_id_attr: 

35 Name of the attribute used to identify the source model for 

36 blending. 

37 """ 

38 self.model_id_attr = model_id_attr 

39 self.lower_threshold = 0.01 

40 self.upper_threshold = 0.99 

41 self.phase_operator = { 

42 "rain": self._rain_phase, 

43 "sleet": self._sleet_phase, 

44 "snow": self._snow_phase, 

45 } 

46 

47 @staticmethod 

48 def _validate_snow_fraction(snow_fraction: Cube) -> None: 

49 """Ensures that the input snow-fraction field has appropriate name 

50 (snow_fraction), units (1) and data (between 0 and 1 inclusive). 

51 

52 Args: 

53 snow_fraction 

54 

55 Raises 

56 ValueError: 

57 If any of the above are not True. 

58 """ 

59 if snow_fraction.name() != "snow_fraction": 

60 raise ValueError( 

61 f"Expected cube named 'snow_fraction', not {snow_fraction.name()}" 

62 ) 

63 if f"{snow_fraction.units}" != "1": 

64 raise ValueError(f"Expected cube with units '1', not {snow_fraction.units}") 

65 if np.ma.is_masked(snow_fraction.data): 

66 raise NotImplementedError("SignificantPhaseMask cannot handle masked data") 

67 if np.any((snow_fraction.data < 0) | (snow_fraction.data > 1)): 

68 raise ValueError( 

69 f"Expected cube data to be in range 0 <= x <= 1. " 

70 f"Found max={snow_fraction.data.max()}; min={snow_fraction.data.min()}" 

71 ) 

72 

73 def _rain_phase(self, snow_fraction_data: Cube) -> ndarray: 

74 """Calculates the rain_phase data""" 

75 return np.where(snow_fraction_data <= self.lower_threshold, 1, 0) 

76 

77 def _snow_phase(self, snow_fraction_data: Cube) -> ndarray: 

78 """Calculates the snow_phase data""" 

79 return np.where(snow_fraction_data >= self.upper_threshold, 1, 0) 

80 

81 def _sleet_phase(self, snow_fraction_data: Cube) -> ndarray: 

82 """Calculates the sleet_phase data""" 

83 return np.where( 

84 (self.lower_threshold < snow_fraction_data) 

85 & (snow_fraction_data < self.upper_threshold), 

86 1, 

87 0, 

88 ) 

89 

90 def process(self, snow_fraction: Cube, phase: str) -> Cube: 

91 """ 

92 Make significant-phase-mask cube for the specified phase. 

93 

94 Args: 

95 snow_fraction: 

96 The input snow-fraction data to derive the phase mask from. 

97 phase: 

98 One of "rain", "sleet" or "snow". This is the phase mask that will be 

99 returned. 

100 

101 Returns: 

102 The requested phase mask containing 1 where that phase is dominant 

103 and 0 elsewhere. Dimensions will be identical to snow-fraction. 

104 """ 

105 self._validate_snow_fraction(snow_fraction) 

106 

107 try: 

108 data = self.phase_operator[phase](snow_fraction.data).astype(np.int8) 

109 except KeyError: 

110 raise KeyError( 

111 f"Requested phase mask '{phase}' not in {list(self.phase_operator.keys())}" 

112 ) 

113 phase_mask = create_new_diagnostic_cube( 

114 f"{phase}_mask", 

115 "1", 

116 snow_fraction, 

117 generate_mandatory_attributes( 

118 [snow_fraction], model_id_attr=self.model_id_attr 

119 ), 

120 data=data, 

121 ) 

122 return phase_mask
