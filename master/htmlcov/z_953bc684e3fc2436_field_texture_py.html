5"""Script to calculate whether or not the input field texture exceeds a given threshold.""" 

6 

7from improver import cli

9 

10@cli.clizefy 

11@cli.with_output 

12def process( 

13 cube: cli.inputcube, 

14 *, 

15 nbhood_radius: float = 20000.0, 

16 textural_threshold: float = 0.05, 

17 diagnostic_threshold: float = 0.8125,

19): 

20 """Calculates field texture for a given neighbourhood radius. 

21 

22 The field texture is an assessment of the transitions/edges within a 

23 neighbourhood of a grid point to indicate whether the field is rough 

24 or smooth. 

25 

26 Args: 

27 cube (iris.cube.Cube): 

28 The diagnostic for which texture is to be assessed. For example cloud 

29 area fraction where transitions between cloudy regions and cloudless 

30 regions will be diagnosed. Defaults set assuming cloud area fraction 

31 cube. 

32 

33 nbhood_radius (float): 

34 The neighbourhood radius in metres within which the number of potential 

35 transitions should be calculated. This forms the denominator in the 

36 calculation of the ratio of actual to potential transitions that indicates a 

37 field's texture. A larger radius should be used for diagnosing larger-scale 

38 textural features. Default value set to 10000.0 assuming cloud area fraction 

39 cube. 

40 

41 textural_threshold (float): 

42 A unit-less threshold value that defines the ratio value above which 

43 the field is considered rough and below which the field is considered 

44 smoother. Default value set to 0.05 assuming cloud area fraction cube. 

45 

46 diagnostic_threshold (float): 

47 The diagnostic threshold for which field texture will be calculated. 

48 A ValueError is raised if this threshold is not present on the input 

49 cube. Default value set to 0.8125 corresponding to 6 oktas, assuming 

50 cloud area fraction cube. 

51 

52 model_id_attr (str): 

53 Name of the attribute used to identify the source model for 

54 blending. 

55 

56 Returns: 

57 iris.cube.Cube: 

58 A field texture cube containing values between 0 and 1, where 0 

59 indicates no transitions and 1 indicates the maximum number of 

60 possible transitions has been achieved, within the neighbourhood of 

61 the grid point. 

62 """ 

63 

64 from improver.utilities.textural import FieldTexture 

65 

66 field_texture = FieldTexture( 

67 nbhood_radius, 

68 textural_threshold, 

69 diagnostic_threshold, 

70 model_id_attr=model_id_attr, 

71 )(cube) 

72 return field_texture
