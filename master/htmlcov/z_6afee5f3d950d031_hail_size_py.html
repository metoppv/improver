30class HailSize(BasePlugin): 

31 """Plugin to calculate the diameter of the hail stones from input cubes 

32 cloud condensation level (ccl) temperature, cloud condensation level pressure, 

33 temperature on pressure levels, the height of the wet bulb freezing level 

34 above sea level and orography. 

35 

36 From these, the values for three other cubes are calculated: 

37 - Temperature of the environment at 268.15K (-5 Celsius) and the 

38 pressure level where this occurs. 

39 - Temperature after a saturated ascent from ccl pressure to the 

40 pressure of the environment at 268.15K (-5 Celsius). 

41 - Temperature after a dry adiabatic descent from the pressure of the 

42 environment at 268.15K (-5 Celsius) to the ccl pressure. 

43 

44 From these, two indexes are calculated as: 

45 - Temperature after a dry adiabatic descent - the temperature of the 

46 atmosphere at 268.15K 

47 - Temperature after a saturated ascent - the temperature of the 

48 atmosphere at 268.15K 

49 

50 These indexes can then be used to extract values of hail size depending on 

51 the wet bulb freezing altitude. The wet bulb freezing altitude is calculated 

52 by subtracting the orography from the wet bulb freezing altitude above sea level. 

53 

54 If the wet bulb freezing altitude is between 3350m and 4400m then the indexes are used 

55 to extract an initial hail value from the first table. A second table 

56 is then accessed to reduce the hail size. The second table is stored as a dictionary, with the 

57 key being the wet bulb freezing altitude and each column in the associated 

58 arrays referring to the previously calculated hail diameter being less than a pre-defined 

59 value. An updated hail_size is then extracted and stored. 

60 

61 If the wet_bulb_freezing_altitude is greater than 4400m then the hail size is set to 

62 0 and if the wet bulb_freezing_altitude is less than 3350m then the originally calculated 

63 hail size is not altered. 

64 

65 Both tables are taken from Hand and Cappelluti (2011) which are a tabular versions of the 

66 graphs from Fawbush and Miller (1953) 

67 

68 If the cloud condensation level is invalid, or masked, this indicates no convection and 

69 the hail size will be zero. 

70 

71 

72 References 

73 - Hand, W., and G. Cappelluti. 2011. “A global hail climatology using the UK 

74 Met Office convection diagnosis procedure (CDP) and model analyses.” 

75 Meteorological Applications 18: 446-458. https://doi.org/10.1002/met.236 

76 - Fawbush, E.J., and R.C. Miller. 1953. “A method for forecasting hailstone size 

77 at the earth's surface.” Bulletin of the American Meteorological Society 34: 235-244. 

78 https://doi.org/10.1175/1520-0477-34.6.235 

79 """

96 def nomogram_values() -> np.ndarray: 

97 """Sets-up an array of a table containing possible diameter of hail stones(mm). 

98 It is a transposed version of the table in Hand and Cappelluti (2011). 

99 

100 The axes of the table are as follows: 

101 

102 - Horizontal axis is calculated from two values: the temperature after a 

103 dry adiabatic descent from the pressure of atmosphere at 268.15K to the 

104 cloud condensation level pressure and the temperature of the atmosphere 

105 at 268.15K. Each column represents a value calculated as the temperature 

106 after the dry adiabatic descent minus the temperature of the atmosphere 

107 at 268.15K rounded to the nearest 0.5K. 

108 - The vertical axis is also calculated from two values: the temperature after 

109 a saturated ascent from the ccl pressure to the pressure of environment at 

110 268.15K and the temperature of the atmosphere at 268.25K. 

111 Each row is represented by a value calculated as the temperature after 

112 the saturated ascent minus the temperature of the atmosphere at 268.15K 

113 rounded to the nearest 5K. 

114 """ 

115 

116 lookup_nomogram = np.array( 

117 [ 

118 [0, 0, 0, 2, 2, 5, 5, 5, 5, 5], 

119 [0, 0, 0, 2, 5, 5, 5, 5, 10, 10], 

120 [0, 0, 2, 2, 5, 5, 10, 10, 15, 15], 

121 [0, 0, 2, 2, 5, 10, 15, 15, 20, 20], 

122 [0, 0, 2, 2, 10, 15, 20, 20, 20, 20], 

123 [0, 2, 2, 5, 15, 20, 20, 20, 25, 25], 

124 [0, 2, 5, 10, 20, 20, 25, 25, 30, 30], 

125 [2, 2, 10, 15, 20, 25, 30, 30, 35, 35], 

126 [2, 5, 10, 20, 25, 30, 35, 35, 40, 40], 

127 [2, 5, 15, 20, 30, 35, 40, 40, 45, 45], 

128 [2, 5, 15, 20, 30, 40, 40, 40, 45, 50], 

129 [2, 10, 20, 25, 35, 40, 45, 45, 50, 50],

150 @staticmethod 

151 def updated_nomogram() -> Tuple[List, List, np.array]: 

152 """Sets up a dictionary of updated hail diameter values (mm). 

153 

154 The dictionary keys are the height of the wet bulb freezing level (m) where, 

155 when accessing at some height value, it should be rounded to the nearest lower value 

156 (e.g. 3549m should access 3350m key). 

157 

158 Each key has an associated list in which each element is a new hail diameter based on 

159 the original hail size that was calculated from nomogram_values table. 

160 Specifically each column associated hail size (mm) is [<5,<10,<20,<25,<50,<75,<100,<125]. 

161 The largest possible value where the equality still holds should be used. 

162 

163 If the wet bulb freezing height is less than 3350m then the original hail size is used. 

164 If the wet bulb freezing height is greater than 4400m then all hail sizes are set to 0. 

165 """ 

166 lookup_dict = { 

167 3350: [0, 5, 10, 15, 25, 50, 65, 75], 

168 3550: [0, 0, 5, 10, 20, 20, 25, 30], 

169 3750: [0, 0, 0, 5, 10, 15, 15, 15], 

170 3950: [0, 0, 0, 0, 5, 10, 10, 10], 

171 4150: [0, 0, 0, 0, 0, 0, 5, 5], 

172 4400: [0, 0, 0, 0, 0, 0, 0, 0], 

173 } 

174 hail_groups = [5, 10, 20, 25, 50, 75, 100, 125] 

175 

176 return ( 

177 list(lookup_dict.keys()), 

178 hail_groups, 

179 np.array(list(lookup_dict.values())), 

180 )

190 """Checks the size and units of input cubes and enforces the standard coord order 

191 

192 Args: 

193 ccl_temperature:

196 Cube of cloud condensation level pressure

198 Cube of environment temperature on pressure levels

205 temperature_on_pressure.coord(axis=axis).name() for axis in "yx"

208 for cube in [ 

209 ccl_temperature, 

210 ccl_pressure, 

211 temperature_on_pressure, 

212 wet_bulb_zero_asl, 

213 orography, 

214 ]: 

215 enforce_coordinate_ordering(cube, coord_order) 

216 

217 temp_slice = next(temperature_on_pressure.slices_over("pressure")) 

218 try: 

219 wb_slice = next(wet_bulb_zero_asl.slices_over("realization")) 

220 except CoordinateNotFoundError: 

221 wb_slice = wet_bulb_zero_asl 

222 assert_spatial_coords_match([wb_slice, orography])

227 ccl_temperature.convert_units("K") 

228 ccl_pressure.convert_units("Pa") 

229 temperature_on_pressure.convert_units("K")

288 def get_hail_size( 

289 self, vertical: np.ndarray, horizontal: np.ndarray, wet_bulb_zero: np.ndarray 

290 ) -> np.ndarray: 

291 """Uses the lookup_table and the vertical and horizontal indexes calculated 

292 to extract and store values from the lookup nomogram. 

293 

294 The hail size will be set to 0 if 

295 1) there are masked data points, 

296 2) vertical or horizontal values are negative, 

297 3) the wet bulb freezing altitude is greater that 4400m. 

298 

299 If the wet bulb freezing altitude is greater that 3300m then the hail_size is reduced. 

300 

301 Args: 

302 vertical: 

303 An n dimensional array containing the values used to calculate the vertical indexes 

304 horizontal: 

305 An n dimensional array containing the values used to calculate the horizontal 

306 indexes 

307 wet_bulb_zero: 

308 An n dimensional array containing the height of the wet bulb freezing level 

309 Returns: 

310 an n dimension array of values for the diameter of hail (mm) 

311 """ 

312 

313 lookup_table = self.nomogram_values() 

314 

315 # Rounds the calculated horizontal value to the nearest 5 which is 

316 # then turned into a relevant index for accessing the appropriate column. 

317 # Rounds the calculated vertical values to the nearest 0.5 which is then 

318 # turned into a relevant index for accessing the appropriate row. 

319 horizontal_rounded = np.around(horizontal / 5, decimals=0) - 1 

320 vertical_rounded = np.around(vertical * 2, decimals=0) 

321 

322 # clips index values to not be longer than the table 

323 vertical_clipped = np.clip(vertical_rounded, None, len(lookup_table) - 1) 

324 horizontal_clipped = np.clip(horizontal_rounded, None, len(lookup_table[0]) - 1) 

325 

326 vertical_clipped = np.ma.where( 

327 (vertical_rounded >= 0) & (horizontal_rounded >= 0), vertical_clipped, 0 

328 ).filled(0) 

329 horizontal_clipped = np.ma.where( 

330 (vertical_rounded >= 0) & (horizontal_rounded >= 0), horizontal_clipped, 0 

331 ).filled(0) 

332 

333 hail_size = lookup_table[ 

334 vertical_clipped.astype(int), horizontal_clipped.astype(int) 

335 ] 

336 hail_size = np.where( 

337 wet_bulb_zero >= 3300, 

338 self.updated_hail_size(hail_size, wet_bulb_zero), 

339 hail_size, 

340 ) 

341 return hail_size

343 def updated_hail_size( 

344 self, hail_size: np.array, wet_bulb_height: np.array 

345 ) -> np.array: 

346 """Uses the updated_nomogram values dictionary to access an updated hail size 

347 based on the original predicted hail size and a wet bulb freezing height. 

348 

349 Args: 

350 hail_size: 

351 Integers of hail diameter value taken from the original nomogram 

352 wet_bulb_height: 

353 Floats of the height of the wet bulb freezing level 

354 Returns: 

355 An updated value for the hail diameter (mm) 

356 """ 

357 

358 vectorised = np.vectorize(lambda n: bisect_right(self._wbzh_keys, n)) 

359 height_index = np.array(vectorised(wet_bulb_height) - 1).astype(int) 

360 

361 vectorised = np.vectorize(lambda n: bisect_right(self._hail_groups, n)) 

362 hail_index = vectorised(hail_size) 

363 

364 updated_hail_size = self._updated_values[height_index, hail_index] 

365 

366 return np.int8(updated_hail_size)

368 def hail_size_data( 

369 self, 

370 temperature_at_268: Cube, 

371 pressure_at_268: Cube, 

372 ccl_pressure: Cube, 

373 ccl_temperature: Cube, 

374 humidity_mixing_ratio_at_ccl: np.array, 

375 wet_bulb_zero: Cube, 

376 ) -> np.ndarray: 

377 """Gets temperature of environment at 268.15K, temperature after a dry adiabatic descent 

378 from the pressure of air at 268.15K to ccl pressure and the temperature 

379 after a saturated ascent from ccl pressure to the pressure of air at 268.15K. 

380 From these values it calculates vertical and horizontal indices. It also masks 

381 data where the ccl_temperature is below 268.15K. 

382 

383 Args: 

384 temperature_at_268: 

385 Cube of the temperature of the environment at 268.15K 

386 pressure_at_268: 

387 Cube of the pressure of the environment at 268.15K 

388 ccl_pressure: 

389 Cube of cloud condensation level pressure 

390 ccl_temperature: 

391 Cube of cloud condensation level pressure 

392 humidity_mixing_ratio_at_ccl: 

393 Array of humidity mixing ratio at the pressure of the environment at the CCL 

394 wet_bulb_zero: 

395 Cube of the height of the wet-bulb freezing level 

396 Returns: 

397 An n dimensional array of diameter of hail stones (m) 

398 """ 

399 

400 # temperature_at_268 is big-B in Hand (2011). 

401 # ccl_temperature is big-C in Hand (2011). 

402 # temp_dry is little-c in Hand (2011). 

403 temp_dry = self.dry_adiabatic_descent_to_ccl( 

404 ccl_pressure, temperature_at_268, pressure_at_268 

405 ) 

406 

407 # temp_saturated_ascent is little-b in Hand (2011). 

408 temp_saturated_ascent = self.temperature_after_saturated_ascent_from_ccl( 

409 ccl_temperature, ccl_pressure, pressure_at_268, humidity_mixing_ratio_at_ccl 

410 ) 

411 

412 # horizontal is c - B in Hand (2011). 

413 horizontal = temp_dry.data - temperature_at_268.data 

414 # vertical is b - B in Hand (2011). 

415 vertical = temp_saturated_ascent.data - temperature_at_268.data 

416 

417 temperature_mask = np.ma.masked_less(ccl_temperature.data, 268.15) 

418 vertical_masked = np.ma.masked_where(np.ma.getmask(temperature_mask), vertical) 

419 horizontal_masked = np.ma.masked_where( 

420 np.ma.getmask(temperature_mask), horizontal 

421 ) 

422 

423 hail_size = self.get_hail_size( 

424 vertical_masked, horizontal_masked, wet_bulb_zero.data 

425 ) 

426 hail_size = hail_size / 1000 

427 hail_size = hail_size.astype("float32") 

428 

429 return hail_size

466 def process(self, *cubes: Union[Cube, CubeList]) -> Cube: 

467 """ 

468 Main entry point of this class 

469 

470 Args: 

471 cubes 

472 air_temperature: 

473 Cube of the cloud condensation level temperature 

474 air_pressure_at_condensation_level: 

475 Cube of the cloud condensation level pressure. 

476 air_temperature_at_condensation_level: 

477 Cube of temperature on pressure levels 

478 wet_bulb_freezing_level_altitude: 

479 Cube of the height of the wet-bulb freezing level above sea level 

480 surface_altitude: 

481 Cube of the orography height. 

482 Returns: 

483 Cube of hail diameter (m) 

484 """ 

485 cubes = as_cubelist(*cubes) 

486 ( 

487 temperature_on_pressure, 

488 ccl_pressure, 

489 ccl_temperature, 

490 wet_bulb_zero_height_asl, 

491 orography, 

492 ) = cubes.extract( 

493 [ 

494 "air_temperature", 

495 "air_pressure_at_condensation_level", 

496 "air_temperature_at_condensation_level", 

497 "wet_bulb_freezing_level_altitude", 

498 "surface_altitude", 

499 ] 

500 ) 

501 

502 self.check_cubes( 

503 ccl_temperature, 

504 ccl_pressure, 

505 temperature_on_pressure, 

506 wet_bulb_zero_height_asl, 

507 orography, 

508 ) 

509 extract_pressure = ExtractLevel( 

510 value_of_level=268.15, positive_correlation=True 

511 ) 

512 pressure_at_268 = extract_pressure(temperature_on_pressure) 

513 

514 temperature_at_268 = next(temperature_on_pressure.slices_over(["pressure"])) 

515 temperature_at_268.rename("temperature_of_atmosphere_at_268.15K") 

516 temperature_at_268.remove_coord("pressure") 

517 temperature = np.full_like( 

518 temperature_at_268.data, extract_pressure.value_of_level, dtype=np.float32 

519 ) 

520 temperature = np.ma.masked_where(np.ma.getmask(pressure_at_268), temperature) 

521 temperature_at_268.data = temperature 

522 

523 attributes = {} 

524 if self.model_id_attr: 

525 attributes[self.model_id_attr] = temperature_on_pressure.attributes[ 

526 self.model_id_attr 

527 ] 

528 del temperature_on_pressure 

529 

530 humidity_mixing_ratio_at_ccl = saturated_humidity( 

531 ccl_temperature.data, ccl_pressure.data 

532 ) 

533 

534 wet_bulb_zero_height = wet_bulb_zero_height_asl - orography 

535 

536 hail_size = self.hail_size_data( 

537 temperature_at_268, 

538 pressure_at_268, 

539 ccl_pressure, 

540 ccl_temperature, 

541 humidity_mixing_ratio_at_ccl, 

542 wet_bulb_zero_height, 

543 ) 

544 

545 hail_cube = self.make_hail_cube( 

546 hail_size, ccl_temperature, ccl_pressure, attributes 

547 ) 

548 enforce_coordinate_ordering(hail_cube, self.final_order) 

549 return hail_cube
