6"""Script to calculate continuous phase change level.""" 

7 

8from improver import cli 

9 

10 

11@cli.clizefy 

12@cli.with_output 

13def process( 

14 *cubes: cli.inputcube_nolazy, 

15 phase_change, 

16 grid_point_radius=2, 

17 horizontal_interpolation=True, 

18 model_id_attr: str = None, 

19): 

20 """Height of precipitation phase change relative to sea level. 

21 

22 Calculated as a continuous 2D field by finding the height above sea level 

23 at which the integral of wet bulb temperature matches an empirical 

24 threshold that is expected to correspond with the phase change. 

25 

26 Args: 

27 cubes (iris.cube.CubeList or list of iris.cube.Cube): 

28 containing: 

29 wet_bulb_temperature (iris.cube.Cube): 

30 Cube of wet bulb temperatures on height levels. 

31 wet_bulb_integral (iris.cube.Cube): 

32 Cube of wet bulb temperature integrals calculated 

33 vertically downwards to height levels. 

34 orography (iris.cube.Cube): 

35 Cube of the orography height in m. 

36 land_sea_mask (iris.cube.Cube): 

37 Cube containing the binary land-sea mask. Land points are 

38 set to 1, sea points are set to 0. 

39 phase_change (str): 

40 The desired phase change for which the altitude should be 

41 returned. Options are: 

42 

43 snow-sleet - the melting of snow to sleet. 

44 sleet-rain - the melting of sleet to rain. 

45 hail-rain - the melting of hail to rain. 

46 grid_point_radius (int): 

47 The radius in grid points used to calculate the maximum 

48 height of the orography in a neighbourhood to determine points that 

49 should be excluded from interpolation for being too close to the 

50 orographic feature where high-resolution models can give highly 

51 localised results. Zero uses central point only (neighbourhood is disabled). 

52 One uses central point and one in each direction. Two goes two points etc. 

53 

54 horizontal_interpolation (bool): 

55 If True apply horizontal interpolation to fill in holes in 

56 the returned phase-change-level that occur because the level 

57 falls below the orography. If False these areas will be masked. 

58 model_id_attr (str): 

59 Name of the attribute used to identify the source model for blending. 

60 

61 Returns: 

62 iris.cube.Cube: 

63 Processed Cube of phase change altitude relative to sea level. 

64 """ 

65 from improver.psychrometric_calculations.psychrometric_calculations import ( 

66 PhaseChangeLevel, 

67 ) 

68 

69 plugin = PhaseChangeLevel( 

70 phase_change=phase_change, 

71 grid_point_radius=grid_point_radius, 

72 horizontal_interpolation=horizontal_interpolation, 

73 model_id_attr=model_id_attr, 

74 ) 

75 result = plugin(*cubes) 

76 return result
